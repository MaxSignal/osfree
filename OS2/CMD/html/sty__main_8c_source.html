81/****************************************************************
82 * Name: main()
83 *
84 * Description: Entry point of program.
85 *
86 * Concepts: Obtains anchor block handle and creates message
87 * queue. Calls the initialization routine.
88 * Creates the main frame window which creates the
89 * main client window. Polls the message queue
90 * via Get/Dispatch Msg loop. Upon exiting the
91 * loop, exits.
92 *
93 * API's : WinInitilize
94 * DosBeep
95 * WinCreateMsgQueue
96 * WinTerminate
97 * WinCreateStdWindow
98 * WinSetWindowText
99 * WinGetMsg
100 * WinDispatchMsg
101 *
102 * Parameters: None
103 *
104 * Returns:
105 * 1 - if sucessful execution completed
106 * 0 - if error
107\****************************************************************/
108int main (VOID)
109{
110 QMSG qmsg; /* message structure */
111 ULONG ulCtlData; /* frame control data */
112
113 hab = WinInitialize(0);
114
115 if(!hab)
116 {
117 DosBeep(BEEP_WARN_FREQ, BEEP_WARN_DUR);
118 return(RETURN_ERROR);
119 }
120
121 hmq = WinCreateMsgQueue(hab, 0);
122
123 if(!hmq)
124 {
125 DosBeep(BEEP_WARN_FREQ, BEEP_WARN_DUR);
126 WinTerminate(hab);
127 return(RETURN_ERROR);
128 }
129
130 if(!Init())
131 {
132 MessageBox(HWND_DESKTOP, IDMSG_INITFAILED, MB_OK | MB_ERROR, TRUE);
133 return(RETURN_ERROR);
134 }
135
136 /* NOTE: clean up from here is handled by the DosExitList processing */
137
138 /* create the main window */
139 ulCtlData = FCF_STANDARD;
140
141 hwndMainFrame = WinCreateStdWindow(HWND_DESKTOP, WS_VISIBLE,
142 (PVOID)&ulCtlData, (PSZ)szAppName,
143 (PSZ)NULL, WS_VISIBLE,
144 (HMODULE)NULL, IDR_MAIN,
145 (PHWND)&hwndMain);
146
147 if(!hwndMainFrame)
148 {
149 MessageBox(HWND_DESKTOP, IDMSG_MAINWINCREATEFAILED, MB_OK | MB_ERROR,
150 TRUE);
151 return(RETURN_ERROR);
152 }
153
154 WinSetWindowText(hwndMainFrame, szAppName);
155
156 InitHelp();
157
158 /* Get/Dispatch Message loop */
159 while(WinGetMsg(hab, (PQMSG)&qmsg, 0, 0, 0))
160 WinDispatchMsg(hab, (PQMSG)&qmsg);
161
162 DestroyHelpInstance();
163
164 return(RETURN_SUCCESS);
165
166} /* End of main porcedure */

170/****************************************************************
171 * Name: MainWndProc
172 *
173 * Description : Window procedure for the main clent window.
174 *
175 * Concepts : Processes the messages sent to the main client
176 * window. This routine processes the basic
177 * messages all client windows should process and
178 * passes all others onto UserWndProc where the
179 * developer can process any others.
180 *
181 * API's : None
182 *
183 * Parameters : hwnd - Window handle to which message is addressed
184 * msg - Message type
185 * mp1 - First message parameter
186 * mp2 - Second message parameter
187 *
188 * Returns: Return values are determined by each message
189 *
190 ****************************************************************/
191MRESULT EXPENTRY MainWndProc(HWND hwnd, USHORT msg, MPARAM mp1, MPARAM mp2)
192{
193 switch(msg)
194 {
195 case WM_CREATE:
196 return(InitMainWindow(hwnd, mp1, mp2));
197 break;
198
199 case WM_PAINT:
200 MainPaint(hwnd);
201 break;
202
203 case WM_COMMAND:
204 MainCommand(mp1, mp2);
205 break;
206
207 case WM_INITMENU:
208 InitMenu(mp1, mp2);
209 break;
210
211 case HM_QUERY_KEYS_HELP:
212 return (MRESULT)PANEL_HELPKEYS; /* return id of key help panel */
213 break;
214
215 /*
216 * Any messages not processed are passed on
217 * to the user's window proc. It is
218 * responsible for passing any messages it
219 * doesn't handle onto WinDefWindowProc()
220 */
221 default:
222 return(UserWndProc(hwnd, msg, mp1, mp2));
223 break;
224
225 }
226 return (MRESULT)0; /* all window procedures should return 0 as a default */
227
228} /* End of MainWndProc() */

230/*********************************************************************
231 * Name: MessageBox
232 *
233 * Description : Message Box procedure
234 *
235 * Concepts : Displays the warning message box with the message
236 * given in idMsg retrived from the message table;
237 * Called wherever an error occurs and a message should
238 * be displayed to the user
239 *
240 * API's : WinLoadString
241 * WinLoadMessage
242 * WinAlarm
243 * WinMessageBox
244 *
245 * Parameters : hwndOwner - window handle of the owner of the
246 * message box
247 * idMsg - id of message to be retrieved from
248 * resource file
249 * fsStyle - message box style
250 * fBeep - flag to sound alarm
251 *
252 * Returns: SHORT
253 *
254\****************************************************************/
255SHORT MessageBox(HWND hwndOwner, SHORT idMsg, SHORT fsStyle, BOOL fBeep)
256{
257 CHAR szText[MESSAGELEN];
258 CHAR szTitle[MESSAGELEN];
259 PSZ pszTitle = NULL;
260
261 if(!WinLoadMessage(hab, (HMODULE)NULL, idMsg, MESSAGELEN, (PSZ)szText))
262 {
263 WinAlarm(HWND_DESKTOP, WA_ERROR);
264 return MBID_ERROR;
265 }
266
267 if(fBeep)
268 WinAlarm(HWND_DESKTOP, WA_ERROR);
269
270 if ( (fsStyle & MB_WARNING) /* use warning title ? */
271 && WinLoadString(hab, (HMODULE)0, IDS_WARNING, MESSAGELEN, szTitle) )
272 pszTitle = szTitle;
273 else
274 pszTitle = NULL; /* default to error title */
275
276 return(WinMessageBox(HWND_DESKTOP, hwndOwner, szText, pszTitle,
277 IDD_MSGBOX, fsStyle));
278
279} /* End of MessageBox() */
