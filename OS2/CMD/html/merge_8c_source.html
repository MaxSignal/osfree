24/*
25 * Attempt to merge two adjacent segments.
26 * Currently only supports striped segments on AREA_PV.
27 * Returns success if successful, in which case 'first'
28 * gets adjusted to contain both areas.
29 */
30static int _merge(struct lv_segment *first, struct lv_segment *second)
31{
32 if (!first || !second || first->segtype != second->segtype ||
33 !first->segtype->ops->merge_segments) return 0;
34
35 return first->segtype->ops->merge_segments(first, second);
36}

38int lv_merge_segments(struct logical_volume *lv)
39{
40 struct list *segh, *t;
41 struct lv_segment *current, *prev = NULL;
42
43 if (lv->status & LOCKED || lv->status & PVMOVE)
44 return 1;
45
46 list_iterate_safe(segh, t, &lv->segments) {
47 current = list_item(segh, struct lv_segment);
48
49 if (_merge(prev, current))
50 list_del(&current->list);
51 else
52 prev = current;
53 }
54
55 return 1;
56}

58/*
59 * Verify that an LV's segments are consecutive, complete and don't overlap.
60 */
61int check_lv_segments(struct logical_volume *lv, int complete_vg)
62{
63 struct lv_segment *seg, *seg2;
64 uint32_t le = 0;
65 unsigned seg_count = 0;
66 int r = 1;
67 uint32_t area_multiplier, s;
68
69 //list_iterate_items(seg, &lv->segments) {
70 list_iterate_items(seg, struct lv_segment, &lv->segments) {
71 seg_count++;
72 if (seg->le != le) {
73 log_error("LV %s invalid: segment %u should begin at "
74 "LE %" PRIu32 " (found %" PRIu32 ").",
75 lv->name, seg_count, le, seg->le);
76 r = 0;
77 }
78
79 area_multiplier = segtype_is_striped(seg->segtype) ?
80 seg->area_count : 1;
81
82 if (seg->area_len * area_multiplier != seg->len) {
83 log_error("LV %s: segment %u has inconsistent "
84 "area_len %u",
85 lv->name, seg_count, seg->area_len);
86 r = 0;
87 }
88
89 if (complete_vg && seg->log_lv) {
90 if (!seg_is_mirrored(seg)) {
91 log_error("LV %s: segment %u has log LV but "
92 "is not mirrored",
93 lv->name, seg_count);
94 r = 0;
95 }
96
97 if (!(seg->log_lv->status & MIRROR_LOG)) {
98 log_error("LV %s: segment %u log LV %s is not "
99 "a mirror log",
100 lv->name, seg_count, seg->log_lv->name);
101 r = 0;
102 }
103
104 if (!(seg2 = first_seg(seg->log_lv)) ||
105 seg2->mirror_seg != seg) {
106 log_error("LV %s: segment %u log LV does not "
107 "point back to mirror segment",
108 lv->name, seg_count);
109 r = 0;
110 }
111 }
112
113 if (complete_vg && seg->status & MIRROR_IMAGE) {
114 if (!seg->mirror_seg ||
115 !seg_is_mirrored(seg->mirror_seg)) {
116 log_error("LV %s: segment %u mirror image "
117 "is not mirrored",
118 lv->name, seg_count);
119 r = 0;
120 }
121 }
122
123 for (s = 0; s < seg->area_count; s++) {
124 if (seg_type(seg, s) == AREA_UNASSIGNED) {
125 log_error("LV %s: segment %u has unassigned "
126 "area %u.",
127 lv->name, seg_count, s);
128 r = 0;
129 } else if (seg_type(seg, s) == AREA_PV) {
130 if (!seg_pvseg(seg, s) ||
131 seg_pvseg(seg, s)->lvseg != seg ||
132 seg_pvseg(seg, s)->lv_area != s) {
133 log_error("LV %s: segment %u has "
134 "inconsistent PV area %u",
135 lv->name, seg_count, s);
136 r = 0;
137 }
138 } else {
139 if (!seg_lv(seg, s) ||
140 seg_lv(seg, s)->vg != lv->vg ||
141 seg_lv(seg, s) == lv) {
142 log_error("LV %s: segment %u has "
143 "inconsistent LV area %u",
144 lv->name, seg_count, s);
145 r = 0;
146 }
147
148 if (complete_vg && seg_lv(seg, s) &&
149 (seg_lv(seg, s)->status & MIRROR_IMAGE) &&
150 (!(seg2 = find_seg_by_le(seg_lv(seg, s),
151 seg_le(seg, s))) ||
152 seg2->mirror_seg != seg)) {
153 log_error("LV %s: segment %u mirror "
154 "image %u missing mirror ptr",
155 lv->name, seg_count, s);
156 r = 0;
157 }
158
159/* FIXME I don't think this ever holds?
160 if (seg_le(seg, s) != le) {
161 log_error("LV %s: segment %u has "
162 "inconsistent LV area %u "
163 "size",
164 lv->name, seg_count, s);
165 r = 0;
166 }
167 */
168 }
169 }
170
171 le += seg->len;
172 }
173
174 if (le != lv->le_count) {
175 log_error("LV %s: inconsistent LE count %u != %u",
176 lv->name, le, lv->le_count);
177 r = 0;
178 }
179
180 return r;
181}

183/*
184 * Split the supplied segment at the supplied logical extent
185 * NB Use LE numbering that works across stripes PV1: 0,2,4 PV2: 1,3,5 etc.
186 */
187static int _lv_split_segment(struct logical_volume *lv, struct lv_segment *seg,
188 uint32_t le)
189{
190 struct lv_segment *split_seg;
191 uint32_t s;
192 uint32_t offset = le - seg->le;
193 uint32_t area_offset;
194
195 if (!seg_can_split(seg)) {
196 log_error("Unable to split the %s segment at LE %" PRIu32
197 " in LV %s", seg->segtype->name, le, lv->name);
198 return 0;
199 }
200
201 /* Clone the existing segment */
202 if (!(split_seg = alloc_lv_segment(lv->vg->cmd->mem, seg->segtype,
203 seg->lv, seg->le, seg->len,
204 seg->status, seg->stripe_size,
205 seg->log_lv,
206 seg->area_count, seg->area_len,
207 seg->chunk_size, seg->region_size,
208 seg->extents_copied))) {
209 log_error("Couldn't allocate cloned LV segment.");
210 return 0;
211 }
212
213 if (!str_list_dup(lv->vg->cmd->mem, &split_seg->tags, &seg->tags)) {
214 log_error("LV segment tags duplication failed");
215 return 0;
216 }
217
218 /* In case of a striped segment, the offset has to be / stripes */
219 area_offset = offset;
220 if (seg_is_striped(seg))
221 area_offset /= seg->area_count;
222
223 split_seg->area_len -= area_offset;
224 seg->area_len = area_offset;
225
226 split_seg->len -= offset;
227 seg->len = offset;
228
229 split_seg->le = seg->le + seg->len;
230
231 /* Adjust the PV mapping */
232 for (s = 0; s < seg->area_count; s++) {
233 seg_type(split_seg, s) = seg_type(seg, s);
234
235 /* Split area at the offset */
236 switch (seg_type(seg, s)) {
237 case AREA_LV:
238 seg_lv(split_seg, s) = seg_lv(seg, s);
239 seg_le(split_seg, s) =
240 seg_le(seg, s) + seg->area_len;
241 log_debug("Split %s:%u[%u] at %u: %s LE %u", lv->name,
242 seg->le, s, le, seg_lv(seg, s)->name,
243 seg_le(split_seg, s));
244 break;
245
246 case AREA_PV:
247 if (!(seg_pvseg(split_seg, s) =
248 assign_peg_to_lvseg(seg_pv(seg, s),
249 seg_pe(seg, s) +
250 seg->area_len,
251 seg_pvseg(seg, s)->len -
252 seg->area_len,
253 split_seg, s))) {
254 stack;
255 return 0;
256 }
257 log_debug("Split %s:%u[%u] at %u: %s PE %u", lv->name,
258 seg->le, s, le,
259 dev_name(seg_dev(seg, s)),
260 seg_pe(split_seg, s));
261 break;
262
263 case AREA_UNASSIGNED:
264 log_error("Unassigned area %u found in segment", s);
265 return 0;
266 }
267 }
268
269 /* Add split off segment to the list _after_ the original one */
270 list_add_h(&seg->list, &split_seg->list);
271
272 return 1;
273}

275/*
276 * Ensure there's a segment boundary at the given logical extent
277 */
278int lv_split_segment(struct logical_volume *lv, uint32_t le)
279{
280 struct lv_segment *seg;
281
282 if (!(seg = find_seg_by_le(lv, le))) {
283 log_error("Segment with extent %" PRIu32 " in LV %s not found",
284 le, lv->name);
285 return 0;
286 }
287
288 /* This is a segment start already */
289 if (le == seg->le)
290 return 1;
291
292 if (!_lv_split_segment(lv, seg, le)) {
293 stack;
294 return 0;
295 }
296
297 if (!vg_validate(lv->vg)) {
298 stack;
299 return 0;
300 }
301
302 return 1;
303}
