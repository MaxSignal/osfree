14typedef ULONG (EXPENTRY KSHOOK)( HAB, BOOL, PQMSG );

16int main( void )
17{
18 int rval = 0;
19 char szBuf[260];
20 HMODULE hModule;
21 QMSG msg;
22 HAB hHAB;
23 HMQ hHMQ;
24 STHOOK *pfnStartHook = 0L;
25 KSHOOK *pfnKeystackHookProc = 0L;
26
27 hHAB = WinInitialize( 0 );
28 hHMQ = WinCreateMsgQueue( hHAB, 0L );
29
30 // Look for our DLL in the LIBPATH
31 if (( DosLoadModule( szBuf, sizeof( szBuf ), "JPOS2DLL", &hModule ) != 0 ) ||
32 ( DosQueryProcAddr( hModule, 0, "STARTHOOK", (PFN *)&pfnStartHook ) != 0 ) || ( DosQueryProcAddr( hModule, 0, "KEYSTACKHOOKPROC", (PFN *)&pfnKeystackHookProc ) != 0 )) {
33 rval = WinGetLastError( hHAB );
34 goto KeystackExit;
35 }
36
37 // call .DLL
38 (*pfnStartHook)( hHMQ );
39
40 // Install the Journal Playback hook.
41 if ( WinSetHook( hHAB, NULLHANDLE, HK_JOURNALPLAYBACK, (PFN)pfnKeystackHookProc, hModule ) == 0 )
42 rval = WinGetLastError( hHAB );
43
44 else {
45
46 while ( WinGetMsg( hHAB, &msg, 0, 0, 0 )) {
47 if ( msg.msg == WM_USER )
48 break;
49 WinDispatchMsg( hHAB, &msg );
50 }
51
52 WinReleaseHook( hHAB, 0, HK_JOURNALPLAYBACK, (PFN)pfnKeystackHookProc, hModule );
53 }
54
55KeystackExit:
56 WinDestroyMsgQueue( hHMQ );
57 WinTerminate( hHAB );
58
59 return rval;
60}
