66BOOL parse_cmd(char *cmd)
67{
68 unsigned int i, rc;
69 int Argc; // argument count
70 char *Argv[10]; // argument pointers
71 char *pszTemp;
72
73 if (strlen(cmd)==0)
74 {
75 return FALSE;
76 }
77
78 if (strlen(cmd)==2)
79 if (cmd[1]==':')
80 {
81 if (QueryDriveReady(cmd[0]-'a'+1))
82 DosSetDefaultDisk(cmd[0]-'a'+1);
83 return FALSE;
84 }
85
86 while (( *cmd == ' ' ) || ( *cmd == '\t' ))
87 cmd++;
88
89 // loop through arguments
90 for ( Argc = 0; (( Argc < 10 ) && ( *cmd )); Argc++ ) {
91
92 pszTemp = cmd; // save start of argument
93
94 while ( isdelim( *cmd ) == 0 ) {
95 if ( *cmd )
96 cmd++;
97 }
98
99 // check flag for terminators
100 if (( *cmd != '\0' ))
101 *cmd++ = '\0';
102
103 // update the Argv pointer
104 Argv[ Argc ] = pszTemp;
105
106 // skip delimiters for Argv[2] onwards
107 while (( *cmd ) && ( isdelim( *cmd )))
108 cmd++;
109 }
110
111 Argv[Argc] = NULL;
112
113 if (!strcmp(Argv[0], "exit"))
114 {
115 return TRUE;
116 }
117
118 for (i=0;i<5;i++)
119 {
120 if (!strcmp(Argv[0], commands[i].cmdname))
121 {
122 (*(commands[i].func))(Argc, Argv);
123 return FALSE;
124 }
125 }
126
127 execute_external(Argc, Argv);
128
129 return FALSE;
130}
