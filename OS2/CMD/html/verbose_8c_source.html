57static HFILE hLogfile = 2; /* stderr? */
58static unsigned iLoglevel = 1; /* only fatal errors */

60static char const achLevel[] = "!*+:# ";

64void
65Verbose(unsigned level,char *fmt,...)
66{
67 APIRET rc;
68
69 if( level <= iLoglevel )
70 {
71 va_list argp;
72 char *str = malloc(strlen(fmt)+200);
73 char *buf = malloc(LOGBUFSIZ);
74 ULONG written;
75 time_t now = 0;
76 struct tm * timep;
77
78 if( level >= strlen(achLevel) )
79 str[0] = ' ';
80 else
81 str[0] = achLevel[level];
82
83 time( &now );
84 timep = localtime( &now );
85 strftime( &str[1], sizeof(str)-30, " %d %b %H:%M:%S ", timep );
86
87 strcat( str, fmt ); /* append format string */
88 strcat( str, "\r\n" ); /* append newline */
89
90 va_start(argp,fmt);
91 vsprintf( buf, str, argp );
92 rc = DosWrite( hLogfile, buf, strlen(buf), &written );
93 va_end(argp);
94
95 free( buf );
96 free( str );
97 }
98}
