22static void _pretty_print(struct rx_node *rx, int depth)
23{
24 int i;
25 for (i = 0; i < depth; i++)
26 printf(" ");
27
28 /* display info about the node */
29 switch (rx->type) {
30 case CAT:
31 printf("Cat");
32 break;
33
34 case OR:
35 printf("Or");
36 break;
37
38 case STAR:
39 printf("Star");
40 break;
41
42 case PLUS:
43 printf("Plus");
44 break;
45
46 case QUEST:
47 printf("Quest");
48 break;
49
50 case CHARSET:
51 printf("Charset : ");
52 for (i = 0; i < 256; i++) {
53 if (bit(rx->charset, i) && isprint(i))
54 printf("%c", (char) i);
55 }
56 break;
57
58 default:
59 printf("Unknown type");
60 }
61 printf("\n");
62
63 if (rx->left)
64 _pretty_print(rx->left, depth + 1);
65
66 if (rx->right)
67 _pretty_print(rx->right, depth + 1);
68}
