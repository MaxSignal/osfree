18static int _move_pv(struct volume_group *vg_from, struct volume_group *vg_to,
19 char *pv_name)
20{
21 struct pv_list *pvl;
22 struct physical_volume *pv;
23
24 if (!(pvl = find_pv_in_vg(vg_from, pv_name))) {
25 log_error("Physical volume %s not in volume group %s",
26 pv_name, vg_from->name);
27 return 0;
28 }
29
30 list_del(&pvl->list);
31 list_add(&vg_to->pvs, &pvl->list);
32
33 vg_from->pv_count--;
34 vg_to->pv_count++;
35
36 pv = list_item(pvl, struct pv_list)->pv;
37
38 vg_from->extent_count -= pv_pe_count(pv);
39 vg_to->extent_count += pv_pe_count(pv);
40
41 vg_from->free_count -= pv_pe_count(pv) - pv_pe_alloc_count(pv);
42 vg_to->free_count += pv_pe_count(pv) - pv_pe_alloc_count(pv);
43
44 return 1;
45}

60static int _move_lvs(struct volume_group *vg_from, struct volume_group *vg_to)
61{
62 struct list *lvh, *lvht;
63 struct logical_volume *lv;
64 struct lv_segment *seg;
65 struct physical_volume *pv;
66 struct volume_group *vg_with;
67 unsigned s;
68
69 list_iterate_safe(lvh, lvht, &vg_from->lvs) {
70 lv = list_item(lvh, struct lv_list)->lv;
71
72 if ((lv->status & SNAPSHOT))
73 continue;
74
75 if ((lv->status & MIRRORED))
76 continue;
77
78 /* Ensure all the PVs used by this LV remain in the same */
79 /* VG as each other */
80 vg_with = NULL;
81 list_iterate_items(seg, struct lv_segment, &lv->segments) {
82 for (s = 0; s < seg->area_count; s++) {
83 /* FIXME Check AREA_LV too */
84 if (seg_type(seg, s) != AREA_PV)
85 continue;
86
87 pv = seg_pv(seg, s);
88 if (vg_with) {
89 if (!pv_is_in_vg(vg_with, pv)) {
90 log_error("Can't split Logical "
91 "Volume %s between "
92 "two Volume Groups",
93 lv->name);
94 return 0;
95 }
96 continue;
97 }
98
99 if (pv_is_in_vg(vg_from, pv)) {
100 vg_with = vg_from;
101 continue;
102 }
103 if (pv_is_in_vg(vg_to, pv)) {
104 vg_with = vg_to;
105 continue;
106 }
107 log_error("Physical Volume %s not found",
108 dev_name(pv_dev(pv)));
109 return 0;
110 }
111
112 }
113
114 if (vg_with == vg_from)
115 continue;
116
117 /* Move this LV */
118 list_del(lvh);
119 list_add(&vg_to->lvs, lvh);
120
121 vg_from->lv_count--;
122 vg_to->lv_count++;
123 }
124
125 /* FIXME Ensure no LVs contain segs pointing at LVs in the other VG */
126
127 return 1;
128}

130static int _move_snapshots(struct volume_group *vg_from,
131 struct volume_group *vg_to)
132{
133 struct list *lvh, *lvht;
134 struct logical_volume *lv;
135 struct lv_segment *seg;
136 int cow_from = 0;
137 int origin_from = 0;
138
139 list_iterate_safe(lvh, lvht, &vg_from->lvs) {
140 lv = list_item(lvh, struct lv_list)->lv;
141
142 if (!(lv->status & SNAPSHOT))
143 continue;
144
145 list_iterate_items(seg, struct lv_segment, &lv->segments) {
146 cow_from = _lv_is_in_vg(vg_from, seg->cow);
147 origin_from = _lv_is_in_vg(vg_from, seg->origin);
148
149 if (cow_from && origin_from)
150 continue;
151 if ((!cow_from && origin_from) ||
152 (cow_from && !origin_from)) {
153 log_error("Can't split snapshot %s between"
154 " two Volume Groups", seg->cow->name);
155 return 0;
156 }
157 }
158
159 /* Move this snapshot */
160 list_del(lvh);
161 list_add(&vg_to->lvs, lvh);
162
163 vg_from->snapshot_count--;
164 vg_to->snapshot_count++;
165 }
166
167 return 1;
168}

170static int _move_mirrors(struct volume_group *vg_from,
171 struct volume_group *vg_to)
172{
173 struct list *lvh, *lvht;
174 struct logical_volume *lv;
175 struct lv_segment *seg;
176 int i, seg_in, log_in;
177
178 list_iterate_safe(lvh, lvht, &vg_from->lvs) {
179 lv = list_item(lvh, struct lv_list)->lv;
180
181 if (!(lv->status & MIRRORED))
182 continue;
183
184 seg = first_seg(lv);
185
186 seg_in = 0;
187 for (i = 0; i < seg->area_count; i++)
188 if (_lv_is_in_vg(vg_to, seg_lv(seg, i)))
189 seg_in++;
190
191 log_in = (!seg->log_lv || _lv_is_in_vg(vg_to, seg->log_lv));
192
193 if ((seg_in && seg_in < seg->area_count) ||
194 (seg_in && seg->log_lv && !log_in) ||
195 (!seg_in && seg->log_lv && log_in)) {
196 log_error("Can't split mirror %s between "
197 "two Volume Groups", lv->name);
198 return 0;
199 }
200
201 if (seg_in == seg->area_count && log_in) {
202 list_del(lvh);
203 list_add(&vg_to->lvs, lvh);
204
205 vg_from->lv_count--;
206 vg_to->lv_count++;
207 }
208 }
209
210 return 1;
211}

213int vgsplit(struct cmd_context *cmd, int argc, char **argv)
214{
215 char *vg_name_from, *vg_name_to;
216 struct volume_group *vg_to, *vg_from;
217 int opt;
218 int active;
219 int consistent = 1;
220
221 if (argc < 3) {
222 log_error("Existing VG, new VG and physical volumes required.");
223 return EINVALID_CMD_LINE;
224 }
225
226 vg_name_from = skip_dev_dir(cmd, argv[0], NULL);
227 vg_name_to = skip_dev_dir(cmd, argv[1], NULL);
228 argc -= 2;
229 argv += 2;
230
231 if (!validate_name(vg_name_from)) {
232 log_error("Volume group name \"%s\" is invalid",
233 vg_name_from);
234 return ECMD_FAILED;
235 }
236
237 if (!strcmp(vg_name_to, vg_name_from)) {
238 log_error("Duplicate volume group name \"%s\"", vg_name_from);
239 return ECMD_FAILED;
240 }
241
242 log_verbose("Checking for volume group \"%s\"", vg_name_from);
243 if (!lock_vol(cmd, vg_name_from, LCK_VG_WRITE)) {
244 log_error("Can't get lock for %s", vg_name_from);
245 return ECMD_FAILED;
246 }
247
248 if (!(vg_from = vg_read(cmd, vg_name_from, NULL, &consistent)) || !consistent) {
249 log_error("Volume group \"%s\" doesn't exist", vg_name_from);
250 unlock_vg(cmd, vg_name_from);
251 return ECMD_FAILED;
252 }
253
254 if (!vg_check_status(vg_from, CLUSTERED | EXPORTED_VG |
255 RESIZEABLE_VG | LVM_WRITE)) {
256 unlock_vg(cmd, vg_name_from);
257 return ECMD_FAILED;
258 }
259
260 log_verbose("Checking for volume group \"%s\"", vg_name_to);
261 if (!lock_vol(cmd, vg_name_to, LCK_VG_WRITE | LCK_NONBLOCK)) {
262 log_error("Can't get lock for %s", vg_name_to);
263 unlock_vg(cmd, vg_name_from);
264 return ECMD_FAILED;
265 }
266
267 consistent = 0;
268 if ((vg_to = vg_read(cmd, vg_name_to, NULL, &consistent))) {
269 /* FIXME Remove this restriction */
270 log_error("Volume group \"%s\" already exists", vg_name_to);
271 goto error;
272 }
273
274 if (!validate_vg_name(cmd, vg_name_to)) {
275 log_error("New volume group name \"%s\" is invalid",
276 vg_name_to);
277 goto error;
278 }
279
280 if ((active = lvs_in_vg_activated(vg_from))) {
281 /* FIXME Remove this restriction */
282 log_error("Logical volumes in \"%s\" must be inactive",
283 vg_name_from);
284 goto error;
285 }
286
287 /* Set metadata format of original VG */
288 /* FIXME: need some common logic */
289 cmd->fmt = vg_from->fid->fmt;
290
291 /* Create new VG structure */
292 if (!(vg_to = vg_create(cmd, vg_name_to, vg_from->extent_size,
293 vg_from->max_pv, vg_from->max_lv,
294 vg_from->alloc, 0, NULL)))
295 goto error;
296
297 if (vg_from->status & CLUSTERED)
298 vg_to->status |= CLUSTERED;
299
300 /* Archive vg_from before changing it */
301 if (!archive(vg_from))
302 goto error;
303
304 /* Move PVs across to new structure */
305 for (opt = 0; opt < argc; opt++) {
306 if (!_move_pv(vg_from, vg_to, argv[opt]))
307 goto error;
308 }
309
310 /* Move required LVs across, checking consistency */
311 if (!(_move_lvs(vg_from, vg_to)))
312 goto error;
313
314 /* Move required snapshots across */
315 if (!(_move_snapshots(vg_from, vg_to)))
316 goto error;
317
318 /* Move required mirrors across */
319 if (!(_move_mirrors(vg_from, vg_to)))
320 goto error;
321
322 /* Split metadata areas and check if both vgs have at least one area */
323 if (!(vg_split_mdas(cmd, vg_from, vg_to)) && vg_from->pv_count) {
324 log_error("Cannot split: Nowhere to store metadata for new Volume Group");
325 goto error;
326 }
327
328 /* Set proper name for all PVs in new VG */
329 if (!vg_rename(cmd, vg_to, vg_name_to))
330 goto error;
331
332 /* store it on disks */
333 log_verbose("Writing out updated volume groups");
334
335 /* Write out new VG as EXPORTED */
336 vg_to->status |= EXPORTED_VG;
337
338 if (!archive(vg_to))
339 goto error;
340
341 if (!vg_write(vg_to) || !vg_commit(vg_to))
342 goto error;
343
344 backup(vg_to);
345
346 /* Write out updated old VG */
347 if (vg_from->pv_count) {
348 if (!vg_write(vg_from) || !vg_commit(vg_from))
349 goto error;
350
351 backup(vg_from);
352 }
353
354 /* Remove EXPORTED flag from new VG */
355 consistent = 1;
356 if (!(vg_to = vg_read(cmd, vg_name_to, NULL, &consistent)) || !consistent) {
357 log_error("Volume group \"%s\" became inconsistent: please "
358 "fix manually", vg_name_to);
359 goto error;
360 }
361
362 vg_to->status &= ~EXPORTED_VG;
363
364 if (!vg_write(vg_to) || !vg_commit(vg_to))
365 goto error;
366
367 backup(vg_to);
368
369 unlock_vg(cmd, vg_name_from);
370 unlock_vg(cmd, vg_name_to);
371
372 log_print("Volume group \"%s\" successfully split from \"%s\"",
373 vg_to->name, vg_from->name);
374 return ECMD_PROCESSED;
375
376 error:
377 unlock_vg(cmd, vg_name_from);
378 unlock_vg(cmd, vg_name_to);
379 return ECMD_FAILED;
380}
