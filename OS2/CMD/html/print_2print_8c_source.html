102/* ------------------------------------------------------------------------- *
103 * Main program *
104 * ------------------------------------------------------------------------- */
105int main( int argc, char *argv[] )
106{
107 FILESTATUS3 fs3; // file info structure
108 PSZ psz, // used for string manipulation
109 pszFile = NULL; // input filename argument
110 CHAR szPort[ CCH_MAXPORT ] = {0}, // output printer port or device name
111 szPdr[ CCHMAXPATH ] = {0}; // FQN of port driver
112 BOOL fSyntax = FALSE, // syntax error
113 fList = FALSE; // user requested a list of devices
114 HMODULE hPdr = NULLHANDLE; // module handle for port driver
115 HFILE hfFile = NULLHANDLE; // file handle for input file
116 ULONG a, // arg index counter
117 ulAppend, // byte-append argument as parsed
118 ulAction, // action reported by DosOpen/PrtOpen
119 flOpFn, // function flags for DosOpen/PrtOpen
120 flOpMd; // mode flags for DosOpen/PrtOpen
121 APIRET rc; // return code from Dos**
122#ifdef ENABLE_QUEUED
123 USHORT usType = US_TYPE_PORT; // type of output device
124#endif
125
126
127 // Parse the input arguments
128 for ( a = 1; a < argc; a++ ) {
129 psz = argv[ a ];
130 if ( *psz == '/' || *psz == '-') {
131 psz++;
132 if ( !(*psz) ) continue;
133 switch ( *psz ) {
134 case 'B':
135 case 'b':
136 if ( sscanf( psz+1, ":%u", cbBuffer ) != 1 )
137 cbBuffer = CB_BUFFER;
138 break;
139 case 'D':
140 case 'd':
141 if ( sscanf( psz+1, ":%255s", szPort ) != 1 )
142 fSyntax = TRUE;
143 break;
144 case 'F':
145 case 'f':
146 fAppendF = TRUE;
147 break;
148 case 'L':
149 case 'l':
150 fList = TRUE;
151 break;
152#ifdef ENABLE_QUEUED
153 case 'P':
154 case 'p':
155 usType = US_TYPE_QUEUE;
156 if ( sscanf( psz+1, ":%255s", szPort ) != 1 )
157 fSyntax = TRUE;
158 break;
159 case 'Q':
160 case 'q':
161 usType = US_TYPE_QUEUE;
162 fList = TRUE;
163 break;
164#endif
165 case 'V':
166 case 'v':
167 fDebug = TRUE;
168 break;
169 case 'X':
170 case 'x':
171 if ( sscanf( psz+1, ":%x", ulAppend ) == 1 ) {
172 fAppendX = TRUE;
173 bAppend = (BYTE) ulAppend;
174 }
175 break;
176 case 'Z':
177 case 'z':
178 fAppendX = TRUE;
179 bAppend = 0x04;
180 break;
181 }
182 }
183 else if ( !pszFile )
184 pszFile = psz;
185 else
186 fSyntax = TRUE;
187 }
188 if ( fList ) {
189#ifdef ENABLE_QUEUED
190 if ( usType == US_TYPE_QUEUE )
191 ListQueues();
192 else
193#endif
194 ListPorts();
195 return 0;
196 }
197
198 if ( !pszFile || !szPort[0] || fSyntax ) {
199 printf("EPRINT - Enhanced PRINT version %s\n\n", SZ_VERSION );
200 printf("Usage: EPRINT <file> /D:<port> [ /B:<size> ] [ /F ] [ /V ] [ /X:<##> ] [ /Z ]\n");
201#ifdef ENABLE_QUEUED
202 printf(" EPRINT <file> /P:<queue> [ /B:<size> ] [ /F ] [ /V ] [ /X:<##> ] [ /Z ]\n");
203#endif
204 printf(" EPRINT /L\n");
205#ifdef ENABLE_QUEUED
206 printf(" EPRINT /Q\n");
207#endif
208 printf("\n /B:<size> Set maximum read/write buffer size (default: 64 KB).\n");
209 printf(" /D:<port> Attempt to print <file> to port <port>.\n");
210 printf(" /F Append a form feed control byte (^L) to the print job.\n");
211 printf(" /L Show a list of valid ports known to the system.\n");
212#ifdef ENABLE_QUEUED
213 printf(" /P:<queue> Attempt to print <filename> to printer queue <queue>.\n");
214 printf(" /Q Show list of printer queues defined on the system.\n");
215#endif
216 printf(" /V Verbose mode, show extra diagnostic information.\n");
217 printf(" /X:<##> Append the hexadecimal byte value <##> to the end of the print\n");
218 printf(" job. May not be used together with /Z.\n");
219 printf(" /Z Append an end-of-transmission byte (^D) to the end of the print\n");
220 printf(" job; equivalent to \"/X:04\". May not be used together with /X.\n\n");
221 return 1;
222 }
223
224
225#ifdef ENABLE_QUEUED
226 if ( usType != US_TYPE_QUEUE )
227#endif
228 {
229 // We don't support the FILE port (there's really no point)
230 if ( !strcmpi( szPort, "FILE") ) {
231 fprintf( stderr, "Use of the FILE port is not supported by this command.\n");
232 return 1;
233 }
234
235 // Make sure the port exists, and get the path/name of its port driver
236 if ( !ResolvePort( szPort, szPdr )) {
237 fprintf( stderr, "The port %s was not found.\n", szPort );
238 return 2;
239 }
240 }
241
242 // Open the input file
243 flOpFn = OPEN_ACTION_OPEN_IF_EXISTS | OPEN_ACTION_FAIL_IF_NEW;
244 flOpMd = OPEN_FLAGS_FAIL_ON_ERROR | OPEN_FLAGS_SEQUENTIAL |
245 OPEN_ACCESS_READONLY | OPEN_SHARE_DENYNONE;
246 rc = DosOpen( pszFile, &hfFile, &ulAction, 0L, 0L, flOpFn, flOpMd, 0L );
247 if ( rc != NO_ERROR || !hfFile ) {
248 fprintf( stderr, "Failed to open input file %s: RC=%u\n", pszFile, rc );
249 return ( rc );
250 }
251
252 // If the file is smaller than our default buffer size, just use the file size
253 rc = DosQueryFileInfo( hfFile, FIL_STANDARD, &fs3, sizeof( fs3 ));
254 if (( rc == NO_ERROR ) && ( fs3.cbFile < cbBuffer ))
255 cbBuffer = fs3.cbFile;
256
257
258#ifdef ENABLE_QUEUED
259 if ( usType == US_TYPE_QUEUE ) {
260 rc = PrintQueuedJob( hfFile, pszFile, szPort );
261
262 }
263 else
264#endif
265 {
266
267 // Now try writing the contents of the input file to the port
268 rc = WritePort_Orthodox( hfFile, szPort );
269
270 #ifdef ENABLE_DIRECT
271 if ( rc == ERROR_OPEN_FAILED ) {
272
273 /* The PrtOpen function failed (this seems to happen with some devices).
274 * As a fallback, we will try to to register and call the port driver's
275 * open/write/close functions directly.
276 */
277 if ( *szPdr ) {
278 // Load the port driver DLL
279 rc = DosLoadModule( NULL, 0, szPdr, &hPdr );
280 if ( !hPdr || ( rc != NO_ERROR )) {
281 fprintf( stderr, "Failed to load port driver %s: DosLoadModule() returned error %u\n", szPdr, rc );
282 if ( hfFile ) DosClose( hfFile );
283 return ( rc );
284 }
285 // If there's an entrypoint for SplPdOpen, we can proceed
286 rc = DosQueryProcAddr( hPdr, 0, "SPLPDOPEN", &SplPdOpen );
287 if ( rc == NO_ERROR ) {
288 printf("Opening port %s in direct access mode.\n", szPort );
289 rc = WritePort_Direct( hfFile, szPort, hPdr );
290 }
291 else {
292 fprintf( stderr, "Failed to open output port %s: PrtOpen() returned error 110\n", szPort );
293 fprintf( stderr, "Port driver does not appear to support SplPdOpen: DosQueryProcAddr() returned %u\n", rc );
294 }
295 // Release the port driver module
296 DosFreeModule( hPdr );
297 }
298 }
299 #endif
300
301 }
302
303 // Close the input file
304 if ( hfFile ) DosClose( hfFile );
305
306 // And we're done
307 return ( rc );
308}

312 * ListPorts *
313 * *
314 * Output a list of ports known to the system. *
315 * *
316 * PARAMETERS: NONE *
317 * *
318 * RETURNS: N/A *
319 * ------------------------------------------------------------------------- */
320void ListPorts( void )
321{
322 PPRPORTINFO1 pInfo = NULL;
323 PVOID pbuf = NULL;
324 ULONG cbBuf,
325 cTotal,
326 cReturned,
327 cbNeeded,
328 i;
329 SPLERR rc;
330
331
332 // Query the amount of available data
333 rc = SplEnumPort( NULL, 1, pbuf, 0L, &cReturned, &cTotal, &cbNeeded, NULL );
334 if (( rc != ERROR_MORE_DATA ) && ( rc != NERR_BufTooSmall )) {
335 fprintf( stderr, "Error getting port information: SplEnumPort() returned error %u\n", rc );
336 return;
337 }
338
339 // Now get the actual data
340 pbuf = malloc( cbNeeded );
341 if ( !pbuf ) {
342 fprintf( stderr, "Memory allocation error.\n", rc );
343 return;
344 }
345 cbBuf = cbNeeded;
346 rc = SplEnumPort( NULL, 1, pbuf, cbBuf,
347 &cReturned, &cTotal, &cbNeeded, NULL );
348 if ( rc == NO_ERROR ) {
349 pInfo = (PPRPORTINFO1) pbuf;
350 printf("------------------------------------------------------------------------------\n");
351 printf("Port Name Port Driver\n");
352 printf("------------------------------------------------------------------------------\n");
353 for ( i = 0; i < cReturned; i++ ) {
354 if ( !pInfo->pszPortName ) continue;
355 printf("%-33s", pInfo->pszPortName );
356 if ( pInfo->pszPortDriverPathName )
357 printf("%s", pInfo->pszPortDriverPathName );
358 printf("\n");
359 pInfo++;
360 }
361 printf("------------------------------------------------------------------------------\n");
362 }
363 free( pbuf );
364}

367/* ------------------------------------------------------------------------- *
368 * ResolvePort *
369 * *
370 * Confirm that the requested port name exists, and identify the port driver *
371 * that owns it. The port name is matched case-insensitively; on return *
372 * the contents of pszPort will be updated if necessary to reflect the *
373 * proper capitalization. *
374 * *
375 * PARAMETERS: *
376 * PSZ pszPort : The requested port name. (IO) *
377 * PSZ pszDriver: The fully-qualified name of the port driver (O) *
378 * *
379 * RETURNS: BOOL *
380 * TRUE if the port was found; pszDriver will contain the port driver FQN. *
381 * FALSE if the port was not found or an error occurred. *
382 * ------------------------------------------------------------------------- */
383BOOL ResolvePort( PSZ pszPort, PSZ pszDriver )
384{
385 PPRPORTINFO1 pInfo = NULL;
386 PVOID pbuf = NULL;
387 ULONG cbBuf,
388 cTotal,
389 cReturned,
390 cbNeeded,
391 i;
392 SPLERR rc;
393 BOOL fFound = FALSE;
394
395
396 if ( !pszPort || !pszDriver ) return FALSE;
397
398 // Query the amount of available data
399 rc = SplEnumPort( NULL, 1, pbuf, 0L, &cReturned, &cTotal, &cbNeeded, NULL );
400 if (( rc != ERROR_MORE_DATA ) && ( rc != NERR_BufTooSmall )) {
401 fprintf( stderr, "Error getting port information: SplEnumPort() returned error %u\n", rc );
402 return FALSE;
403 }
404
405 // Now get the actual data
406 pbuf = malloc( cbNeeded );
407 if ( !pbuf ) {
408 fprintf( stderr, "Memory allocation error.\n");
409 return FALSE;
410 }
411 cbBuf = cbNeeded;
412 rc = SplEnumPort( NULL, 1, pbuf, cbBuf,
413 &cReturned, &cTotal, &cbNeeded, NULL );
414 if ( rc == NO_ERROR ) {
415 pInfo = (PPRPORTINFO1) pbuf;
416 for ( i = 0; i < cReturned; i++ ) {
417 if ( pInfo->pszPortName &&
418 ( stricmp( pszPort, pInfo->pszPortName )) == 0 )
419 {
420 fFound = TRUE;
421 strcpy( pszPort, pInfo->pszPortName );
422 if ( pInfo->pszPortDriverPathName ) {
423 strcpy( pszDriver, pInfo->pszPortDriverPathName );
424 if ( fDebug ) printf("Found port %s --> %s\n", pszPort, pszDriver );
425 }
426 else if ( fDebug ) printf("Found port %s (unknown driver)\n", pszPort );
427 break;
428 }
429 pInfo++;
430 }
431 }
432
433 free( pbuf );
434 return fFound;
435}

438/* ------------------------------------------------------------------------- *
439 * WritePort_Orthodox *
440 * *
441 * Write the contents of the input file to the output port, using the *
442 * Prt*** API from PMSPL.DLL (which is the 'proper' way of doing things). *
443 * *
444 * PARAMETERS: *
445 * HFILE hfFile: Handle of the input file (as returned from DosOpen). (I) *
446 * PSZ pszPort : The requested port name. (I) *
447 * *
448 * RETURNS: ULONG *
449 * 0 on success, system error code otherwise. *
450 * ------------------------------------------------------------------------- */
451ULONG WritePort_Orthodox( HFILE hfFile, PSZ pszPort )
452{
453 HFILE hfPort = NULLHANDLE; // file handle for output port
454 PBYTE pbuf; // read/write buffer
455 ULONG ulAction, // action reported by DosOpen/PrtOpen
456 flOpFn, // function flags for DosOpen/PrtOpen
457 flOpMd, // mode flags for DosOpen/PrtOpen
458 cbRead, // number of bytes read from file
459 cbWritten, // number of bytes written to port
460 ulrc; // return code from Prt**
461 APIRET rc; // return code from Dos**
462
463
464 // Open the port
465 flOpFn = OPEN_ACTION_OPEN_IF_EXISTS | OPEN_ACTION_FAIL_IF_NEW;
466 flOpMd = OPEN_FLAGS_NONSPOOLED | OPEN_FLAGS_FAIL_ON_ERROR |
467 OPEN_ACCESS_WRITEONLY | OPEN_SHARE_DENYNONE;
468 ulrc = PrtOpen( pszPort, &hfPort, &ulAction, 0L, FILE_NORMAL, flOpFn, flOpMd, 0L );
469 if ( ulrc != NO_ERROR || !hfPort ) {
470#ifdef ENABLE_DIRECT
471 // (We will handle ERROR_OPEN_FAILED specially after returning)
472 if ( ulrc != ERROR_OPEN_FAILED )
473#endif
474 fprintf( stderr, "Failed to open output port %s: PrtOpen() returned error %u\n", pszPort, ulrc );
475 return ( ulrc );
476 }
477
478 // Allocate our read/write buffer
479 pbuf = malloc( cbBuffer );
480 if ( !pbuf ) {
481 fprintf( stderr, "Memory allocation error.\n");
482 return ERROR_NOT_ENOUGH_MEMORY;
483 }
484
485 // Read from the file and write to the port until there is no more data
486 do {
487 rc = DosRead( hfFile, pbuf, cbBuffer, &cbRead );
488 if ( rc || !cbRead ) continue;
489 if ( fDebug ) printf("Read %u/%u bytes from file [%u] : ", cbRead, cbBuffer, rc );
490
491 ulrc = PrtWrite( hfPort, pbuf, cbRead, &cbWritten );
492 if ( fDebug ) printf("Wrote %u bytes to port [%u]\n", cbWritten, ulrc );
493
494 /* Currently we just retry once if the write was incomplete.
495 * Eventually this should probably be made more robust.
496 */
497 if ( ulrc == NO_ERROR && ( cbWritten < cbRead )) {
498 cbRead = cbRead - cbWritten;
499 ulrc = PrtWrite( hfPort, (PBYTE)(pbuf+cbWritten), cbRead, &cbWritten );
500 if ( fDebug ) printf("Wrote %u bytes to port [%u]\n", cbWritten, ulrc );
501 if ( ulrc == NO_ERROR && ( cbWritten < cbRead ))
502 ulrc = ERROR_MORE_DATA;
503 }
504
505 } while (( ulrc == NO_ERROR ) && ( rc == NO_ERROR ) && ( cbRead == cbBuffer ));
506
507 if ( rc )
508 fprintf( stderr, "Error reading input file: DosRead() returned error %u\n", rc );
509
510 if ( !rc && !ulrc && fAppendF ) {
511 *pbuf = 0x0C;
512 ulrc = PrtWrite( hfPort, pbuf, 1, &cbWritten );
513 }
514 if ( !rc && !ulrc && fAppendX ) {
515 *pbuf = bAppend;
516 ulrc = PrtWrite( hfPort, pbuf, 1, &cbWritten );
517 }
518
519 if ( ulrc ) {
520 fprintf( stderr, "Error writing to port: PrtWrite() returned error %u\n", ulrc );
521 rc = ulrc;
522 }
523
524 // Close the port
525 PrtClose( hfPort );
526
527 free( pbuf );
528 return ( rc );
529}

532#ifdef ENABLE_DIRECT
533/* ------------------------------------------------------------------------- *
534 * WritePort_Direct *
535 * *
536 * Write the contents of the input file to the output port by directly *
537 * registering and using the SplPd*** entrypoints from the port driver. *
538 * This is needed by some ports for which the Prt*** functions don't seem *
539 * to work. *
540 * *
541 * PARAMETERS: *
542 * HFILE hfFile : Handle of the input file (from DosOpen). (I) *
543 * PSZ pszPort : The requested port name. (I) *
544 * HMODULE hmPort: Handle of the port driver (from DosLoadModule). (I) *
545 * *
546 * RETURNS: ULONG *
547 * 0 on success, system error code otherwise. *
548 * ------------------------------------------------------------------------- */
549ULONG WritePort_Direct( HFILE hfFile, PSZ pszPort, HMODULE hmPort )
550{
551 FILESTATUS3 fs3; // file info structure
552 PRTOPENSTRUCT0 spOpen = {0}; // information returned by SplPdOpen
553 PRTSW prtsw = {0}; // BIDI information returned by SplPdQuery
554 BOOL fProt = FALSE; // use SplPdProtWrite instead of SplPdWrite
555 HFILE hfPort = NULLHANDLE; // handle to port
556 PBYTE pbuf; // read/write buffer
557 ULONG flDevice, // device flags returned by SplPdOpen
558 cbRead, // number of bytes read from file
559 cbWritten, // number of bytes written to port
560 ulrc; // return code from SplPd**
561 APIRET rc; // return code from Dos**
562
563
564
565 // If the file is smaller than our default buffer size, just use the file size
566 rc = DosQueryFileInfo( hfFile, FIL_STANDARD, &fs3, sizeof( fs3 ));
567 if (( rc == NO_ERROR ) && ( fs3.cbFile < cbBuffer ))
568 cbBuffer = fs3.cbFile;
569
570 // Call SplPdQuery to find out which write function we need to use
571 rc = DosQueryProcAddr( hmPort, 0, "SPLPDQUERY", &SplPdQuery );
572 if ( rc == NO_ERROR ) {
573 rc = SplPdQuery( pszPort, TYPE_SHORT_WAIT, BIDI_Q_SW, NULL, 0L, &prtsw, &cbRead );
574 if (( rc == NO_ERROR ) && ( prtsw.flJob & PRTSW_JOB_WRAPPER_REQUIRED )) {
575 if ( fDebug ) printf("Port driver requested use of SplPdProtWrite()\n");
576 rc = DosQueryProcAddr( hmPort, 0, "SPLPDPROTWRITE", &SplPdProtWrite );
577 if (( rc != NO_ERROR ) || !SplPdProtWrite ) {
578 fprintf( stderr, "Failed to load entrypoint for SplPdProtWrite: DosQueryProcAddr() returned %u\n", rc );
579 return ( rc );
580 }
581 fProt = TRUE;
582 }
583 }
584
585 // Load the remaining functions (note: SplPdOpen was registered earlier)
586 rc = DosQueryProcAddr( hmPort, 0, "SPLPDWRITE", &SplPdWrite );
587 if ( rc != NO_ERROR ) {
588 fprintf( stderr, "Failed to load entrypoint for SplPdWrite: DosQueryProcAddr() returned %u\n", rc );
589 return ( rc );
590 }
591 rc = DosQueryProcAddr( hmPort, 0, "SPLPDCLOSE", &SplPdClose );
592 if ( rc != NO_ERROR ) {
593 fprintf( stderr, "Failed to load entrypoint for SplPdClose: DosQueryProcAddr() returned %u\n", rc );
594 return ( rc );
595 }
596
597 // Open the output port
598 ulrc = SplPdOpen( pszPort, &hfPort, &flDevice, &spOpen );
599 if ( ulrc != NO_ERROR || !hfPort ) {
600 fprintf( stderr, "Failed to open output port %s: SplPdOpen() returned %u\n", pszPort, ulrc );
601 return ( rc );
602 }
603
604 // Allocate our read/write buffer
605 pbuf = malloc( cbBuffer );
606 if ( !pbuf ) {
607 fprintf( stderr, "Memory allocation error.\n");
608 return ERROR_NOT_ENOUGH_MEMORY;
609 }
610
611 // Read from the file and write to the port
612 do {
613 rc = DosRead( hfFile, pbuf, cbBuffer, &cbRead );
614 if ( rc || !cbRead ) continue;
615 if ( fDebug ) printf("Read %u/%u bytes from file [%u] : ", cbRead, cbBuffer, rc );
616 ulrc = fProt ?
617 SplPdProtWrite( hfPort, pszPort, SplPdWrite,
618 NULL, pbuf, cbRead, &cbWritten ) :
619 SplPdWrite( hfPort, pbuf, cbRead, &cbWritten );
620 if ( fDebug ) printf("Wrote %u/%u bytes to port [%u]\n", cbWritten, cbRead, ulrc );
621
622 /* Currently we just retry once if the write was incomplete.
623 * Eventually this should probably be made more robust.
624 */
625 if ( ulrc == NO_ERROR && ( cbWritten < cbRead )) {
626 cbRead = cbRead - cbWritten;
627 ulrc = fProt ?
628 SplPdProtWrite( hfPort, pszPort, SplPdWrite, NULL,
629 (PBYTE)(pbuf+cbWritten), cbRead, &cbWritten ) :
630 SplPdWrite( hfPort, (PBYTE)(pbuf+cbWritten), cbRead, &cbWritten );
631 if ( fDebug ) printf("Writing %u bytes to port [%u] : ", cbWritten, ulrc );
632 if ( ulrc == NO_ERROR && ( cbWritten < cbRead ))
633 ulrc = ERROR_MORE_DATA;
634 }
635
636 } while (( ulrc == NO_ERROR ) && ( rc == NO_ERROR ) && ( cbRead == cbBuffer ));
637
638 if ( rc )
639 fprintf( stderr, "Error reading input file: DosRead returned error %u\n", rc );
640
641 if ( !rc && !ulrc && fAppendF ) {
642 *pbuf = 0x0C;
643 ulrc = fProt ? SplPdProtWrite( hfPort, pszPort, SplPdWrite,
644 NULL, pbuf, 1, &cbWritten ) :
645 SplPdWrite( hfPort, pbuf, 1, &cbWritten );
646 }
647 if ( !rc && !ulrc && fAppendX ) {
648 *pbuf = bAppend;
649 ulrc = fProt ? SplPdProtWrite( hfPort, pszPort, SplPdWrite,
650 NULL, pbuf, 1, &cbWritten ) :
651 SplPdWrite( hfPort, pbuf, 1, &cbWritten );
652 }
653
654 if ( ulrc ) {
655 fprintf( stderr, "Error writing to port: %s() returned error %u\n",
656 fProt? "SplPdProtWrite": "SplPdWrite", ulrc );
657 rc = ulrc;
658 }
659
660 // Close the port
661 SplPdClose( hfPort );
662
663 free( pbuf );
664 return ( rc );
665}
666#endif

730/* ------------------------------------------------------------------------- *
731 * PrintQueuedJob *
732 * *
733 * Prints the contents of the input file to the specified print queue by *
734 * sending it to the spooler. *
735 * *
736 * PARAMETERS: *
737 * HFILE hfFile: Handle of the input file (as returned from DosOpen). (I) *
738 * PSZ pszFile : The fully-qualified name of the input file. (I) *
739 * PSZ pszDest : The requested printer queue name. (I) *
740 * *
741 * RETURNS: ULONG *
742 * 0 on success, system error code otherwise. *
743 * ------------------------------------------------------------------------- */
744ULONG PrintQueuedJob( HFILE hfFile, PSZ pszFile, PSZ pszDest )
745{
746 HAB hab; // system HAB
747 HSPL hspl; // handle to spooler queue
748 DEVOPENSTRUC qmo = {0}; // device open structure for spooler
749 PBYTE pbuf; // read/write buffer
750 ULONG cbRead, // number of bytes read from file
751 ulJob, // job ID
752 ulrc; // return code from SplQm*
753 time_t tt; // job submission time (seconds since epoch)
754 APIRET rc; // return code from Dos*
755
756
757 hab = WinInitialize( 0 );
758
759 // Open the spooler queue
760 qmo.pszLogAddress = pszDest;
761 qmo.pszDriverName = "IBMNULL";
762 qmo.pszComment = "EPRINT";
763 //qmo.pszDataType = "PM_Q_RAW";
764 hspl = SplQmOpen("*", 4L, (PQMOPENDATA) &qmo );
765 if ( hspl == SPL_ERROR ) {
766 ulrc = ERRORIDERROR( WinGetLastError( hab ));
767 if ( ulrc == PMERR_SPL_QUEUE_NOT_FOUND )
768 fprintf( stderr, "The print queue \"%s\" was not found.\n", pszDest );
769 else
770 fprintf( stderr, "Failed to open print queue: SplQmOpen() returned error 0x%X\n", ulrc );
771 goto finish;
772 }
773
774 // Signal the start of the print job
775 if ( ! SplQmStartDoc( hspl, pszFile )) {
776 ulrc = ERRORIDERROR( WinGetLastError( hab ));
777 fprintf( stderr, "Failed to initialize job: SplQmStartDoc() returned error 0x%X\n", ulrc );
778 goto close;
779 }
780
781 // Allocate our read/write buffer (there is a max of 64KB with SplQmWrite)
782 if ( cbBuffer > 0xFFFF ) cbBuffer = 0xFFFF;
783 pbuf = malloc( cbBuffer );
784 if ( !pbuf ) {
785 fprintf( stderr, "Memory allocation error.\n");
786 ulrc = ERROR_NOT_ENOUGH_MEMORY;
787 goto close;
788 }
789
790 // Now copy the contents of the input file to the queue
791 do {
792 rc = DosRead( hfFile, pbuf, cbBuffer, &cbRead );
793 if ( rc || !cbRead ) continue;
794 if ( fDebug ) printf("Read %u/%u bytes from file [%u] : ", cbRead, cbBuffer, rc );
795
796 if ( SplQmWrite( hspl, cbRead, pbuf )) {
797 if ( fDebug ) printf("Wrote %u bytes to spooler\n", cbRead );
798 }
799 else {
800 ulrc = ERRORIDERROR( WinGetLastError( hab ));
801 fprintf( stderr, "Failed to write to spooler: SplQmWrite() returned error 0x%X\n", ulrc );
802 break;
803 }
804
805 } while (( rc == NO_ERROR ) && ( cbRead == cbBuffer ));
806 free( pbuf );
807
808 // Signal job completion
809 ulJob = SplQmEndDoc( hspl );
810 if ( ulJob == SPL_ERROR ) {
811 ulrc = ERRORIDERROR( WinGetLastError( hab ));
812 fprintf( stderr, "Failed to complete job: SplQmEndDoc() returned error 0x%X\n", ulrc );
813 goto close;
814 }
815
816 printf("%s has been submitted to print queue \"%s\" as job %u.\n", pszFile, pszDest, ulJob );
817 if ( fDebug ) {
818 ulrc = SplQueryJob( NULL, pszDest, ulJob, 3, NULL, 0, &cbRead );
819 if (( ulrc == NERR_BufTooSmall ) || ( ulrc == ERROR_MORE_DATA )) {
820 PPRJINFO3 pInfo;
821 pbuf = malloc( cbRead );
822 ulrc = SplQueryJob( NULL, pszDest, ulJob, 3, pbuf, cbRead, &cbRead );
823 if ( !ulrc ) {
824 pInfo = (PPRJINFO3) pbuf;
825 tt = (time_t) pInfo->ulSubmitted;
826 printf(" Priority: %u\n", pInfo->uPriority );
827 printf(" Position: %u\n", pInfo->uPosition );
828 printf(" Status: %u\n", pInfo->fsStatus & PRJ_QSTATUS );
829 if ( pInfo->pszStatus && *(pInfo->pszStatus) )
830 printf(" Message: %s\n", pInfo->pszStatus );
831 printf(" Driver: %s\n", pInfo->pszDriverName );
832 printf(" Submitted: %s", asctime( gmtime( &tt )));
833 printf(" Size: %u\n", pInfo->ulSize );
834 printf(" Document: %s\n", pInfo->pszDocument );
835 }
836 free( pbuf );
837 }
838 if ( ulrc )
839 fprintf( stderr, "Failed to query job: SplQueryJob() returned %u\n", ulrc );
840 }
841
842close:
843 // Close the queue
844 if ( ! SplQmClose( hspl )) {
845 ulrc = ERRORIDERROR( WinGetLastError( hab ));
846 fprintf( stderr, "Failed to close spooler: SplQmClose() returned error 0x%X\n", ulrc );
847 }
848
849finish:
850 // Clean up and return
851 WinTerminate( hab );
852 return ( ulrc );
853}
