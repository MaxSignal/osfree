45 /*
46 *@@ INIFILE_HEADER:
47 * header of OS/2 INI file. This is
48 * what an OS/2 .INI file starts with
49 * at byte 0.
50 *
51 * An OS/2 INI file has the following layout:
52 *
53 * -- INIFILE_HEADER
54 * -- first INIFILE_APP, if any
55 * -- name of first application (zero-terminated),
56 * of which the first INIFILE_APP.offAppName
57 * has the offset
58 * -- first INIFILE_KEY of first app
59 * -- name of first key (zero-terminated),
60 * of which the first INIFILE_KEY.offKeyName
61 * has the offset
62 * -- data block of first key, of which the first
63 * INIFILE_KEY.offKeyData has the offset;
64 * the length is in INIFILE_KEY.lenKeyData
65 *
66 * -- subsequent INIFILE_KEY's, names, data, if any
67 *
68 * -- subsequent INIFILE_APP's, if any
69 * ...
70 *
71 *@@added V0.9.5 (2000-08-10) [umoeller]
72 */
73
74 typedef struct _INIFILE_HEADER
75 {
76 ULONG magic; // Magic Footprint, Always $FFFFFFFF (?)
77 ULONG offFirstApp; // Offset of First Application in File
78 ULONG lenFile; // Length of INI File
79 ULONG filler1; // Always $00000000 (?)
80 ULONG filler2; // Always $00000000 (?)
81 } INIFILE_HEADER, *PINIFILE_HEADER;
82
83 /*
84 *@@ INIFILE_APP:
85 * application entry in OS/2 INI file.
86 * The first application comes right after
87 * INIFILE_HEADER. After INIFILE_APP, the
88 * Prf* functions store the app name and
89 * the keys which belong to this application
90 * (INIFILE_KEY).
91 *
92 *@@added V0.9.5 (2000-08-10) [umoeller]
93 */
94
95 typedef struct _INIFILE_APP
96 {
97 ULONG offNextApp; // Offset of Next Application in File
98 // (0 No More Apps)
99 ULONG offFirstKeyInApp; // Offset of Application's First Key Entry
100 ULONG filler1; // Always $00000000 (?)
101 USHORT lenAppName; // Length of Application Name
102 // (incl. terminating \0)
103 USHORT _lenAppName; // Always same as above (?)
104 ULONG offAppName; // Offset of ASCIIZ Application Name
105 } INIFILE_APP, *PINIFILE_APP;
106
107 /*
108 *@@ INIFILE_KEY:
109 * key entry in OS/2 INI file.
110 * The first key in an application comes right after
111 * its INIFILE_APP. After INIFILE_KEY, the Prf*
112 * functions store the key name and finally the
113 * data for that key.
114 *
115 *@@added V0.9.5 (2000-08-10) [umoeller]
116 */
117
118 typedef struct _INIFILE_KEY
119 {
120 ULONG offNextKeyInApp; // Offset of Next Key in Application
121 // (0 = No More Keys)
122 ULONG filler1; // Always $00000000 (?)
123 USHORT lenKeyName; // Length of Key Name (incl. terminating \0)
124 USHORT _lenKeyName; // Always same as above (?)
125 ULONG offKeyName; // Offset of ASCIIZ Key Name
126 USHORT lenKeyData; // Length of Key Data
127 USHORT _lenKeyData; // Always same as above (?)
128 ULONG offKeyData; // Offset of Key Data (ASCII(Z) or Binary)
129 } INIFILE_KEY, *PINIFILE_KEY;
