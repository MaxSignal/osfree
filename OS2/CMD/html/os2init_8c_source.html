37void InitOS( int argc, char **argv )
38{
39 char *arg;
40 ULONG ulTimes = 0L;
41 char *errmsg, *pszOpts;
42
43 (void)_setmode( STDIN, O_TEXT );
44 (void)_setmode( STDOUT, O_TEXT );
45 (void)_setmode( STDERR, O_TEXT );
46
47 // initialize the critical variables
48 memset( &cv, '\0', sizeof(cv) );
49 cv.bn = -1;
50
51 QueryCountryInfo();
52 InitSystemVariables(); // initialize 4OS2 global variables
53
54 (void)VioSetAnsi( ANSI_ON, 0 ); // enable ANSI escape sequences
55
56 SetCurSize( 0 ); // set the default cursor shape
57
58 // get shell #, initialize INI data, process any inherited INI data
59 init_ini();
60
61 // support the CMD.EXE SHELL syntax for specifying a COMSPEC
62 // directory - if first arg on line is a directory, it must
63 // be a COMSPEC request
64 if (( argv[1] != NULL ) && ( argv[1][0] != gpIniptr->SwChr ) && ( is_dir( argv[1] ))) {
65 strcpy( gszCmdline, strupr(*(++argv) ));
66 mkdirname( gszCmdline, OS2_NAME );
67 } else
68 strcpy( (char *)gszCmdline, _pgmptr );
69
70 // save _pgmptr (its location in the environment will be overwritten)
71 _pgmptr = (char *)strdup( gszCmdline );
72
73 // get program name (for session title)
74 OS2_NAME = strdup( fname_part( gszCmdline ));
75
76 // get title; if none, change to "4OS2 Full Screen" or "4OS2 Window"
77 gszSessionTitle[0] = '\0';
78 update_task_list( NULL );
79
80 if ( setjmp( cv.env ) == -1 )
81 _exit( ERROR_EXIT );
82
83 RemoveQuotes( pszCmdLineOpts );
84 for ( argc = 0; (( arg = ntharg( pszCmdLineOpts, argc )) != NULL ); argc++) {
85next_start_arg:
86 // check for .INI filename
87 if ( *arg == '@' ) {
88 // parse the user's file - if not an .INI file, must
89 // be something like "@COPY ..."
90 if ( process_ini( arg ))
91 break;
92 } else if (( *arg == gpIniptr->SwChr ) || ( *arg == '-' )) {
93 // kludge to detect "//inistuff" because ntharg()
94 // only returns the first "/"
95 switch ( _ctoupper( gpNthptr[1] )) {
96 case '/': // INI file line
97 // skip leading '/'
98 argc++;
99 arg = gpNthptr;
100
101 // be sure default INI file has been processed
102 // so // overrides
103 default_ini();
104
105 // copy so IniLine can add an extra null
106 strcpy( gszCmdline, first_arg( arg+2 ));
107 if ( IniLine( gszCmdline, &gaInifile, 0, 0, 0, &errmsg ))
108 error( ERROR_4DOS_BAD_DIRECTIVE, arg );
109 break;
110
111 case 'C': // transient load
112 gnTransient = 1;
113
114 // kludge for people who do "/Ccommand"
115 if ( arg[2] ) {
116 arg += 2;
117 gpNthptr += 2;
118 goto next_start_arg;
119 }
120
121 // check for a "4OS2 /C 4OS2" & turn it into
122 // a "4OS2"
123 if (( errmsg = ntharg( pszCmdLineOpts, argc+1 )) != NULL )
124 errmsg = fname_part( errmsg);
125
126 if (( errmsg != NULL ) && ( cv.bn < 0 ) && ( stricmp( errmsg,OS2_NAME) == 0 )) {
127 gnTransient = 0;
128 argc++;
129 }
130
131 break;
132
133 case 'K': // permanent load in OS/2
134 gpIniptr->ShellLevel = 0;
135
136 // OS/2 passes STARTUP.CMD as "/kSTARTUP.CMD"
137 if ( isalnum( arg[2] )) {
138 arg += 2;
139 gpNthptr += 2;
140 goto next_start_arg;
141 }
142
143 break;
144
145 case 'L': // set local alias & history lists
146 // be sure default INI file has been processed
147 // so /L overrides
148 arg = gpNthptr;
149 default_ini();
150
151 switch ( _ctoupper( arg[2] )) {
152 case 'A':
153 gpIniptr->LocalAliases = 1;
154 break;
155 case 'D':
156 gpIniptr->LocalDirHistory = 1;
157 break;
158 case 'H':
159 gpIniptr->LocalHistory = 1;
160 break;
161 default:
162 gpIniptr->LocalAliases = 1;
163 gpIniptr->LocalHistory = 1;
164 gpIniptr->LocalDirHistory = 1;
165 }
166
167 break;
168
169 case 'Q': // for compatibility w/CMD.EXE
170 break;
171
172 case 'S':
173 // don't set up a ^C / ^BREAK handler; they
174 // will be ignored
175 nSignalHandler = 1;
176 break;
177
178 default:
179 // kludge for "4OS2 -c ..."
180 if ( *arg == '-' )
181 goto args_done;
182 error( ERROR_INVALID_PARAMETER, arg );
183 }
184 } else
185 break;
186 }
187
188args_done:
189 pszOpts = gpNthptr;
190
191 // process the default INI file if necessary
192 default_ini();
193
194 // set the default cursor shape (must be after INI file is processed)
195 SetCurSize(0 );
196
197 // display the passed command line if requested
198 if (gpIniptr->INIDebug & INI_DB_SHOWTAIL) {
199 qprintf( STDERR, TAILIS, NAME_HEADER, getcmd( gszCmdline ));
200 (void)GetKeystroke( EDIT_NO_ECHO | EDIT_ECHO_CRLF );
201 }
202
203 // duplicate CMD.EXE IF bug
204 if ( gpIniptr->DupBugs ) {
205 // set IF processing
206 commands[ findcmd( "IF", 1 )].pflag |= CMD_DETACH_LINE;
207 }
208
209 // get the international format chars
210 QueryCountryInfo();
211
212 // init environment, alias & history lists
213 init_env();
214
215 // set up ^C and ^BREAK handling
216 HoldSignals();
217
218 if (nSignalHandler == 2) {
219 // set signal handler (structure must be on main() stack)
220 pExceptionStruct->prev_structure = 0;
221 pExceptionStruct->ExceptionHandler = (_ERR *)&BreakHandler;
222 (void)DosSetExceptionHandler( pExceptionStruct );
223 (void)DosSetSignalExceptionFocus( SIG_SETFOCUS, &ulTimes );
224 }
225
226 // if non-transient, display signon message
227 if ( gnTransient == 0 ) {
228 // Set the window position if requested
229 set_window();
230
231 // display copyright
232 DisplayCopyright();
233 }
234
235 // get startup disk for 4START and 4EXIT
236 AUTOSTART[0] = AUTOEXIT[0] = gpIniptr->BootDrive;
237
238 // execute 4START.BTM / 4START.CMD
239 find_4files( AUTOSTART );
240
241 // execute remainder of command line
242 if ( pszOpts != NULL ) {
243 pszOpts = skipspace( pszOpts );
244 if ( *pszOpts ) {
245 RemoveQuotes( pszOpts );
246 command( pszOpts, ( gnTransient == 0 ));
247 }
248 }
249}

252// support CMD.EXE syntax: CMD /C ""a" b c" (with exception
253// for embedded whitespace: START /pgm "t start.cmd")
254static void RemoveQuotes( char *line )
255{
256 int nLength;
257
258 nLength = ( strlen( line ) - 1 );
259 if (( line[0] == '"' ) && ( line[ nLength ] == '"' ) && ( is_file( first_arg( line )) == 0 )) {
260 // remove the trailing double quote
261 line[ nLength ] = '\0';
262
263 // remove the leading double quote
264 strcpy( line, line + 1 );
265 }
266}

269// initialize the OS/2 environment, alias, and history space
270static void init_env( void )
271{
272 static char szAliasSharename[32] = SHR_4OS2_ALIAS;
273 static char szHistorySharename[32] = SHR_4OS2_HISTORY;
274 static char szDirHistSharename[32] = SHR_4OS2_DIRHIST;
275 PUINT puSize;
276
277 // make sure default environment size is big enough (minimum 4K)
278 if ( QueryMemSize( glpEnvironment) < ENVIRONMENT_SIZE )
279 (void)ReallocMem( glpEnvironment, ENVIRONMENT_SIZE );
280
281 // try to open shared segments for the alias & history lists. If
282 // they don't exist, create them. (We create shared segs for the
283 // alias & history lists, so they can be shared in all sessions.)
284 // If we started with "4OS2 /L[a|d|h]", then use local (shared
285 // only for that session) lists for aliases and/or history.
286
287 // create an extension consisting of the session number
288 if ( gpIniptr->LocalAliases != 0 )
289 sprintf( strend( szAliasSharename ), ".%03d", gpLIS->sgCurrent );
290
291 if ( gpIniptr->LocalHistory != 0 )
292 sprintf( strend( szHistorySharename ), ".%03d", gpLIS->sgCurrent );
293
294 if ( gpIniptr->LocalDirHistory != 0 )
295 sprintf( strend( szDirHistSharename ), ".%03d", gpLIS->sgCurrent );
296
297 if ( DosGetNamedSharedMem( (PVOID)&glpAliasList, szAliasSharename, (PAG_READ | PAG_WRITE) )) {
298 // allocate 4K out of a 64K sparse memory block
299 if (( DosAllocSharedMem( (PVOID)&glpAliasList, szAliasSharename, 65535L, (PAG_READ | PAG_WRITE))) || ( DosSetMem( glpAliasList, ALIAS_SIZE, (PAG_COMMIT | PAG_READ | PAG_WRITE))))
300 _exit( error(ERROR_NOT_ENOUGH_MEMORY, NULL ));
301
302 // initialize the alias list to 0's
303 memset( glpAliasList, '\0', ALIAS_SIZE );
304 }
305
306 // kludge for different entries used in OPTION
307 gpIniptr->DirHistoryNew = gpIniptr->DirHistorySize;
308 gpIniptr->HistoryNew = gpIniptr->HistorySize;
309
310 if ( DosGetNamedSharedMem( (PVOID)&glpHistoryList, szHistorySharename, (PAG_READ | PAG_WRITE) )) {
311 // default history size is 1K
312 if ( DosAllocSharedMem( (PVOID)&glpHistoryList, szHistorySharename, gpIniptr->HistorySize + sizeof(UINT), (PAG_READ | PAG_WRITE | PAG_COMMIT) ))
313 _exit( error( ERROR_NOT_ENOUGH_MEMORY, NULL ));
314
315 // initialize the history list to 0's
316 memset( glpHistoryList, '\0', gpIniptr->HistorySize + sizeof(UINT) );
317
318 // set the history size in the shared area
319 puSize = (PUINT)glpHistoryList;
320 *puSize = (UINT)gpIniptr->HistorySize;
321 } else {
322 // set size of existing shared memory
323 puSize = (PUINT)glpHistoryList;
324 gpIniptr->HistorySize = *puSize;
325 }
326
327 glpHistoryList += sizeof(UINT);
328
329 if ( DosGetNamedSharedMem( (PVOID)&glpDirHistory, szDirHistSharename, (PAG_READ | PAG_WRITE) )) {
330 // default directory history size is 256 bytes
331 if ( DosAllocSharedMem( (PVOID)&glpDirHistory, szDirHistSharename, gpIniptr->DirHistorySize + sizeof(UINT), (PAG_READ | PAG_WRITE | PAG_COMMIT) ))
332 _exit( error( ERROR_NOT_ENOUGH_MEMORY, NULL ));
333
334 // initialize the directory history list to 0's
335 memset( glpDirHistory, '\0', gpIniptr->DirHistorySize + sizeof(UINT) );
336 puSize = (PUINT)glpDirHistory;
337 *puSize = (UINT)gpIniptr->DirHistorySize;
338 } else {
339 // set size of existing shared memory
340 puSize = (PUINT)glpDirHistory;
341 gpIniptr->DirHistorySize = *puSize;
342 }
343
344 glpDirHistory += sizeof(UINT);
345
346 // create queue for child sessions to notify us when they exit
347 sprintf( gszQueueName, "\\QUEUES\\4OS32\\TQ%03d.%03d", gpLIS->sgCurrent, gpLIS->pidCurrent);
348
349 (void)DosCreateQueue( &ghQueueHandle, (QUE_FIFO | QUE_CONVERT_ADDRESS), gszQueueName );
350
351 // set DescriptionName
352 if ( gpIniptr->DescriptName != INI_EMPTYSTR )
353 sprintf( DESCRIPTION_FILE, "%.63s", gpIniptr->StrData + gpIniptr->DescriptName );
354
355 // set COMSPEC
356 sprintf( gszCmdline, COMSPEC_OS2, COMSPEC, _pgmptr );
357 add_variable( gszCmdline );
358}

537// initialize the INI file data
538static void init_ini( void )
539{
540 unsigned int i;
541 PCH prev_shell_sharedmem = 0L;
542 char shell_sharename[32];
543 char szFname[MAXFILENAME + 1];
544 int save_shell;
545 PCH prev_ini_data;
546 INIFILE *prev_ini;
547 ULONG prev_size;
548 int valid = 0;
549
550 // get shell number
551 for ( i = 0; ; i++ ) {
552 // if shared memory for shell i exists, save its pointer so we
553 // can use it to inherit the INI file data
554 // if not, i is our shell number -- reserve a shared memory segment
555 // for it
556 sprintf( shell_sharename, "\\SHAREMEM\\4OS2.%03d\\SHELL%03d.%03d", VER_BUILD, gpLIS->sgCurrent, i );
557
558 if ( DosGetNamedSharedMem( (PVOID)&prev_shell_sharedmem, shell_sharename, (PAG_READ | PAG_WRITE)) != 0 ) {
559 if ( DosAllocSharedMem( (PVOID)&shell_sharedmem, shell_sharename, INI_TOTAL_BYTES, (PAG_READ | PAG_WRITE | PAG_COMMIT)) == 0 )
560 memset( shell_sharedmem, '\0', INI_TOTAL_BYTES );
561 break;
562 }
563 }
564
565 // set shell number, clear out dummy log file name
566 gaInifile.ShellNum = (char)i;
567
568 if (( gaInifile.ShellNum != 0 ) && ( prev_shell_sharedmem != 0L )) {
569 // we have an old INI file, see if we can inherit it (signature must
570 // match and string / key data must fit in our data areas)
571 prev_ini = (INIFILE *)(prev_ini_data = prev_shell_sharedmem);
572 prev_size = QueryMemSize( prev_shell_sharedmem );
573 valid = ( prev_size > (ULONG)sizeof(INIFILE) ) && ( prev_ini->INISig == INI_SIG ) && ( prev_ini->INIBuild == VER_BUILD ) && ( prev_ini->StrUsed <= INI_STRMAX ) && ( prev_ini->KeyUsed <= INI_KEYMAX );
574 }
575
576 if ( valid ) {
577 // old INI file data seems valid so inherit it
578 save_shell = gaInifile.ShellNum;
579 memmove( &gaInifile, prev_ini_data, sizeof(INIFILE) );
580 prev_ini_data += sizeof(INIFILE);
581 memmove( gaIniStrings, prev_ini_data, prev_ini->StrUsed );
582 prev_ini_data += prev_ini->StrUsed;
583 memmove( gaIniKeys, prev_ini_data, ( 2 * sizeof(unsigned int) * prev_ini->KeyUsed ));
584 gaInifile.ShellNum = save_shell;
585 } else {
586 // no inheritance, clear the INI file data
587 gaInifile.StrUsed = 0;
588 gaInifile.KeyUsed = INI_DEFKEYS;
589 IniClear(&gaInifile );
590 gaInifile.INIBuild = VER_BUILD;
591 gaInifile.INISig = INI_SIG;
592 gaInifile.BootDrive = gchSysBootDrive;
593 }
594
595 // set up pointer to strings area
596 gaInifile.StrData = gaIniStrings;
597 gaInifile.StrMax = INI_STRMAX;
598
