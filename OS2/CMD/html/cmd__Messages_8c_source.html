30/* A little note: a standard 80x25 screenful is 2000 bytes worth, so we */
31/* should always be safe with a buffer of 2Kb (AFAIK there are no */
32/* messages occupying more than one screenful). To be sure one can */
33/* test the cbMsgL returned, if it is > 2048 then the message has been */
34/* truncated. */
35
36/* example 1: a call to show a message that has no parameters: */
37/* */
38/* cmd_prntmsg(NULL, 0L, 1001L,"OSO001.MSG"); */
39/* */
40/* */
41/* example 2: a call to show a message that has 1 parameter: */
42/* */
43/* CHAR *pTable[1]; */
44/* CHAR fname[CCHMAXPATH] = ""; */
45/* */
46/* strcpy(fname, "C:\\CONFIG.SYS"); */
47/* pTable[0] = fname; */
48/* cmd_prntmsg(pTable, 1L, 1480L,"OSO001.MSG"); */
49/* */

51int cmd_prntmsg(PCHAR *pTable, ULONG cTable, ULONG ulMsgID,PSZ pszFileName)
52{
53 UCHAR pBuf[MAX_MESSAGE] = "";
54 ULONG cbMsgL;
55 APIRET ulrc;
56
57 ulrc = DosGetMessage(pTable, cTable, pBuf, sizeof(pBuf), ulMsgID, pszFileName, &cbMsgL);
58
59 if (ulrc != NO_ERROR)
60 printf(all_GetSystemErrorMessage(ulrc));
61 else
62 printf("%s", pBuf);
63
64 return ulrc;
65}
