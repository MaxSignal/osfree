105/*
106 *@@ operator[]:
107 * returns the ul'th character of the
108 * string or 0 if ul is too large.
109 *
110 * Warning: this counts bytes, not UTF-8
111 * characters.
112 */
113
114char BSUString::operator[](const size_type ul) // in: character offset
115 const
116{
117 STRINGLOCK;
118 if ( (_pBuf)
119 && (_pBuf->_str.psz)
120 && (ul < _pBuf->_str.ulLength)
121 )
122 return (_pBuf->_str.psz[ul]);
123
124 return ('\0');
125}

127/*
128 *@@ assign:
129 * assigns another BSUString to this. This
130 * is the only direct assign that is supported
131 * for BSUString (and gets called thru
132 * BSUString::operator= as well); otherwise
133 * you will have to use one of the
134 * assignUtf8 methods.
135 *
136 * Returns *this.
137 */
138
139BSUString& BSUString::assign(const BSUString &s)
140{
141 STRINGLOCK;
142 // avoid self assignment
143 if (_pBuf != s._pBuf)
144 {
145 FreeBuf();
146 CopyFrom(s);
147 }
148
149 return (*this);
150}

248/*
249 *@@ append:
250 * appends the given BSUString to this.
251 *
252 * Also used internally by BSUString::operator+=.
253 *
254 * Returns *this.
255 */
256
257BSUString& BSUString::append(const BSUString &s)
258{
259 STRINGLOCK;
260 size_type ulSourceLen;
261 if (ulSourceLen = s.size())
262 {
263 // assert that we have a modifiable
264 // copy with enough space
265 reserve(ulSourceLen + size() + 1);
266 xstrcats(&_pBuf->_str, &s._pBuf->_str);
267 }
268
269 return (*this);
270}

304 *@@added V0.9.18 (2002-03-08) [umoeller]

307BSUString& BSUString::appendCP(BSUniCodec *pCodec,
308 const BSString &str)
309{
310 STRINGLOCK;
311 if (str.length())
312 {
313 BSUString ustr;
314 ustr.assignCP(pCodec, str);
315 append(ustr);
316 }
317
318 return (*this);

321/*
322 *@@ compareI:
323 * case-insensitive compare.
324 *
325 * Compares this against the given BSUString,
326 * without respect to case, with full
327 * Unicode support by calling encicmp().
328 *
329 * Returns 0 if the two strings are the
330 * same, -1 if "this" is smaller, 1 if

333 *@@added V0.9.20 (2002-07-03) [umoeller]
