173/* This structure is basically a concatenation of the LVM Drive_Control_Data and
174 * Drive_Information_Record structures, which places all the information
175 * together in one place for convenience. We make some modifications to the
176 * actual values, however; for instance, converting the size value from sectors
177 * to megabytes.
178 */
179typedef struct _LVM_Disk_Info {
180 CARDINAL32 iNumber; // OS/2 drive number of the disk
181 CARDINAL32 iSize; // total size of the disk
182 DoubleWord iSerial; // disk serial number
183 ADDRESS handle; // LVM handle identifying this disk
184 CARDINAL32 iCylinders; // number of cylinders
185 CARDINAL32 iHeads; // number of heads per cylinder
186 CARDINAL32 iSectors; // number of sectors per track
187 BOOLEAN fPRM; // disk is partitioned removable media
188 BYTE reserved1[ 3 ]; // (reserved field for alignment)
189 CARDINAL32 iTotalFree; // total number of free sectors on the disk
190 CARDINAL32 iLargestFree; // size of the largest block of free space, in sectors
191 BOOLEAN fCorrupt; // disk partitioning information is invalid
192 BOOLEAN fUnusable; // disk MBR cannot be accessed
193 BOOLEAN fIOErr; // last I/O operation on the disk failed
194 BOOLEAN fBigFloppy; // disk is a "large floppy" device
195 char szName[ DISK_NAME_SIZE ]; // user-assigned name of the disk
196} LVMDISKINFO, *PLVMDISKINFO;
197

199/* Similarly, this structure concatenates the fields from Volume_Control_Data
200 * and Volume_Information_Record structures. Again, we convert the size from
201 * sectors to MiB when populating the structure.
202 */
203typedef struct _LVM_Volume_Info {
204 DoubleWord iSerial; // serial number of this volume
205 ADDRESS handle; // handle of this volume
206 BOOLEAN fCompatibility; // volume is a compatibility volume
207 BYTE bDevice; // type of device on which volume resides
208 BYTE reserved1[ 2 ]; // (reserved field for alignment)
209 CARDINAL32 iSize; // size of this volume
210 CARDINAL32 iPartitions; // number of partitions in this volume
211 CARDINAL32 iConflict; // a drive letter conflict occurred
212 BYTE reserved2; // (actually a duplicate of fCompatibility)
213 BOOLEAN fBootable; // volume is bootable
214 char cPreference; // preferred (user-selected) drive letter
215 char cLetter; // actual current drive letter
216 char cInitial; // drive letter when LVM was opened
217 BOOLEAN fNew; // volume was created during this session
218 BYTE iStatus; // bootability status
219 BYTE reserved3; // (reserved field for alignment)
220 char szName[ VOLUME_NAME_SIZE ]; // user-assigned volume name
221 char szFS[ FILESYSTEM_NAME_SIZE ]; // name of filesystem on this volume
222} LVMVOLUMEINFO, *PLVMVOLUMEINFO;
