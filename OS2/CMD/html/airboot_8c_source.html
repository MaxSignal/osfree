30#define AIRBOOT_INSTALLER "AIRBOOT2.EXE" // AiR-BOOT installer executable
31#define AIRBOOT_INSTPARMS "/CID" // AiR-BOOT installer parameters

33#define BOFF_AIRBOOT_SIG 0x2 // offset of AiR-BOOT signature in MBR
34#define ACH_AIRBOOT_SIG "AiRBOOT" // contents of a valid AiR-BOOT signature
35#define CCH_AIRBOOT_SIG 7 // length of a valid AiR-BOOT signature
36
37#define AB_PROMPT_MAXZ (( STRING_RES_MAXZ * 4 ) + 16 )

40/* ------------------------------------------------------------------------- *
41 * AirBoot_IsInstalled() *
42 * *
43 * Check to see if AiR-BOOT is installed by scanning the MBR on disk 1 for *
44 * the AiR-BOOT signature. *
45 * *
46 * ARGUMENTS: *
47 * PCARDINAL32 pError : Pointer to LVM return code. *
48 * *
49 * RETURNS: BOOL *
50 * TRUE if the AiR-BOOT (i.e. its signature) was found in the MBR. *
51 * FALSE if AiR-BOOT was not found in the MBR, or if an error occurred. *
52 * ------------------------------------------------------------------------- */
53BOOL AirBoot_IsInstalled( PCARDINAL32 pError )
54{
55 PSZ pchSig; // AiR-BOOT signature bytes
56 BYTE abSector[ BYTES_PER_SECTOR ] = {0}; // buffer for MBR contents
57
58 LvmReadSectors( 1, 0, 1, &abSector, pError );
59 if ( *pError == LVM_ENGINE_NO_ERROR ) {
60 pchSig = (PSZ) abSector + BOFF_AIRBOOT_SIG;
61 if ( strncmp( pchSig, ACH_AIRBOOT_SIG, CCH_AIRBOOT_SIG ) == 0 )
62 return TRUE;
63 }
64 return FALSE;
65}

68/* ------------------------------------------------------------------------- *
69 * AirBoot_GetInstaller() *
70 * *
71 * Determine whether or not the AiR-BOOT installer is available, and define *
72 * the pathname as appropriate. If the installer is not found, disable *
73 * AiR-BOOT in the global flags. *
74 * *
75 * ARGUMENTS: *
76 * PDVMGLOBAL pGlobal: Pointer to global program data *
77 * *
78 * RETURNS: BOOL *
79 * TRUE if the AiR-BOOT installer was found, FALSE otherwise *
80 * ------------------------------------------------------------------------- */
81BOOL AirBoot_GetInstaller( PDVMGLOBAL pGlobal )
82{
83 APIRET rc;
84
85 rc = DosSearchPath( SEARCH_ENVIRONMENT | SEARCH_CUR_DIRECTORY, "PATH",
86 AIRBOOT_INSTALLER, pGlobal->szABEXE, CCHMAXPATH+1 );
87 if ( rc )
88 memset( pGlobal->szABEXE, 0, CCHMAXPATH+1 );
89
90 if ( !pGlobal->szABEXE[0] ) {
91 pGlobal->fsProgram &= ~FS_APP_ENABLE_AB;
92 return FALSE;
93 }
94
95 return TRUE;
96}

99/* ------------------------------------------------------------------------- *
100 * AirBoot_Install *
101 * *
102 * Executes the AiR-BOOT install program (in silent/CID mode). *
103 * *
104 * ARGUMENTS: *
105 * HWND hwnd: application client window handle *
106 * *
107 * RETURNS: n/a *
108 * ------------------------------------------------------------------------- */
109void AirBoot_Install( HWND hwnd )
110{
111 PDVMGLOBAL pGlobal; // pointer to global program data
112 PPROGDETAILS pDetails; // structure used by WinStartApp()
113 PSZ psz; // offset within path string
114 UCHAR szPath[ CCHMAXPATH + 1 ], // various string buffers
115 szRes1[ STRING_RES_MAXZ ],
116 szRes2[ STRING_RES_MAXZ ],
117 szError[ STRING_APP_ERROR_MAXZ ],
118 szBuffer[ AB_PROMPT_MAXZ ] = {0};
119 ULONG ulErr, // WinGetLastError() error code
120 flStyle; // WinMessageBox() style flags
121 CARDINAL32 rc; // LVM return code
122
123
124 pGlobal = WinQueryWindowPtr( hwnd, 0 );
125 if ( !pGlobal || !pGlobal->szABEXE[0] )
126 return;
127
128 // Prompt for confirmation
129 WinLoadString( pGlobal->hab, pGlobal->hmri,
130 IDS_AIRBOOT_INSTALL_TITLE, STRING_RES_MAXZ, szRes1 );
131 WinLoadString( pGlobal->hab, pGlobal->hmri,
132 IDS_AIRBOOT_INSTALL_1, STRING_RES_MAXZ, szBuffer );
133 WinLoadString( pGlobal->hab, pGlobal->hmri,
134 IDS_AIRBOOT_INSTALL_2, STRING_RES_MAXZ, szRes2 );
135 strncat( szBuffer, szRes2, AB_PROMPT_MAXZ );
136 if ( pGlobal->fsEngine & FS_ENGINE_PENDING ) {
137 WinLoadString( pGlobal->hab, pGlobal->hmri,
138 IDS_SAVE_WARNING, STRING_RES_MAXZ, szRes2 );
139 strncat( szBuffer, szRes2, AB_PROMPT_MAXZ );
140 strncat( szBuffer, "\r\n\r\n", AB_PROMPT_MAXZ );
141 flStyle = MB_YESNO | MB_WARNING | MB_MOVEABLE;
142 }
143 else
144 flStyle = MB_YESNO | MB_QUERY | MB_MOVEABLE;
145 WinLoadString( pGlobal->hab, pGlobal->hmri,
146 IDS_PROCEED_CONFIRM, STRING_RES_MAXZ, szRes2 );
147 strncat( szBuffer, szRes2, AB_PROMPT_MAXZ );
148 strncat( szBuffer, "\r\n\r\n", AB_PROMPT_MAXZ );
149
150 if ( WinMessageBox( HWND_DESKTOP, hwnd, szBuffer, szRes1,
151 0, flStyle ) != MBID_YES )
152 return;
153
154 // Save any changes still pending
155 if ( pGlobal->fsEngine & FS_ENGINE_PENDING ) {
156 LvmCommit( &rc );
157 if ( pGlobal->pLog ) {
158 fprintf( pGlobal->pLog, "-------------------------------------------------------------------------------\n");
159 fprintf( pGlobal->pLog, "LVM CHANGES COMMITTED: %u\n", rc );
160 }
161 if ( rc != LVM_ENGINE_NO_ERROR ) {
162 PopupEngineError( NULL, rc, hwnd, pGlobal->hab, pGlobal->hmri );
163 return;
164 }
165 SetModified( hwnd, FALSE );
166 }
167
168 // Clear our data structures and close the LVM engine
169 LVM_FreeData( pGlobal );
170 LVM_Stop( pGlobal );
171
172 // Run the Air-Boot install utility (external program)
173 strncpy( szPath, pGlobal->szABEXE, CCHMAXPATH );
174 if (( psz = strrchr( szPath, '\\')) != NULL ) *psz = 0;
175 else szPath[0] = 0;
176 pDetails = (PPROGDETAILS) calloc( 1, sizeof(PROGDETAILS) );
177 pDetails->Length = sizeof( PROGDETAILS );
178 pDetails->progt.progc = PROG_DEFAULT;
179 pDetails->progt.fbVisible = SHE_INVISIBLE;
180 pDetails->pszExecutable = strdup( pGlobal->szABEXE );
181 pDetails->pszParameters = strdup( AIRBOOT_INSTPARMS );
182 pDetails->pszStartupDir = ( szPath[0] ? szPath : NULL );
183 pDetails->swpInitial.fl = SWP_HIDE;
184
185 pGlobal->happAB = WinStartApp( hwnd, pDetails, NULL, NULL,
186 SAF_INSTALLEDCMDLINE | SAF_STARTCHILDAPP );
187 if ( pGlobal->happAB == NULLHANDLE ) {
188 // Show an error if the program failed to start
189 ulErr = WinGetLastError( pGlobal->hab );
190 WinLoadString( pGlobal->hab, pGlobal->hmri,
191 IDS_ERROR_STARTAPP, STRING_RES_MAXZ, szRes1 );
192 WinLoadString( pGlobal->hab, pGlobal->hmri,
193 IDS_ERROR_GENERIC, STRING_RES_MAXZ, szRes2 );
194 sprintf( szError, szRes1, pDetails->pszExecutable,
195 ERRORIDERROR( ulErr ));
196 WinMessageBox( HWND_DESKTOP, HWND_DESKTOP, szError, szRes2,
197 0, MB_OK | MB_ERROR );
198 }
199
200 free( pDetails->pszExecutable );
201 free( pDetails->pszParameters );
202 free( pDetails );
203}

207/* ------------------------------------------------------------------------- *
208 * AirBoot_InstallerExit *
209 * *
210 * Interprets the result of executing the AiR-BOOT install program. Called *
211 * on WM_APPTERMINATENOTIFY from the main window handler. *
212 * *
213 * ARGUMENTS: *
214 * HWND hwnd : application client window handle *
215 * ULONG rc : return code from install program *
216 * PDVMGLOBAL pGlobal: Pointer to global program data *
217 * *
218 * RETURNS: n/a *
219 * ------------------------------------------------------------------------- */
220void AirBoot_InstallerExit( HWND hwnd, ULONG rc, PDVMGLOBAL pGlobal )
221{
222 UCHAR szRes1[ STRING_RES_MAXZ ],
223 szRes2[ STRING_RES_MAXZ ],
224 szRes3[ STRING_RES_MAXZ ],
225 szError[ STRING_APP_ERROR_MAXZ ] = {0};
226 ULONG ulID;
227
228
229 // Restart the LVM engine
230 LVM_Start( pGlobal->pLog, pGlobal->hab, pGlobal->hmri );
231 LVM_InitData( hwnd, pGlobal );
232
233 // Check to see whether AiR-BOOT was actually installed successfully
234 if ( pGlobal->fsEngine & ~FS_ENGINE_AIRBOOT ) {
235 switch ( rc ) {
236 case 1: ulID = IDS_ERROR_IO; break;
237 case 2: ulID = IDS_ERROR_GEOMETRY; break;
238 case 3: ulID = IDS_ERROR_INSUFFICIENT; break;
239 default: ulID = IDS_ERROR_OTHER; break;
240 }
241 WinLoadString( pGlobal->hab, pGlobal->hmri,
242 IDS_AIRBOOT_FAILED_TITLE, STRING_RES_MAXZ, szRes1 );
243 WinLoadString( pGlobal->hab, pGlobal->hmri,
244 IDS_AIRBOOT_FAILED, STRING_RES_MAXZ, szRes2 );
245 WinLoadString( pGlobal->hab, pGlobal->hmri,
246 ulID, STRING_RES_MAXZ, szRes3 );
247 sprintf( szError, szRes2, szRes3, rc );
248 WinMessageBox( HWND_DESKTOP, hwnd, szError, szRes1,
249 0, MB_OK | MB_ERROR | MB_MOVEABLE );
250 }
251 else {
252 WinSendMsg( pGlobal->hwndMenu, MM_SETITEMATTR,
253 MPFROM2SHORT( ID_AIRBOOT_REMOVE, TRUE ),
254 MPFROM2SHORT( MIA_DISABLED, 0 ));
255 WinLoadString( pGlobal->hab, pGlobal->hmri,
256 IDS_SUCCESS_TITLE, STRING_RES_MAXZ, szRes1 );
257 WinLoadString( pGlobal->hab, pGlobal->hmri,
258 IDS_AIRBOOT_INSTALLED, STRING_RES_MAXZ, szRes2 );
259 WinMessageBox( HWND_DESKTOP, hwnd, szRes2, szRes1, 0,
260 MB_OK | MB_INFORMATION | MB_MOVEABLE );
261 }
262
263 pGlobal->happAB = NULLHANDLE;
264}

267/* ------------------------------------------------------------------------- *
268 * AirBoot_Delete() *
269 * *
270 * Forces an immediate rewrite of the Master Boot Record on the first disk *
271 * drive, in order to remove AiR-BOOT. *
272 * *
273 * ARGUMENTS: *
274 * HWND hwnd: application client window handle *
275 * *
276 * RETURNS: N/A *
277 * ------------------------------------------------------------------------- */
278void AirBoot_Delete( HWND hwnd )
279{
280 PDVMGLOBAL pGlobal;
281 UCHAR szRes1[ STRING_RES_MAXZ ],
282 szRes2[ STRING_RES_MAXZ ],
283 szBuffer[ STRING_RES_MAXZ * 2 ];
284 CARDINAL32 rc;
285
286
287 pGlobal = WinQueryWindowPtr( hwnd, 0 );
288 if ( !pGlobal ) return;
289
290 WinLoadString( pGlobal->hab, pGlobal->hmri,
291 IDS_AIRBOOT_DELETE, STRING_RES_MAXZ, szRes1 );
292 WinLoadString( pGlobal->hab, pGlobal->hmri,
293 IDS_NEWMBR_TITLE, STRING_RES_MAXZ, szBuffer );
294 WinLoadString( pGlobal->hab, pGlobal->hmri,
295 IDS_PROCEED_CONFIRM, STRING_RES_MAXZ, szRes2 );
296 strncat( szBuffer, szRes2, STRING_RES_MAXZ * 2 );
297 strncat( szBuffer, "\r\n", STRING_RES_MAXZ * 2 );
298 if ( WinMessageBox( HWND_DESKTOP, hwnd, szBuffer, szRes1, 0,
299 MB_YESNO | MB_WARNING | MB_MOVEABLE ) == MBID_YES )
300 {
301 LvmNewMBR( pGlobal->disks[ 0 ].handle, &rc );
302 if ( pGlobal->pLog ) {
303 fprintf( pGlobal->pLog, "-------------------------------------------------------------------------------\n");
304 fprintf( pGlobal->pLog, "Writing new MBR to disk: 0\nResult: %u\n", rc );
305 }
306 if ( rc != LVM_ENGINE_NO_ERROR )
307 PopupEngineError( NULL, rc, hwnd, pGlobal->hab, pGlobal->hmri );
308 else {
309 SetModified( hwnd, TRUE );
310 pGlobal->fsEngine &= ~FS_ENGINE_AIRBOOT;
311 WinSendMsg( pGlobal->hwndMenu, MM_SETITEMATTR,
312 MPFROM2SHORT( ID_AIRBOOT_REMOVE, TRUE ),
313 MPFROM2SHORT( MIA_DISABLED, MIA_DISABLED ));
314 WinLoadString( pGlobal->hab, pGlobal->hmri,
315 IDS_SUCCESS_TITLE, STRING_RES_MAXZ, szRes1 );
316 WinLoadString( pGlobal->hab, pGlobal->hmri,
317 IDS_NEWMBR_OK, STRING_RES_MAXZ, szRes2 );
318 WinMessageBox( HWND_DESKTOP, hwnd, szRes2, szRes1, 0,
319 MB_OK | MB_INFORMATION | MB_MOVEABLE );
320 }
321 }
