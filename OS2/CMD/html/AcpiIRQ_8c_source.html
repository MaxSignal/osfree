37ACPI_STATUS
38IRQPrintResourceInfo(
39 ACPI_RESOURCE *Resource,
40 void *Context)
41{
42 ULONG i;
43 switch (Resource->Type)
44 {
45 case ACPI_RESOURCE_TYPE_IRQ:
46 printf("used ");
47 for (i=0; i < Resource->Data.Irq.InterruptCount; i++)
48 printf(" IRQ%-2.d ", Resource->Data.Irq.Interrupts[i]);
49 if (Resource->Data.Irq.InterruptCount)
50 {
51 printf("triggerred by %s, ",Resource->Data.Irq.Triggering == ACPI_LEVEL_SENSITIVE? "Level":"Edge");
52 printf("polarity %s",Resource->Data.Irq.Polarity == ACPI_ACTIVE_LOW? "Low":"High");
53 //if (Resource->Data.Irq.Sharable) printf(", Sharable");
54 }
55 break;
56 case ACPI_RESOURCE_TYPE_EXTENDED_IRQ:
57 printf("PrCm %d",Resource->Data.ExtendedIrq.ProducerConsumer);
58 for (i=0; i < Resource->Data.ExtendedIrq.InterruptCount; i++)
59 printf(" IRQ%d ", Resource->Data.ExtendedIrq.Interrupts[i]);
60 if (Resource->Data.ExtendedIrq.InterruptCount)
61 {
62 printf("triggerred by %s, ",Resource->Data.ExtendedIrq.Triggering == ACPI_LEVEL_SENSITIVE? "Level":"Edge");
63 printf("polarity %s",Resource->Data.ExtendedIrq.Polarity == ACPI_ACTIVE_LOW? "Low":"High");
64 //if (Resource->Data.ExtendedIrq.Sharable) printf(", Sharable");
65 }
66 printf(" SourceIdx %d",Resource->Data.ExtendedIrq.ResourceSource.Index);
67 if (Resource->Data.ExtendedIrq.ResourceSource.StringLength)
68 printf("%s",Resource->Data.ExtendedIrq.ResourceSource.StringPtr);
69 break;
70 default:
71 break;
72 }
73
74 return AE_OK;
75}

105ULONG
106PrintPackage(
107 ULONG Count,
108 ACPI_OBJECT *oObj)
109{
110 ULONG j, i, Cnt;
111 ACPI_OBJECT *Obj;
112 char DevName[5];
113 ACPI_BUFFER Buffer;
114
115 printf("==================================================\n");
116 Obj = (ACPI_OBJECT *)oObj->Package.Elements;
117 for (Cnt=0,j = 0; j < Count; j++,Cnt++ ,oObj++)
118 {
119 Obj = (ACPI_OBJECT *)oObj->Package.Elements;
120 printf(" Device 0x%4.4x ", ( Obj->Integer.Value >> 16));
121 if ((Obj->Integer.Value & 0xffff) == 0xffff)
122 printf("Function ALL ");
123 else
124 printf("Function 0x%4.4x ",Obj->Integer.Value);
125 Obj++;
126 printf(" INT%c ",IntPin[Obj->Integer.Value]);
127 Obj++;
128 printf(" Route to ");
129 if (Obj->Reference.Handle)
130 {
131 Buffer.Length = 5;
132 Buffer.Pointer = DevName;
133 AcpiTkGetName(Obj->Reference.Handle, ACPI_SINGLE_NAME, &Buffer);
134 printf(" Device [");
135 for (i=0; i < 4; i++)
136 printf("%c",DevName[i]);
137 printf("] ");
138 AcpiTkWalkResources(Obj->Reference.Handle, METHOD_NAME__CRS, IRQPrintResourceInfo, NULL);
139 printf("\n");
140 Obj++;
141 }
142 else
143 {
144 Obj++;
145 printf(" IRQ%d\n",Obj->Integer.Value);
146 }
147 Obj++;
148 fflush(stdout);
149 }
150
151 printf("==================================================\n");
152 return Cnt;
153}

157ACPI_STATUS
158internal_PrintObjectInfo(
159 ACPI_HANDLE ObjHandle,
160 UINT32 NestingLevel,
161 void *Context,
162 void **ReturnValue)
163{
164 int i;
165 ACPI_BUFFER Buffer;
166 ACPI_DEVICE_INFO *DevInfo = NULL;
167 char achName[5];
168 ACPI_BUFFER Results;
169 ACPI_OBJECT Obj, *oObj;
170 ACPI_STATUS Status;
171
172 Results.Length = sizeof(Object);
173 Results.Pointer = &Object;
174
175 if (NestingLevel > 1)
176 {
177 for (i=0; i < NestingLevel-1 ; i++)
178 printf(" ");
179 }
180 printf("|\n");
181
182 if (NestingLevel > 1)
183 {
184 for (i=0; i<NestingLevel-1 ; i++)
185 printf(" ");
186 }
187
188 printf("+*");
189 memset(achName, 0, 5);
190 Buffer.Length = 5;
191 Buffer.Pointer = achName;
192 AcpiTkGetName(ObjHandle, ACPI_SINGLE_NAME, &Buffer);
193 printf("%s", achName); fflush(stdout);
194
195 AcpiTkGetObjectInfoAlloc3(ObjHandle, &DevInfo);
196 printf(" T:[%s]", AcpiTkUtGetTypeName(DevInfo->Type));
197 if (DevInfo->Valid & ACPI_VALID_HID)
198 printf(" HID:[%.9s]", &(DevInfo->HardwareId.String));
199 if (DevInfo->Valid & ACPI_VALID_CID)
200 printf(" CID:[%s]", &(DevInfo->HardwareId.String));
201 if (DevInfo->Valid & ACPI_VALID_UID)
202 printf(" UID:[%.9s]", &(DevInfo->UniqueId.String));
203 if (DevInfo->Valid & ACPI_VALID_ADR)
204 printf(" ADR:[0x%x]", DevInfo->Address);
205 if (DevInfo->Valid & ACPI_VALID_SXDS)
206 {
207 printf(" SxD State:[0x%02x:%02x:%02x:%02x]",
208 DevInfo->HighestDstates[0],
209 DevInfo->HighestDstates[1],
210 DevInfo->HighestDstates[2],
211 DevInfo->HighestDstates[3]);
212 }
213 printf("\n");
214 fflush(stdout);
215
216 Status = AcpiTkEvaluateObject(ObjHandle,METHOD_NAME__PRT,NULL,&Results);
217 if (Status == AE_OK)
218 {
219 oObj=(ACPI_OBJECT *)Object[0].Package.Elements;
220 i += PrintPackage((ULONG)Object[0].Package.Count,oObj);
221 }
222
223 return AE_OK;
224}

228void
229AcpiWalkNameSpaceTst(
230 ULONG PIC)
231{
232 ACPI_STATUS Status;
233 ULONG Lev;
234 ACPI_OBJECT Obj;
235 ACPI_OBJECT_LIST Params;
236
237 Params.Count = 1;
238 Params.Pointer = &Obj;
239 Obj.Type = ACPI_TYPE_INTEGER;
240 Obj.Integer.Value = PIC;
241
242 Status = AcpiTkEvaluateObject(NULL,"\\_PIC",&Params,NULL);
243 if (Status != AE_OK)
244 {
245 printf("Error turn on APIC, status %x\n",Status);
246 }
247
248 printf("Show ACPI IRQ routing\n");
249 printf("@\n");
250 Status = AcpiTkWalkNamespace(ACPI_TYPE_DEVICE, // All object types
251 ACPI_ROOT_OBJECT, // Start object
252 ACPI_UINT32_MAX, // Max depth
253 internal_PrintObjectInfo, // Callback
254 &Lev, NULL); // Callback param and poitner to return value
255 printf("Status 0x%x %d - %s\n",Status,Status,AcpiTkStatusToStr(Status));
256}
