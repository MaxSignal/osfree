75int id_create(struct id *id)
76{
77 int randomfile;
78 unsigned i;
79 size_t len = sizeof(id->uuid);
80
81 memset(id->uuid, 0, len);
82 if ((randomfile = open("/dev/urandom", O_RDONLY)) < 0) {
83 log_sys_error("open", "id_create: /dev/urandom");
84 return 0;
85 }
86
87 if (read(randomfile, id->uuid, len) != (ssize_t) len) {
88 log_sys_error("read", "id_create: /dev/urandom");
89 if (close(randomfile))
90 stack;
91 return 0;
92 }
93 if (close(randomfile))
94 stack;
95
96 /*
97 * Skip out the last 2 chars in randomized creation for LVM1
98 * backwards compatibility.
99 */
100 for (i = 0; i < len; i++)
101 id->uuid[i] = _c[id->uuid[i] % (sizeof(_c) - 3)];
102
103 return 1;
104}

106/*
107 * The only validity check we have is that
108 * the uuid just contains characters from
109 * '_c'. A checksum would have been nice :(
110 */
111static void _build_inverse(void)
112{
113 char *ptr;
114
115 if (_built_inverse)
116 return;
117
118 memset(_inverse_c, 0, sizeof(_inverse_c));
119
120 for (ptr = _c; *ptr; ptr++)
121 _inverse_c[(int) *ptr] = (char) 0x1;
122}

124int id_valid(struct id *id)
125{
126 int i;
127
128 _build_inverse();
129
130 for (i = 0; i < ID_LEN; i++)
131 if (!_inverse_c[id->uuid[i]]) {
132 log_err("UUID contains invalid character");
133 return 0;
134 }
135
136 return 1;
137}

146int id_write_format(const struct id *id, char *buffer, size_t size)
147{
148 int i, tot;
149
150 static unsigned group_size[] = { 6, 4, 4, 4, 4, 4, 6 };
151
152 assert(ID_LEN == 32);
153
154 /* split into groups separated by dashes */
155 if (size < (32 + 6 + 1)) {
156 log_err("Couldn't write uuid, buffer too small.");
157 return 0;
158 }
159
160 for (i = 0, tot = 0; i < 7; i++) {
161 memcpy(buffer, id->uuid + tot, group_size[i]);
162 buffer += group_size[i];
163 tot += group_size[i];
164 *buffer++ = '-';
165 }
166
167 *--buffer = '\0';
168 return 1;
169}

171int id_read_format(struct id *id, const char *buffer)
172{
173 int out = 0;
174
175 /* just strip out any dashes */
176 while (*buffer) {
177
178 if (*buffer == '-') {
179 buffer++;
180 continue;
181 }
182
183 if (out >= ID_LEN) {
184 log_err("Too many characters to be uuid.");
185 return 0;
186 }
187
188 id->uuid[out++] = *buffer++;
189 }
190
191 if (out != ID_LEN) {
192 log_err("Couldn't read uuid, incorrect number of characters.");
193 return 0;
194 }
195
196 return id_valid(id);
197}
