17// parsing flag, where bit flags are:
18// CMD_EXPAND_VARS - expand variables
19// CMD_EXPAND_REDIR - do redirection
20// CMD_STRIP_QUOTES - strip quoting

144// display the enabled internal commands
145int cmds_cmd(int argc, char **argv)
146{
147 unsigned int i, nColumn;
148
149 argc = GetScrCols();
150 for ( i = 0, nColumn = 0; ( i < NUMCMDS ); i++ ) {
151
152 // make sure command hasn't been disabled (SETDOS /I-cmd)
153 if ((commands[i].pflag & CMD_DISABLED) == 0) {
154
155 printf( FMT_STR, commands[i].cmdname );
156
157 // get width of display
158 if (( ++nColumn % ( argc / 12 )) != 0 )
159 printf( FMT_LEFT_STR, 12 - strlen( commands[i].cmdname ), NULLSTR );
160 else
161 crlf();
162 }
163 }
164
165 if (( nColumn % ( argc / 12 )) != 0 )
166 crlf();
167
168 return 0;
169}

172// do binary search to find command in internal command table & return index
173int findcmd( char *cmd, int eflag )
174{
175 static char DELIMS[] = "%10[^ \t;,.\"`\\+=<>|]";
176 int low, high, mid, cond;
177 char szInternalName[12];
178
179 // set the current compound command character & switch character
180 DELIMS[5] = gpIniptr->CmdSep;
181 DELIMS[6] = gpIniptr->SwChr;
182
183 // extract the command name (first argument)
184 // (including nasty kludge for nasty people who do "echo:"
185 // and a minor kludge for "y:")
186 DELIMS[7] = (char)(( cmd[1] == ':' ) ? ' ' : ':' );
187 sscanf( cmd, DELIMS, szInternalName );
188
189 // do a binary search for the command name
190 for ( low = 0, high = ( NUMCMDS - 1 ); ( low <= high ); ) {
191
192 mid = ( low + high ) / 2;
193
194 if (( cond = _stricmp( szInternalName, commands[mid].cmdname )) < 0)
195 high = mid - 1;
196 else if ( cond > 0 )
197 low = mid + 1;
198 else {
199 // kludge for trailing '\' (i.e., "TEXT\")
200 if ((( commands[mid].pflag & CMD_BACKSLASH_OK ) == 0 ) && ( cmd[ strlen(szInternalName) ] == '\\' ))
201 return -1;
202 return (((( commands[mid].pflag & CMD_DISABLED) == 0 ) || eflag ) ? mid : -1 );
203 }
204 }
205
206 return -1;
207}
