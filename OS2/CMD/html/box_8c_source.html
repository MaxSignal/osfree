77/*#define DEBUG 1*/
78/***********************************************************************/
79#ifdef HAVE_PROTO
80void box_operations(short action,CHARTYPE reset,bool boverlay,CHARTYPE fillchar)
81#else
82void box_operations(action,reset,boverlay,fillchar)
83short action;
84CHARTYPE reset;
85bool boverlay;
86CHARTYPE fillchar;
87#endif
88/***********************************************************************/
89{
90/*--------------------------- local data ------------------------------*/
91 BOXP boxp;
92 short rc=RC_OK;
93 unsigned short y=0,x=0;
94 LENGTHTYPE offset=0;
95 short save_mark_type=MARK_VIEW->mark_type;
96 bool same_view=FALSE;
97/*--------------------------- processing ------------------------------*/
98 TRACE_FUNCTION("box.c: box_operations");
99 /*
100 * This procedure is for copying, deleting, filling, moving and
101 * overlaying box blocks. Box blocks consist of BOX, WORD and COLUMN
102 * blocks.
103 */
104 post_process_line(CURRENT_VIEW,CURRENT_VIEW->focus_line,(_LINE *)NULL,TRUE);
105 /*
106 * If the command was issued on the command line then the destination
107 * line is the current line and the destination column is 0;
108 */
109 if (CURRENT_VIEW->current_window == WINDOW_COMMAND)
110 {
111 if (CURRENT_VIEW->current_line == 0L)
112 boxp.dst_start_line = 1L;
113 else
114 boxp.dst_start_line = CURRENT_VIEW->current_line;
115 boxp.dst_start_col = 0;
116 }
117 else
118 {
119 if (CURRENT_VIEW->focus_line == 0L)
120 boxp.dst_start_line = 1L;
121 else
122 boxp.dst_start_line = CURRENT_VIEW->focus_line;
123 getyx(CURRENT_WINDOW,y,x);
124 boxp.dst_start_col = x + CURRENT_VIEW->verify_col-1;
125 }
126 /*
127 * Set up the source and destination views...
128 */
129 boxp.src_view = MARK_VIEW;
130 boxp.dst_view = CURRENT_VIEW;
131 if ( MARK_VIEW == CURRENT_VIEW )
132 same_view = TRUE;
133
134 boxp.mark_type = MARK_VIEW->mark_type;
135
136 /*
137 * CUA block can have start position after end position
138 */
139 if ( boxp.mark_type == M_CUA
140 && ( MARK_VIEW->mark_start_line * max_line_length ) + MARK_VIEW->mark_start_col > (MARK_VIEW->mark_end_line * max_line_length ) + MARK_VIEW->mark_end_col )
141 {
142 boxp.src_start_line = MARK_VIEW->mark_end_line;
143 boxp.src_start_col = MARK_VIEW->mark_end_col-1;
144 boxp.src_end_line = MARK_VIEW->mark_start_line;
145 boxp.src_end_col = MARK_VIEW->mark_start_col-1;
146 }
147 else
148 {
149 boxp.src_start_line = MARK_VIEW->mark_start_line;
150 boxp.src_start_col = MARK_VIEW->mark_start_col-1;
151 boxp.src_end_line = MARK_VIEW->mark_end_line;
152 boxp.src_end_col = MARK_VIEW->mark_end_col-1;
153 }
154
155 boxp.num_cols = MARK_VIEW->mark_end_col - MARK_VIEW->mark_start_col + 1;
156 /*
157 * If the block type is COLUMN, the number of lines to operate on is
158 * the number of lines in the source file and the destination start
159 * line is line 1. Reset these values set above.
160 */
161 if (MARK_VIEW->mark_type == M_COLUMN)
162 {
163 boxp.num_lines = MARK_VIEW->file_for_view->number_lines - boxp.src_start_line +1L;
164 boxp.dst_start_line = 1L;
165 }
166 else
167 boxp.num_lines = boxp.src_end_line - boxp.src_start_line +1L;
168
169 /*
170 * Find the current LINE pointer for both the source and destination
171 * lines.
172 */
173 boxp.curr_src = lll_find(MARK_FILE->first_line,MARK_FILE->last_line,boxp.src_start_line,MARK_FILE->number_lines);
174 if (action != BOX_D)
175 boxp.curr_dst = lll_find(CURRENT_FILE->first_line,CURRENT_FILE->last_line,boxp.dst_start_line,CURRENT_FILE->number_lines);
176 /*
177 * Call the appropriate box function...
178 */
179 boxp.action = action;
180 switch(action)
181 {
182 case BOX_D:
183 rc = box_delete(&boxp);
184 break;
185 case BOX_M:
186 rc = box_move(&boxp,boverlay);
187 break;
188 case BOX_C:
189 rc = box_move(&boxp,boverlay);
190 break;
191 case BOX_F:
192 rc = box_fill(&boxp,fillchar);
193 break;
194 }
195 /*
196 * Increment the alteration counts...
197 */
198 increment_alt(CURRENT_FILE);
199 /*
200 * Set the parameters in the MARK_VIEW to OFF, and
201 * redraw the view if different from the current view
202 */
203 MARK_VIEW->marked_line = MARK_VIEW->marked_col = FALSE;
204 MARK_VIEW->mark_type = M_NONE;
205 MARK_VIEW = (VIEW_DETAILS *)NULL;
206 if ( !same_view
207 && display_screens > 1)
208 {
209 build_screen(other_screen);
210 display_screen(other_screen);
211 }
212 /*
213 * If we are not resetting the block, set up block markers...
214 */
215 if (reset != SOURCE_BLOCK_RESET)
216 {
217 if (boxp.src_start_col < boxp.dst_start_col
218 && action == BOX_M)
219 offset = boxp.num_cols;
220 MARK_VIEW = CURRENT_VIEW;
221 MARK_VIEW->mark_start_line = boxp.dst_start_line;
222 MARK_VIEW->mark_end_line = boxp.dst_start_line+boxp.num_lines-1L;
223 MARK_VIEW->mark_start_col = boxp.dst_start_col+1-offset;
224 if ( (save_mark_type == M_STREAM || save_mark_type == M_CUA )
225 && boxp.num_lines != 1)
226 MARK_VIEW->mark_end_col = boxp.src_end_col+1;
227 else
228 MARK_VIEW->mark_end_col = boxp.dst_start_col+boxp.num_cols-offset;
229 MARK_VIEW->mark_type = save_mark_type;
230 MARK_VIEW->marked_line = MARK_VIEW->marked_col = TRUE;
231 wmove(CURRENT_WINDOW,y,x-offset);
232 }
233 pre_process_line(CURRENT_VIEW,CURRENT_VIEW->focus_line,(_LINE *)NULL);
234 build_screen(current_screen);
235 display_screen(current_screen); /* should only call this is the marked block is in view */
236 TRACE_RETURN();
237 return;
238}
239/***********************************************************************/
240#ifdef HAVE_PROTO
241static short box_delete(BOXP *prm)
242#else
243static short box_delete(prm)
244BOXP *prm;
245#endif
246/***********************************************************************/
247{
248/*--------------------------- local data ------------------------------*/
249 LINETYPE i=0L;
250 LENGTHTYPE j=0;
251 LENGTHTYPE num_to_move=0,length_diff=0;
252 short rc=0;
253 bool shadow_found = FALSE;
254 bool advance_line_ptr = TRUE;
255 _LINE *curr=NULL;
256/*--------------------------- processing ------------------------------*/
257 TRACE_FUNCTION("box.c: box_delete");
258 curr = prm->curr_src;
259 for (i=0L;i<prm->num_lines;i++)
260 {
261 advance_line_ptr = TRUE;
262 rc = processable_line(prm->src_view,prm->src_start_line+i,curr);
263 switch(rc)
264 {
265 case LINE_SHADOW:
266 shadow_found = TRUE;
267 break;
268 case LINE_TOF:
269 case LINE_EOF:
270 break;
271 default:
272 if ( (prm->mark_type == M_STREAM || prm->mark_type == M_CUA )
273 && prm->src_start_line != prm->src_end_line)
274 {
275 if (i == 0)
276 {
277 /*
278 * first line of stream block
279 * delete from start column to end of line
280 */
281 if (prm->src_start_col < curr->length)
282 {
283 add_to_recovery_list(curr->line,curr->length);
284 curr->length = prm->src_start_col;
285 }
286 }
287 else if (i+1 == prm->num_lines) /* last line of stream block */
288 {
289 if (curr->length > 0)
290 {
291 add_to_recovery_list(curr->line,curr->length);
292 curr->length = (LENGTHTYPE)(prm->src_end_col < curr->length-1) ? curr->length - prm->src_end_col - 1: 0;
293 for (j=0;j<curr->length;j++)
294 *(curr->line+(LENGTHTYPE)(LINETYPE)j) = *(curr->line+prm->src_end_col+j+1);
295 }
296 }
297 else
298 {
299 curr = delete_LINE(MARK_FILE->first_line,MARK_FILE->last_line,curr,DIRECTION_FORWARD);
300 MARK_FILE->number_lines--;
301 advance_line_ptr = FALSE;
302 }
303 }
304 else
305 {
306 if ((LINETYPE)curr->length >= (LINETYPE)MARK_VIEW->mark_start_col)
307 {
308 num_to_move = (LENGTHTYPE)max((LINETYPE)curr->length - (LINETYPE)MARK_VIEW->mark_end_col,0L);
309 length_diff = (LENGTHTYPE)min((LINETYPE)MARK_VIEW->mark_end_col-(LINETYPE)MARK_VIEW->mark_start_col+1L,
310 (LINETYPE)curr->length-(LINETYPE)MARK_VIEW->mark_start_col+1L);
311 }
312 else
313 num_to_move = length_diff = 0;
314 if (length_diff != 0)
315 {
316 add_to_recovery_list(curr->line,curr->length);
317 curr->length = (LENGTHTYPE)((LINETYPE)curr->length-(LINETYPE)length_diff);
318 for (j=0;j<num_to_move;j++)
319 *(curr->line+(LENGTHTYPE)((LINETYPE)j+(LINETYPE)MARK_VIEW->mark_start_col-1L)) = *(curr->line+prm->src_start_col+j+prm->num_cols);
320 *(curr->line+curr->length) = '\0';/* null terminate */
321 }
322 }
323 }
324 if (advance_line_ptr)
325 curr = curr->next; /* this should NEVER go past the end */
326 }
327 /*
328 * if the delete involved lines and there were no shadow lines involved
329 * join the first line with the last line of the block
330 */
331 if ( (prm->mark_type == M_STREAM || prm->mark_type == M_CUA )
332 && prm->src_start_line != prm->src_end_line
333 && !shadow_found)
334 {
335 curr = prm->curr_src;
336 if (curr->next->length > 0)
337 {
338 curr->line = (CHARTYPE *)(*the_realloc)((void *)curr->line,(curr->length+curr->next->length)*sizeof(CHARTYPE));
339 if (curr->line == NULL)
340 {
341 display_error(30,(CHARTYPE *)"",FALSE);
342 TRACE_RETURN();
343 return(RC_OUT_OF_MEMORY);
344 }
345 memcpy((DEFCHAR*)curr->line+curr->length,(DEFCHAR*)curr->next->line,curr->next->length);
346 curr->length += curr->next->length;
347 if (curr->length > max_line_length)
348 {
349 display_error(0,(CHARTYPE*)"Truncated",FALSE);
350 curr->length = max_line_length;
351 }
352 *(curr->line+curr->length) = '\0'; /* do we need to do this anymore ? */
353 }
354 curr = delete_LINE(MARK_FILE->first_line,MARK_FILE->last_line,curr->next,DIRECTION_FORWARD);
355 MARK_FILE->number_lines--;
356 }
357 TRACE_RETURN();
358 return(RC_OK);
359}
360/***********************************************************************/
361#ifdef HAVE_PROTO
362static short box_move(BOXP *prm,bool boverlay)
363#else
364static short box_move(prm,boverlay)
365BOXP *prm;
366bool boverlay;
367#endif
368/***********************************************************************/
369{
370/*--------------------------- local data ------------------------------*/
371 _LINE *first_save=NULL,*save_src=NULL,*temp_src=NULL;
372 LINETYPE save_src_start_line=0L;
373 LENGTHTYPE save_src_start_col=0;
374/*--------------------------- processing ------------------------------*/
375 TRACE_FUNCTION("box.c: box_move");
376 save_src_start_col = prm->src_start_col;
377 save_src_start_line = prm->src_start_line;
378 save_src = prm->curr_src;
379 box_copy_to_temp(prm); /* copy box from file to temp */
380 if (prm->dst_start_col <= prm->src_start_col+prm->num_cols
381 && prm->action == BOX_M)
382 {
383 temp_src = prm->curr_src; /* save the LINE pointer to temp src lines */
384 prm->curr_src = save_src; /* point to file src lines */
385 box_delete(prm);
386 prm->curr_src = temp_src; /* point to temp src lines */
387 }
388 prm->src_start_line = 0L;
389 prm->src_start_col = 0;
390 if ( (prm->mark_type == M_STREAM || prm->mark_type == M_CUA )
391 && prm->num_lines != 1)
392 box_copy_stream_from_temp(prm,boverlay);/* copy box from temp to file */
393 else
394 box_copy_from_temp(prm,boverlay); /* copy box from temp to file */
395 first_save = lll_free(first_save);
396 prm->src_start_line = save_src_start_line;
397 prm->src_start_col = save_src_start_col;
398 prm->curr_src = save_src;
399 if (prm->dst_start_col > prm->src_start_col+prm->num_cols
400 && prm->action == BOX_M)
401 box_delete(prm);
402
403 TRACE_RETURN();
404 return(RC_OK);
405}
406/***********************************************************************/
407#ifdef HAVE_PROTO
408static short box_copy_to_temp(BOXP *prm)
409#else
410static short box_copy_to_temp(prm)
411BOXP *prm;
412#endif
413/***********************************************************************/
414{
415/*--------------------------- local data ------------------------------*/
416 _LINE *first_save=NULL,*save_src=NULL,*tmp=NULL;
417 LINETYPE i=0L;
418 short rc=RC_OK;
419/*--------------------------- processing ------------------------------*/
420 TRACE_FUNCTION("box.c: box_copy_to_temp");
421 tmp = prm->curr_src;
422 for (i=0L;i<prm->num_lines;i++)
423 {
424 if (processable_line(prm->src_view,prm->src_start_line+i,tmp) == LINE_LINE)
425 {
426 memset(rec,' ',max_line_length);
427#if 1
428 if (prm->mark_type == M_STREAM
429 || prm->mark_type == M_CUA)
430 {
431 int mystart=0,mynum=tmp->length;
432 memset(trec,' ', max_line_length);
433 memcpy(trec,tmp->line,tmp->length);
434 if (i == 0)
435 {
436 mystart = prm->src_start_col;
437 if (mystart > tmp->length)
438 mystart = mynum = 0;
439 else
440 mynum = tmp->length - mystart;
441 }
442 if (i+1 == prm->num_lines)
443 mynum = prm->src_end_col - mystart + 1;
444 memcpy(rec, trec+mystart,mynum);
445 rec_len = mynum;
446 if ((save_src = add_LINE(first_save,save_src,rec,rec_len,0,TRUE)) == (_LINE *)NULL)
447 {
448 TRACE_RETURN();
449 return(RC_OUT_OF_MEMORY);
450 }
451 }
452 else
453 {
454 memcpy(rec,tmp->line,tmp->length);
455 rec_len = tmp->length;
456 if ((save_src = add_LINE(first_save,save_src,rec+prm->src_start_col,prm->num_cols,0,TRUE)) == (_LINE *)NULL)
457 {
458 TRACE_RETURN();
459 return(RC_OUT_OF_MEMORY);
460 }
461 }
462
463#else
464 memcpy(rec,tmp->line,tmp->length);
465 rec_len = tmp->length;
466 if ((save_src = add_LINE(first_save,save_src,
467 rec+prm->src_start_col,prm->num_cols,0,TRUE)) == (_LINE *)NULL)
468 {
469 TRACE_RETURN();
470 return(RC_OUT_OF_MEMORY);
471 }
472#endif
473 if (first_save == (_LINE *)NULL)
474 first_save = save_src;
475 }
476 tmp = tmp->next;
477 }
478 prm->curr_src = first_save;
479 TRACE_RETURN();
480 return(rc);
481}
482/***********************************************************************/
483#ifdef HAVE_PROTO
484static short box_copy_from_temp(BOXP *prm,bool boverlay)
485#else
486static short box_copy_from_temp(prm,boverlay)
487BOXP *prm;
488bool boverlay;
489#endif
490/***********************************************************************/
491{
492/*--------------------------- local data ------------------------------*/
493 LINETYPE dst_lineno=0L;
494 LENGTHTYPE j=0;
495 CHARTYPE chr=0;
496 short rc=RC_OK,line_type=0;
497/*--------------------------- processing ------------------------------*/
498 TRACE_FUNCTION("box.c: box_copy_from_temp");
499 dst_lineno = prm->dst_start_line;
500 while(prm->curr_src)
501 {
502 while(1)
503 {
504 line_type = processable_line(prm->dst_view,dst_lineno,prm->curr_dst);
505 if (line_type == LINE_LINE)
506 break;
507/* if (line_type == LINE_TOF_EOF) MH12 */
508 if (line_type == LINE_TOF
509 || line_type == LINE_EOF)
510 {
511 if ((prm->curr_dst = add_LINE(CURRENT_FILE->first_line,prm->curr_dst->prev,
512 (CHARTYPE *)"",0,0,TRUE)) == (_LINE *)NULL)
513 {
514 TRACE_RETURN();
515 return(RC_OUT_OF_MEMORY);
516 }
517 CURRENT_FILE->number_lines++;
518 break;
519 }
520 dst_lineno++;
521 prm->curr_dst = prm->curr_dst->next;
522 }
523
524 pre_process_line(prm->dst_view,dst_lineno,prm->curr_dst);/* copy dest line into rec */
525/* MH 300499 pre_process_line(prm->dst_view,dst_lineno,(_LINE *)NULL); *//* copy dest line into rec */
526 for (j=0;j<prm->num_cols;j++)
527 {
528 if (prm->src_start_col+j+1 > prm->curr_src->length)
529 chr = (CHARTYPE)' ';
530 else
531 chr = (CHARTYPE)*(prm->curr_src->line+prm->src_start_col+j);
532 if (boverlay)
533 rec[prm->dst_start_col+j] = chr;
534 else
535 meminschr(rec,chr,prm->dst_start_col+j,max_line_length,rec_len++);
536 }
537 rc = memrevne(rec,' ',max_line_length);
538 if (rc == (-1))
539 rec_len = 0;
540 else
541 rec_len = rc+1;
542 post_process_line(prm->dst_view,dst_lineno,(_LINE *)NULL,FALSE);
543 prm->curr_src = prm->curr_src->next; /* this should NEVER go past the end */
544 prm->curr_dst = prm->curr_dst->next; /* this should NEVER go past the end */
545 dst_lineno++;
546 }
547 prm->num_lines = dst_lineno - prm->dst_start_line;
548 TRACE_RETURN();
549 return(RC_OK);
550}
551/***********************************************************************/
552#ifdef HAVE_PROTO
553static short box_copy_stream_from_temp(BOXP *prm,bool boverlay)
554#else
555static short box_copy_stream_from_temp(prm,boverlay)
556BOXP *prm;
557bool boverlay;
558#endif
559/***********************************************************************/
560{
561/*--------------------------- local data ------------------------------*/
562 LINETYPE dst_lineno=0L,i=0L;
563 short line_type=0;
564 int mystart=0,mynumcols=0;
565 bool full_line=TRUE;
566 _LINE *curr=NULL,*last_line=NULL;
567/*--------------------------- processing ------------------------------*/
568 TRACE_FUNCTION("box.c: box_copy_stream_from_temp");
569 if (boverlay)
570 {
571 /*
572 * In overlay mode treat the stream block similar to a box block
573 */
574 dst_lineno = prm->dst_start_line;
575 i = 0;
576 while(prm->curr_src)
577 {
578 while(1)
579 {
580 line_type = processable_line(prm->dst_view,dst_lineno,prm->curr_dst);
581 if (line_type == LINE_LINE)
582 {
583 break;
584 }
585/* if (line_type == LINE_TOF_EOF) MH12 */
586 if (line_type == LINE_TOF
587 || line_type == LINE_EOF)
588 {
589 if ((prm->curr_dst = add_LINE(CURRENT_FILE->first_line,prm->curr_dst->prev,
590 (CHARTYPE *)"",0,0,TRUE)) == (_LINE *)NULL)
591 {
592 TRACE_RETURN();
593 return(RC_OUT_OF_MEMORY);
594 }
595 CURRENT_FILE->number_lines++;
596 break;
597 }
598 dst_lineno++;
599 prm->curr_dst = prm->curr_dst->next;
600 }
601
602 pre_process_line(prm->dst_view,dst_lineno,prm->curr_dst);/* copy dest line into rec */
603/* MH 300499 pre_process_line(prm->dst_view,dst_lineno,(_LINE *)NULL); *//* copy dest line into rec */
604 mystart=0;
605 mynumcols=prm->curr_src->length;
606 full_line=TRUE;
607 if (i == 0)
608 {
609 mystart = prm->src_start_col;
610 full_line = FALSE;
611 }
612 if (i+1 == prm->num_lines)
613 full_line = FALSE;
614 memcpy(rec+mystart,prm->curr_src->line,prm->curr_src->length);
615 if (full_line)
616 {
617 rec_len = prm->curr_src->length;
618 }
619 else
620 {
621 rec_len = max(rec_len,prm->curr_src->length+mystart);
622 }
623
624 post_process_line(prm->dst_view,dst_lineno,(_LINE *)NULL,FALSE);
625 prm->curr_src = prm->curr_src->next; /* this should NEVER go past the end */
626 prm->curr_dst = prm->curr_dst->next; /* this should NEVER go past the end */
627 dst_lineno++;
628 i++;
629 }
630 prm->num_lines = dst_lineno - prm->dst_start_line;
631 }
632 else
633 {
634 /*
635 * In non-overlay mode treat the stream block similar to a line block
636 */
637 /*
638 * This function copies saved lines from a multi-line stream block.
639 * The line to which the stream block is copied is split at the
640 * cursor position.
641 * The first source line is appended to the current line.
642 * The remainder of the lines are added as new lines, except the last.
643 * The last line is pre-pended to that portion of the line that was
644 * split.
645 */
646 dst_lineno = prm->dst_start_line;
647 while(1)
648 {
649 line_type = processable_line(prm->dst_view,dst_lineno,prm->curr_dst);
650 if (line_type == LINE_LINE)
651 {
652 break;
653 }
654/* if (line_type == LINE_TOF_EOF) MH12 */
655 if (line_type == LINE_TOF
656 || line_type == LINE_EOF)
657 {
658 if ((prm->curr_dst = add_LINE(CURRENT_FILE->first_line,prm->curr_dst->prev,
659 (CHARTYPE *)"",0,0,TRUE)) == (_LINE *)NULL)
660 {
661 TRACE_RETURN();
