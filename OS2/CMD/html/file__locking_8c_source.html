46static int _release_lock(const char *file, int unlock)
47{
48 struct lock_list *ll;
49 struct list *llh, *llt;
50
51 struct stat buf1, buf2;
52
53 list_iterate_safe(llh, llt, &_lock_list) {
54 ll = list_item(llh, struct lock_list);
55
56 if (!file || !strcmp(ll->res, file)) {
57 list_del(llh);
58 if (unlock) {
59 log_very_verbose("Unlocking %s", ll->res);
60 if (flock(ll->lf, LOCK_NB | LOCK_UN))
61 log_sys_error("flock", ll->res);
62 }
63
64 if (!flock(ll->lf, LOCK_NB | LOCK_EX) &&
65 !stat(ll->res, &buf1) &&
66 !fstat(ll->lf, &buf2) &&
67 !memcmp(&buf1.st_ino, &buf2.st_ino, sizeof(ino_t)))
68 if (unlink(ll->res))
69 log_sys_error("unlink", ll->res);
70
71 if (close(ll->lf) < 0)
72 log_sys_error("close", ll->res);
73
74 dm_free(ll->res);
75 dm_free(llh);
76
77 if (file)
78 return 1;
79 }
80 }
81
82 return 0;
83}

95static void _remove_ctrl_c_handler()
96{
97 siginterrupt(SIGINT, 0);
98 if (!_handler_installed)
99 return;
100
101 _handler_installed = 0;
102
103 sigprocmask(SIG_SETMASK, &_fullsigset, NULL);
104 if (signal(SIGINT, _oldhandler) == SIG_ERR)
105 log_sys_error("signal", "_remove_ctrl_c_handler");
106}

108static void _trap_ctrl_c(int sig) // __attribute((unused)))
109{
110 _remove_ctrl_c_handler();
111 log_error("CTRL-c detected: giving up waiting for lock");
112}

114static void _install_ctrl_c_handler()
115{
116 _handler_installed = 1;
117
118 if ((_oldhandler = signal(SIGINT, _trap_ctrl_c)) == SIG_ERR) {
119 _handler_installed = 0;
120 return;
121 }
122
123 sigprocmask(SIG_SETMASK, &_intsigset, NULL);
124 siginterrupt(SIGINT, 1);
125}

127static int _lock_file(const char *file, int flags)
128{
129 int operation;
130 int r = 1;
131 int old_errno;
132
133 struct lock_list *ll;
134 struct stat buf1, buf2;
135 char state;
136
137 switch (flags & LCK_TYPE_MASK) {
138 case LCK_READ:
139 operation = LOCK_SH;
140 state = 'R';
141 break;
142 case LCK_WRITE:
143 operation = LOCK_EX;
144 state = 'W';
145 break;
146 case LCK_UNLOCK:
147 return _release_lock(file, 1);
148 default:
149 log_error("Unrecognised lock type: %d", flags & LCK_TYPE_MASK);
150 return 0;
151 }
152
153 if (!(ll = dm_malloc(sizeof(struct lock_list))))
154 return 0;
155
156 if (!(ll->res = dm_strdup(file))) {
157 dm_free(ll);
158 return 0;
159 }
160
161 ll->lf = -1;
162
163 log_very_verbose("Locking %s %c%c", ll->res, state,
164 flags & LCK_NONBLOCK ? ' ' : 'B');
165 do {
166 if ((ll->lf > -1) && close(ll->lf))
167 log_sys_error("close", file);
168
169 if ((ll->lf = open(file, O_CREAT | O_APPEND | O_RDWR, 0777))
170 < 0) {
171 log_sys_error("open", file);
172 goto err;
173 }
174
175 if ((flags & LCK_NONBLOCK))
176 operation |= LOCK_NB;
177 else
178 _install_ctrl_c_handler();
179
180 r = flock(ll->lf, operation);
181 old_errno = errno;
182 if (!(flags & LCK_NONBLOCK))
183 _remove_ctrl_c_handler();
184
185 if (r) {
186 errno = old_errno;
187 log_sys_error("flock", ll->res);
188 goto err;
189 }
190
191 if (!stat(ll->res, &buf1) && !fstat(ll->lf, &buf2) &&
192 !memcmp(&buf1.st_ino, &buf2.st_ino, sizeof(ino_t)))
193 break;
194 } while (!(flags & LCK_NONBLOCK));
195
196 list_add(&_lock_list, &ll->list);
197 return 1;
198
199 err:
200 dm_free(ll->res);
201 dm_free(ll);
202 return 0;
203}

205static int _file_lock_resource(struct cmd_context *cmd, const char *resource,
206 int flags)
207{
208 char lockfile[PATH_MAX];
209
210 assert(resource);
211
212 switch (flags & LCK_SCOPE_MASK) {
213 case LCK_VG:
214 if (!*resource)
215 dm_snprintf(lockfile, sizeof(lockfile),
216 "%s/P_orphans", _lock_dir);
217 else
218 dm_snprintf(lockfile, sizeof(lockfile),
219 "%s/V_%s", _lock_dir, resource);
220
221 if (!_lock_file(lockfile, flags))
222 return 0;
223
224 switch (flags & LCK_TYPE_MASK) {
225 case LCK_UNLOCK:
226 lvmcache_unlock_vgname(resource);
227 break;
228 default:
229 lvmcache_lock_vgname(resource,
230 (flags & LCK_TYPE_MASK) ==
231 LCK_READ);
232 }
233 break;
234 case LCK_LV:
235 switch (flags & LCK_TYPE_MASK) {
236 case LCK_UNLOCK:
237 log_debug("Unlocking LV %s", resource);
238 if (!lv_resume_if_active(cmd, resource))
239 return 0;
240 break;
241 case LCK_NULL:
242 log_debug("Locking LV %s (NL)", resource);
243 if (!lv_deactivate(cmd, resource))
244 return 0;
245 break;
246 case LCK_READ:
247 log_debug("Locking LV %s (R)", resource);
248 if (!lv_activate_with_filter(cmd, resource, 0))
249 return 0;
250 break;
251 case LCK_WRITE:
252 log_debug("Locking LV %s (W)", resource);
253 if (!lv_suspend_if_active(cmd, resource))
254 return 0;
255 break;
256 case LCK_EXCL:
257 log_debug("Locking LV %s (EX)", resource);
258 if (!lv_activate_with_filter(cmd, resource, 1))
259 return 0;
260 break;
261 default:
262 break;
263 }
264 break;
265 default:
266 log_error("Unrecognised lock scope: %d",
267 flags & LCK_SCOPE_MASK);
268 return 0;
269 }
270
271 return 1;
272}

274int init_file_locking(struct locking_type *locking, struct cmd_context *cmd)
275{
276 locking->lock_resource = _file_lock_resource;
277 locking->reset_locking = _reset_file_locking;
278 locking->fin_locking = _fin_file_locking;
279 locking->flags = 0;
280
281 /* Get lockfile directory from config file */
282 strncpy(_lock_dir, find_config_tree_str(cmd, "global/locking_dir",
283 DEFAULT_LOCK_DIR),
284 sizeof(_lock_dir));
285
286 if (!create_dir(_lock_dir))
287 return 0;
288
289 /* Trap a read-only file system */
290 if ((access(_lock_dir, R_OK | W_OK | X_OK) == -1) && (errno == EROFS))
291 return 0;
292
293 list_init(&_lock_list);
294
295 if (sigfillset(&_intsigset) || sigfillset(&_fullsigset)) {
296 log_sys_error("sigfillset", "init_file_locking");
297 return 0;
298 }
299
300 if (sigdelset(&_intsigset, SIGINT)) {
301 log_sys_error("sigdelset", "init_file_locking");
302 return 0;
303 }
304
305 return 1;
306}
