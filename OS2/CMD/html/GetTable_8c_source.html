13#include <os2.h>
14#include <acpi.h>
15#include <acpiapi.h>
16#include <stdio.h>
17#include <stdlib.h>
18#include <string.h>
19
20/*
21 * Main program for get ACPI table
22*/
23void
24main(
25 int argc,
26 char *argv[])
27{
28 ACPI_STATUS Status;
29 ACPI_TABLE_HEADER Table;
30 ACPI_TABLE_HEADER *Tbl = NULL;
31 FILE *OutTbl = NULL;
32 ACPI_TK_VERSION ApiBuffer;
33
34 if (AcpiTkValidateVersion(ACPI_TK_VERSION_MAJOR, ACPI_TK_VERSION_MINOR)) {
35 printf("This program is not compatible with the version of ACPI that is installed.\n");
36
37 /* display the version information */
38 ApiBuffer.Size = sizeof(ACPI_TK_VERSION);
39 if (AcpiTkGetVersion(&ApiBuffer)) exit(1);
40 if (ApiBuffer.Size != sizeof(ACPI_TK_VERSION)) exit(1);
41
42 printf("The version of ACPI installed is %u.%u\n", ApiBuffer.PSD.Major, ApiBuffer.PSD.Minor);
43 printf("The API version is %u.%u, but version %u.%u is needed.\n",
44 ApiBuffer.Api.Major, ApiBuffer.Api.Minor, ACPI_TK_VERSION_MAJOR, ACPI_TK_VERSION_MINOR);
45 }
46
47 if (argc < 2)
48 {
49 printf ( "Usage: GetTable TableName\n");
50 return;
51 }
52
53 printf("Look table \"%s\"\n",argv[1]);
54 Status = AcpiTkGetTableHeader (argv[1],0,&Table);
55 if (Status != AE_OK)
56 {
57 printf ( "Error call ACPI CA %x - %s\n",Status,AcpiTkStatusToStr(Status));
58 return;
59 }
60
61 printf ("Signature : %4.4s\n",Table.Signature);
62 printf ("Length : %d bytes\n",Table.Length); /* Length of table in bytes, including this header */
63 printf ("Revision : %d\n",Table.Revision); /* ACPI Specification minor version # */
64 printf ("Checksum ; 0x%x\n",Table.Checksum); /* To make sum of entire table == 0 */
65 printf ("OemId : %6.6s\n",Table.OemId); /* ASCII OEM identification */
66 printf ("Table OemId : %8.8s\n",Table.OemTableId); /* ASCII OEM table identification */
67 printf ("OemRevision : 0x%x\n",Table.OemRevision); /* OEM revision number */
68 printf ("AslCompilerId : %4.4s\n",Table.AslCompilerId); /* ASCII ASL compiler vendor ID */
69 printf ("AslCompilerRevision: 0x%x\n",Table.AslCompilerId); /* ASL compiler version */
70
71 Tbl = malloc ( Table.Length );
72 Status = AcpiTkGetTable (argv[1],0,&Tbl);
73 printf ("Status:%x addr %x\n",Status,Tbl);
74 printf ("Signature : %4.4s\n",Tbl->Signature);
75
76 if ((OutTbl = fopen ( "table.dat","wb")) != NULL)
77 {
78 fwrite(Tbl,Table.Length,1,OutTbl);
79 fclose(OutTbl);
80 }
81 else
82 printf ("Error write table\n");
83}
