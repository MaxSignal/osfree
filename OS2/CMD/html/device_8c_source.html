45static int _is_partitionable(struct device *dev)
46{
47 int parts = max_partitions(MAJOR(dev->dev));
48
49 if ((parts <= 1) || (MINOR(dev->dev) % parts))
50 return 0;
51
52 return 1;
53}

55static int _has_partition_table(struct device *dev)
56{
57 int ret = 0;
58 unsigned p;
59 uint16_t buf[SECTOR_SIZE/sizeof(uint16_t)];
60 uint16_t *part_magic;
61 struct partition *part;
62
63 if (!dev_open(dev)) {
64 stack;
65 return -1;
66 }
67
68 if (!dev_read(dev, UINT64_C(0), sizeof(buf), &buf)) {
69 stack;
70 goto out;
71 }
72
73 /* FIXME Check for other types of partition table too */
74
75 /* Check for msdos partition table */
76 part_magic = buf + PART_MAGIC_OFFSET/sizeof(buf[0]);
77 if ((*part_magic == xlate16(PART_MAGIC))) {
78 part = (struct partition *) (buf + PART_OFFSET/sizeof(buf[0]));
79 for (p = 0; p < 4; p++, part++) {
80 /* Table is invalid if boot indicator not 0 or 0x80 */
81 if ((part->boot_ind & 0x7f)) {
82 ret = 0;
83 break;
84 }
85 /* Must have at least one non-empty partition */
86 if (part->nr_sects)
87 ret = 1;
88 }
89 }
90
91 out:
92 if (!dev_close(dev))
93 stack;
94
95 return ret;
96}

98int is_partitioned_dev(struct device *dev)
99{
100 if (!_is_partitionable(dev))
101 return 0;
102
103 return _has_partition_table(dev);

144int partition_type_is_lvm(struct dev_mgr *dm, struct device *d)
145{
146 int pt;
147
148 pt = _get_partition_type(dm, d);
149
150 if (!pt) {
151 if (is_whole_disk(dm, d))
152 /* FIXME: Overloaded pt=0 in error cases */
153 return 1;
154 else {
155 log_error
156 ("%s: missing partition table "
157 "on partitioned device", d->name);
158 return 0;
159 }
160 }
161
162 if (is_whole_disk(dm, d)) {
163 log_error("%s: looks to possess partition table", d->name);
164 return 0;
165 }
166
167 /* check part type */
168 if (pt != LVM_PARTITION && pt != LVM_NEW_PARTITION) {
169 log_error("%s: invalid partition type 0x%x "
170 "(must be 0x%x)", d->name, pt, LVM_NEW_PARTITION);
171 return 0;
172 }
173
174 if (pt == LVM_PARTITION) {
175 log_error
176 ("%s: old LVM partition type found - please change to 0x%x",
177 d->name, LVM_NEW_PARTITION);
178 return 0;
179 }
180
181 return 1;
182}

184int _get_partition_type(struct dev_mgr *dm, struct device *d)
185{
186 int pv_handle = -1;
187 struct device *primary;
188 ssize_t read_ret;
189 ssize_t bytes_read = 0;
190 char *buffer;
191 unsigned short *s_buffer;
192 struct partition *part;
193 loff_t offset = 0;
194 loff_t extended_offset = 0;
195 int part_sought;
196 int part_found = 0;
197 int first_partition = 1;
198 int extended_partition = 0;
199 int p;
200
201 if (!(primary = dev_primary(dm, d))) {
202 log_error
203 ("Failed to find main device containing partition %s",
204 d->name);
205 return 0;
206 }
207
208 if (!(buffer = dm_malloc(SECTOR_SIZE))) {
209 log_error("Failed to allocate partition table buffer");
210 return 0;
211 }
212
213 /* Get partition table */
214 if ((pv_handle = open(primary->name, O_RDONLY)) < 0) {
215 log_error("%s: open failed: %s", primary->name,
216 strerror(errno));
217 return 0;
218 }
219
220 s_buffer = (unsigned short *) buffer;
221 part = (struct partition *) (buffer + 0x1be);
222 part_sought = MINOR_PART(dm, d);
223
224 do {
225 bytes_read = 0;
226
227 if (llseek(pv_handle, offset * SECTOR_SIZE, SEEK_SET) == -1) {
228 log_error("%s: llseek failed: %s",
229 primary->name, strerror(errno));
230 return 0;
231 }
232
233 while ((bytes_read < SECTOR_SIZE) &&
234 (read_ret =
235 read(pv_handle, buffer + bytes_read,
236 SECTOR_SIZE - bytes_read)) != -1)
237 bytes_read += read_ret;
238
239 if (read_ret == -1) {
240 log_error("%s: read failed: %s", primary->name,
241 strerror(errno));
242 return 0;
243 }
244
245 if (s_buffer[255] == 0xAA55) {
246 if (is_whole_disk(dm, d))
247 return -1;
248 } else
249 return 0;
250
251 extended_partition = 0;
252
253 /* Loop through primary partitions */
254 for (p = 0; p < 4; p++) {
255 if (part[p].sys_ind == DOS_EXTENDED_PARTITION ||
256 part[p].sys_ind == LINUX_EXTENDED_PARTITION
257 || part[p].sys_ind == WIN98_EXTENDED_PARTITION) {
258 extended_partition = 1;
259 offset = extended_offset + part[p].start_sect;
260 if (extended_offset == 0)
261 extended_offset = part[p].start_sect;
262 if (first_partition == 1)
263 part_found++;
264 } else if (first_partition == 1) {
265 if (p == part_sought) {
266 if (part[p].sys_ind == 0) {
267 /* missing primary? */
268 return 0;
269 }
270 } else
271 part_found++;
272 } else if (!part[p].sys_ind)
273 part_found++;
274
275 if (part_sought == part_found)
276 return part[p].sys_ind;
277
278 }
279 first_partition = 0;
280 }
281 while (extended_partition == 1);
282
283 return 0;
284}
