42/*
43 * The format instance is given a directory path upon creation.
44 * Each file in this directory whose name is of the form
45 * '(.*)_[0-9]*.vg' is a config file (see lib/config.[hc]), which
46 * contains a description of a single volume group.
47 *
48 * The prefix ($1 from the above regex) of the config file gives
49 * the volume group name.
50 *
51 * Backup files that have expired will be removed.
52 */
53
54/*
55 * A list of these is built up for our volume group. Ordered
56 * with the least recent at the head.
57 */
58struct archive_file {
59 struct list list;
60
61 char *path;
62 uint32_t index;
63};

65/*
66 * Extract vg name and version number from a filename.
67 */
68static int _split_vg(const char *filename, char *vgname, size_t vg_size,
69 uint32_t *ix)
70{
71 size_t len, vg_len;
72 const char *dot, *underscore;
73
74 len = strlen(filename);
75 if (len < 7)
76 return 0;
77
78 dot = (filename + len - 3);
79 if (strcmp(".vg", dot))
80 return 0;
81
82 if (!(underscore = strrchr(filename, '_')))
83 return 0;
84
85 if (sscanf(underscore + 1, "%u", ix) != 1)
86 return 0;
87
88 vg_len = underscore - filename;
89 if (vg_len + 1 > vg_size)
90 return 0;
91
92 strncpy(vgname, filename, vg_len);
93 vgname[vg_len] = '\0';
94
95 return 1;
96}

133/*
134 * Returns a list of archive_files.
135 */
136static struct list *_scan_archive(struct dm_pool *mem,
137 const char *vgname, const char *dir)
138{
139 int i, count;
140 uint32_t ix;
141 char vgname_found[64], *path;
142 struct dirent **dirent;
143 struct archive_file *af;
144 struct list *results;
145
146 if (!(results = dm_pool_alloc(mem, sizeof(*results)))) {
147 stack;
148 return NULL;
149 }
150
151 list_init(results);
152
153 /* Sort fails beyond 5-digit indexes */
154 if ((count = scandir(dir, &dirent, NULL, alphasort)) < 0) {
155 log_err("Couldn't scan the archive directory (%s).", dir);
156 return 0;
157 }
158
159 for (i = 0; i < count; i++) {
160 if (!strcmp(dirent[i]->d_name, ".") ||
161 !strcmp(dirent[i]->d_name, ".."))
162 continue;
163
164 /* check the name is the correct format */
165 if (!_split_vg(dirent[i]->d_name, vgname_found,
166 sizeof(vgname_found), &ix))
167 continue;
168
169 /* is it the vg we're interested in ? */
170 if (strcmp(vgname, vgname_found))
171 continue;
172
173 if (!(path = _join_file_to_dir(mem, dir, dirent[i]->d_name))) {
174 stack;
175 goto out;
176 }
177
178 /*
179 * Create a new archive_file.
180 */
181 if (!(af = dm_pool_alloc(mem, sizeof(*af)))) {
182 log_err("Couldn't create new archive file.");
183 results = NULL;
184 goto out;
185 }
186
187 af->index = ix;
188 af->path = path;
189
190 /*
191 * Insert it to the correct part of the list.
192 */
193 _insert_archive_file(results, af);
194 }
195
196 out:
197 for (i = 0; i < count; i++)
198 free(dirent[i]);
199 free(dirent);
200
201 return results;
202}

204static void _remove_expired(struct list *archives, uint32_t archives_size,
205 uint32_t retain_days, uint32_t min_archive)
206{
207 struct archive_file *bf;
208 struct stat sb;
209 time_t retain_time;
210
211 /* Make sure there are enough archives to even bother looking for
212 * expired ones... */
213 if (archives_size <= min_archive)
214 return;
215
216 /* Convert retain_days into the time after which we must retain */
217 retain_time = time(NULL) - (time_t) retain_days *SECS_PER_DAY;
218
219 /* Assume list is ordered newest first (by index) */
220 //list_iterate_back_items(bf, archives) {
221 list_iterate_back_items(bf, struct archive_file, archives) {
222 /* Get the mtime of the file and unlink if too old */
223 if (stat(bf->path, &sb)) {
224 log_sys_error("stat", bf->path);
225 continue;
226 }
227
228 if (sb.st_mtime > retain_time)
229 return;
230
231 log_very_verbose("Expiring archive %s", bf->path);
232 if (unlink(bf->path))
233 log_sys_error("unlink", bf->path);
234
235 /* Don't delete any more if we've reached the minimum */
236 if (--archives_size <= min_archive)
237 return;
238 }
239}

241int archive_vg(struct volume_group *vg,
242 const char *dir, const char *desc,
243 uint32_t retain_days, uint32_t min_archive)
244{
245 int i, fd, renamed = 0;
246 uint32_t ix = 0;
247 struct archive_file *last;
248 FILE *fp = NULL;
249 char temp_file[PATH_MAX], archive_name[PATH_MAX];
250 struct list *archives;
251
252 /*
253 * Write the vg out to a temporary file.
254 */
255 if (!create_temp_name(dir, temp_file, sizeof(temp_file), &fd)) {
256 log_err("Couldn't create temporary archive name.");
257 return 0;
258 }
259
260 if (!(fp = fdopen(fd, "w"))) {
261 log_err("Couldn't create FILE object for archive.");
262 if (close(fd))
263 log_sys_error("close", temp_file);
264 return 0;
265 }
266
267 if (!text_vg_export_file(vg, desc, fp)) {
268 stack;
269 if (fclose(fp))
270 log_sys_error("fclose", temp_file);
271 return 0;
272 }
273
274 if (fclose(fp)) {
275 log_sys_error("fclose", temp_file);
276 /* Leave file behind as evidence of failure */
277 return 0;
278 }
279
280 /*
281 * Now we want to rename this file to <vg>_index.vg.
282 */
283 if (!(archives = _scan_archive(vg->cmd->mem, vg->name, dir))) {
284 stack;
285 return 0;
286 }
287
288 if (list_empty(archives))
289 ix = 0;
290 else {
291 last = list_item(list_first(archives), struct archive_file);
292 ix = last->index + 1;
293 }
294
295 for (i = 0; i < 10; i++) {
296 if (dm_snprintf(archive_name, sizeof(archive_name),
297 "%s/%s_%05u.vg", dir, vg->name, ix) < 0) {
298 log_error("Archive file name too long.");
299 return 0;
300 }
301
302 if ((renamed = lvm_rename(temp_file, archive_name)))
303 break;
304
305 ix++;
306 }
307
308 if (!renamed)
309 log_error("Archive rename failed for %s", temp_file);
310
311 _remove_expired(archives, list_size(archives) + renamed, retain_days,
312 min_archive);
313
314 return 1;
315}
