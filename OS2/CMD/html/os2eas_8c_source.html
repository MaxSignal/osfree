51/****************************************************************\
52 * *
53 * Name: ReadEAs(filename) *
54 * *
55 * Purpose: Call EMX to Query a file's EA names and values. *
56 * *
57 * Usage: *
58 * *
59 * Method : NOTE: This routine does NOT prevent other *
60 * processes from accessing the file's EAs while it is*
61 * reading them in, or while the program is editing *
62 * them. *
63 * *
64 * Returns: 0 if it successfully reads in the EAs. Upon exit*
65 * , CurrentFileEAs holds the EAs for the current *
66 * file. Any EAs previously saved will be destroyed. *
67 * *
68\****************************************************************/
69bool ReadEAs(CHARTYPE *filename)
70{
71 if (CurrentFileEAs != NULL) /* should not happen */
72 {
73 _ead_destroy(CurrentFileEAs);
74 CurrentFileEAs = NULL;
75 }
76
77 if ((CurrentFileEAs = _ead_create()) == NULL)
78 return (1);
79 if (_ead_read(CurrentFileEAs, filename, 0, 0) != 0)
80 {
81 _ead_destroy(CurrentFileEAs);
82 CurrentFileEAs = NULL;
83 return (1);
84 }
85 return(0);
86}

88/****************************************************************\
89 * *
90 * Name: WriteEAs(filename) *
91 * *
92 * Purpose: Writes all EAs of CurrentFileEAs to the file *
93 * and destroys the contents of CurrentFileEAs. *
94 * CurrentFileEAs may be empty. *
95 * You should never nest ReadEAs und WriteEAs. *
96 * *
97 * Returns: 0 if successful, 1 otherwise *
98 * *
99\****************************************************************/
100bool WriteEAs(CHARTYPE *filename)
101{
102 bool retval;
103
104 if (CurrentFileEAs == NULL)
105 return (1);
106 if (_ead_write(CurrentFileEAs, filename, 0, 0) < 0)
107 retval = 1;
108 else
109 retval = 0;
110
111 _ead_destroy(CurrentFileEAs);
112 CurrentFileEAs = NULL;
113 return (retval);
114}

120 directly to those of the FEA structure. Note however, that both szName

133};

163/****************************************************************\
164 * *
165 * Name: ReadEAs(filename) *
166 * *
167 * Purpose: Call DOS to Query a file's EA names and values. *
168 * *
169 * Usage: *
170 * *
171 * Method : Routine does NOT do full memory error trapping. *
172 * NOTE: This routine does NOT prevent other *
173 * processes from accessing the file's EAs while it is*
174 * reading them in, or while the program is editing *
175 * them. *
176 * *
177 * Returns: 0 if it successfully reads in the EAs. Upon exit*
178 * , globalp HoldFEA points to a linked list of the *
179 * EAs for the current file. *
180 * *
181\****************************************************************/
182bool ReadEAs(CHARTYPE *filename)
183{
184 CHAR *pAllocc=NULL; /* Holds the FEA struct returned by DosEnumAttribute */
185 /* used to create the GEA2LIST for DosQueryPathInfo */
186 CHAR *pBigAlloc=NULL; /* Temp buffer to hold each EA as it is read in */
187 USHORT cbBigAlloc=0; /* Size of buffer */
188
189 ULONG ulEntryNum = 1; /* count of current EA to read (1-relative) */
190 ULONG ulEnumCnt; /* Number of EAs for Enum to return, always 1 */
191
192 HOLDFEA *pLastIn=0; /* Points to last EA added, so new EA can link */
193 HOLDFEA *pNewFEA=NULL; /* Struct to build the new EA in */
194
195 FEA2 *pFEA; /* Used to read from Enum's return buffer */
196 GEA2LIST *pGEAList;/*Ptr used to set up buffer for DosQueryPathInfo call */
197 EAOP2 eaopGet; /* Used to call DosQueryPathInfo */
198 /* Allocate enough room for any GEA List */
199 GetMem((PPVOID)&pAllocc, MAX_GEA);
200 pFEA = (FEA2 *) pAllocc; /* pFEA always uses pAlloc buffer */
201
202 pHoldFEA = '\0'; /* Reset the pointer for the EA linked list */
203
204 while(1) /* Loop continues until there are no more EAs */
205 {
206 ulEnumCnt = 1; /* Only want to get one EA at a time */
207 if(DosEnumAttribute(Ref_ASCIIZ, /* Read into pAlloc Buffer */
208 (CHAR *)filename, /* Note that this does not */
209 ulEntryNum, /* get the aValue field, */
210 pAllocc, /* so DosQueryPathInfo must*/
211 MAX_GEA, /* be called to get it. */
212 &ulEnumCnt,
213 (LONG) GetInfoLevel1))
214 break; /* There was some sort of error */
215
216 if(ulEnumCnt != 1) /* All the EAs have been read */
217 break;
218
219 ulEntryNum++;
220
221 GetMem((PPVOID)&pNewFEA, sizeof(HOLDFEA));
222
223 if (pNewFEA == '\0') /* Out of memory */
224 {
225 DosFreeMem(pAllocc);
226 Free_FEAList(pHoldFEA);
227 return (1);
228 }
229
230 pNewFEA->cbName = pFEA->cbName; /* Fill in the HoldFEA structure */
231 pNewFEA->cbValue= pFEA->cbValue;
232 pNewFEA->fEA = pFEA->fEA;
233 pNewFEA->next = '\0';
234
235 GetMem((PPVOID)&pNewFEA->szName, pFEA->cbName +1); /* Allocate for 2 arrays */
236 GetMem((PPVOID)&pNewFEA->aValue, pFEA->cbValue);
237
238 if (!pNewFEA->szName || !pNewFEA->aValue) /* Out of memory */
239 {
240 if(pNewFEA->szName)
241 DosFreeMem(pNewFEA->szName);
242 if(pNewFEA->aValue)
243 DosFreeMem(pNewFEA->aValue);
244
245 DosFreeMem(pAllocc);
246 DosFreeMem(pNewFEA);
247
248 Free_FEAList(pHoldFEA);
249 return (1);
250 }
251 strcpy(pNewFEA->szName,pFEA->szName); /* Copy in EA Name */
252
253 cbBigAlloc = sizeof(FEA2LIST) + pNewFEA->cbName+1 +
254 pNewFEA->cbValue;
255 GetMem((PPVOID)&pBigAlloc, cbBigAlloc);
256 if (pBigAlloc == '\0')
257 {
258 DosFreeMem(pNewFEA->szName);
259 DosFreeMem(pNewFEA->aValue);
260 DosFreeMem(pAllocc);
261 DosFreeMem(pNewFEA);
262 Free_FEAList(pHoldFEA);
263 return (1);
264 }
265
266 pGEAList = (GEA2LIST *) pAllocc; /* Set up GEAList structure */
267
268 pGEAList->cbList = sizeof(GEA2LIST) + pNewFEA->cbName; /* +1 for NULL */
269 pGEAList->list[0].oNextEntryOffset = 0L;
270 pGEAList->list[0].cbName = pNewFEA->cbName;
271 strcpy(pGEAList->list[0].szName,pNewFEA->szName);
272
273 eaopGet.fpGEA2List = (GEA2LIST FAR *) pAllocc;
274 eaopGet.fpFEA2List = (FEA2LIST FAR *) pBigAlloc;
275
276 eaopGet.fpFEA2List->cbList = cbBigAlloc;
277
278 DosQueryPathInfo((CHAR *)filename, /* Get the complete EA info */
279 FIL_QUERYEASFROMLIST,
280 (PVOID) &eaopGet,
281 sizeof(EAOP2));
282 memcpy(pNewFEA->aValue, /* Copy the value to HoldFEA */
283 pBigAlloc+sizeof(FEA2LIST)+pNewFEA->cbName,
284 pNewFEA->cbValue);
285
286 DosFreeMem(pBigAlloc); /* Release the temp Enum buffer */
287
288#if 0
289 if(!CheckEAIntegrity(pNewFEA->aValue,pNewFEA->cbValue)) /* Bad EA */
290 {
291 FreeMem(pNewFEA->szName);
292 FreeMem(pNewFEA->aValue);
293 FreeMem(pNewFEA);
294 continue; /* Don't add this EA to linked list */
295 }
296#endif
297
298 if(pHoldFEA == '\0') /* If first EA, set pHoldFEA */
299 pHoldFEA = pNewFEA;
300 else /* Otherwise, add to end of list */
301 pLastIn->next = pNewFEA;
302
303 pLastIn = pNewFEA; /* Update the end of the list */
304 }
305 DosFreeMem(pAllocc); /* Free up the GEA buf for DosEnum */
306 return (0);
307}

309/****************************************************************\
310 * *
311 * Name: WriteEAs(filename) *
312 * *
313 * Purpose: This routine updates the EAs on disk to reflect *
314 * their current condition in memory. First, all EAs *
315 * in the delete list are removed from the disk, then *
316 * all EAs in the pHoldFEA list are written to disk. *
317 * Usage : *
318 * Method : This routine is not bulletproof as it does not get *
319 * exclusive access to the file EAs, nor does it *
320 * handle out of disk space sort of errors. Also, *
321 * memory fetches are not fully error trapped. *
322 * *
323 * Returns: 0 if successful, 1 otherwise *
324 * *
325\****************************************************************/
326bool WriteEAs(CHARTYPE *filename)
327{
328 HOLDFEA *pHFEA=pHoldFEA;
329 EAOP2 eaopWrite;
330 CHAR aBuf[MAX_GEA],*aPtr=NULL;
331 FEA2 *pFEA = (FEA2 *) &aBuf[sizeof(ULONG)];
332 USHORT usMemNeeded=0, usRet=0;
333 ULONG *pulPtr=(ULONG *)aBuf;/* Initally points to top of FEALIST */
334
335 eaopWrite.fpFEA2List=(FEA2LIST FAR *)aBuf; /* Setup fields that won't */
336 pFEA->fEA = 0; /* change for the delete */
337 pFEA->cbValue = 0; /* calls to DosSetPathInfo */
338
339 while(pHFEA) /* Go through each HoldFEA */
340 {
341 usMemNeeded = sizeof(FEA2LIST) + pHFEA->cbName+1 + pHFEA->cbValue;
342 GetMem((PPVOID)&aPtr, usMemNeeded);
343 if (aPtr == NULL) /* GetMem failed to allocate */
344 {
345 Free_FEAList(pHoldFEA);
346 return(1);
347 }
348
349 eaopWrite.fpFEA2List = (FEA2LIST FAR *) aPtr;/* Fill in eaop struct */
350 eaopWrite.fpFEA2List->cbList = usMemNeeded;
351
352 eaopWrite.fpFEA2List->list[0].fEA = pHFEA->fEA;
353 eaopWrite.fpFEA2List->list[0].cbName = pHFEA->cbName;
354 eaopWrite.fpFEA2List->list[0].cbValue = pHFEA->cbValue;
355
356 strcpy(eaopWrite.fpFEA2List->list[0].szName, pHFEA->szName);
357 memcpy(eaopWrite.fpFEA2List->list[0].szName + pHFEA->cbName+1,
358 pHFEA->aValue, pHFEA->cbValue);
359
360 usRet = DosSetPathInfo((CHAR *)filename, /* Write out the EA */
361 FIL_QUERYEASIZE,
362 (PVOID) &eaopWrite,
363 sizeof(EAOP2),
364 DSPI_WRTTHRU);
365 if (usRet)
366 {
367 DosFreeMem(aPtr); /* Free up the FEALIST struct */
368 Free_FEAList(pHoldFEA);
369 return(1);
370 }
371 DosFreeMem(aPtr); /* Free up the FEALIST struct */
372
373 pHFEA = pHFEA->next; /* point to next EA */
374 }
375 Free_FEAList(pHoldFEA); /* free up all memory associated with EAs */
376 return(0);
377}

379/****************************************************************\
380 * *
381 * Name: Free_FEAList(pFEA) *
382 * *
383 * Purpose: This routine frees up the current list of EAs by *
384 * deallocating the space used by the szName and *
385 * aValue fields, then deallocating the HoldFEA struct*
386 * Next, it deletes the EAName space. *
387 * Usage : *
388 * Method: Note that EAS always passes in pHoldFEA. *
389 * which is unnecessary since they are global*
390 * pointers;however, this is done to make the routine *
391 * more flexible by allowing multiple linked lists to *
392 * exist. *
393 * *
394 * Returns: VOID. The two linked lists passed in are cleaned *
395 * out though. *
396 * *
397\****************************************************************/
398static VOID Free_FEAList(HOLDFEA *pFEA)
399{
400 HOLDFEA *next; /* Holds the next field since we free the structure */
401 /* before reading the current next field */
402
403 while(pFEA)
404 {
405 next = pFEA->next;
406 if(pFEA->szName) /* Free if non-NULL name */
407 DosFreeMem(pFEA->szName);
408 if(pFEA->aValue) /* Free if non-NULL value */
409 DosFreeMem(pFEA->aValue);
410
411 DosFreeMem(pFEA); /* Free HoldFEA struct */
412 pFEA = next;
413 }
414 return;
415}

417/****************************************************************\
418 * *
419 * Name: GetMem (ppv, cb) *
420 * *
421 * Purpose: This routine returns a pointer to a available* *
422 * memory space. *
423 * *
424 * Usage : *
425 * Method : Routine should be bullet proof as it does its own *
426 * error checking. It assumes that hwnd points to the *
427 * correct window with the name listbox in it. *
428 * *
429 * Returns: VOID, and allocated memory. *
430 * *
431\****************************************************************/
432static VOID GetMem (PVOID *ppv, ULONG cb)
433{
434 BOOL f;
435
436 f =(BOOL) DosAllocMem(ppv, cb, fPERM|PAG_COMMIT);
437 if (f)
438 {
439 *ppv = NULL;
440 return;
441 }
442 return;
443}
