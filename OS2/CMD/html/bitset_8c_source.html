21dm_bitset_t dm_bitset_create(struct dm_pool *mem, unsigned num_bits)
22{
23 unsigned n = (num_bits / DM_BITS_PER_INT) + 2;
24 size_t size = sizeof(int) * n;
25 dm_bitset_t bs;
26
27 if (mem)
28 bs = dm_pool_zalloc(mem, size);
29 else
30 bs = dm_malloc(size);
31
32 if (!bs)
33 return NULL;
34
35 *bs = num_bits;
36
37 if (!mem)
38 dm_bit_clear_all(bs);
39
40 return bs;
41}

48void dm_bit_union(dm_bitset_t out, dm_bitset_t in1, dm_bitset_t in2)
49{
50 int i;
51 for (i = (in1[0] / DM_BITS_PER_INT) + 1; i; i--)
52 out[i] = in1[i] | in2[i];

58static inline int _test_word(uint32_t test, int bit)
59{
60 while (bit < (int) DM_BITS_PER_INT) {
61 if (test & (0x1 << bit))
62 return bit;
63 bit++;
64 }
65
66 return -1;
67}

69int dm_bit_get_next(dm_bitset_t bs, int last_bit)
70{
71 int bit, word;
72 uint32_t test;
73
74 last_bit++; /* otherwise we'll return the same bit again */
75
76 /*
77 * bs[0] holds number of bits
78 */
79 while (last_bit < (int) bs[0]) {
80 word = last_bit >> INT_SHIFT;
81 test = bs[word + 1];
82 bit = last_bit & (DM_BITS_PER_INT - 1);
83
84 if ((bit = _test_word(test, bit)) >= 0)
85 return (word * DM_BITS_PER_INT) + bit;
86
87 last_bit = last_bit - (last_bit & (DM_BITS_PER_INT - 1)) +
88 DM_BITS_PER_INT;
89 }
90
91 return -1;
92}
