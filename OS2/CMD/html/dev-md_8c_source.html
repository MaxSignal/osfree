22#define MD_SB_MAGIC 0xa92b4efc
23#define MD_RESERVED_BYTES (64 * 1024)
24#define MD_RESERVED_SECTORS (MD_RESERVED_BYTES / 512)
25#define MD_NEW_SIZE_SECTORS(x) ((x & ~(MD_RESERVED_SECTORS - 1)) \
26 - MD_RESERVED_SECTORS)

28/*
29 * Returns -1 on error
30 */
31int dev_is_md(struct device *dev, uint64_t *sb)
32{
33 int ret = 0;
34
35#ifdef linux
36
37 uint64_t size, sb_offset;
38 uint32_t md_magic;
39
40 if (!dev_get_size(dev, &size)) {
41 stack;
42 return -1;
43 }
44
45 if (size < MD_RESERVED_SECTORS * 2)
46 return 0;
47
48 if (!dev_open(dev)) {
49 stack;
50 return -1;
51 }
52
53 sb_offset = MD_NEW_SIZE_SECTORS(size) << SECTOR_SHIFT;
54
55 /* Check if it is an md component device. */
56 /* Version 1 is little endian; version 0.90.0 is machine endian */
57 if (dev_read(dev, sb_offset, sizeof(uint32_t), &md_magic) &&
58 ((md_magic == xlate32(MD_SB_MAGIC)) ||
59 (md_magic == MD_SB_MAGIC))) {
60 if (sb)
61 *sb = sb_offset;
62 ret = 1;
63 }
64
65 if (!dev_close(dev))
66 stack;
67
68#endif
69 return ret;
70}
