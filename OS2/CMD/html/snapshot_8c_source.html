32static int _snap_text_import(struct lv_segment *seg, const struct config_node *sn,
33 struct dm_hash_table *pv_hash) //__attribute((unused)))
34{
35 uint32_t chunk_size;
36 const char *org_name, *cow_name;
37 struct logical_volume *org, *cow;
38
39 seg->lv->status |= SNAPSHOT;
40
41 if (!get_config_uint32(sn, "chunk_size", &chunk_size)) {
42 log_error("Couldn't read chunk size for snapshot.");
43 return 0;
44 }
45
46 log_suppress(1);
47
48 if (!(cow_name = find_config_str(sn, "cow_store", NULL))) {
49 log_suppress(0);
50 log_error("Snapshot cow storage not specified.");
51 return 0;
52 }
53
54 if (!(org_name = find_config_str(sn, "origin", NULL))) {
55 log_suppress(0);
56 log_error("Snapshot origin not specified.");
57 return 0;
58 }
59
60 log_suppress(0);
61
62 if (!(cow = find_lv(seg->lv->vg, cow_name))) {
63 log_error("Unknown logical volume specified for "
64 "snapshot cow store.");
65 return 0;
66 }
67
68 if (!(org = find_lv(seg->lv->vg, org_name))) {
69 log_error("Unknown logical volume specified for "
70 "snapshot origin.");
71 return 0;
72 }
73
74 if (!vg_add_snapshot(seg->lv->vg->fid, seg->lv->name, org, cow,
75 &seg->lv->lvid, seg->len, chunk_size)) {
76 stack;
77 return 0;
78 }
79
80 return 1;
81}

92#ifdef DEVMAPPER_SUPPORT
93static int _snap_target_percent(void **target_state __attribute((unused)),
94 struct dm_pool *mem __attribute((unused)),
95 struct cmd_context *cmd __attribute((unused)),
96 struct lv_segment *seg __attribute((unused)),
97 char *params, uint64_t *total_numerator,
98 uint64_t *total_denominator, float *percent)
99{
100 float percent2;
101 uint64_t numerator, denominator;
102
103 if (strchr(params, '/')) {
104 if (sscanf(params, "%" PRIu64 "/%" PRIu64,
105 &numerator, &denominator) == 2) {
106 *total_numerator += numerator;
107 *total_denominator += denominator;
108 }
109 } else if (sscanf(params, "%f", &percent2) == 1) {
110 *percent += percent2;
111 *percent /= 2;
112 }
113
114 return 1;
115}
