41/***********************************************************************/
42#ifdef HAVE_PROTO
43RESERVED *add_reserved_line(CHARTYPE *spec,CHARTYPE *line,short base,short off,COLOUR_ATTR *attr)
44#else
45RESERVED *add_reserved_line(spec,line,base,off,attr)
46CHARTYPE *spec,*line;
47short base;
48short off;
49COLOUR_ATTR *attr;
50#endif
51/***********************************************************************/
52{
53 RESERVED *curr=NULL;
54 CHARTYPE *templine=line;
55
56 TRACE_FUNCTION( "reserved.c:add_reserved_line" );
57 /*
58 * First check if the row already has a reserved line on it...
59 */
60 if ( ( curr = find_reserved_line( current_screen, FALSE, 0, base, off ) ) != NULL )
61 delete_reserved_line( base, off );
62 curr = rll_add( CURRENT_FILE->first_reserved, CURRENT_FILE->first_reserved, sizeof(RESERVED) );
63 if ( CURRENT_FILE->first_reserved == NULL )
64 CURRENT_FILE->first_reserved = curr;
65 if ( templine == NULL )
66 templine = (CHARTYPE *)"";
67 if ( ( curr->line = (CHARTYPE *)(*the_malloc)( ( strlen( (DEFCHAR *)templine ) + 1 ) * sizeof(CHARTYPE) ) ) == NULL )
68 {
69 display_error( 30, (CHARTYPE *)"", FALSE );
70 TRACE_RETURN();
71 return(NULL);
72 }
73 if ( ( curr->disp = (CHARTYPE *)(*the_malloc)( ( strlen( (DEFCHAR *)templine ) + 1 ) * sizeof(CHARTYPE) ) ) == NULL )
74 {
75 display_error( 30, (CHARTYPE *)"", FALSE );
76 TRACE_RETURN();
77 return(NULL);
78 }
79 if ( ( curr->spec = (CHARTYPE *)(*the_malloc)( ( strlen( (DEFCHAR *)spec ) + 1 ) * sizeof(CHARTYPE) ) ) == NULL )
80 {
81 display_error( 30, (CHARTYPE *)"", FALSE );
82 TRACE_RETURN();
83 return(NULL);
84 }
85 if ( ( curr->attr = (COLOUR_ATTR *)(*the_malloc)( sizeof(COLOUR_ATTR) ) ) == NULL )
86 {
87 display_error( 30, (CHARTYPE *)"", FALSE );
88 TRACE_RETURN();
89 return(NULL);
90 }
91 strcpy( (DEFCHAR *)curr->line, (DEFCHAR *)templine );
92 strcpy( (DEFCHAR *)curr->spec, (DEFCHAR *)spec );
93 curr->length = strlen( (DEFCHAR *)templine );
94 curr->base = base;
95 curr->off = off;
96 memcpy( curr->attr, attr, sizeof(COLOUR_ATTR) );
97 parse_reserved_line( curr );
98 TRACE_RETURN();
99 return(curr);
100}
101/***********************************************************************/
102#ifdef HAVE_PROTO
103RESERVED *find_reserved_line(CHARTYPE scrno,bool find_by_row,ROWTYPE row,short base,short off)
104#else
105RESERVED *find_reserved_line(scrno,find_by_row,row,base,off)
106CHARTYPE scrno;
107bool find_by_row;
108ROWTYPE row;
109short base,off;
110#endif
111/***********************************************************************/
112{
113 RESERVED *curr=SCREEN_FILE(scrno)->first_reserved;
114
115 TRACE_FUNCTION( "reserved.c:find_reserved_line" );
116 while( curr != NULL )
117 {
118 if ( find_by_row )
119 {
120 if ( curr->base == POSITION_TOP
121 && row == curr->off - 1 )
122 break;
123 if ( curr->base == POSITION_BOTTOM
124 && row == ( curr->off + screen[scrno].rows[WINDOW_FILEAREA] ) )
125 break;
126 if ( curr->base == POSITION_MIDDLE
127 && row == ( curr->off + (screen[scrno].rows[WINDOW_FILEAREA]/2) ) - 1 )
128 break;
129 }
130 else
131 {
132 if ( curr->base == base
133 && curr->off == off )
134 break;
135 }
136 curr = curr->next;
137 }
138 TRACE_RETURN();
139 return(curr);
140}
141/***********************************************************************/
142#ifdef HAVE_PROTO
143short delete_reserved_line(short base,short off)
144#else
145short delete_reserved_line(base,off)
146short base,off;
147#endif
148/***********************************************************************/
149{
150 RESERVED *curr=NULL;
151
152 TRACE_FUNCTION( "reserved.c:delete_reserved_line" );
153 if ( ( curr = find_reserved_line( current_screen, FALSE, 0, base, off ) ) == NULL )
154 {
155 display_error( 64, (CHARTYPE *)"", FALSE );
156 TRACE_RETURN();
157 return(RC_NO_LINES_CHANGED);
158 }
159 if ( curr->line != NULL )
160 (*the_free)(curr->line);
161 if ( curr->disp != NULL )
162 (*the_free)(curr->disp);
163 if ( curr->spec != NULL )
164 (*the_free)(curr->spec);
165 if ( curr->attr != NULL )
166 (*the_free)(curr->attr);
167 rll_del( &CURRENT_FILE->first_reserved, NULL, curr, DIRECTION_FORWARD );
168 TRACE_RETURN();
169 return(RC_OK);
170}
