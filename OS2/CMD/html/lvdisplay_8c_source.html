18static int _lvdisplay_single(struct cmd_context *cmd, struct logical_volume *lv,
19 void *handle)
20{
21 if (!arg_count(cmd, all_ARG) && !lv_is_visible(lv))
22 return ECMD_PROCESSED;
23
24 if (arg_count(cmd, colon_ARG))
25 lvdisplay_colons(lv);
26 else {
27 lvdisplay_full(cmd, lv, handle);
28 if (arg_count(cmd, maps_ARG))
29 lvdisplay_segments(lv);
30 }
31
32 return ECMD_PROCESSED;
33}

35int lvdisplay(struct cmd_context *cmd, int argc, char **argv)
36{
37 if (arg_count(cmd, columns_ARG)) {
38 if (arg_count(cmd, colon_ARG) || arg_count(cmd, maps_ARG)) {
39 log_error("Incompatible options selected");
40 return EINVALID_CMD_LINE;
41 }
42 return lvs(cmd, argc, argv);
43 } else if (arg_count(cmd, aligned_ARG) ||
44 arg_count(cmd, noheadings_ARG) ||
45 arg_count(cmd, options_ARG) ||
46 arg_count(cmd, separator_ARG) ||
47 arg_count(cmd, sort_ARG) || arg_count(cmd, unbuffered_ARG)) {
48 log_error("Incompatible options selected");
49 return EINVALID_CMD_LINE;
50 }
51
52 if (arg_count(cmd, colon_ARG) && arg_count(cmd, verbose_ARG)) {
53 log_error("Options -v and -c are incompatible");
54 return EINVALID_CMD_LINE;
55 }
56
57 return process_each_lv(cmd, argc, argv, LCK_VG_READ, NULL,
58 &_lvdisplay_single);
59}
