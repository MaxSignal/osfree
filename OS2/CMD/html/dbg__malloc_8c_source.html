61void *dm_malloc_aux_debug(size_t s, const char *file, int line)
62{
63 struct memblock *nb;
64 size_t tsize = s + sizeof(*nb) + sizeof(unsigned long);
65
66 if (s > 50000000) {
67 log_error("Huge memory allocation (size %" PRIsize_t
68 ") rejected - metadata corruption?", s);
69 return 0;
70 }
71
72 if (!(nb = malloc(tsize))) {
73 log_error("couldn't allocate any memory, size = %" PRIsize_t,
74 s);
75 return 0;
76 }
77
78 /* set up the file and line info */
79 nb->file = file;
80 nb->line = line;
81
82 dm_bounds_check();
83
84 /* setup fields */
85 nb->magic = nb + 1;
86 nb->length = s;
87 nb->id = ++_mem_stats.block_serialno;
88 nb->next = 0;
89
90 /* stomp a pretty pattern across the new memory
91 and fill in the boundary bytes */
92 {
93 char *ptr = (char *) (nb + 1);
94 size_t i;
95 for (i = 0; i < s; i++)
96 *ptr++ = i & 0x1 ? (char) 0xba : (char) 0xbe;
97
98 for (i = 0; i < sizeof(unsigned long); i++)
99 *ptr++ = (char) nb->id;
100 }
101
102 nb->prev = _tail;
103
104 /* link to tail of the list */
105 if (!_head)
106 _head = _tail = nb;
107 else {
108 _tail->next = nb;
109 _tail = nb;
110 }
111
112 _mem_stats.blocks_allocated++;
113 if (_mem_stats.blocks_allocated > _mem_stats.blocks_max)
114 _mem_stats.blocks_max = _mem_stats.blocks_allocated;
115
116 _mem_stats.bytes += s;
117 if (_mem_stats.bytes > _mem_stats.mbytes)
118 _mem_stats.mbytes = _mem_stats.bytes;
119
120 /* log_debug("Allocated: %u %u %u", nb->id, _mem_stats.blocks_allocated,
121 _mem_stats.bytes); */
122
123 return nb + 1;
124}

126void dm_free_aux(void *p)
127{
128 char *ptr;
129 size_t i;
130 struct memblock *mb = ((struct memblock *) p) - 1;
131 if (!p)
132 return;
133
134 dm_bounds_check();
135
136 /* sanity check */
137 assert(mb->magic == p);
138
139 /* check data at the far boundary */
140 ptr = ((char *) mb) + sizeof(struct memblock) + mb->length;
141 for (i = 0; i < sizeof(unsigned long); i++)
142 if (*ptr++ != (char) mb->id)
143 assert(!"Damage at far end of block");
144
145 /* have we freed this before ? */
146 assert(mb->id != 0);
147
148 /* unlink */
149 if (mb->prev)
150 mb->prev->next = mb->next;
151 else
152 _head = mb->next;
153
154 if (mb->next)
155 mb->next->prev = mb->prev;
156 else
157 _tail = mb->prev;
158
159 mb->id = 0;
160
161 /* stomp a different pattern across the memory */
162 ptr = ((char *) mb) + sizeof(struct memblock);
163 for (i = 0; i < mb->length; i++)
164 *ptr++ = i & 1 ? (char) 0xde : (char) 0xad;
165
166 assert(_mem_stats.blocks_allocated);
167 _mem_stats.blocks_allocated--;
168 _mem_stats.bytes -= mb->length;
169
170 /* free the memory */
171 free(mb);
172}

174void *dm_realloc_aux(void *p, unsigned int s, const char *file, int line)
175{

189int dm_dump_memory_debug(void)
190{
191 unsigned long tot = 0;
192 struct memblock *mb;
193 char str[32];
194 size_t c;
195
196 if (_head)
197 log_very_verbose("You have a memory leak:");
198
199 for (mb = _head; mb; mb = mb->next) {
200 for (c = 0; c < sizeof(str) - 1; c++) {
201 if (c >= mb->length)
202 str[c] = ' ';
203 else if (*(char *)((char *)mb->magic + c) == '\0')
204 str[c] = '\0';
205 else if (*(char *)((char *)mb->magic + c) < ' ')
206 str[c] = '?';
207 else
208 str[c] = *(char *)((char *)mb->magic + c);
209 }
210 str[sizeof(str) - 1] = '\0';
211
212 dm_log(_LOG_INFO, mb->file, mb->line,
213 "block %d at %p, size %" PRIsize_t "\t [%s]",
214 mb->id, mb->magic, mb->length, str);
215 tot += mb->length;
216 }
217
218 if (_head)
219 log_very_verbose("%ld bytes leaked in total", tot);
220
221 return 1;
222}

224void dm_bounds_check_debug(void)
225{
226 struct memblock *mb = _head;
227 while (mb) {
228 size_t i;
229 char *ptr = ((char *) (mb + 1)) + mb->length;
230 for (i = 0; i < sizeof(unsigned long); i++)
231 if (*ptr++ != (char) mb->id)
232 assert(!"Memory smash");
233
234 mb = mb->next;
235 }
236}
