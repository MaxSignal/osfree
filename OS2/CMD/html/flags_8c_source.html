66static struct flag *_get_flags(int type)
67{
68 switch (type) {
69 case VG_FLAGS:

71
72 case PV_FLAGS:
73 return _pv_flags;

75 case LV_FLAGS:
76 return _lv_flags;
77 }
78
79 log_err("Unknown flag set requested.");
80 return NULL;

83/*
84 * Converts a bitset to an array of string values,
85 * using one of the tables defined at the top of
86 * the file.
87 */
88int print_flags(uint32_t status, int type, char *buffer, size_t size)
89{
90 int f, first = 1;
91 struct flag *flags;
92
93 if (!(flags = _get_flags(type))) {
94 stack;
95 return 0;
96 }
97
98 if (!emit_to_buffer(&buffer, &size, "["))
99 return 0;
100
101 for (f = 0; flags[f].mask; f++) {
102 if (status & flags[f].mask) {
103 status &= ~flags[f].mask;
104
105 /* Internal-only flag? */
106 if (!flags[f].description)
107 continue;
108
109 if (!first) {
110 if (!emit_to_buffer(&buffer, &size, ", "))
111 return 0;
112 } else
113 first = 0;
114
115 if (!emit_to_buffer(&buffer, &size, "\"%s\"",
116 flags[f].description))
117 return 0;
118 }
119 }
120
121 if (!emit_to_buffer(&buffer, &size, "]"))
122 return 0;
123
124 if (status)
125 log_error("Metadata inconsistency: Not all flags successfully "
126 "exported.");
127
128 return 1;
129}

131int read_flags(uint32_t *status, int type, struct config_value *cv)
132{
133 int f;
134 uint32_t s = 0;
135 struct flag *flags;
136
137 if (!(flags = _get_flags(type))) {
138 stack;
139 return 0;
140 }
141
142 if (cv->type == CFG_EMPTY_ARRAY)
143 goto out;
144
145 while (cv) {
146 if (cv->type != CFG_STRING) {
147 log_err("Status value is not a string.");
148 return 0;
149 }
150
151 for (f = 0; flags[f].description; f++)
152 if (!strcmp(flags[f].description, cv->v.str)) {
153 s |= flags[f].mask;
154 break;
155 }
156
157 if (!flags[f].description) {
158 log_err("Unknown status flag '%s'.", cv->v.str);
159 return 0;
160 }
161
162 cv = cv->next;
163 }
164
165 out:
166 *status = s;
167 return 1;
168}
