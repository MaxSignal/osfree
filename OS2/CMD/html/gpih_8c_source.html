83/*
84 *@@gloss: GPI_rectangles GPI rectangles
85 * OS/2 PM (and GPI) uses two types of rectangles. This is rarely
86 * mentioned in the documentation, so a word is in order here.
87 *
88 * In general, graphics operations involving device coordinates
89 * (such as regions, bit maps and bit blts, and window management)
90 * use inclusive-exclusive rectangles. In other words, with
91 * those rectangles, xRight - xLeft is the same as the width
92 * of the rectangle (and yTop - yBottom = height).
93 *
94 * All other graphics operations, such as GPI functions that
95 * define paths, use inclusive-inclusive rectangles.
96 *
97 * This can be a problem with mixing Win and Gpi functions. For
98 * example, WinQueryWindowRect returns an inclusive-exclusive
99 * rectangle (tested V0.9.7 (2000-12-20) [umoeller]).
100 *
101 * WinFillRect expects an inclusive-exclusive rectangle, so it
102 * will work with a rectangle from WinQueryWindowRect directly.
103 *
104 * By contrast, the GpiBox expects an inclusive-inclusive rectangle.
105 */

121/*
122 *@@ gpihIsPointInRect:
123 * like WinPtInRect, but doesn't need a HAB.
124 *
125 * NOTE: as opposed to WinPtInRect, prcl is
126 * considered inclusive, that is, TRUE is
127 * returned even if x or y are exactly
128 * the same as prcl->xRight or prcl->yTop.
129 *
130 *@@added V0.9.9 (2001-02-28) [umoeller]
131 */
132
133BOOL gpihIsPointInRect(PRECTL prcl,
134 LONG x,
135 LONG y)
136{
137 if (prcl)
138 return ( (x >= prcl->xLeft)
139 && (x <= prcl->xRight)
140 && (y >= prcl->yBottom)
141 && (y <= prcl->yTop)
142 );
143
144 return FALSE;
145}

240/*
241 *@@ gpihManipulateRGB:
242 * this changes an RGB color value
243 * by multiplying each color component
244 * (red, green, blue) with dFactor.
245 *
246 * Each color component is treated separately,
247 * so if overflows occur (because dFactor
248 * is > 1), this does not affect the other
249 * components.
250 *
251 *@@changed V0.9.11 (2001-04-25) [umoeller]: changed prototype to use a double now
252 */
253
254VOID gpihManipulateRGB(PLONG plColor, // in/out: RGB color
255 double dFactor) // in: factor (> 1 makes brigher, < 1 makes darker)
256{
257 PBYTE pb = (PBYTE)plColor;
258
259 // in memory, the bytes are blue, green, red, unused
260
261 // blue
262 ULONG ul = (ULONG)( (double)(*pb) * dFactor
263 );
264 if (ul > 255)
265 *pb = 255;
266 else
267 *pb = (BYTE)ul;
268
269 // green
270 ul = (ULONG)( (double)(*(++pb)) * dFactor
271 );
272 if (ul > 255)
273 *pb = 255;
274 else
275 *pb = (BYTE)ul;
276
277 // red
278 ul = (ULONG)( (double)(*(++pb)) * dFactor
279 );
280 if (ul > 255)
281 *pb = 255;
282 else
283 *pb = (BYTE)ul;
284}

481 prcl->yBottom++;
482 prcl->xRight--;

568 *@@ gpihCharStringPosAt:
569 * wrapper for GpiCharStringPosAt.
570 * Since that function is limited to 512 characters
571 * (according to GPIREF; on my Warp 4 FP13, I actually
572 * get some 3000 characters... whatever this is),
573 * this splits the string into 512 byte chunks and
574 * calls GpiCharStringPosAt accordingly.
575 *
576 *@@added V0.9.3 (2000-05-06) [umoeller]
577 *@@changed V0.9.20 (2002-08-10) [umoeller]: fixed underline for bitmap fonts, which never worked
578 */
579
580LONG gpihCharStringPosAt(HPS hps, // in: presentation space for output
581 PPOINTL pptlStart,
582 PRECTL prclRect,
583 ULONG flOptions,
584 LONG lCount,
585 PCH pchString)
586{
587 LONG lHits = 0,
588 lCountLeft = lCount;
589 PCH pchThis = pchString;
590
591 GpiMove(hps, pptlStart);
592
593 if (lCount)
594 {
595 do
596 {
597 LONG lCountThis = lCountLeft;
598 if (lCountLeft >= 512)
599 lCountThis = 512;
600
601 lHits = GpiCharStringPos(hps,
602 prclRect,
603 0, // flOptions,
604 lCountThis,
605 pchThis,
606 0);
607
608 pchThis += 512;
609 lCountLeft -= 512;
610 } while (lCountLeft > 0);
611 }
612
613 // I can't get underscore to work with bitmap fonts,
614 // so I'm doing it manually always now
615 // V0.9.20 (2002-08-10) [umoeller]
616 if (flOptions & CHS_UNDERSCORE)
617 {
618 POINTL ptl2, ptlSave;
619 GpiQueryCurrentPosition(hps, &ptlSave);
620 ptl2.x = pptlStart->x;
621 ptl2.y = pptlStart->y - 2;
622 GpiMove(hps, &ptl2);
623 ptl2.x = ptlSave.x;
624 GpiLine(hps, &ptl2);
625 GpiMove(hps, &ptlSave);
626 }
627
628 return lHits;
629}

679/*
680 *@@ gpihDrawString:
681 * replacement for WinDrawText that can still align
682 * properly with multi-line strings.
683 *
684 * fl works as with WinDrawText, that is:
685 *
686 * -- specify one of DT_LEFT, DT_CENTER, DT_RIGHT;
687 *
688 * -- specifiy one of DT_TOP, DT_VCENTER, DT_BOTTOM.
689 *
690 * The alignment definitions are:
691 *
692 * -- DT_LEFT 0x00000000
693 * -- DT_CENTER 0x00000100
694 * -- DT_RIGHT 0x00000200
695 * -- DT_TOP 0x00000000
696 * -- DT_VCENTER 0x00000400
697 * -- DT_BOTTOM 0x00000800
698 *
699 * Other flags:
700 *
701 * -- DT_QUERYEXTENT 0x00000002 (not supported)
702 * -- DT_UNDERSCORE 0x00000010 (not supported)
703 * -- DT_STRIKEOUT 0x00000020 (not supported)
704 * -- DT_TEXTATTRS 0x00000040 (always enabled)
705 * -- DT_EXTERNALLEADING 0x00000080 (not supported)
706 * -- DT_HALFTONE 0x00001000 (not supported)
707 * -- DT_MNEMONIC 0x00002000 (not supported)
708 * -- DT_WORDBREAK 0x00004000 (always enabled)
709 * -- DT_ERASERECT 0x00008000 (not supported)
710 *
711 *@@added V1.0.1 (2003-01-17) [umoeller]
712 */
713
714VOID gpihDrawString(HPS hps, // in: presentation space for output
715 PCSZ pcsz, // in: string to test
716 PRECTL prcl, // in: clipping rectangle (inclusive!)
717 ULONG fl, // in: alignment flags
718 PFONTMETRICS pfm)
719{
720 PCSZ pThis = pcsz;
721 POINTL ptlRun,
722 ptlUse;
723 LONG cxRect,
724 cyRect,
725 cyString;
726
727 if (!pThis || !prcl || !pfm)
728 return;
729
730 ptlRun.x = prcl->xLeft;
731 ptlRun.y = prcl->yTop;
732
733 cxRect = prcl->xRight - prcl->xLeft + 1;
734 cyRect = prcl->yTop - prcl->yBottom + 1;
735
736 // vertical alignment:
737 if (fl & (DT_VCENTER | DT_BOTTOM))
738 {
739 ULONG cLines = strhCount(pcsz, '\n') + 1;
740 cyString = cLines * (pfm->lMaxBaselineExt + pfm->lMaxDescender + pfm->lExternalLeading);
741
742 if (fl & DT_VCENTER)
743 ptlRun.y = (cyRect - cyString) / 2;
744 else
745 ptlRun.y = cyString;
746 }
747
748 while (*pThis)
749 {
750 ULONG lenThis;
751 PCSZ pNext = strhFindEOL(pThis, &lenThis);
752
753 ptlRun.y -= pfm->lMaxBaselineExt;
754
755 if (lenThis)
756 {
757 // horizontal alignment:
758 if (!(fl & (DT_CENTER | DT_RIGHT)))
759 ptlUse.x = ptlRun.x;
760 else
761 {
762 POINTL aptl[TXTBOX_COUNT];
763 LONG cxString;
764 GpiQueryTextBox(hps,
765 lenThis,
766 (PCH)pThis,

1476 * of those fonts to perform a "real" match.
1477 *
1478 * If that fails, we allow GPI to do a "close" match based
1479 * on the input values. This might result in the system
1480 * default font (System Proportional) to be found, in the
1481 * worst case. But even then, a new LCID is returned.
1482 *
1483 * The "close match" comes in especially when using the
1484 * font attributes (bold, italics) and we are unable to
1485 * find the correct outline font for that. Unfortunately,
1486 * the information in FONTMETRICS.fsSelection is wrong,
1487 * wrong, wrong for the large majority of fonts. (For
1488 * example, I get the "bold" flag set for regular fonts,
1489 * and vice versa.) So we attempt to use the other fields,
1490 * but this doesn't always help. Not even Netscape gets
1491 * this right.
1492 *
1493 * <B>Font faces:</B>
1494 *
1495 * This is terribly complicated as well. You need to
1496 * differentiate between "true" emphasis faces (that
1497 * is, bold and italics are implemented thru separate
1498 * font files) and the ugly GPI simulation, which simply
1499 * makes the characters wider or shears them.
1500 *
1501 * This function even finds true "bold" and "italic" faces
1502 * for outline fonts. To do this, always specify the "family"
1503 * name as pszFaceName (e.g. "Courier" instead of "Courier
1504 * Bold") and set the flags for usFormat (e.g. FATTR_SEL_BOLD).
1505 * Note that this implies that you need call this function
1506 * twice to create two logical fonts for regular and bold faces.
1507 *
1508 * If a "true" emphasis font is not found, the GPI simulation
1509 * is enabled.
1510 *
1511 * <B>Remarks:</B>
1512 *
1513 * 1) This function _always_ creates a new logical font,
1514 * whose ID is returned, even if the specified font
1515 * was not found or a "close match" was performed by
1516 * GPI. As a result, do not call this function twice
1517 * for the same font specification, because there are
1518 * only 255 logical font IDs for each process.
1519 *
1520 * 2) Since this function always creates an LCID,
1521 * you should _always_ free the LCID later.
1522 * This is only valid if the font is no longer selected
1523 * into any presentation space. So use these calls:
1524 + GpiSetCharSet(hps, LCID_DEFAULT);
1525 + GpiDeleteSetId(hps, lLCIDReturnedByThis);
1526 *
1527 * 3) Using this function, bitmap fonts will have priority
1528 * over outline fonts of the same face name. This is
1529 * how the WPS does it too. This is most obvious with
1530 * the "Helv" font, which exists as a bitmap font for
1531 * certain point sizes only.
1532 *
1533 * 4) Since logical font IDs are shared across the
1534 * process, a mutex is requested while the lcids are
1535 * being queried and/or manipulated. In other words,
1536 * this func is now thread-safe (V0.9.9).
1537 *
1538 * This calls gpihQueryNextFontID in turn to find the
1539 * next free lcid. See remarks there.
1540 *

1544 * like this (according to PMREF):

1547 + º

1560 + ÈÍ # º
1561 + # º lMaxDescender
1562 + ______________ ####______________________________ Í¼

1565 * In turn, lMaxBaselineExt is lMaxAscender + lMaxDescender.
1566 *
1567 * Soooo... to find out about the optimal line spacing, GPIREF
