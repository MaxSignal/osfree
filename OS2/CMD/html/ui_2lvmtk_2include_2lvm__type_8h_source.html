87/* The following enum defines the various LVM classes to which a "feature" may belong.
88An LVM Plugin is used to implement a "feature", so "plugin" and "feature" are really synonyms. */
89typedef enum _LVM_Classes {
90Partition_Class, /* For "features" which must operate on a partition level - i.e. Bad Block Relocation. */
91Aggregate_Class, /* For "features" which combine partitions into a single logical entity - i.e. Drive Linking. */
92Volume_Class /* For "features" which operate best on a volume level - i.e. encryption, mirroring etc. */
93} LVM_Classes;
94
95/* An LVM plugin may belong to one or more classes. For each class to which it belongs, certain attributes must be defined.
96This structure tracks those attributes for a class. */
97typedef struct _LVM_Class_Attributes{
98BOOLEAN ClassMember; /* TRUE if a member of this class, FALSE otherwise. */
99BOOLEAN GlobalExclusive; /* TRUE if this plugin can not work with any other plugin - i.e. it
100must be the only "feature" on the volume, besides the built in feature of BBR. */
101BOOLEAN TopExclusive; /* TRUE if this plugin must be the topmost plugin in this class. */
102BOOLEAN BottomExclusive; /* TRUE if this plugin must be the bottommost plugin in this class. */
103BOOLEAN ClassExclusive; /* TRUE if this plugin will not work with any other plugin in this class. */
104CARDINAL32 Weight_Factor; /* A value between 1 and 100 which is used to guide the LVM interfaces when attempting to
105establish a default ordering for plugins within this class. A value of 1
106indicates that this plugin wants to be as close to the bottom of the plugins
107in this class as possible. A value of 100 means that this plugin wants to
108be as close to being the topmost plugin in this class as possible. This value
109is only used if none of the "exclusive" flags are set. */
110} LVM_Class_Attributes;

141/* The following define the maximum length of the names which can be used to represent a feature in LVM. The
142maximum name length for a feature is 30 characters plus the trailing NULL character. For command line parsing,
143though, a shorter name is preferable! Thus, the "short" name for a feature will be limited to 10 characters
144plus the trailing NULL character. The "short" name will be used for command line purposes, while the regular
145name will be used by all other interfaces. */
146#define MAX_FEATURE_NAME_LENGTH 31
147#define MAX_FEATURE_SHORT_NAME_LENGTH 11
148#define MAX_OEM_INFO_LENGTH 255

150/* The following definitions are used to control and access the various "features" available through the LVM Engine, such as Drive Linking and BBR. */
151typedef struct _Feature_ID_Data {
152char Name[MAX_FEATURE_NAME_LENGTH]; /* Feature Name, for use in menus and command line parsing. */
153char Short_Name[MAX_FEATURE_SHORT_NAME_LENGTH]; /* The name/code used to represent this feature during command line parsing. */
154char OEM_Info[MAX_OEM_INFO_LENGTH]; /* Name and copyright info. of the manufacturer, i.e. IBM, Vinca, etc. */
155CARDINAL32 ID; /* Numeric Feature ID. */
156CARDINAL32 Major_Version_Number; /* The version number of this feature. */
157CARDINAL32 Minor_Version_Number; /* The version number of this feature. */
158CARDINAL32 LVM_Major_Version_Number; /* The version of LVM that this feature was designed to work with. */
159CARDINAL32 LVM_Minor_Version_Number; /* The version of LVM that this feature was designed to work with. */
160LVM_Classes Preferred_Class; /* The class from which this "feature" prefers to be chosen. Encryption can be performed
161at the partition level or the volume level, and may therefore belong to both the
162Partition_Class and the Volume_Class. However, it is preferrable for it to be used
163on the volume level instead of at the partition level. Thus, its perferred class would
164be the Volume_Class, but it would still be a member of both the Volume_Class and the
165Partition_Class. */
166LVM_Class_Attributes ClassData[MAXIMUM_LVM_CLASSES]; /* The attributes for each of the LVM classes that this "feature" is in. */
167LVM_Interface_Support Interface_Support[MAXIMUM_LVM_INTERFACE_TYPES]; /* The functions and classes for each of the video modes that LVM can run it. */
168} Feature_ID_Data;
