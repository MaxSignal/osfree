26const char *text_vgname_import(const struct format_type *fmt,
27 struct device *dev,
28 off_t offset, uint32_t size,
29 off_t offset2, uint32_t size2,
30 checksum_fn_t checksum_fn, uint32_t checksum,
31 struct id *vgid, uint32_t *vgstatus,
32 char **creation_host)
33{
34 struct config_tree *cft;
35 struct text_vg_version_ops **vsn;
36 const char *vgname = NULL;
37
38 static int _text_import_initialised = 0;
39
40 if (!_text_import_initialised) {
41 _text_vsn_list[0] = text_vg_vsn1_init();
42 _text_vsn_list[1] = NULL;
43 _text_import_initialised = 1;
44 }
45
46 if (!(cft = create_config_tree(NULL, 0)))
47 return_NULL;
48
49 if ((!dev && !read_config_file(cft)) ||
50 (dev && !read_config_fd(cft, dev, offset, size,
51 offset2, size2, checksum_fn, checksum)))
52 goto_out;
53
54 /*
55 * Find a set of version functions that can read this file
56 */
57 for (vsn = &_text_vsn_list[0]; *vsn; vsn++) {
58 if (!(*vsn)->check_version(cft))
59 continue;
60
61 if (!(vgname = (*vsn)->read_vgname(fmt, cft, vgid, vgstatus,
62 creation_host)))
63 goto_out;
64
65 break;
66 }
67
68 out:
69 destroy_config_tree(cft);
70 return vgname;
71}

73struct volume_group *text_vg_import_fd(struct format_instance *fid,
74 const char *file,
75 struct device *dev,
76 off_t offset, uint32_t size,
77 off_t offset2, uint32_t size2,
78 checksum_fn_t checksum_fn,
79 uint32_t checksum,
80 time_t *when, char **desc)
81{
82 struct volume_group *vg = NULL;
83 struct config_tree *cft;
84 struct text_vg_version_ops **vsn;
85
86 static int _text_vg_import_initialised = 0;
87
88 if (!_text_vg_import_initialised) {
89 _text_vsn_list[0] = text_vg_vsn1_init();
90 _text_vsn_list[1] = NULL;
91 _text_vg_import_initialised = 1;
92 }
93
94 *desc = NULL;
95 *when = 0;
96
97 if (!(cft = create_config_tree(file, 0)))
98 return_NULL;
99
100 if ((!dev && !read_config_file(cft)) ||
101 (dev && !read_config_fd(cft, dev, offset, size,
102 offset2, size2, checksum_fn, checksum))) {
103 log_error("Couldn't read volume group metadata.");
104 goto out;
105 }
106
107 /*
108 * Find a set of version functions that can read this file
109 */
110 for (vsn = &_text_vsn_list[0]; *vsn; vsn++) {
111 if (!(*vsn)->check_version(cft))
112 continue;
113
114 if (!(vg = (*vsn)->read_vg(fid, cft))) {
115 stack;
116 goto out;
117 }
118
119 (*vsn)->read_desc(fid->fmt->cmd->mem, cft, when, desc);
120 break;
121 }
122
123 out:
124 destroy_config_tree(cft);
125 return vg;
126}
