393ACPI_STATUS
394GetGpeNotify(
395 ACPI_HANDLE ObjHandle,
396 UINT32 NestingLevel,
397 void *Context,
398 void **ReturnValue)
399{
400 ACPI_BUFFER Buffer;
401 char DevName[5];
402 ULONG GpeNumber, GpeTriggered, i, Flag;
403
404 Buffer.Length = 5;
405 Buffer.Pointer = DevName;
406 AcpiTkGetName(ObjHandle, ACPI_SINGLE_NAME, &Buffer);
407 DevName[5] = 0;
408
409 if ((DevName[1] == 'L') || (DevName[1] == 'l'))
410 {
411 GpeTriggered = ACPI_GPE_LEVEL_TRIGGERED;
412 }
413
414 if ((DevName[1] == 'E') || (DevName[1] == 'e'))
415 {
416 GpeTriggered = ACPI_GPE_EDGE_TRIGGERED;
417 }
418
419 DevName[0] = '0';
420 DevName[1] = 'X';
421 GpeNumber = strtoul(DevName,NULL,0);
422 Event[GpeNumber].GpeNdl = ObjHandle;
423 Event[GpeNumber].GpeUsed = 1;
424 Event[GpeNumber].GpeTriggered = GpeTriggered;
425 printf("GPE0x%2.2X triggered by %s \n",GpeNumber,
426 Event[GpeNumber].GpeTriggered == ACPI_GPE_EDGE_TRIGGERED ? "Edge" : "Level"
427 );
428
429 Flag = 0;
430 for (i = 0; i < DeviceCount; i++)
431 {
432 if (Device[i].GpeNumber != GpeNumber)
433 continue;
434 if (!Flag)
435 {
436 printf(" Using by device:\n\n");
437 Flag++;
438 }
439 Buffer.Length = 5;
440 Buffer.Pointer = DevName;
441 AcpiTkGetName(Device[i].Device, ACPI_SINGLE_NAME, &Buffer);
442 DevName[5] = 0;
443 printf(" [%s] S%dD %s %s\n",DevName, Device[i].SxDState,Device[i].PnPid, Device[i].PnPidInfo == NULL? "" :Device[i].PnPidInfo);
444 }
445 if (Flag)
446 printf("\n");
447
448 return(AE_OK);
449}

453ACPI_STATUS
454GetGpeDevice(
455 ACPI_HANDLE ObjHandle,
456 UINT32 NestingLevel,
457 void *Context,
458 void **ReturnValue)
459{
460 char DevName[5];
461 ACPI_BUFFER Results, Buffer;
462 ACPI_OBJECT *Obj;
463 ACPI_STATUS Status;
464 ULONG i,ndev;
465 ACPI_DEVICE_INFO *DevInfo = NULL;
466
467 Results.Length = sizeof(Object);
468 Results.Pointer = Object;
469
470 Status = AcpiTkEvaluateObject(ObjHandle,"_PRW",NULL,&Results);
471 if (Status != AE_OK)
472 return AE_OK;
473
474 Buffer.Length = 5;
475 Buffer.Pointer = DevName;
476 AcpiTkGetName(ObjHandle, ACPI_SINGLE_NAME, &Buffer);
477 DevName[5] = 0;
478
479 Obj = (ACPI_OBJECT *)Object;
480 if ((Obj->Type == ACPI_TYPE_PACKAGE) && (Obj->Package.Count == 2))
481 {
482 Device[DeviceCount].Device = ObjHandle;
483 Device[DeviceCount].GpeNumber = Obj->Package.Elements[0].Integer.Value;
484 Device[DeviceCount].SxDState = Obj->Package.Elements[1].Integer.Value;
485 Device[DeviceCount].PnPidInfo = NULL;
486 Device[DeviceCount].PnPid[0] = '\0';
487
488 Status = AcpiTkGetObjectInfoAlloc3(ObjHandle, &DevInfo);
489 if (Status)
490 {
491 DeviceCount++;
492 return AE_OK;
493 }
494 sprintf(Device[DeviceCount].PnPid,"%7.7s",&DevInfo->HardwareId);
495 ndev = sizeof(HidCodeInfo)/sizeof(struct Hid_Code_Info);
496 for (i = 0; i < ndev; i++)
497 {
498 if (!stricmp((char *)&(DevInfo->HardwareId), HidCodeInfo[i].pHid) )
499 {
