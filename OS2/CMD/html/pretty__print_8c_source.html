31void dump_lv(struct logical_volume *lv, FILE *fp)
32{
33 int i;
34
35 fprintf(fp, "logical_volume {\n");

37 fprintf(fp, "\tsize = %llu\n", lv->size);
38 fprintf(fp, "\tle_count = %u\n", lv->le_count);
39
40 fprintf(fp, "\tmap {\n");
41 for (i = 0; i < lv->le_count; i++) {
42 struct physical_volume *pv = lv->map[i].pv;
43
44 fprintf(fp, "\t\tpv = '%s', ",
45 pv ? pv->dev->name : "null ???");
46 fprintf(fp, "\textent = %u\n", lv->map[i].pe);
47 }
48 fprintf(fp, "\t}\n}\n\n");
49}

51void dump_vg(struct volume_group *vg, FILE *fp)
52{
53 struct list_head *tmp;
54
55 fprintf(fp, "volume_group {\n");
56 fprintf(fp, "\tname = '%s'\n", vg->name);
57 fprintf(fp, "\textent_size = %llu\n", vg->extent_size);
58 fprintf(fp, "\textent_count = %d\n", vg->extent_count);
59 fprintf(fp, "\tfree_count = %d\n", vg->free_count);
60 fprintf(fp, "\tmax_lv = %d\n", vg->max_lv);
61 fprintf(fp, "\tmax_pv = %d\n", vg->max_pv);
62 fprintf(fp, "\tpv_count = %d\n", vg->pv_count);
63 fprintf(fp, "\tlv_count = %d\n", vg->lv_count);
64 fprintf(fp, "}\n\n");
65
66 list_for_each(tmp, &vg->pvs) {
67 struct pv_list *pvl = list_entry(tmp, struct pv_list, list);
68 dump_pv(&pvl->pv, fp);
69 }
70
71 list_for_each(tmp, &vg->lvs) {
72 struct lv_list *lvl = list_entry(tmp, struct lv_list, list);
73 dump_lv(&lvl->lv, fp);
74 }
75}
