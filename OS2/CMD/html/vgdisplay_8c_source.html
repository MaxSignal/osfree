18static int vgdisplay_single(struct cmd_context *cmd, const char *vg_name,
19 struct volume_group *vg, int consistent,
20 void *handle) // __attribute((unused)))
21{
22 /* FIXME Do the active check here if activevolumegroups_ARG ? */
23 if (!vg) {
24 log_error("Volume group \"%s\" doesn't exist", vg_name);
25 return ECMD_FAILED;
26 }
27
28 if (!consistent)
29 log_error("WARNING: Volume group \"%s\" inconsistent", vg_name);
30
31 vg_check_status(vg, EXPORTED_VG);
32
33 if (arg_count(cmd, colon_ARG)) {
34 vgdisplay_colons(vg);
35 return ECMD_PROCESSED;
36 }
37
38 if (arg_count(cmd, short_ARG)) {
39 vgdisplay_short(vg);
40 return ECMD_PROCESSED;
41 }
42
43 vgdisplay_full(vg); /* was vg_show */
44
45 if (arg_count(cmd, verbose_ARG)) {
46 vgdisplay_extents(vg);
47
48 process_each_lv_in_vg(cmd, vg, NULL, NULL, NULL,
49 &lvdisplay_full);
50
51 log_print("--- Physical volumes ---");
52 process_each_pv_in_vg(cmd, vg, NULL, NULL, &pvdisplay_short);
53 }
54
55 check_current_backup(vg);
56
57 return ECMD_PROCESSED;
58}
