31// used by %@EXEC[] in all platforms
32int DoINT2E( char *cmd2e )
33{
34 int rval;
35 CRITICAL_VARS save_cv;
36
37 // save the critical variables
38 memmove( &save_cv, &cv, sizeof(cv) );
39
40 // ( re)initialize the critical vars
41 rval = cv.bn;
42 memset( &cv, '\0', sizeof(cv) );
43 cv.bn = rval;
44
45 // we have to force the CALL flag to 1 to keep from clobbering the
46 // current batch file if the program calls another batch file
47 // from within an INT 2Eh
48 cv.call_flag |= 1;
49
50 strcpy( gszCmdline, cmd2e );
51 rval = command( gszCmdline, 0 );
52
53 // restore the critical vars
54 memmove( &cv, &save_cv, sizeof(cv) );
55
56 return rval;
57}

60// find 4START & 4EXIT
61void find_4files( char *szFileName )
62{
63 PIPE_INHERIT *pPipeSave;
64 char *arg;
65 int nSaveErrorLevel, nSaveCallFlag;
66 PCH feptr;
67
68 // first, check .INI data for FSPath (no fallback - the file
69 // HAS to be there or it will be ignored)
70 if ( gpIniptr->FSPath != INI_EMPTYSTR ) {
71
72 strcpy( gszCmdline, ( gpIniptr->StrData + gpIniptr->FSPath ));
73
74 // force trailing backslash on path name
75 mkdirname( gszCmdline, szFileName+3 );
76
77 } else {
78
79 // look in the COMSPEC directory
80 if (( feptr = get_variable( COMSPEC )) == 0L )
81 return;
82
83 // remove "4xxx.xxx" & replace w/ 4START or 4EXIT
84 sprintf( gszCmdline, FMT_FAR_PREC_STR, MAXFILENAME-1, feptr );
85 insert_path( gszCmdline, szFileName+3, gszCmdline );
86 }
87
88 // if not found in COMSPEC, look in the boot root directory
89 if ((( arg = searchpaths( gszCmdline, NULL, TRUE )) == NULL ) && ( gpIniptr->FSPath == INI_EMPTYSTR ) && ( _stricmp( gszCmdline, szFileName ) != 0 ))
90 arg = searchpaths( szFileName, NULL, TRUE );
91
92 if ( arg != NULL ) {
93
94 // check for LFN name
95 AddQuotes( arg );
96
97 if (( pszCmdLineOpts != NULL ) && ( *pszCmdLineOpts ))
98 sprintf( gszCmdline, "%s %s", arg, pszCmdLineOpts );
99 else
100 strcpy( gszCmdline, arg );
101
102 nSaveErrorLevel = gnErrorLevel;
103
104 // treat it as a CALL to keep batch_cmd() from overwriting
105 // batch arguments inside a pipe
106 nSaveCallFlag = cv.call_flag;
107 cv.call_flag |= 1;
108
109 // save pipe shared memory so it's not closed yet
110 pPipeSave = gpPipeParent;
111 gpPipeParent = 0L;
112 ProcessCommands( gszCmdline, NULLSTR );
113 gpPipeParent = pPipeSave;
114 cv.call_flag = nSaveCallFlag;
115 gnErrorLevel = nSaveErrorLevel;
116 }
117}

120// batch file processor
121int BatchCLI( void )
122{
123 int ch, rval = 0;
124
125 // if offset == -1, we processed a CANCEL or QUIT
126 for ( cv.exception_flag = 0; ( bframe[cv.bn].offset >= 0L ); ) {
127
128 if (( setjmp( cv.env ) == -1 ) || ( cv.exception_flag )) {
129
130 // check for stack overflow
131 if ( cv.exception_flag & BREAK_STACK_OVERFLOW )
132 ch = ALL_CHAR;
133
134 // check for "ON ERROR / ERRORMSG" request
135 else if ( cv.exception_flag & BREAK_ON_ERROR ) {
136
137 cv.exception_flag = 0;
138 strcpy( gszCmdline, (( bframe[cv.bn].OnError != NULL ) ? bframe[cv.bn].OnError : bframe[cv.bn].OnErrorMsg ));
139 goto BatchExecCmd;
140
141 } else {
142
143 // Reset ^C so we come back here if the user types
144 // ^C in response to the (Y/N)? prompt
145 cv.exception_flag = 0;
146
147 // Got ^C internally or from an external program.
148
149 // check for "ON BREAK" request
150 if ( bframe[ cv.bn ].OnBreak != NULL ) {
151 strcpy( gszCmdline, bframe[cv.bn].OnBreak );
152 goto BatchExecCmd;
153 }
154
155 EnableSignals();
156
157 // console is redirected - write to display &
158 // read from keyboard
159 sprintf( gszCmdline, CANCEL_BATCH_JOB, bframe[cv.bn].pszBatchName );
160 WriteTTY( gszCmdline );
161
162 for ( ; ; ) {
163
164 ch = GetKeystroke( EDIT_NO_ECHO | EDIT_BIOS_KEY | EDIT_UC_SHIFT);
165
166 if ( isprint( ch )) {
167 WriteTTY( (char *)&ch );
168 if (( ch == YES_CHAR ) || ( ch == NO_CHAR ) || ( ch == ALL_CHAR ))
169 break;
170 WriteTTY( "\b" );
171 }
172
173 honk();
174 }
175
176 WriteTTY( "\r\n" );
177 }
178
179 cv.exception_flag = 0;
180
181 if ( ch != NO_CHAR ) {
182
183 // CANCEL ALL nested batch files
184 // by setting the file pointers to EOF
185 if ( ch == ALL_CHAR ) {
186
187 for ( ch = 0; ( ch < cv.bn ); ch++ )
188 bframe[ch].offset = -1L;
189
190 // force abort in things like GLOBAL
191 cv.exception_flag = CTRLC;
192 }
193
194 // cancel current batch file
195 bframe[ cv.bn ].offset = -1L;
196 rval = CTRLC;
197
198 // clear DO / IFF parsing flags
199 cv.f.lFlags = 0L;
200 break;
201 }
202
203 continue;
204 }
205
206 // Reset ^C handling
207 EnableSignals();
208
209 if ( open_batch_file() == 0 )
210 break;
211
212 // get file input
213 if ( getline( bframe[cv.bn].bfd, gszCmdline, MAXLINESIZ-1, EDIT_COMMAND ) == 0 )
214 break;
215 bframe[cv.bn].uBatchLine++;
216BatchExecCmd:
217
218 // parse & execute the command. BATCH_RETURN is a special
219 // code sent by functions like GOSUB/RETURN to end recursion
220 if ( command( gszCmdline, 2 ) == BATCH_RETURN )
221 break;
222 }
223
224 if ( setjmp( cv.env ) == -1 )
225 rval = CTRLC;
226
227 close_batch_file();
228
229 // disable signal handling momentarily
230 HoldSignals();
231
232 return rval;
233}

236// open the batch file and seek to the saved offset
237int open_batch_file( void )
238{
239 if (( bframe[ cv.bn ].bfd < 0 ) && (( bframe[ cv.bn ].flags & BATCH_REXX ) == 0 )) {
240
241 // Batch file was closed (either by us or Netware)
242 while (( bframe[ cv.bn ].bfd = _sopen( bframe[ cv.bn ].pszBatchName, (O_RDONLY | O_BINARY), SH_DENYWR) ) < 0 ) {
243
244 // if missing file is on fixed disk, assume worst & give up
245 if ( bframe[ cv.bn ].pszBatchName[0] > 'B' ) {
246 error( ERROR_4DOS_MISSING_BATCH, bframe[ cv.bn ].pszBatchName );
247 return 0;
248 }
249
250 qprintf( STDERR, INSERT_DISK, bframe[ cv.bn ].pszBatchName );
251 (void)GetKeystroke( EDIT_NO_ECHO | EDIT_ECHO_CRLF );
252 }
253
254 // seek to saved position
255 _lseek( bframe[cv.bn].bfd, bframe[cv.bn].offset, SEEK_SET );
256 }
257
258 return 1;
259}

273// get command line from console or file
274int getline( int fd, char *line, int nMaxSize, int nEditFlag )
275{
276 int i;
277 int fPipe = 0;
278
279 // get STDIN input (if not redirected or CTTY'd) from egets()
280 if (( fd == STDIN ) && ( QueryIsConsole( STDIN ))) {
281
282 // KEYS OFF?
283 if ( gpIniptr->LineIn ) {
284
285 STRINGINBUF Length;
286
287 Length.cb = (( nMaxSize > 255 ) ? 255 : nMaxSize );
288 Length.cchIn = 0;
289 *line = '\0';
290 (void)KbdStringIn( line, &Length, 0, 0 );
291 line[ Length.cchIn ] = '\0';
292 crlf();
293
294 return ( Length.cchIn );
295
296 } else
297 return ( egets( line, nMaxSize, nEditFlag ));
298 }
299
300 // if it's an in-memory batch file, read the far buffer
301 if ( fd == IN_MEMORY_FILE )
302 (void)memmove( line, &(bframe[cv.bn].in_memory_buffer[bframe[cv.bn].offset]), nMaxSize );
303
304 else {
305 // check to see if it's a pipe
306 if (( fPipe = QueryIsPipeHandle( fd )) != 0 ) {
307
308 unsigned int bytes_read;
309 int rval;
310
311 // we have to read anonymous pipes a byte at a
312 // time, since we can't peek or rewind them
313 i = 0;
314 do {
315 rval = FileRead( fd, line+i, 1, &bytes_read );
316 if ( bytes_read == 0 )
317 break;
318 if ( line[ i++ ] == '\n' )
319 break;
320 } while (( rval == 0 ) && ( i < nMaxSize ));
321 nMaxSize = i;
322
323 } else
324 nMaxSize = _read( fd, line, nMaxSize );
325 }
326
327 // get a line and set the file pointer to the next line
328 for ( i = 0; ; i++, line++ ) {
329
330 if (( i >= nMaxSize ) || ( *line == EoF ))
331 break;
332
333 if ( *line == '\r' ) {
334
335 // skip a LF following a CR or LF
336 if (( ++i < nMaxSize ) && ( line[1] == '\n' ))
337 i++;
338 break;
339
340 } else if ( *line == '\n' ) {
341 i++;
342 break;
343 }
344 }
345
346 // truncate the line
347 *line = '\0';
348
349 if ( i >= 0 ) {
350
351 // save the next line's position
352 if (( fd == IN_MEMORY_FILE ) || (( cv.bn >= 0 ) && ( fd == bframe[cv.bn].bfd )))
353 bframe[cv.bn].offset += i;
354 if ( fPipe == 0 ) // only do a seek on disk files, not pipes
355 if ( fd != IN_MEMORY_FILE )
356 (void)_lseek( fd, (long)( i - nMaxSize ), SEEK_CUR );
357 }
358
359 return i;
360}

366// Parse the command line, perform alias & variable expansion & redirection,
367// and execute it.
368int command( char *line, int fOptions )
369{
370 extern char *pszBatchDebugLine;
371
372 static int fStepOver = 0;
373 char *pszStartLine;
374 int i, rval = 0, fEcho = 0, fStep;
375 char condition, cEoL;
376 REDIR_IO redirect;
377
378 if (( line == NULL ) || ( *line == '\0' ) || ( *line == ':' ))
379 return 0;
380
381 // clear stdin, stdout, stderr, & pipe flags
382 memset( &redirect, '\0', sizeof(REDIR_IO) );
383
384 // echo the line if at the beginning of the line, and either:
385 // 1. In a batch file and ECHO is on
386 // 2. VERBOSE is on
387 if ( line == gszCmdline )
388 fEcho = (( cv.bn >= 0 ) ? bframe[cv.bn].echo_state : cv.verbose );
389
390 // history logging
391 if (( gpIniptr->HistLogOn ) && ( fOptions & 1 ))
392 (void)_log_entry( line, 1 );
393
394 for ( ; ; ) {
395
396 fStep = 0;
397
398 // check for ^C or stack overflow
399 if (( setjmp( cv.env ) == -1 ) || ( cv.exception_flag )) {
400 rval = CTRLC;
401 break;
402 }
403
404 // ( re)enable ^C and ^BREAK
405 EnableSignals();
406
407 // reset the default switch character (things like IF kill it!)
408 gpIniptr->SwChr = '/';
409
410 if (( rval == ABORT_LINE) || ( rval == BATCH_RETURN ) || (( cv.bn >= 0 ) && ( bframe[cv.bn].offset < 0L )))
411 break;
412
413 // kludge for WordPerfect Office bug ("COMMAND /C= ...")
414 strip_leading( line, "= \t\r\n,;" );
415 if ( *line == '\0' )
416 break;
417
418 // get the international format chars
419 QueryCountryInfo();
420
421 // kludge to check for a leading `
422 // (for example, "if "%a" == "%b" `a1 ^ a2`)
423 if ( *line == '`' ) {
424 if (( pszStartLine = scan( ++line, "`", QUOTES + 1 )) == BADQUOTES )
425 break;
426 // strip trailing `
427 if ( *pszStartLine != '\0' )
428 strcpy( pszStartLine, pszStartLine + 1 );
429 }
430
431 pszStartLine = strcpy( gszCmdline, line );
432
433 // perform alias expansion
434 if (( rval = alias_expand( pszStartLine )) != 0 )
435 break;
436
437 // check for leading '@' (no echo)
438 if (( *pszStartLine == '@' ) || ( strnicmp( pszStartLine, "*@", 2 ) == 0 )) {
439
440 if ( *pszStartLine == '*' )
441 strcpy( pszStartLine+1, skipspace( pszStartLine+2 ));
442 else {
443 pszStartLine++;
444 strcpy( pszStartLine, skipspace( pszStartLine ));
445 }
446
447 // check for "persistent echo" flag
448 fEcho &= 2;
449 fSetCMDLINE = 0;
450
451 } else
452 fSetCMDLINE = 1;
453
454 // skip blank lines or those beginning with REM or a label
455 if (( *pszStartLine == ':' ) || (( line = first_arg( pszStartLine )) == NULL ))
456 break;
457
458 // check for IFF / THEN / ELSEIFF / ENDIFF condition
459 // (if waiting for ENDIFF or ELSEIFF, ignore the command)
460 if (( iff_parsing( line, pszStartLine ) != 0 ) || (( rval = do_parsing( line )) != 0 )) {
461
462 // check for DO / END condition
463 if ( rval == BATCH_RETURN )
464 break;
465
466 // skip this command - look for next compound command
467 // (ignoring pipes & conditionals)
468 i = gpIniptr->CmdSep;
469
470 if (( line = scan( pszStartLine, (char *)&i, QUOTES_PARENS )) == BADQUOTES )
471 break;
472 if ( *line )
473 line++;
474 continue;
475 }
476
477 // check for ?"prompt" command single-stepping (Novell DOS 7)
478 if (( *pszStartLine == '?' ) && ( QueryIsConsole( STDIN ))) {
479
480 line = skipspace( pszStartLine + 1 );
481
482 // "? > ..." a kludge for Mike B.!
483 if (( *line != '\0' ) && ( *line != '>' ) && ( *line != '|' ) && ( *line != gpIniptr->CmdSep)) {
484
485 char *pszPrompt = line;
486
487 if ( *line == '"' ) {
488 // get prompt text enclosed in quotes
489 pszPrompt++;
490 line++;
491 while (( *line != '\0' ) && ( *line != '"' ))
492 line++;
493 if ( *line == '"' )
494 *line++ = '\0';
495 }
496
497 if ( QueryInputChar( pszPrompt, YES_NO ) == NO_CHAR )
498 break;
499 strcpy( pszStartLine, skipspace( line ));
500 }
501 }
502
503 // IFF may have removed command
504 if (( line = first_arg( pszStartLine )) == NULL )
505 break;
506
507 // skip a REM
508 // kludge for dumdums who do "rem > file"
509 if (( _stricmp( "REM", line ) == 0 ) && (*(skipspace( pszStartLine+3 )) != '>' )) {
510
511 // echo the REM lines if ECHO is ON
512 if ( fEcho )
513 printf( FMT_STR_CRLF, pszStartLine );
514 break;
515 }
516
517 // check last argument for an escape char (line continuation)
518 if (( rval = continued_line( pszStartLine )) != 0 )
519 break;
520
521 // Do variable expansion (except for certain internal commands)
522 // var_expand() will ignore variables inside command groups
523 if ((( i = findcmd( pszStartLine, 0 )) < 0 ) || ( commands[i].pflag & CMD_EXPAND_VARS )) {
524 if ((( rval = var_expand( pszStartLine, 0 )) != 0 ) || (*(pszStartLine = skipspace( pszStartLine )) == '\0' ))
525 break;
526 }
527
528 line = (( *(skipspace( pszStartLine )) == '(' ) ? QUOTES_PARENS : QUOTES );
529
530 // Trailing command group in internal command? ("IF a==b (")
531 if (( i >= 0 ) && ( commands[i].pflag & CMD_GROUPS )) {
532
533 // check last argument for a " ("
534 line = strend( pszStartLine ) - 1;
535 if (( *line == '(' ) && ( isdelim( line[ -1 ] )) && (( rval = command_groups( line, 0 )) != 0 ))
536 break;
537
538 // for commands like IF, IFF, FOR, etc. watch for ( )'s
539 line = QUOTES_PARENS;
540 }
541
542 // check for batch single-stepping
543 if (( fOptions & 2 ) && ( cv.bn >= 0 ) && ( gpIniptr->SingleStep ) && ( fStepOver == 0 ) && ( QueryIsConsole( STDIN ))) {
544
545 extern int BatchDebugger( void );
546
547 pszBatchDebugLine = pszStartLine;
548 rval = BatchDebugger( );
549
550 if (( rval == ERROR_EXIT ) || ( rval == 'Q' )) {
551 for ( rval = cv.bn; ; rval-- ) {
552 bframe[ rval ].offset = -1L;
553 if ( rval == 0 )
554 break;
555 }
556 break;
557 } else if ( rval == 'J' ) {
558 // 'Jump'?
559// Future enhancement - jump to any line in batch file
560 rval = 0;
561 break;
562 } else if ( rval == 'S' ) {
563 // 'S'(tep over)
564 fStep = fStepOver = 1;
565 } else if (( rval == 'O' ) || ( rval == ESCAPE ))
566 gpIniptr->SingleStep = 0;
567
568 // default is 'T'(race into)
569 }
570
571 // DETACH passes entire line to child
572 if (( i >= 0 ) && ( commands[i].pflag & CMD_DETACH_LINE ))
573 line = strend( pszStartLine );
574 else
575 // get compound command char, pipe, conditional, or EOL
576 if (( line = scan( pszStartLine, NULL, line )) == BADQUOTES )
577 break;
578
579 rval = 0;
580 cEoL = condition = 0;
581
582 // terminate command & save special char (if any)
583 if ( *line ) {
584 cEoL = *line;
585 *line++ = '\0';
586 }
587
588 // command ECHOing
589 if (( fEcho ) && ( *pszStartLine ))
590 printf( FMT_STR_CRLF, pszStartLine );
591
592 // command logging
593 if ( gpIniptr->LogOn )
594 (void)_log_entry( pszStartLine, 0 );
595
596 // process compound command char, pipe, or conditional
597 if ( cEoL == '|' ) {
598
599 if ( *line == '|' ) {
600
601 // conditional OR (a || b)
602 condition |= OR_CONDITION;
603 line++;
604
605 } else {
606
607 char *arg;
608
609 // check last argument for a group (" (")
610 // and pass the group to the child process
611 arg = strend( line ) - 1;
612 if (( *arg == '(' ) && ( isdelim( arg[ -1 ] )) && (( rval = command_groups( arg, 0 )) != 0 ))
613 break;
614
615 if (( rval = open_pipe( &redirect, line )) != 0 )
616 break;
617 line--;
618 }
619
620 } else if (( cEoL == '&' ) && ( *line == '&' )) {
621
622 // conditional AND (a && b)
623 condition |= AND_CONDITION;
624 line++;
625 }
626
627 // do I/O redirection except for some internal commands
628 // ( redir() will check for command grouping)
629 if ((( gpIniptr->Expansion & EXPAND_NO_REDIR ) == 0 ) && (( i < 0 ) || ( commands[i].pflag & CMD_EXPAND_REDIR ))) {
630
631 if ( redir( pszStartLine, &redirect )) {
632 rval = ERROR_EXIT;
633 break;
634 }
635 }
636
637 // strip leading delimiters & trailing whitespace
638 strip_leading( pszStartLine, ", \t\n\f\r" );
639 strip_trailing( pszStartLine, " \t\r\n" );
640
641 rval = ProcessCommands( pszStartLine, line );
642
643 // change icon label back to the default
644 if ( i < 0 )
645 update_task_list(( cv.bn < 0 ) ? NULL : bframe[cv.bn].pszBatchName );
646
647 // check for ^C or stack overflow
648 if (( setjmp( cv.env ) == -1 ) || ( cv.exception_flag )) {
649 rval = CTRLC;
650 break;
651 }
652
653 if ( fStep ) {
654 // reenable batch debugger after stepping over
655 // a CALL, GOSUB, etc.
656 fStep = fStepOver = 0;
657 }
658
659 // clean everything up
660 line = gszCmdline;
661
662 // check for funny return value from IFF (this kludge allows
663 // you to pipe to a command following the IFF; otherwise
664 // IFF would eat the pipe itself)
665 if ( rval == 0x666 ) {
666 rval = 0;
667 continue;
668 }
669
670 // if conditional command failed, skip next command(s)
671 while ((( rval != 0 ) && ( condition & AND_CONDITION )) || (( rval == 0 ) && ( condition & OR_CONDITION ))) {
672
673 // check for command grouping
674 line = skipspace( line );
675 if (( line = scan( line, NULL, ((*line == '(' ) ? QUOTES_PARENS : QUOTES) )) == BADQUOTES)
676 break;
677
678 if (( *line == '&' ) && ( line[1] == '&' )) {
679 condition = AND_CONDITION;
680 line += 2;
681 } else if (( *line == '|' ) && ( line[1] == '|' )) {
682 condition = OR_CONDITION;
683 line += 2;
684 } else
685 break;
686 }
687
688 ClosePipe( line );
689 // clean up redirection & check pipes from last command
690 unredir( &redirect, &rval );
691 }
692
693 // clean up I/O redirection
694 unredir( &redirect, &rval );
695
696 if ( fOptions & 1 )
697 crlf();
698
699 // disable signal handling momentarily
700 HoldSignals();
701
702 if ( fStep )
703 fStepOver = 0;
704
705 return rval;
706}

709static int ProcessCommands( char *pszStartLine, char *line )
710{
711 char *pszCmdName;
712 char *arg;
713 int i, rval = 0;
714
715 if ( *pszStartLine != '\0' ) {
716
717 // save the line continuation (if any) onto the stack
718 arg = _alloca( strlen( line ) + 1 );
719 line = strcpy( arg, line );
720
721 if ( *pszStartLine == '(' ) {
722
723 pszStartLine = strcpy( gszCmdline, pszStartLine );
724
725 // Process command groups
726 if (( rval = command_groups( pszStartLine, 1 )) == 0 ) {
727 PIPE_INHERIT *pPipeSave;
728
729 // don't return to parent pipe process in the middle of a
730 // command group!
731 pPipeSave = gpPipeParent;
732 gpPipeParent = 0L;
733 rval = command( pszStartLine, 0 );
734 gpPipeParent = pPipeSave;
735 }
736
737 } else if (( pszStartLine[1] == ':' ) && ( isdelim( pszStartLine[2] ))) {
738
739 // a disk change request
740 rval = __cd( pszStartLine, CD_CHANGEDRIVE | CD_SAVELASTDIR | CD_NOFUZZY );
741
742 // try internal commands
743 } else if (( rval = InternalCommands( pszStartLine )) == -1) {
744 // if it's a directory name, do a fast CDD
745 // Have to break on first '/' (annoying UNIX users) because
746 // too many people do things like "program/a"
747 if (( gpIniptr->UnixPaths ) || ( strnicmp( pszStartLine, "../", 3 ) == 0 ))
748 pszCmdName = ntharg( pszStartLine, 0x800 );
749 else
750 pszCmdName = first_arg( pszStartLine );
751
752 i = ( strlen( pszCmdName ) - 1 );
753
754 // check for implicit directory change (trailing '\')
755 // call "parse_line" because we need to process any
756 // escaped characters
757 if (( i >= 0 ) && (( pszCmdName[i] == '\\' ) || ( pszCmdName[i] == '/' )))
758 rval = parse_line( "cdd", pszStartLine, cdd_cmd, CMD_STRIP_QUOTES );
759 else {
760 // search for external command
761 rval = ExternalCommands( pszStartLine, pszCmdName );
762 }
763 }
764 }
765
766 // restore remainder of line
767 strcpy( gszCmdline, line );
768
769 return rval;
770}

773// search for an internal command, & execute it if found
774static int InternalCommands( char *pszStartLine )
775{
776 extern int nRexxError;
777 int i;
778 char *pszCmdName;
779 int rval = 0;
780
781 // search for an internal command
782 if (( i = findcmd( pszStartLine, 0 )) < 0 )
783 return i;
784
785 // save the command name (for things like USAGE)
786 gpInternalName = commands[i].cmdname;
787
788 // check for "command /?" & display help if found
789 // else, call the internal function
790 if ((( pszCmdName = first_arg( pszStartLine + strlen( gpInternalName ))) != NULL ) && (pszCmdName[0] == gpIniptr->SwChr ) && (pszCmdName[1] == '?' ) && (pszCmdName[2] == '\0' ))
791 return ( _help( gpInternalName, pszCmdName ));
792
793 // check if command only allowed in a batch file
794 if (( commands[i].pflag & CMD_ONLY_BATCH ) && ( cv.bn == -1 )) {
795 error( ERROR_4DOS_ONLY_BATCH, gpInternalName );
796 return USAGE_ERR;
797 }
798
799 pszWindowTitle = pszStartLine;
800 rval = parse_line( pszStartLine, (char *)(pszStartLine+strlen( gpInternalName)), commands[i].func, commands[i].pflag );
801
802 // ON ERROR usually returns a CTRLC, which isn't really accurate
803 if (( rval == CTRLC ) && ( cv.exception_flag & BREAK_ON_ERROR ))
804 rval = gnInternalErrorLevel;
805 else if ( rval == BATCH_RETURN_RETURN )
806 rval = BATCH_RETURN;
807 else if (( rval == BATCH_RETURN ) || (( gnInternalErrorLevel = rval ) == ABORT_LINE ) || ( gnInternalErrorLevel == 0x666 )) {
808 // clear internal error level flag
809 gnInternalErrorLevel = 0;
810 }
811
812 nRexxError = gnInternalErrorLevel;
813
814 // for some reason CMD.EXE sets ERRORLEVEL on some
815 // (but not all!) internal commands
816 if (( commands[i].pflag & CMD_SET_ERRORLEVEL ) && ( rval != ABORT_LINE ) && ( rval != BATCH_RETURN ))
817 gnErrorLevel = rval;
818 return rval;
819}

822// search for an external command or batch file, & execute it if found
823static int ExternalCommands( char *pszStartLine, char *pszCmdName )
824{
825 char *ptr, *pszSave;
826 PCH feptr;
827 int i, rval = 0;
828
829 pszSave = pszStartLine;
830
831 // search for external command, & set batch name pointer
832 *(scan( pszCmdName + 1, "`=", QUOTES + 1 )) = '\0';
833
834 // nitwit JPI Modula-2 escapes first char of command!
835 EscapeLine( pszCmdName );
836
837 ptr = _alloca( strlen( pszCmdName ) + 1 );
838 gpBatchName = pszCmdName = strcpy( ptr, pszCmdName );
839
840 // Wildcards not allowed in program names; note that we have to kludge
841 // the check for '[' because of Netware stupidity with names
842 // like "[:\dos\program.exe"
843 ptr = pszCmdName;
844 if (( *ptr == '[' ) && ( ptr[1] == ':' ))
845 ptr += 2;
846
847 // if not a COM, EXE, BTM, or BAT (or CMD), or
848 // a user-defined "executable extension", return w/error
849 if (( strpbrk( ptr, WILD_CHARS ) == NULL ) && (( ptr = searchpaths( pszCmdName, NULL, TRUE )) != NULL )) {
850
851 // DOS only allows a 127 character command line, so save the
852 // full command line to var CMDLINE. If command line preceded
853 // by a '@', remove CMDLINE from the environment
854 if ( fSetCMDLINE ) {
855 strins( pszStartLine, CMDLINE_VAR );
856 add_variable( pszStartLine );
857 strcpy( pszStartLine, pszStartLine + strlen( CMDLINE_VAR ));
858 } else
859 add_variable( CMDLINE_VAR );
860 pszWindowTitle = pszStartLine;
861 pszStartLine += strlen( pszCmdName );
862
863 pszCmdName = mkfname( ptr, 0 );
864 if (( ptr = ext_part( pszCmdName )) == NULL )
865 ptr = NULLSTR;
866
867 // OS/2 2.0 runs a DOS box on a .BAT file
868 if (( _stricmp( ptr, COM ) == 0 ) || ( _stricmp( ptr, EXE ) == 0 ) || ( _stricmp( ptr, BAT ) == 0 ))
869 {
870 gnErrorLevel = rval = parse_line( pszCmdName, pszStartLine, external, CMD_STRIP_QUOTES | CMD_CLOSE_BATCH );
871 }
872 else if ( _stricmp( ptr, CMD ) == 0 ) {
873 // check .CMD files for REXX or EXTPROC
874 rval = process_rexx( pszCmdName, pszStartLine, FALSE );
875 } else if ( _stricmp( ptr, BTM ) == 0 )
876 rval = parse_line( pszCmdName, pszStartLine, batch, CMD_STRIP_QUOTES | CMD_ADD_NULLS | CMD_CLOSE_BATCH | CMD_UCASE_CMD );
877
878 else {
879
880 // is it an executable extension?
881 if ( *(feptr = executable_ext( ptr )) != '\0' ) {
882
883 AddQuotes( pszCmdName );
884
885 // check for length & save remainder of command line
886 i = strlen( feptr ) + strlen( pszCmdName ) + 1;
887
888 if (( i + strlen( pszStartLine )) >= MAXLINESIZ )
889 return ( error( ERROR_4DOS_COMMAND_TOO_LONG, NULL ));
890
891 memmove( ( gszCmdline + i ), pszStartLine, (MAXLINESIZ - i) );
892 sprintf( gszCmdline, "%Fs %s%s", feptr, pszCmdName, gszCmdline+i );
893
894 // go back so we can support aliases, internal and
895 // external commands
896 rval = command( gszCmdline, 0 );
897
898 } else {
899 // OS/2 CMD.EXE allows you to execute program
900 // names like "program.pgm"
901 // NT will execute ASSOC'd file extensions
902 gnErrorLevel = rval = parse_line( pszCmdName, pszStartLine, external, (CMD_STRIP_QUOTES | CMD_CLOSE_BATCH) );
903 }
904 }
905
906 } else {
907 // check for UNKNOWN_CMD alias
908 if ( get_alias( UNKNOWN_COMMAND ) != 0L ) {
909
910 static int nUCLoop = 0;
911
912 if ( ++nUCLoop > 10 )
913 rval = error( ERROR_4DOS_UNKNOWN_CMD_LOOP, pszSave );
914
915 else {
916
917 if (( strlen( pszSave ) + 14 ) > MAXLINESIZ )
918 return ( error( ERROR_4DOS_COMMAND_TOO_LONG, pszSave ));
919 strcpy( gszCmdline, pszSave );
920 strins( gszCmdline, " " );
921 strins( gszCmdline, UNKNOWN_COMMAND );
922
923 // go back so we can support aliases, internal
924 // and external commands
925 rval = command( gszCmdline, 0 );
926 }
927
928 nUCLoop = 0;
929
930 } else {
931 // unknown command
932 gnErrorLevel = rval = error( ERROR_4DOS_UNKNOWN_COMMAND, pszCmdName );
933 }
934 }
935
936 return rval;
937}

940// process command grouping & get additional lines if necessary
941static int command_groups( char *pszStartLine, int remove_parens )
942{
943 int rval;
944 char *ptr;
945
946 // first get the command group; then remove leading & trailing parens
947 // (except when retrieving continuation lines)
948 for ( ; ; ) {
949
950 if (( ptr = scan( pszStartLine, ")", QUOTES_PARENS )) == BADQUOTES)
951 return ERROR_EXIT;
952
953 if ( *ptr == '\0' ) {
954
955 // go back & get some more input
956 // ("line" has to be empty to get here!)
957
958 // don't remove ( ) 's if a line continuation is found
959 remove_parens = 0;
960
961 strcat( pszStartLine, " " );
962 ptr = strend( pszStartLine );
963
964 if (( rval = ( CMDBUFSIZ - (UINT)(( ptr + 2) - gszCmdline))) <= 0 ) {
965 error( ERROR_4DOS_COMMAND_TOO_LONG, NULL );
966 return BATCH_RETURN;
967 }
968
969 if ( rval > ( MAXLINESIZ - 1 ))
970 rval = MAXLINESIZ - 1;
971
972 // if not a batch file, prompt for more input
973 if ( cv.bn < 0 ) {
974
975 printf( COMMAND_GROUP_MORE );
976 rval = getline( STDIN, ptr, rval, EDIT_COMMAND );
977
978 // add it to history list
979 addhist( ptr );
980
981 // history logging
982 if ( gpIniptr->HistLogOn )
983 (void)_log_entry( ptr, 1 );
984
985 } else {
986 rval = getline( bframe[ cv.bn ].bfd, ptr, rval, EDIT_COMMAND );
987 bframe[cv.bn].uBatchLine++;
988 }
989
990 if ( rval == 0 ) {
991 error( ERROR_4DOS_UNBALANCED_PARENS, pszStartLine );
992 return BATCH_RETURN;
993 }
994
995 // ignore blank and comment lines
996 if (( first_arg( ptr ) == NULL ) || ( _stricmp( first_arg( ptr ), "REM" ) == 0 )) {
997 ptr[-1] = '\0';
998 continue;
999 }
1000
1001 // skip leading whitespace
1002 strcpy( ptr, skipspace( ptr ) );
1003 if (( ptr[-2] != '(' ) && ( *ptr != ')' )) {
1004 strins( ptr, " " );
1005 *ptr = gpIniptr->CmdSep;
1006 }
1007
1008 } else {
1009
1010 // remove the first ( ) set
1011 if ( remove_parens ) {
1012 strcpy( ptr, ptr + 1 );
1013 strcpy( pszStartLine, pszStartLine + 1 );
1014 }
1015
1016 return 0;
1017 }
1018 }
1019}

1073// parse the command line into Argc, Argv format
1074int parse_line( char *pszCmdName, char *pszLine, int (*cmd)(int, char **), int pflag)
1075{
1076 static int Argc; // argument count
1077 static char *Argv[ARGMAX]; // argument pointers
1078 static char *pszTemp;
1079 static char cQuote;
1080
1081 // pflag bit arguments are:
1082 // CMD_STRIP_QUOTES - strip quoting
1083 // CMD_ADD_NULLS - add terminators to each argument
1084 // CMD_CLOSE_BATCH - close open batch file before executing command
1085
1086 // if it's an external or special case internal, close the batch file
1087 if ( pflag & CMD_CLOSE_BATCH )
1088 close_batch_file();
1089
1090 // set command / program name
1091 if ( pflag & CMD_UCASE_CMD )
1092 strupr( pszCmdName );
1093 Argv[0] = pszCmdName;
1094
1095 pszLine = skipspace( pszLine );
1096
1097 // loop through arguments
1098 for ( Argc = 1; (( Argc < ARGMAX ) && ( *pszLine )); Argc++ ) {
1099
1100 pszTemp = pszLine; // save start of argument
1101
1102 while ( isdelim( *pszLine ) == 0 ) {
1103
1104 if ((( gpIniptr->Expansion & EXPAND_NO_QUOTES) == 0 ) && ((*pszLine == SINGLE_QUOTE) || (*pszLine == DOUBLE_QUOTE))) {
1105
1106 cQuote = *pszLine;
1107
1108 // strip a single quote
1109 if (( *pszLine == SINGLE_QUOTE ) && ( pflag & CMD_STRIP_QUOTES ))
1110 strcpy( pszLine, pszLine + 1 );
1111 else
1112 pszLine++;
1113
1114 // arg ends after matching close quote
1115 while (( *pszLine ) && ( *pszLine != cQuote )) {
1116
1117 // collapse any "escaped" characters
1118 if ( pflag & CMD_STRIP_QUOTES ) {
1119 // CMD.EXE kludge to NOT collapse
1120 // escape chars inside double quotes
1121 if ( cQuote != DOUBLE_QUOTE )
1122 escape( pszLine );
1123 }
1124
1125 if ( *pszLine )
1126 pszLine++;
1127 }
1128
1129 // strip a single quote
1130 if (( *pszLine == SINGLE_QUOTE ) && ( pflag & CMD_STRIP_QUOTES ))
1131 strcpy( pszLine, pszLine + 1 );
1132 else if ( *pszLine )
1133 pszLine++;
1134
1135 } else {
1136
1137 // collapse any "escaped" characters
1138 if ( pflag & CMD_STRIP_QUOTES )
1139 escape( pszLine );
1140 if ( *pszLine )
1141 pszLine++;
1142 }
1143 }
1144
1145 // check flag for terminators
1146 if ((( pflag & CMD_ADD_NULLS ) != 0 ) && ( *pszLine != '\0' ))
1147 *pszLine++ = '\0';
1148
1149 // update the Argv pointer
1150 Argv[ Argc ] = pszTemp;
1151
1152 // skip delimiters for Argv[2] onwards
1153 while (( *pszLine ) && ( isdelim( *pszLine )))
1154 pszLine++;
1155 }
1156
1157 Argv[Argc] = NULL;
1158
1159 // test if only processing line, not calling func
1160 if ( cmd == NULL )
1161 return 0;
1162
1163 // update title bar (unless inside a batch file)
1164 if ((( cv.bn < 0 ) && ( cmd == external )) || ( cmd == batch ))
1165 update_task_list( Argv[0] );
1166
1167 // call the requested function (indirectly)
1168 return (*cmd)( Argc, Argv );
1169}

1172// search paths for file -- return pathname if found, NULL otherwise
1173char * searchpaths( char *filename, char *pszSearchPath, int fSearchCurrentDir )
1174{
1175 static char szFileName[MAXFILENAME + 32]; // expanded filename
1176 char *pOldExe, *pNewExe;
1177 int i, n, nPass = 0;
1178 PCH feptr, fenvptr = 0L, lpPathExt = 0L, lpExt;
1179 FILESEARCH dir;
1180 int fHPFS;
1181
1182 // build a legal filename ( right filename & extension size)
1183 insert_path( szFileName, fname_part( filename ), filename );
1184
1185 // check for existing extension
1186 pOldExe = ext_part( szFileName );
1187
1188 fHPFS = ifs_type( szFileName );
1189
1190 // check for relative path spec
1191 if ( strstr( szFileName, "..." ) != NULL ) {
1192
1193 if ( mkfname( szFileName, 0 ) == NULL )
1194 return NULL;
1195
1196 // no path searches if path already specified!
1197 } else if ( path_part( filename ) != NULL )
1198 nPass = -1;
1199
1200 else {
1201
1202 if ( pszSearchPath != NULL )
1203 fenvptr = pszSearchPath;
1204
1205 else if (( fenvptr = get_variable( PATH_VAR )) != 0L ) {
1206
1207 // if path has a ";.", DON'T search the current directory first!
1208 for ( feptr = fenvptr; (( feptr = strchr( feptr, ';' )) != 0L ) ; feptr++ ) {
1209 if (( feptr[1] == '.' ) && (( feptr[2] == ';' ) || ( feptr[2] == '\0' )))
1210 goto next_path;
1211 }
1212 }
1213
1214 // don't look in the current directory?
1215 if ( fSearchCurrentDir == FALSE )
1216 goto next_path;
1217 }
1218
1219 if (( pszSearchPath == NULL ) && ( gpIniptr->PathExt ))
1220 lpPathExt = get_variable( PATHEXT );
1221
1222 // search the PATH for the file
1223 for ( ; ; ) {
1224
1225 if ( pOldExe != NULL ) {
1226 if ( is_file( szFileName ))
1227 return szFileName;
1228 }
1229
1230 // if LFN/HPFS/NTFS, support odd names like "file1.comp.exe"
1231 if (( pOldExe == NULL ) || ( fHPFS )) {
1232
1233 // if no extension, try an executable one
1234 pNewExe = strend( szFileName );
1235
1236 // first check with "filename.*" for any possible match
1237 strcpy( pNewExe, WILD_EXT );
1238 i = 0;
1239
1240 if ( is_file( szFileName )) {
1241 lpExt = lpPathExt;
1242 for ( i = 0; ( nPass <= 0 ); i++ ) {
1243
1244 if ( lpPathExt ) {
1245
1246 if ( *lpExt == '\0' )
1247 break;
1248
1249 // get next PATHEXT argument
1250 sscanf( lpExt, "%*[;,=]%31[^;,=]%n", pNewExe, &n );
1251 lpExt += n;
1252
1253 } else {
1254 if ( executables[i] == NULL )
1255 break;
1256 strcpy( pNewExe, executables[i] );
1257 }
1258
1259 // look for the file
1260 if ( is_file( szFileName ))
1261 return szFileName;
1262 }
1263
1264 // check for user defined executable extensions
1265 feptr = glpEnvironment;
1266
1267 for ( n = 0; (( feptr != 0L ) && ( *feptr != '\0' )); feptr = next_env( feptr )) {
1268
1269 if ( *feptr == '.' ) {
1270
1271 // add extension & look for filename
1272 sscanf( feptr, "%13[^=]", pNewExe );
1273
1274 // search directory for possible wildcard match
1275 if ( find_file( FIND_FIRST, szFileName, 0x2107, &dir, szFileName ) != NULL )
1276 return szFileName;
1277 }
1278 }
1279 }
1280 }
1281
1282next_path:
1283 if ( fenvptr == 0L )
1284 return NULL;
1285
1286 // get next PATH argument, skipping delimiters
1287 sscanf( fenvptr, "%*[;,=]%260[^;,=]%n", szFileName, &n );
1288
1289 // check for end of path
1290 if ( szFileName[0] == '\0' ) {
1291 return NULL;
1292 }
1293
1294 fenvptr += n;
1295
1296 // make new directory search name
1297 mkdirname( szFileName, fname_part( filename ));
1298
1299 // make sure specified drive is ready
1300 if (( is_net_drive( szFileName ) == 0 ) && ( QueryDriveReady( gcdisk( szFileName )) == 0 ))
1301 goto next_path;
1302 }
1303}

1306// display the DOS / OS2 command line prompt
1307void show_prompt( void )
1308{
1309 extern char gaPushdQueue[];
1310
1311 int c;
1312 char *eptr;
1313 char *ptr;
1314 PCH feptr;
1315
1316 UCHAR pBuf[81] = "";
1317 ULONG cbMsgL;
1318 APIRET ulrc;
1319
1320 // odd kludge for UNC names
1321 if (( gnCurrentDisk = _getdrive()) < 0 )
1322 gnCurrentDisk = 0;
1323
1324 // get prompt from environment, or use default if none
1325 if (( feptr = get_variable( PROMPT_NAME )) != 0L ) {
1326
1327 // expand any environment variables in the PROMPT string
1328 strcpy( gszCmdline, feptr );
1329 (void)var_expand( gszCmdline, 1 );
1330 eptr = gszCmdline;
1331
1332 } else {
1333 eptr = ((( gnCurrentDisk == 0 ) || ( gnCurrentDisk > 2 )) ? "[$p]" : "[$n]");
1334 }
1335
1336 for ( ; ( *eptr != '\0' ); eptr++ ) {
1337
1338 if ((( c = *eptr ) == '$' ) && ( eptr[1] )) {
1339
1340 // special character follows a '$'
1341 switch ( _ctoupper( *(++eptr ) )) {
1342 case 'A': // ampersand
1343 c = '&';
1344 break;
1345
1346 case 'B': // vertical bar
1347 c = '|';
1348 break;
1349
1350 case 'C': // open paren
1351 c = '(';
1352 break;
1353
1354 case 'D': // current date
1355 if ( *eptr == 'd' ) {
1356 // "Sat 01-01-92"
1357 printf( "%.4s%s", gdate(0 ), gdate(1) );
1358 } else {
1359 // "Sat Jan 1, 1992"
1360 qputs( gdate( 0 ) );
1361 }
1362
1363 continue;
1364
1365 case 'E': // ESCAPE (for ANSI sequences)
1366 c = ESC;
1367 break;
1368
1369 case 'F': // close paren
1370 c = ')';
1371 break;
1372
1373 case 'G':
1374 c = '>';
1375 break;
1376
1377 case 'H': // destructive backspace
1378 qputs( "\010 \010" );
1379 continue;
1380
1381 case 'I': // display Program Selector help
1382 {
1383 int i, n;
1384 VIOMODEINFO vioMode;
1385
1386 vioMode.cb = sizeof( vioMode );
1387 (void)VioGetMode( &vioMode, 0 );
1388
1389 // set attribute (inverse white on mono screen,
1390 // white on blue for color screen)
1391 i = ( vioMode.fbType == 0 ) ? 0x70 : 0x1F;
1392
1393 ulrc = DosGetMessage(NULL, 0, pBuf, sizeof(pBuf), 1492, "OSO001.MSG", &cbMsgL);
1394 if (ulrc != NO_ERROR)
1395 printf(all_GetSystemErrorMessage(ulrc));
1396 (void)VioWrtCharStrAtt( pBuf, strlen( pBuf ), 0, 0, (PBYTE)&i, 0 );
1397
1398 // kludge if we're on line 0
1399 GetCurPos( &i, &n );
1400 if ( i == 0 )
1401 SetCurPos( 1, n );
1402 }
1403 // ignore $i in TCMD
1404 continue;
1405
1406 case 'L':
1407 c = '<';
1408 break;
1409
1410 case 'M': // current time w/o seconds
1411 ptr = gtime( *eptr == 'm' );
1412
1413 // remove seconds (but preserve "a" or "p")
1414 strcpy( ptr + 5, ptr + 8 );
1415 qputs( ptr );
1416 continue;
1417
1418 case 'N': // default drive letter
1419 if ( gnCurrentDisk == 0 )
1420 continue;
1421 c = gnCurrentDisk + 64;
1422 break;
1423
1424 case 'P': // current directory
1425 if (( ptr = gcdir( NULL, 1 )) != NULL ) {
1426
1427 if ( *eptr == 'P' ) {
1428 // HPFS & NTFS preserve case in names,
1429 // so just leave it alone
1430 if ( ifs_type( ptr ) != 0 ) {
1431 qputs( ptr );
1432 continue;
1433 }
1434 }
1435
1436 qputs((( *eptr == 'p' ) ? strlwr( ptr ) : strupr( ptr )));
1437 }
1438 continue;
1439
1440 case 'Q': // Equals sign
1441 c = '=';
1442 break;
1443
1444 case 'R': // errorlevel
1445 printf( FMT_INT, gnErrorLevel );
1446 continue;
1447
1448 case 'S': // print space
1449 c = ' ';
1450 break;
1451
1452 case 'T': // current time
1453 qputs( gtime( *eptr == 't' ) );
1454 continue;
1455
1456 case 'V': // OS version number
1457 printf( gszOsVersion );
1458 continue;
1459
1460 case 'X': // default dir for specified drive
1461 if (( ptr = gcdir( (( eptr[1] ) ? eptr + 1 : NULL ), 1 )) != NULL )
1462 qputs(( *eptr == 'x' ) ? ptr : strupr( ptr ));
1463
1464
