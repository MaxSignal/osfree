156/* This module contains all possible exception codes for ACPI_STATUS */
157
158/*
159 * Exception code classes
160 */
161#define AE_CODE_ENVIRONMENTAL 0x0000 /* General ACPICA environment */
162#define AE_CODE_PROGRAMMER 0x1000 /* External ACPICA interface caller */
163#define AE_CODE_ACPI_TABLES 0x2000 /* ACPI tables */
164#define AE_CODE_AML 0x3000 /* From executing AML code */
165#define AE_CODE_CONTROL 0x4000 /* Internal control codes */
166
167#define AE_CODE_MAX 0x4000
168#define AE_CODE_MASK 0xF000
169
170/*
171 * Macros to insert the exception code classes
172 */
173#define EXCEP_ENV(code) ((ACPI_STATUS) (code | AE_CODE_ENVIRONMENTAL))
174#define EXCEP_PGM(code) ((ACPI_STATUS) (code | AE_CODE_PROGRAMMER))
175#define EXCEP_TBL(code) ((ACPI_STATUS) (code | AE_CODE_ACPI_TABLES))
176#define EXCEP_AML(code) ((ACPI_STATUS) (code | AE_CODE_AML))
177#define EXCEP_CTL(code) ((ACPI_STATUS) (code | AE_CODE_CONTROL))
178
179/*
180 * Exception info table. The "Description" field is used only by the
181 * ACPICA help application (acpihelp).
182 */
183typedef struct acpi_exception_info
184{
185 char *Name;
186
187#if defined (ACPI_HELP_APP) || defined (ACPI_ASL_COMPILER)
188 char *Description;
189#endif
190} ACPI_EXCEPTION_INFO;
191
192#if defined (ACPI_HELP_APP) || defined (ACPI_ASL_COMPILER)
193#define EXCEP_TXT(Name,Description) {Name, Description}
194#else
195#define EXCEP_TXT(Name,Description) {Name}
196#endif
197
198
199/*
200 * Success is always zero, failure is non-zero
201 */
202#define ACPI_SUCCESS(a) (!(a))
203#define ACPI_FAILURE(a) (a)
204
205#define AE_OK (ACPI_STATUS) 0x0000
206
207#define ACPI_ENV_EXCEPTION(Status) (((Status) & AE_CODE_MASK) == AE_CODE_ENVIRONMENTAL)
208#define ACPI_AML_EXCEPTION(Status) (((Status) & AE_CODE_MASK) == AE_CODE_AML)
209#define ACPI_PROG_EXCEPTION(Status) (((Status) & AE_CODE_MASK) == AE_CODE_PROGRAMMER)
210#define ACPI_TABLE_EXCEPTION(Status) (((Status) & AE_CODE_MASK) == AE_CODE_ACPI_TABLES)
211#define ACPI_CNTL_EXCEPTION(Status) (((Status) & AE_CODE_MASK) == AE_CODE_CONTROL)
