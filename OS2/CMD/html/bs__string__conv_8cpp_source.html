55 *@@ FindEncodingID:

57 *@@added V0.9.18 (2002-03-08) [umoeller]
58 */
59
60static ENCID FindEncodingID(USHORT usCodepage,
61 BOOL *pfDouble) // out: TRUE if double-byte cp
62{
63 ENCBYTECOUNT bc;
64 ENCID id = encFindIdForCodepage(usCodepage, NULL, &bc);
65 if ( id != UNSUPPORTED
66 && ( (bc == SINGLE)
67 || (bc == DOUBLE)
68 )
69 )
70 {
71 if (pfDouble)
72 *pfDouble = (bc == DOUBLE);
73
74 return (id);
75 }
76
77 throw BSUnsupportedCPExcpt(usCodepage);
78}

88 * constructor. Creates the internal conversion

93
94BSUniCodec::BSUniCodec(unsigned short usCodepage) // in: codepage
95 : BSRoot(tBSUniCodec),
96 _usCodepage(usCodepage)
97{
98 ENCID id;
99
100 // WarpIN V1.0.18
101 if (usCodepage == CP_UTF8)
102 {
103 _pCodec = NULL;
104 return;
105 }
106
107 id = FindEncodingID(usCodepage, &_fDouble);
108 if (_fDouble)
109 {
110 APIRET arc;
111 COUNTRYCODE cc = { 0, usCodepage };
112 if (arc = DosQueryDBCSEnv(sizeof(_achDBCS),
113 &cc,
114 _achDBCS))
115 {
116 CHAR sz[200];
117 sprintf(sz,
118 "DosQueryDBCSEnv returned error %d for codepage %d",
119 arc,
120 usCodepage);
121 throw BSExcptBase(sz);
122 }
123 }
124
125 _pCodec = encCreateCodec(id);
126}

151
152static BOOL IsLeadByte(CHAR c, // in: character to test
153 PSZ pachDBCS) // in: DBCS array from DosQueryDBCSEnv
154{
155 while (*pachDBCS)
156 {
157 if ( (c >= *pachDBCS++)
158 && (c <= *pachDBCS++)
159 )
160 return TRUE;
161 }
162
163 return FALSE;

166/*
167 *@@ Codepage2Uni:
168 * converts the given string from codepage-specific
169 * to UTF-8. Used by BSString::assignUtf8 and
170 * others.
171 *
172 *@@changed WarpIN V1.0.18 (2008-09-24) [pr]: added codepage 1208 support @@fixes 1127
173 */
174
175void BSUniCodec::Codepage2Uni(BSUString &ustr, // out: target
176 const char *pcszCP, // in: cp string
177 unsigned long ulLength) // in: length of cp string
178{
179 PCONVERSION pTable = (PCONVERSION)_pCodec;
180
181 // WarpIN V1.0.18
182 if (QueryCodepage() == CP_UTF8)
183 {
184 ustr.assignUtf8(pcszCP);
185 return;
186 }
187
188 XSTRING xstrNew;
189 xstrInit(&xstrNew, ulLength + 1);
190
191 ULONG ul;
192 for (ul = 0;
193 ul < ulLength;
194 ++ul)
195 {
196 unsigned short c = pcszCP[ul];
197
198 if (_fDouble)
199 {
200 // we're using a double-byte codepage:
201 // check if this is a DBCS char
202 if (IsLeadByte(c, _achDBCS))
203 c = (c << 8) | pcszCP[++ul];
204 }
205
206 // convert this char to Unicode, using the current codepage
207 unsigned long ulUni = encChar2Uni(pTable, c);
208
209 if (ulUni == 0xFFFF)
210 // shouldn't happen
211 xstrcatc(&xstrNew, '?');
212 else if (ulUni < 0x80)
213 xstrcatc(&xstrNew, (CHAR)ulUni);
214 else if (ulUni < 0x800)
215 {
216 xstrcatc(&xstrNew, (CHAR)(0xC0 | ulUni>>6));
217 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni & 0x3F));
218 }
219 else if (ulUni < 0x10000)
220 {
221 xstrcatc(&xstrNew, (CHAR)(0xE0 | ulUni>>12));
222 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni>>6 & 0x3F));
223 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni & 0x3F));
224 }
225 /* else if (ulUni < 0x200000)
226 {
227 xstrcatc(&xstrNew, (CHAR)(0xF0 | ulUni>>18));
228 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni>>12 & 0x3F));
229 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni>>6 & 0x3F));
230 xstrcatc(&xstrNew, (CHAR)(0x80 | ulUni & 0x3F));
231 } */
232 else
233 {
234 CHAR sz[100];
235 sprintf(sz,
236 "Unsupported Unicode character %u at offset %u in string",
237 ulUni,
238 ul);
239 throw BSExcptBase(sz);
240 }
241 }
242
243 // copy back
244 ustr._take_from(xstrNew);
245}

247/*
248 *@@ Uni2Codepage:
249 * converts the given string from UTF-8 to
250 * codepage-specific. Used by BSUString::assignCP
251 * and others.
252 *
253 * Characters that are not supported with the
254 * given codepage are replaced by '?'.
255 *
256 *@@changed WarpIN V1.0.18 (2008-09-24) [pr]: added codepage 1208 support @@fixes 1127
257 */
258
259void BSUniCodec::Uni2Codepage(BSString &str,
260 const char *pcszUni,
261 unsigned long ulLength)
262{
263 // WarpIN V1.0.18
264 if (QueryCodepage() == CP_UTF8)
265 {
266 str.assign(pcszUni);
267 return;
268 }
269
270 PCONVERSION pTable = (PCONVERSION)_pCodec;
271
272 XSTRING xstrNew;
273 xstrInit(&xstrNew, ulLength + 1);
274
275 const char *pFirst = pcszUni;
276
277 while (*pcszUni)
278 {
279 unsigned long ulUni = encDecodeUTF8(&pcszUni);
280 unsigned short us = encUni2Char(pTable, ulUni);
281 if (us > 0xFF)
282 {
283 if (us == 0xFFFF)
284 us = '?';
285 else if (_fDouble)
286 {
287 // we're using a double-byte codepage:
288 // store lead byte first
289 xstrcatc(&xstrNew, (CHAR)(us >> 8));
290 }
291 else
292 {
293 // not double-byte codepage: then we can't handle > 0xFF
294 CHAR sz[100];
295 sprintf(sz,
296 "Unsupported Unicode character %u at offset %u in string",
297 ulUni,
298 pcszUni - pFirst);
299 throw BSExcptBase(sz);
300
301 }
302 }
303 xstrcatc(&xstrNew, (CHAR)us);
304 }
305
306 // copy back
307 str._take_from(xstrNew);
308}
