27/*
28 * Description: This module defines the disk structures used by LVM,
29 * including that of the Master Boot Record (MBR) and
30 * Extended Boot Records (EBR).
31 *
32 * Notes: LVM Drive Letter Assignment Tables (DLA_Tables) appear on the
33 * last sector of each track containing a valid MBR or EBR. Since
34 * partitions must be track aligned, any track containing an MBR or
35 * EBR will be almost all empty sectors. We will grab the last
36 * of these empty sectors for our DLT_Tables.
37 *
38 */

143/* The following structure defines the LVM Signature Sector. This is the last sector of every partition which is part of an LVM volume. It gives vital
144 information about the version of LVM used to create the LVM volume that it is a part of, as well as which LVM features (BBR, drive linking, etc.) are
145 active on the volume that this partition is a part of.
146 */
147typedef struct _LVM_Signature_Sector { /* LSS */
148 DoubleWord LVM_Signature1; /* The first part of the magic LVM signature. */
149 DoubleWord LVM_Signature2; /* The second part of the magic LVM signature. */
150 DoubleWord Signature_Sector_CRC; /* 32 bit CRC for this sector. Calculated using 0 for this field. */
151 DoubleWord Partition_Serial_Number; /* The LVM assigned serial number for this partition. */
152 LBA Partition_Start; /* LBA of the first sector of this partition. */
153 LBA Partition_End; /* LBA of the last sector of this partition. */
154 DoubleWord Partition_Sector_Count; /* The number of sectors in this partition. */
155 DoubleWord LVM_Reserved_Sector_Count; /* The number of sectors reserved for use by LVM. */
156 DoubleWord Partition_Size_To_Report_To_User; /* The size of the partition as the user sees it - i.e. (the actual size of the partition - LVM reserved sectors) rounded to a track boundary. */
157 DoubleWord Boot_Disk_Serial_Number; /* The serial number of the boot disk for the system. If the system contains Boot Manager, then this is the serial number of the disk containing the active copy of Boot Manager. */
158 DoubleWord Volume_Serial_Number; /* The serial number of the volume that this partition belongs to. */
159 CARDINAL32 Fake_EBR_Location; /* The location, on disk, of a Fake EBR, if one has been allocated. */
160 Word LVM_Major_Version_Number; /* Major version number of the LVM that created this partition. */
161 Word LVM_Minor_Version_Number; /* Minor version number of the LVM that created this partition. */
162 char Partition_Name[PARTITION_NAME_SIZE]; /* User defined partition name. */
163 char Volume_Name[VOLUME_NAME_SIZE]; /* The name of the volume that this partition belongs to. */
164 LVM_Feature_Data LVM_Feature_Array[MAX_FEATURES_PER_VOLUME]; /* The feature array. This indicates which LVM features, if any, are active on this volume
165 and what order they should be applied in. */
166 char Drive_Letter; /* The drive letter assigned to the volume that this partition is part of. */
167 BOOLEAN Fake_EBR_Allocated; /* If TRUE, then a fake EBR has been allocated. */
168 char Comment[COMMENT_SIZE]; /* User comment. */
169 char Disk_Name[DISK_NAME_SIZE]; /* Added to allow BBR to report the name of a disk when bad sectors are encountered on that disk. */
170 CARDINAL32 Sequence_Number; /* This indicates the order that partitions within a volume are used. This number is 1 based. A 0 here indicates that the volume was made by LVM Ver. 1. */
171 CARDINAL32 Next_Aggregate_Number; /* Used during volume creation and expansion when creating unique names for aggregates. */
172 /* The remainder of the sector is reserved for future use and should be all zero or else the CRC will not come out correctly. */

177typedef struct _Partition_Record { /* PR */
178 Byte Boot_Indicator; /* 80h = active partition. */
179 Byte Starting_Head;
180 Byte Starting_Sector; /* Bits 0-5 are the sector. Bits 6 and 7 are the high order bits of the starting cylinder. */

183 Byte Ending_Head;
184 Byte Ending_Sector;
185 Byte Ending_Cylinder;
