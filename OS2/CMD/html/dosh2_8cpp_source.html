16 *@@ doshIsWarp4:
17 * checks the OS/2 system version number.
18 *
19 * Returns:
20 *
21 * -- 0 (FALSE): OS/2 2.x or Warp 3 is running.
22 *
23 * -- 1: Warp 4.0 is running.
24 *
25 * -- 2: Warp 4.5 kernel is running on Warp 4.0 (Warp 4 FP 13+).
26 *
27 * -- 3: Warp 4.5 is running (WSeB or eCS or ACP/MCP), or even something newer.
28 *
29 *@@changed V0.9.2 (2000-03-05) [umoeller]: reported TRUE on Warp 3 also; fixed
30 *@@changed V0.9.6 (2000-10-16) [umoeller]: patched for speed
31 *@@changed V0.9.9 (2001-04-04) [umoeller]: now returning 2 for Warp 4.5 and above
32 *@@changed V1.0.5 (2006-05-29) [pr]: now returning 3 for Warp 4.5 and above and 2 for
33 * Warp 4.0 FP13+; moved here from dosh.c
34 */
35
36ULONG doshIsWarp4(VOID)
37{
38 static BOOL s_fQueried = FALSE;
39 static ULONG s_ulrc = 0;
40
41 if (!s_fQueried)
42 {
43 // first call:
44 ULONG aulBuf[3];
45
46 DosQuerySysInfo(QSV_VERSION_MAJOR, // 11
47 QSV_VERSION_MINOR, // 12
48 &aulBuf, sizeof(aulBuf));
49 // Warp 3 is reported as 20.30
50 // Warp 4 is reported as 20.40
51 // Aurora is reported as 20.45 (regardless of convenience packs)
52
53 if (aulBuf[0] > 20) // major > 20; not the case with Warp 3, 4, 5
54 s_ulrc = 3;
55 else
56 if (aulBuf[0] == 20) // major == 20
57 if (aulBuf[1] >= 45) // minor >= 45 Warp 4 FP13 or later
58 s_ulrc = 2;
59 else
60 if (aulBuf[1] == 40) // minor == 40 Warp 4 pre-FP13
61 s_ulrc = 1;
62
63 // Now check SYSLEVEL.OS2 to detect between Warp 4 and MCP
64 if (s_ulrc == 2)
65 {
66 CHAR szName[CCHMAXPATH] = "?:\\OS2\\INSTALL\\SYSLEVEL.OS2";
67 ULONG cbFile;
68 PXFILE pFile;
69
70 szName[0] = doshQueryBootDrive();
71 if (!doshOpen(szName,
72 XOPEN_READ_EXISTING | XOPEN_BINARY,
73 &cbFile,
74 &pFile))
75 {
76 CHAR szVersion[2];
77 ULONG ulSize;
78
79 if ( !lvlQueryLevelFileData(pFile->hf,
80 QLD_MINORVERSION,
81 szVersion,

83 &ulSize)
84 && (szVersion[0] >= '5')) // minor >= 5 is MCP
85 s_ulrc = 3;

88 }
89 }
90
91 s_fQueried = TRUE;
92 }
