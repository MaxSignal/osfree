30ULONG
31PrintPackage(
32 ULONG Count,
33 ACPI_OBJECT *oObj)
34{
35 ULONG j,i,Cnt;
36 ACPI_OBJECT *Obj;
37 char DevName[5];
38 ACPI_BUFFER Buffer;
39
40 for (Cnt=0,j=0; j < Count; j++,oObj++,Cnt++)
41 {
42 switch (oObj->Type)
43 {
44 case ACPI_TYPE_INTEGER:
45 printf(" I:[%d] value: 0x%x\n",j,oObj->Integer.Value);
46 fflush(stdout);
47 break;
48 case ACPI_TYPE_STRING:
49 printf(" S:[%d] value: 0x%x %s\n",j,oObj->String.Length,oObj->String.Pointer);
50 break;
51 case ACPI_TYPE_PACKAGE:
52 Obj=(ACPI_OBJECT *)oObj->Package.Elements;
53 printf("P:[%d] Package elements: 0x%x addr:%x\n",j,oObj->Package.Count,oObj->Package.Elements);
54 fflush(stdout);
55 Cnt += PrintPackage((ULONG)oObj->Package.Count,Obj);
56 break;
57 default:
58 printf(" U:[%d] Unknown 0x%x Fill:%x Handle:%x\n",j,oObj->Type,oObj->Reference.ActualType,oObj->Reference.Handle);
59 if (oObj->Reference.Handle)
60 {
61 Buffer.Length = 5;
62 Buffer.Pointer = DevName;
63 AcpiTkGetName(oObj->Reference.Handle, ACPI_SINGLE_NAME, &Buffer);
64 printf(" Device [");
65 for (i=0; i < 4; i++)
66 printf("%c",DevName[i]);
67 printf("]\n");
68 }
69 fflush(stdout);
70 break;
71 }
72 }
73
74 return Cnt;
75
76}
