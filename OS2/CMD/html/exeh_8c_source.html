82/*
83 *@@ exehOpen:
84 * this opens the specified executable file
85 * (which can be an .EXE, .COM, .DLL, or
86 * driver file) for use with the other
87 * exeh* functions.
88 *
89 * Basically this does a plain DosOpen on the
90 * executable file and reads in the various
91 * executable headers manually. Since DosLoadModule
92 * et al is never used, the OS/2 executable loader
93 * is completely circumvented. (Side note:
94 * DosLoadModule cannot be used on EXE files in
95 * the first place.)
96 *
97 * To be more precise, this uses doshOpen internally
98 * and can thus profit from the caching that is
99 * implemented there (V0.9.16).
100 *
101 * If no error occurs, NO_ERROR is returned
102 * and a pointer to a new EXECUTABLE structure
103 * is stored in *ppExec. Consider this pointer a
104 * handle and pass it to exehClose to clean up.
105 *
106 * If NO_ERROR is returned, all the fields through
107 * ulOS are set in EXECUTABLE. The psz* fields
108 * which follow afterwards require an additional
109 * call to exehQueryBldLevel.
110 *
111 * NOTE: If NO_ERROR is returned, the executable
112 * file is kept open by this function. It will
113 * only be closed when you call exehClose.
114 *
115 * If errors occur, this function returns the
116 * following error codes:
117 *
118 * -- ERROR_NOT_ENOUGH_MEMORY: malloc() failed.
119 *
120 * -- ERROR_INVALID_EXE_SIGNATURE (191): header is
121 * neither plain DOS, nor NE, nor LX, nor PE.
122 * The given file probably isn't even an
123 * executable.
124 *
125 * -- ERROR_BAD_EXE_FORMAT (193): header was
126 * recognized, but the header data was
127 * not understood. Also this might be
128 * returned for .COM files which are
129 * not recognized.
130 *
131 * -- ERROR_INVALID_PARAMETER: ppExec is NULL.
132 *
133 * plus those of doshOpen and doshReadAt.
134 *
135 * The following executable types are supported
136 * (see EXECUTABLE for details):
137 *
138 * -- Plain DOS 3.x executable without new header.
139 *
140 * -- New Executable (NE), used by Win16 and
141 * 16-bit OS/2 and still many of today's drivers.
142 *
143 * -- Linear Executable (LX), OS/2 2.x and above.
144 *
145 * -- Portable Executable (PE), used by Win32.
146 *
147 * -- For files with the .COM, .BAT, or .CMD
148 * extensions, this will _not_ open the
149 * executable file, but set flags in EXECUTABLE
150 * only. Most importantly, EXECUTABLE.pDosExeHeader
151 * will be NULL.
152 *
153 * V0.9.12 adds support for NOSTUB executables,
154 * which are new-style executables (NE or LX)
155 * without a leading DOS header. JFS.IFS uses that
156 * format, for example. The executable then starts
157 * directly with the NE or LX header. I am not sure
158 * whether PE supports such beasts as well... if
159 * so, it should be supported too.
160 *
161 * Note that not all of the other exeh* functions
162 * support all of the executable types. See the
163 * respective function descriptions for remarks.
164 *
165 * @@todo:
166 *
167 * win95 \WINDOWS\extract.exe is NE with a non-standard format
168 * win16 \WINDOWS\EXPAND.EXE
169 * win16 \WINDOWS\MSD.EXE"
170 *
171 *@@added V0.9.0 [umoeller]
172 *@@changed V0.9.1 (2000-02-13) [umoeller]: fixed 32-bits flag
173 *@@changed V0.9.7 (2000-12-20) [lafaix]: fixed ulNewHeaderOfs
174 *@@changed V0.9.10 (2001-04-08) [lafaix]: added PE support
175 *@@changed V0.9.10 (2001-04-08) [umoeller]: now setting ppExec only if NO_ERROR is returned
176 *@@changed V0.9.12 (2001-05-03) [umoeller]: added support for NOSTUB newstyle executables
177 *@@changed V0.9.16 (2001-12-08) [umoeller]: now using OPEN_SHARE_DENYWRITE
178 *@@changed V0.9.16 (2001-12-08) [umoeller]: fLibrary was never set, works for LX, NE, and PE now
179 *@@changed V0.9.16 (2001-12-08) [umoeller]: speed optimizations, changed some return codes
180 *@@changed V0.9.16 (2002-01-04) [umoeller]: added fixes for COM, BAT, CMD extensions
181 *@@changed V1.0.0 (2002-08-18) [umoeller]: this was completely broken for files without extensions (os2krnl)
182 */

184APIRET exehOpen(const char* pcszExecutable,
185 PEXECUTABLE* ppExec)
186{
187 APIRET arc = NO_ERROR;
188
189 PEXECUTABLE pExec = NULL;
190
191 PXFILE pFile = NULL;
192 ULONG cbFile = 0;
193 PCSZ pExt;
194 BOOL fOpenFile = FALSE;
195 BOOL fLoadNewHeader = FALSE;
196 ULONG ulNewHeaderOfs = 0; // V0.9.12 (2001-05-03) [umoeller]
197
198 if (!ppExec)
199 return ERROR_INVALID_PARAMETER;
200
201 if (!(pExec = (PEXECUTABLE)malloc(sizeof(EXECUTABLE))))
202 return ERROR_NOT_ENOUGH_MEMORY;
203
204 memset(pExec, 0, sizeof(EXECUTABLE));
205
206 // check some of the default extensions
207 // V0.9.16 (2002-01-04) [umoeller]
208 if (!(pExt = doshGetExtension(pcszExecutable)))
209 {
210 // has no extension: then open file!
211 // fixed V1.0.0 (2002-08-18) [umoeller]
212 fOpenFile = TRUE;
213 }
214 else
215 {
216 // has extension:
217 if (!stricmp(pExt, "COM"))
218 {
219 // I am not willing to find out more about the
220 // .COM executable format, so for this one case,
221 // let OS/2 determine what we have here
222 // (otherwise we do _not_ use DosQueryAppType
223 // because it's quite an expensive call)
224 ULONG ulDosAppType = 0;
225 if (!(arc = DosQueryAppType((PSZ)pcszExecutable, &ulDosAppType)))
226 {
227 if (ulDosAppType & FAPPTYP_DOS) // 0x20
228 pExec->ulOS = EXEOS_DOS3;
229 else
230 {
231 ULONG fl = ulDosAppType & FAPPTYP_WINDOWAPI; // 0x03
232 if ( (fl == FAPPTYP_WINDOWCOMPAT) // 0x02)
233 || (fl == FAPPTYP_NOTWINDOWCOMPAT) // 0x01)
234 )
235 pExec->ulOS = EXEOS_OS2;
236 else
237 arc = ERROR_BAD_EXE_FORMAT;
238 }
239
240 pExec->ulExeFormat = EXEFORMAT_COM;
241 }
242 }
243 else if (!stricmp(pExt, "BAT"))
244 {
245 pExec->ulOS = EXEOS_DOS3;
246 pExec->ulExeFormat = EXEFORMAT_TEXT_BATCH;
247 }
248 else if (!stricmp(pExt, "CMD"))
249 {
250 pExec->ulOS = EXEOS_OS2;
251 pExec->ulExeFormat = EXEFORMAT_TEXT_CMD;
252 }
253 else
254 fOpenFile = TRUE;
255 }
256
257 if ( (fOpenFile) // none of the above
258 && (!(arc = doshOpen((PSZ)pcszExecutable,
259 XOPEN_READ_EXISTING,
260 &cbFile,
261 &pFile)))
262 )
263 {
264 // file opened successfully:
265 pExec->pFile = pFile;
266 pExec->cbDosExeHeader = sizeof(DOSEXEHEADER);
267
268 // read old DOS EXE header
269 if ((arc = doshReadAt(pFile,
270 0,
271 &pExec->cbDosExeHeader, // in/out
272 (PBYTE)&pExec->DosExeHeader,
273 DRFL_FAILIFLESS)))
274 pExec->cbDosExeHeader = 0;
275 else
276 {
277 // now check if we really have a DOS header
278 if (pExec->DosExeHeader.usDosExeID != 0x5a4d)
279 {
280 // arc = ERROR_INVALID_EXE_SIGNATURE;
281
282 // V0.9.12 (2001-05-03) [umoeller]
283 // try loading new header directly; there are
284 // drivers which were built with NOSTUB, and
285 // the exe image starts out with the NE or LX
286 // image directly (try JFS.IFS)
287 fLoadNewHeader = TRUE;
288 // ulNewHeaderOfs is 0 now
289
290 // remove the DOS header info, since we have none
291 // V0.9.12 (2001-05-03) [umoeller]
292 // FREE(pExec->pDosExeHeader);
293 pExec->cbDosExeHeader = 0;
294 }
295 else
296 {
297 // V1.0.3 (2004-10-24) [pr]: Some non-DOS EXEs have a relocation table
298 // offset which is 0 - these were previously identified as DOS EXEs.
299 // we have a DOS header:
300 if ( ( (pExec->DosExeHeader.usRelocTableOfs == 0)
301 || (pExec->DosExeHeader.usRelocTableOfs >= sizeof(DOSEXEHEADER))
302 )
303 && (pExec->DosExeHeader.ulNewHeaderOfs != 0)
304 )
305 {
306 // we have a new header offset:
307 fLoadNewHeader = TRUE;
308 ulNewHeaderOfs = pExec->DosExeHeader.ulNewHeaderOfs;
309 }
310 else
311 {
312 // else DOS:
313 pExec->ulOS = EXEOS_DOS3;
314 pExec->ulExeFormat = EXEFORMAT_OLDDOS;
315 }
316 }
317 }
318 }
319
320 if (fLoadNewHeader)
321 {
322 // either LX or PE or NE:
323 // read in new header...
324 // ulNewHeaderOfs is now either 0 (if no DOS header
325 // was found) or pDosExeHeader->ulNewHeaderOfs
326 // V0.9.12 (2001-05-03) [umoeller]
327
328 // read in the first two bytes to find out
329 // what extended header type we have; note,
330 // PE uses four bytes here
331 CHAR achNewHeaderType[4] = "???";
332 ULONG cbRead = 4;
333
334 if (!(arc = doshReadAt(pFile,
335 ulNewHeaderOfs,
336 &cbRead,
337 achNewHeaderType,
338 DRFL_FAILIFLESS)))
339 {
340 PBYTE pbCheckOS = NULL;
341
342 if (!memcmp(achNewHeaderType, "NE", 2))
343 {
344 // New Executable:
345 pExec->ulExeFormat = EXEFORMAT_NE;
346 cbRead = sizeof(NEHEADER);
347
348 // go read in the complete header then
349 // (doshReadAt has this in the cache)
350 if (!(pExec->pNEHeader = (PNEHEADER)malloc(cbRead)))
351 arc = ERROR_NOT_ENOUGH_MEMORY;
352 else if (!(arc = doshReadAt(pFile,
353 ulNewHeaderOfs,
354 &cbRead,
355 (PBYTE)pExec->pNEHeader,
356 0)))
357 {
358 if (cbRead < sizeof(NEHEADER))
359 arc = ERROR_BAD_EXE_FORMAT;
360 else
361 {
362 pExec->cbNEHeader = cbRead;
363 pbCheckOS = &pExec->pNEHeader->bTargetOS;
364 // set library flag V0.9.16 (2001-12-08) [umoeller]
365 if (pExec->pNEHeader->usFlags & 0x8000)
366 // library:
367 pExec->fLibrary = TRUE;
368 }
369 }
370 }
371 else if ( (!memcmp(achNewHeaderType, "LX", 2))
372 || (!memcmp(achNewHeaderType, "LE", 2))
373 // this is used by SMARTDRV.EXE
374 )
375 {
376 // OS/2 Linear Executable:
377 pExec->ulExeFormat = EXEFORMAT_LX;
378 cbRead = sizeof(LXHEADER);
379
380 // go read in the complete header then
381 // (doshReadAt has this in the cache)
382 if (!(pExec->pLXHeader = (PLXHEADER)malloc(cbRead)))
383 arc = ERROR_NOT_ENOUGH_MEMORY;
384 else if (!(arc = doshReadAt(pFile,
385 ulNewHeaderOfs,
386 &cbRead,
387 (PBYTE)pExec->pLXHeader,
388 0)))
389 {
390 if (cbRead < sizeof(LXHEADER))
391 arc = ERROR_BAD_EXE_FORMAT;
392 else
393 {
394 pExec->cbLXHeader = cbRead;
395 pbCheckOS = (PBYTE)(&pExec->pLXHeader->usTargetOS);
396 // set library flag V0.9.16 (2001-12-08) [umoeller]
397 if (pExec->pLXHeader->ulFlags & 0x8000)
398 // library:
399 pExec->fLibrary = TRUE;
400 }
401 }
402 }
403 else if (!memcmp(achNewHeaderType, "PE\0\0", 4))
404 {
405 pExec->ulExeFormat = EXEFORMAT_PE;
406
407 // PE has a standard header of 24 bytes
408 // plus an extended header, so check
409 // what we've got
410 if (!(pExec->pPEHeader = (PPEHEADER)malloc(sizeof(PEHEADER))))
411 arc = ERROR_NOT_ENOUGH_MEMORY;
412 else
413 {
414 ULONG ulOfs = ulNewHeaderOfs + 4;
415 PPEHEADER pPEHeader = pExec->pPEHeader;
416
417 // null the entire header
418 memset(pExec->pPEHeader,
419 0,
420 sizeof(PEHEADER));
421
422 // copy sig
423 pPEHeader->ulSignature = *((PULONG)&achNewHeaderType);
424
425 // read standard header
426 cbRead = sizeof(IMAGE_FILE_HEADER);
427 if (!(arc = doshReadAt(pFile,
428 ulOfs,
429 &cbRead,
430 (PBYTE)&pPEHeader->FileHeader,
431 0)))
432 {
433 if (cbRead < sizeof(IMAGE_FILE_HEADER))
434 // only if we don't even have the
435 // standard header, return an error
436 arc = ERROR_BAD_EXE_FORMAT;
437 else
438 {
439 pExec->f32Bits = TRUE;
440 pExec->cbPEHeader = 4 + sizeof(PEHEADER); // for now
441
442 if (pPEHeader->FileHeader.fsCharacteristics & IMAGE_FILE_DLL)
443 pExec->fLibrary = TRUE;
444
445 // try extended header
446 ulOfs += sizeof(IMAGE_FILE_HEADER);
447 if ( (cbRead = pPEHeader->FileHeader.usSizeOfOptionalHeader)
448 && (cbRead <= sizeof(IMAGE_OPTIONAL_HEADER))
449 )
450 {
451 if (!(arc = doshReadAt(pFile,
452 ulOfs,
453 &cbRead,
454 (PBYTE)&pPEHeader->OptionalHeader,
455 0)))
456 {
457 if (cbRead != sizeof(IMAGE_OPTIONAL_HEADER))
458 arc = ERROR_BAD_EXE_FORMAT;
459 else switch (pPEHeader->OptionalHeader.usSubsystem)
460 {
461 // case IMAGE_SUBSYSTEM_UNKNOWN: // 0
462 // case IMAGE_SUBSYSTEM_NATIVE: // 1
463 // case IMAGE_SUBSYSTEM_OS2_CUI: // 5
464 // case IMAGE_SUBSYSTEM_POSIX_CUI: // 7
465 // for these we shouldn't set win32
466
467 case IMAGE_SUBSYSTEM_WINDOWS_GUI: // 2 // Windows GUI subsystem
468 pExec->ulOS = EXEOS_WIN32_GUI;
469 break;
470
471 case IMAGE_SUBSYSTEM_WINDOWS_CUI: // 3 // Windows character subsystem
472 pExec->ulOS = EXEOS_WIN32_CLI;
473 break;
474 }
475
476 pExec->cbPEHeader = sizeof(PEHEADER);
477 }
478 }
479 } // end else if (cbRead < sizeof(IMAGE_FILE_HEADER))
480 } // end if (!(arc = doshReadAt(pFile,
481 } // end else if (!(pExec->pPEHeader = (PPEHEADER)malloc(sizeof(PEHEADER))))
482 } // end else if (!memcmp(achNewHeaderType, "PE\0\0", 4))
483 else
484 // strange type:
485 arc = ERROR_INVALID_EXE_SIGNATURE;
486
487 if ((!arc) && (pbCheckOS))
488 {
489 // BYTE to check for operating system
490 // (NE and LX):
491 switch (*pbCheckOS)
492 {
493 case NEOS_OS2:
494 pExec->ulOS = EXEOS_OS2;
495 if (pExec->ulExeFormat == EXEFORMAT_LX)
496 pExec->f32Bits = TRUE;
497 break;
498
499 case NEOS_WIN16:
500 pExec->ulOS = EXEOS_WIN16;
501 break;
502
503 case NEOS_DOS4:
504 pExec->ulOS = EXEOS_DOS4;
505 break;
506
507 case NEOS_WIN386:
508 pExec->ulOS = EXEOS_WIN386;
509 pExec->f32Bits = TRUE;
510 break;
511 }
512 }
513 } // end if (!(arc = doshReadAt(hFile,
514 } // end if (fLoadNewHeader)
515
516 if (arc)
517 // error: clean up
518 exehClose(&pExec);
519 else
520 *ppExec = pExec;
521
522 return arc;
523}

525/*
526 *@@ ParseBldLevel:
527 * called from exehQueryBldLevel to parse the BLDLEVEL string.
528 *
529 * On entry, caller has copied the string into pExec->pszDescription.
530 * The string is null-terminated.
531 *
532 * The BLDLEVEL string comes in at least two basic flavors.
533 *
534 * -- The standard format is:
535 *
536 + @#VENDOR:VERSION#@DESCRIPTION
537 *
538 * DESCRIPTION can have leading spaces, but
539 * need to have them.
540 *
541 * -- However, there is an extended version in that the DESCRIPTION field
542 * is split up even more.
543 *
544 * I have seen two subformats for this.
545 *
546 * -- DANIS506.ADD and some IBM programs have a marker that seems
547 * to be that the description starts out with "##1##".
548 *
549 + ##1## DATETIME BUILDMACHINE:ASD:LANG:CTRY:REVISION:UNKNOWN:FIXPAK@@DESCRIPTION
550 *
551 * The problem is that the DATETIME field comes
552 * in several flavors. IBM uses things like
553 *
554 + "Thu Nov 30 15:30:37 2000 BWBLD228"
555 *
556 * while DANIS506.ADD has
557 *
558 + "15.12.2000 18:22:57 Nachtigall"
559 *
560 * Looks like the date/time string is standardized to have 24 characters then.
561 *
562 * -- IBM TCP/IP executables (try INETD.EXE) have something yet different on.
563 * We now try to parse that format as well, even though bldlevel.exe can't
564 * handle it. (Isn't that utility from IBM too?)
565 *
566 * Here's what I get for inetd.exe:
567 *
568 + ##built 09:16:27 Mon Sep 17 2001 -- On AURORA43;0.1@@ TCP/IP for OS/2: INETD
569 *
570 *@@added V0.9.12 (2001-05-18) [umoeller]
571 *@@changed V0.9.12 (2001-05-19) [umoeller]: added extended BLDLEVEL support
572 *@@changed V1.0.0 (2002-08-18) [umoeller]: added support for IBM TCP/IP format
573 *@@changed V1.0.0 (2002-08-18) [umoeller]: fixed DANIS506 format when an extended field had only one character
574 */
575
576STATIC VOID ParseBldLevel(PEXECUTABLE pExec)
577{
578 PCSZ pStartOfVendor,
579 pStartOfInfo,
580 pEndOfVendor;
581
582 // @#VENDOR:VERSION#@ DESCRIPTION
583 if ( (pStartOfVendor = strstr(pExec->pszDescription, "@#"))
584 && (pStartOfInfo = strstr(pStartOfVendor + 2, "#@"))
585 && (pEndOfVendor = strchr(pStartOfVendor + 2, ':'))
586 )
587 {
588 pExec->pszVendor = strhSubstr(pStartOfVendor + 2,
589 pEndOfVendor);
590 pExec->pszVersion = strhSubstr(pEndOfVendor + 1,
591 pStartOfInfo);
592 // skip "@#" in DESCRIPTION string
593 pStartOfInfo += 2;
594
595 // now check if we have extended DESCRIPTION V0.9.12 (2001-05-19) [umoeller]
596 if (strlen(pStartOfInfo) > 6)
597 {
598 if (!memcmp(pStartOfInfo, "##1##", 5))
599 {
600 // DANIS506.ADD format:
601 // "##1## 2.7.2002 19:32:34 Nachtigall::::6::@@..."
602
603 // parse that beast
604 PCSZ p = pStartOfInfo + 5;
605
606 // get build date/time
607 if (strlen(p) > 24)
608 {
609 // skip leading and trailing spaces
610 // V1.0.0 (2002-08-18) [umoeller]
611 PCSZ pStartOfDT = p,
612 pEndOfDT = p + 24;
613 // date/time seems to be fixed 24 chars in length
614
615 while (*pStartOfDT == ' ')
616 ++pStartOfDT;
617
618 while ( (*pEndOfDT == ' ')
619 && (pEndOfDT > pStartOfDT)
620 )
621 --pEndOfDT;
622
623 pExec->pszBuildDateTime = strhSubstr(pStartOfDT, pEndOfDT + 1);
624
625 /* memcpy(pExec->pszBuildDateTime,
626 p,
627 24);
628 pExec->pszBuildDateTime[24] = '\0';
629 */
630
631 // date/time seems to be fixed 24 chars in length
632 p += 24;
633
634 // now we're at the colon-separated
635 // strings, first of which is the build machine;
636 // skip leading spaces
637 while (*p == ' ')
638 p++;
639
640 if (*p)
641 {
642 ULONG ul;
643 char **papsz[7];
644 papsz[0]=&pExec->pszBuildMachine;
645 papsz[1]=&pExec->pszASD;
646 papsz[2]=&pExec->pszLanguage;
647 papsz[3]=&pExec->pszCountry;
648 papsz[4]=&pExec->pszRevision;
649 papsz[5]=&pExec->pszUnknown;
650 papsz[6]=&pExec->pszFixpak;
651
652
653 for (ul = 0;
654 ul < sizeof(papsz) / sizeof(papsz[0]);
655 ul++)
656 {
657 BOOL fStop = FALSE;
658 PCSZ pNextColon = strchr(p, ':'),
659 pDoubleAt = strstr(p, "@@");
660 if (!pNextColon)
661 {
662 // last item:
663 if (pDoubleAt)
664 pNextColon = pDoubleAt;
665 else
666 pNextColon = p + strlen(p);
667
668 fStop = TRUE;
669 }
670
671 if ( (fStop)
672 || ( (pNextColon)
673 && ( (!pDoubleAt)
674 || (pNextColon < pDoubleAt)
675 )
676 )
677 )
678 {
679 // if (pNextColon > p + 1)
680 // fixed V1.0.0 (2002-08-18) [umoeller]
681 // this failed on fields like "revision"
682 // which only had one character
683 if (pNextColon > p)
684 *(papsz[ul]) = strhSubstr(p, pNextColon);
685 }
686 else
687 break;
688
689 if (fStop)
690 break;
691
692 p = pNextColon + 1;
693 }
694 }
695 }
696
697 if (pStartOfInfo = strstr(p,
698 "@@"))
699 pStartOfInfo += 2;
700 } // end if (!memcmp(pStartOfInfo, "##1##", 5))
701 else if (!memcmp(pStartOfInfo, "##built", 7))
702 {
703 // IBM TCP/IP format:
704 // V1.0.0 (2002-08-18) [umoeller]
705
706 // ##built 09:16:27 Mon Sep 17 2001 -- On AURORA43;0.1@@ TCP/IP for OS/2: INETD
707
708 PCSZ p = pStartOfInfo + 7,
709 p2,
710 p3;
711
712 if (p3 = strchr(p, ';'))
713 {
714 while (*p == ' ')
715 ++p;
716
717 // ##built 09:16:27 Mon Sep 17 2001 -- On AURORA43;0.1@@ TCP/IP for OS/2: INETD
