46// check for SELECT options
47static int InitSelect( SELECT_ARGS *sel, char *start_line, int *mode )
48{
49 int i;
50 char *arg;
51
52 // clear everything to 0
53 memset( (char *)sel, '\0', sizeof(SELECT_ARGS) );
54
55 // reset the DIR flags
56 init_dir();
57
58 // initialize date/time/size ranges
59 if ( GetRange( start_line, &(sel->aSelRanges), 1 ) != 0 )
60 return ERROR_EXIT;
61
62 for ( i = 0; (( arg = ntharg( start_line, i )) != NULL ) && ( *arg == gpIniptr->SwChr ); i++ ) {
63
64 // point past switch character
65 for (arg++; (*arg != '\0'); ) {
66
67 switch ( _ctoupper( *arg++ )) {
68 case 'A':
69 // retrieve based on specified atts
70 arg = GetSearchAttributes( arg );
71
72 // default to everything
73 *mode = 0x417;
74 break;
75
76 case 'D': // don't colorize
77 glDirFlags |= DIRFLAGS_NO_COLOR;
78 break;
79
80 case 'E': // display filenames in upper case
81 glDirFlags |= DIRFLAGS_UPPER_CASE;
82 break;
83
84 case 'H': // hide "." and ".."
85 glDirFlags |= DIRFLAGS_NO_DOTS;
86 break;
87
88 case 'I': // find matching descriptions
89 if ( *arg == '"' )
90 sscanf( ++arg, "%511[^\"]", gszFindDesc );
91 else
92 strcpy( gszFindDesc, arg );
93 arg = NULLSTR;
94 break;
95
96 case 'J': // DOS justify filenames
97 glDirFlags |= DIRFLAGS_JUSTIFY;
98 break;
99
100 case 'L': // display filenames in lower case
101 glDirFlags |= DIRFLAGS_LOWER_CASE;
102 break;
103
104 case 'O': // dir sort order
105 // kludges for DOS 5 format
106 arg = dir_sort_order( arg );
107 break;
108
109 case 'T': // display attributes or time field
110
111 if ( *arg ) {
112
113 if ( *arg == ':' )
114 arg++;
115
116 switch ( _ctoupper( *arg )) {
117 case 'A': // last access
118 gnDirTimeField = 1;
119 break;
120 case 'C': // creation
121 gnDirTimeField = 2;
122 break;
123 case 'W': // last write (default)
124 gnDirTimeField = 0;
125 break;
126 default:
127 goto SelectError;
128 }
129 arg = NULLSTR;
130
131 } else
132 glDirFlags |= DIRFLAGS_SHOW_ATTS;
133 break;
134
135 case 'Z':
136 glDirFlags |= DIRFLAGS_HPFS_TO_FAT;
137 break;
138
139 default: // invalid option
140SelectError:
141 error( ERROR_INVALID_PARAMETER, arg-1 );
142 return ( usage( SELECT_USAGE ));
143 }
144 }
145 }
146
147 // skip the switch statements
148 strcpy( start_line, ((gpNthptr == NULL) ? NULLSTR : gpNthptr));
149
150 return 0;
151}

154// process the SELECT'd filenames
155static int ProcessSelect( SELECT_ARGS *sel, char *szCmdline )
156{
157 unsigned int i, n;
158 int rval = 0;
159 long lSaveDirFlags;
160 char *szFormat;
161
162 // process the marked files in order
163 for ( *szCmdline = '\0', n = 0, i = 1; ((rval != CTRLC) && (i <= sel->marked)); ) {
164
165 if ( setjmp( cv.env ) == -1) {
166 clear_screen();
167 rval = CTRLC;
168 break;
169 }
170
171 if (sel->sfiles[n].file_mark == i) {
172
173 if (sel->open_paren == '(') {
174
175 // substitute the filename & run the
176 // specified command
177
178 // kludge to save flags for "Select DIR ..."
179 lSaveDirFlags = glDirFlags;
180
181 // if LFN/HPFS names with whitespace, quote them
182 if ( glDirFlags & DIRFLAGS_HPFS ) {
183 szFormat = ((( strpbrk( sel->sfiles[n].hpfs_name, " \t,+=:" ) != 0L ) || ( strpbrk( sel->szPath, " \t,+=|<>" ) != 0L )) ? "%s\"%s%Fs\"%s" : "%s%s%Fs%s");
184 sprintf( szCmdline, szFormat, sel->cmd_start, sel->szPath, (( glDirFlags & DIRFLAGS_NT_ALT_NAME ) ? sel->sfiles[n].file_name : sel->sfiles[n].hpfs_name), sel->cmd_tail );
185 } else {
186
187 // reassemble pre-/J name
188 if (( glDirFlags & DIRFLAGS_JUSTIFY ) && ( sel->sfiles[n].file_name[8] == ' ' )) {
189
190 int j;
191
192 sel->sfiles[n].file_name[8] = '.';
193 for ( j = 8; (( j > 0 ) && ( sel->sfiles[n].file_name[j-1] == ' ' )); j-- )
194 ;
195 if ( j < 8 )
196 strcpy( sel->sfiles[n].file_name + j, sel->sfiles[n].file_name + 8 );
197 }
198
199 sprintf( szCmdline, "%s%s%Fs%s", sel->cmd_start, sel->szPath, sel->sfiles[n].file_name, sel->cmd_tail );
200 }
201
202 rval = command( szCmdline, 0 );
203
204 // restore flags (primarily for DIRFLAGS_HPFS)
205 glDirFlags = lSaveDirFlags;
206 *szCmdline = '\0';
207
208 } else {
209 // if LFN/HPFS names with whitespace, quote them
210 if (glDirFlags & DIRFLAGS_HPFS) {
211 szFormat = ((( strpbrk( sel->sfiles[n].hpfs_name, " \t,+=:" ) != 0L ) || ( strpbrk( sel->szPath, " \t,+=:" ) != 0L )) ? "%s\"%Fs\" " : "%s%Fs " );
212 sprintf( strend(szCmdline), szFormat, sel->szPath, sel->sfiles[n].hpfs_name );
213 } else
214 sprintf( strend(szCmdline), "%s%Fs ", sel->szPath, sel->sfiles[n].file_name );
215 }
216
217 } else if (++n < sel->entries)
218 continue;
219
220 // reset index & marker
221 n = 0;
222 i++;
223 }
224
225 // check for [ ] (put everything on a single line) use
226 if (szCmdline[0]) {
227 strins( szCmdline, sel->cmd_start );
228 strcat( szCmdline, sel->cmd_tail );
229 rval = command( szCmdline, 0 );
230 }
231
232 HoldSignals();
233
234 return rval;
235}
