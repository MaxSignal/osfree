51/*
52 Generate tables for a byte-wise 32-bit CRC calculation on the polynomial:
53 x^32+x^26+x^23+x^22+x^16+x^12+x^11+x^10+x^8+x^7+x^5+x^4+x^2+x+1.
54
55 Polynomials over GF(2) are represented in binary, one bit per coefficient,
56 with the lowest powers in the most significant bit. Then adding polynomials
57 is just exclusive-or, and multiplying a polynomial by x is a right shift by
58 one. If we call the above polynomial p, and represent a byte as the
59 polynomial q, also with the lowest power in the most significant bit (so the
60 byte 0xb1 is the polynomial x^7+x^3+x+1), then the CRC is (q*x^32) mod p,
61 where a mod b means the remainder after dividing a by b.
62
63 This calculation is done using the shift-register method of multiplying and
64 taking the remainder. The register is initialized to zero, and for each
65 incoming bit, x^32 is added mod p to the register if the bit is a one (where
66 x^32 mod p is p+x^32 = x^26+...+1), and the register is multiplied mod p by
67 x (which is shifting right by one and adding x^32 mod p if the bit shifted
68 out is a one). We start with the highest power (least significant bit) of
69 q and repeat for all eight bits of q.
70
71 The first (or only) table is simply the CRC of all possible eight bit values.
72 This is all the information needed to generate CRC's on data a byte-at-a-time
73 for all combinations of CRC register values and incoming bytes.
74 The remaining 3 tables (if IZ_CRCOPTIM_UNFOLDTBL is enabled) allow for
75 word-at-a-time CRC calculation, where a word is four bytes.
76*/

115local void make_crc_table()
116{
117 ulg c; /* crc shift register */
118 int n; /* counter for all possible eight bit values */
119 int k; /* byte being shifted into crc apparatus */
120#ifdef DYNALLOC_CRCTAB
121 ulg near *crctab_p; /* temporary pointer to allocated crc_table area */
122#else /* !DYNALLOC_CRCTAB */
123# define crctab_p crc_table
124#endif /* DYNALLOC_CRCTAB */
125
126#ifdef COMPUTE_XOR_PATTERN
127 /* This piece of code has been left here to explain how the XOR pattern
128 * used in the creation of the crc_table values can be recomputed.
129 * For production versions of this function, it is more efficient to
130 * supply the resultant pattern at compile time.
131 */
132 ulg xor; /* polynomial exclusive-or pattern */
133 /* terms of polynomial defining this crc (except x^32): */
134 static ZCONST uch p[] = {0,1,2,4,5,7,8,10,11,12,16,22,23,26};
135
136 /* make exclusive-or pattern from polynomial (0xedb88320L) */
137 xor = 0L;
138 for (n = 0; n < sizeof(p)/sizeof(uch); n++)
139 xor |= 1L << (31 - p[n]);
140#else
141# define xor 0xedb88320L
142#endif
143
144#ifdef DYNALLOC_CRCTAB
145 crctab_p = (ulg near *) nearmalloc (CRC_TBLS*256*sizeof(ulg));
146 if (crctab_p == NULL) {
147 ziperr(ZE_MEM, "crc_table allocation");
148 }
149#endif /* DYNALLOC_CRCTAB */
150
151 /* generate a crc for every 8-bit value */
152 for (n = 0; n < 256; n++) {
153 c = (ulg)n;
154 for (k = 8; k; k--)
155 c = c & 1 ? xor ^ (c >> 1) : c >> 1;
156 crctab_p[n] = REV_BE(c);
157 }
158
159#ifdef IZ_CRCOPTIM_UNFOLDTBL
160 /* generate crc for each value followed by one, two, and three zeros */
161 for (n = 0; n < 256; n++) {
162 c = crctab_p[n];
163 for (k = 1; k < 4; k++) {
164 c = CRC32(c, 0, crctab_p);
165 crctab_p[k*256+n] = c;
166 }
167 }
168#endif /* IZ_CRCOPTIM_UNFOLDTBL */
169
170 MARK_CRCTAB_FILLED;
171}

722#endif /* ?(IZ_CRC_BE_OPTIMIZ || IZ_CRC_LE_OPTIMIZ) */
