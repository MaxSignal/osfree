70struct pool_disk {
71 uint64_t pl_magic; /* Pool magic number */
72 uint64_t pl_pool_id; /* Unique pool identifier */
73 char pl_pool_name[POOL_NAME_SIZE]; /* Name of pool */
74 uint32_t pl_version; /* Pool version */
75 uint32_t pl_subpools; /* Number of subpools in this pool */
76 uint32_t pl_sp_id; /* Subpool number within pool */
77 uint32_t pl_sp_devs; /* Number of data partitions in this subpool */
78 uint32_t pl_sp_devid; /* Partition number within subpool */
79 uint32_t pl_sp_type; /* Partition type */
80 uint64_t pl_blocks; /* Number of blocks in this partition */
81 uint32_t pl_striping; /* Striping size within subpool */
82 /*
83 * If the number of DMEP devices is zero, then the next field **
84 * ** (pl_sp_dmepid) becomes the subpool ID for redirection. In **
85 * ** other words, if this subpool does not have the capability **
86 * ** to do DMEP, then it must specify which subpool will do it **
87 * ** in it's place
88 */
89
90 /*
91 * While the next 3 field are no longer used, they must stay to keep **
92 * ** backward compatibility...........................................
93 */
94 uint32_t pl_sp_dmepdevs;/* Number of dmep devices in this subpool */
95 uint32_t pl_sp_dmepid; /* Dmep device number within subpool */
96 uint32_t pl_sp_weight; /* if dmep dev, pref to using it */
97
98 uint32_t pl_minor; /* the pool minor number */
99 uint32_t pl_padding; /* reminder - think about alignment */
100
101 /*
102 * Even though we're zeroing out 8k at the front of the disk before
103 * writing the label, putting this in
104 */
105 char pl_reserve[184]; /* bump the structure size out to 512 bytes */
106};
