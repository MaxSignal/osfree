26static int _pvresize_single(struct cmd_context *cmd,
27 struct volume_group *vg,
28 struct physical_volume *pv,
29 void *handle)
30{
31 struct pv_list *pvl;
32 int consistent = 1;
33 uint64_t size = 0;
34 uint32_t new_pe_count = 0;
35 struct list mdas;
36 const char *pv_name = dev_name(pv_dev(pv));
37 struct pvresize_params *params = (struct pvresize_params *) handle;
38 const char *vg_name;
39
40 list_init(&mdas);
41
42 params->total++;
43
44 if (!*pv_vg_name(pv)) {
45 vg_name = ORPHAN;
46
47 if (!lock_vol(cmd, vg_name, LCK_VG_WRITE)) {
48 log_error("Can't get lock for orphans");
49 return ECMD_FAILED;
50 }
51
52 if (!(pv = pv_read(cmd, pv_name, &mdas, NULL, 1))) {
53 unlock_vg(cmd, vg_name);
54 log_error("Unable to read PV \"%s\"", pv_name);
55 return ECMD_FAILED;
56 }
57
58 /* FIXME Create function to test compatibility properly */
59 if (list_size(&mdas) > 1) {
60 log_error("%s: too many metadata areas for pvresize",
61 pv_name);
62 unlock_vg(cmd, vg_name);
63 return ECMD_FAILED;
64 }
65 } else {
66 vg_name = pv_vg_name(pv);
67
68 if (!lock_vol(cmd, vg_name, LCK_VG_WRITE)) {
69 log_error("Can't get lock for %s", pv_vg_name(pv));
70 return ECMD_FAILED;
71 }
72
73 if (!(vg = vg_read(cmd, vg_name, NULL, &consistent))) {
74 unlock_vg(cmd, vg_name);
75 log_error("Unable to find volume group of \"%s\"",
76 pv_name);
77 return ECMD_FAILED;
78 }
79
80 if (!vg_check_status(vg, CLUSTERED | EXPORTED_VG | LVM_WRITE)) {
81 unlock_vg(cmd, vg_name);
82 return ECMD_FAILED;
83 }
84
85 if (!(pvl = find_pv_in_vg(vg, pv_name))) {
86 unlock_vg(cmd, vg_name);
87 log_error("Unable to find \"%s\" in volume group \"%s\"",
88 pv_name, vg->name);
89 return ECMD_FAILED;
90 }
91
92 pv = pvl->pv;
93
94 if (!archive(vg))
95 return ECMD_FAILED;
96 }
97
98 if (!(pv->fmt->features & FMT_RESIZE_PV)) {
99 log_error("Physical volume %s format does not support resizing.",
100 pv_name);
101 unlock_vg(cmd, vg_name);
102 return ECMD_FAILED;
103 }
104
105 /* Get new size */
106 if (!dev_get_size(pv_dev(pv), &size)) {
107 log_error("%s: Couldn't get size.", pv_name);
108 unlock_vg(cmd, vg_name);
109 return ECMD_FAILED;
110 }
111
112 if (params->new_size) {
113 if (params->new_size > size)
114 log_warn("WARNING: %s: Overriding real size. "
115 "You could lose data.", pv_name);
116 log_verbose("%s: Pretending size is %" PRIu64 " not %" PRIu64
117 " sectors.", pv_name, params->new_size, pv_size(pv));
118 size = params->new_size;
119 }
120
121 if (size < PV_MIN_SIZE) {
122 log_error("%s: Size must exceed minimum of %ld sectors.",
123 pv_name, PV_MIN_SIZE);
124 unlock_vg(cmd, vg_name);
125 return ECMD_FAILED;
126 }
127
128 if (size < pv_pe_start(pv)) {
129 log_error("%s: Size must exceed physical extent start of "
130 "%" PRIu64 " sectors.", pv_name, pv_pe_start(pv));
131 unlock_vg(cmd, vg_name);
132 return ECMD_FAILED;
133 }
134
135 pv->size = size;
136
137 if (vg) {
138 pv->size -= pv_pe_start(pv);
139 new_pe_count = pv_size(pv) / vg->extent_size;
140
141 if (!new_pe_count) {
142 log_error("%s: Size must leave space for at "
143 "least one physical extent of "
144 "%" PRIu32 " sectors.", pv_name,
145 pv_pe_size(pv));
146 unlock_vg(cmd, vg_name);
147 return ECMD_FAILED;
148 }
149
150 if (!pv_resize(pv, vg, new_pe_count)) {
151 stack;
152 unlock_vg(cmd, vg_name);
153 return ECMD_FAILED;
154 }
155 }
156
157 log_verbose("Resizing volume \"%s\" to %" PRIu64 " sectors.",
158 pv_name, pv_size(pv));
159
160 log_verbose("Updating physical volume \"%s\"", pv_name);
161 if (*pv_vg_name(pv)) {
162 if (!vg_write(vg) || !vg_commit(vg)) {
163 unlock_vg(cmd, pv_vg_name(pv));
164 log_error("Failed to store physical volume \"%s\" in "
165 "volume group \"%s\"", pv_name, vg->name);
166 return ECMD_FAILED;
167 }
168 backup(vg);
169 unlock_vg(cmd, vg_name);
170 } else {
171 if (!(pv_write(cmd, pv, NULL, INT64_C(-1)))) {
172 unlock_vg(cmd, ORPHAN);
173 log_error("Failed to store physical volume \"%s\"",
174 pv_name);
175 return ECMD_FAILED;
176 }
177 unlock_vg(cmd, vg_name);
178 }
179
180 log_print("Physical volume \"%s\" changed", pv_name);
181
182 params->done++;
183
184 return 1;
185}

187int pvresize(struct cmd_context *cmd, int argc, char **argv)
188{
189 struct pvresize_params params;
190 int ret;
191
192 if (!argc) {
193 log_error("Please supply physical volume(s)");
194 return EINVALID_CMD_LINE;
195 }
196
197 if (arg_sign_value(cmd, physicalvolumesize_ARG, 0) == SIGN_MINUS) {
198 log_error("Physical volume size may not be negative");
199 return 0;
200 }
201
202 params.new_size = arg_uint64_value(cmd, physicalvolumesize_ARG,
203 UINT64_C(0)) * 2;
204
205 params.done = 0;
206 params.total = 0;
207
208 ret = process_each_pv(cmd, argc, argv, NULL, &params, _pvresize_single);
209
210 log_print("%d physical volume(s) resized / %d physical volume(s) "
211 "not resized", params.done, params.total - params.done);
212
213 return ret;
214}
