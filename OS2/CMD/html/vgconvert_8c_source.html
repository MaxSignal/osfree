18static int vgconvert_single(struct cmd_context *cmd, const char *vg_name,
19 struct volume_group *vg, int consistent,
20 void *handle) // __attribute((unused)))
21{
22 struct physical_volume *pv, *existing_pv;
23 struct logical_volume *lv;
24 struct lv_list *lvl;
25 uint64_t size = 0;
26 struct list mdas;
27 int pvmetadatacopies = 0;
28 uint64_t pvmetadatasize = 0;
29 uint64_t pe_end = 0, pe_start = 0;
30 struct pv_list *pvl;
31 int change_made = 0;
32 struct lvinfo info;
33 int active = 0;
34
35 if (!vg) {
36 log_error("Unable to find volume group \"%s\"", vg_name);
37 return ECMD_FAILED;
38 }
39
40 if (!consistent) {
41 unlock_vg(cmd, vg_name);
42 dev_close_all();
43 log_error("Volume group \"%s\" inconsistent", vg_name);
44 if (!(vg = recover_vg(cmd, vg_name, LCK_VG_WRITE)))
45 return ECMD_FAILED;
46 }
47
48 if (!vg_check_status(vg, LVM_WRITE | EXPORTED_VG))
49 return ECMD_FAILED;
50
51 if (vg->fid->fmt == cmd->fmt) {
52 log_error("Volume group \"%s\" already uses format %s",
53 vg_name, cmd->fmt->name);
54 return ECMD_FAILED;
55 }
56
57 if (cmd->fmt->features & FMT_MDAS) {
58 if (arg_sign_value(cmd, metadatasize_ARG, 0) == SIGN_MINUS) {
59 log_error("Metadata size may not be negative");
60 return EINVALID_CMD_LINE;
61 }
62
63 pvmetadatasize = arg_uint64_value(cmd, metadatasize_ARG,
64 UINT64_C(0)) * 2;
65 if (!pvmetadatasize)
66 pvmetadatasize =
67 find_config_tree_int(cmd,
68 "metadata/pvmetadatasize",
69 DEFAULT_PVMETADATASIZE);
70
71 pvmetadatacopies = arg_int_value(cmd, metadatacopies_ARG, -1);
72 if (pvmetadatacopies < 0)
73 pvmetadatacopies =
74 find_config_tree_int(cmd,
75 "metadata/pvmetadatacopies",
76 DEFAULT_PVMETADATACOPIES);
77 }
78
79 if (!archive(vg)) {
80 log_error("Archive of \"%s\" metadata failed.", vg_name);
81 return ECMD_FAILED;
82 }
83
84 /* Attempt to change any LVIDs that are too big */
85 if (cmd->fmt->features & FMT_RESTRICTED_LVIDS) {
86 list_iterate_items(lvl, struct lv_list, &vg->lvs) {
87 lv = lvl->lv;
88 if (lv->status & SNAPSHOT)
89 continue;
90 if (lvnum_from_lvid(&lv->lvid) < MAX_RESTRICTED_LVS)
91 continue;
92 if (lv_info(cmd, lv, &info, 0) && info.exists) {
93 log_error("Logical volume %s must be "
94 "deactivated before conversion.",
95 lv->name);
96 active++;
97 continue;
98 }
99 lvid_from_lvnum(&lv->lvid, &lv->vg->id, find_free_lvnum(lv));
100
101 }
102 }
103
104 if (active)
105 return ECMD_FAILED;
106
107 list_iterate_items(pvl, struct pv_list, &vg->pvs) {
108 existing_pv = pvl->pv;
109
110 pe_start = pv_pe_start(existing_pv);
111 pe_end = pv_pe_count(existing_pv) * pv_pe_size(existing_pv)
112 + pe_start - 1;
113
114 list_init(&mdas);
115 if (!(pv = pv_create(cmd->fmt, pv_dev(existing_pv),
116 &existing_pv->id, size,
117 pe_start, pv_pe_count(existing_pv),
118 pv_pe_size(existing_pv), pvmetadatacopies,
119 pvmetadatasize, &mdas))) {
120 log_error("Failed to setup physical volume \"%s\"",
121 dev_name(pv_dev(existing_pv)));
122 if (change_made)
123 log_error("Use pvcreate and vgcfgrestore to "
124 "repair from archived metadata.");
125 return ECMD_FAILED;
126 }
127
128 /* Need to revert manually if it fails after this point */
129 change_made = 1;
130
131 log_verbose("Set up physical volume for \"%s\" with %" PRIu64
132 " available sectors", dev_name(pv_dev(pv)), pv_size(pv));
133
134 /* Wipe existing label first */
135 if (!label_remove(pv_dev(pv))) {
136 log_error("Failed to wipe existing label on %s",
137 dev_name(pv_dev(pv)));
138 log_error("Use pvcreate and vgcfgrestore to repair "
139 "from archived metadata.");
140 return ECMD_FAILED;
141 }
142
143 log_very_verbose("Writing physical volume data to disk \"%s\"",
144 dev_name(pv_dev(pv)));
145 if (!(pv_write(cmd, pv, &mdas,
146 arg_int64_value(cmd, labelsector_ARG,
147 DEFAULT_LABELSECTOR)))) {
148 log_error("Failed to write physical volume \"%s\"",
149 dev_name(pv_dev(pv)));
150 log_error("Use pvcreate and vgcfgrestore to repair "
151 "from archived metadata.");
152 return ECMD_FAILED;
153 }
154 log_verbose("Physical volume \"%s\" successfully created",
155 dev_name(pv_dev(pv)));
156
157 }
158
159 log_verbose("Deleting existing metadata for VG %s", vg_name);
160 if (!vg_remove(vg)) {
161 log_error("Removal of existing metadata for %s failed.",
162 vg_name);
163 log_error("Use pvcreate and vgcfgrestore to repair "
164 "from archived metadata.");
165 return ECMD_FAILED;
166 }
167
168 /* FIXME Cache the label format change so we don't have to skip this */
169 if (test_mode()) {
170 log_verbose("Test mode: Skipping metadata writing for VG %s in"
171 " format %s", vg_name, cmd->fmt->name);
172 return ECMD_PROCESSED;
173 }
174
175 log_verbose("Writing metadata for VG %s using format %s", vg_name,
176 cmd->fmt->name);
177 if (!backup_restore_vg(cmd, vg)) {
178 log_error("Conversion failed for volume group %s.", vg_name);
179 log_error("Use pvcreate and vgcfgrestore to repair from "
180 "archived metadata.");
181 return ECMD_FAILED;
182 }
183 log_print("Volume group %s successfully converted", vg_name);
184
185 backup(vg);
186
187 return ECMD_PROCESSED;
188}

190int vgconvert(struct cmd_context *cmd, int argc, char **argv)
191{
192 if (!argc) {
193 log_error("Please enter volume group(s)");
194 return EINVALID_CMD_LINE;
195 }
196
197 if (arg_int_value(cmd, labelsector_ARG, 0) >= LABEL_SCAN_SECTORS) {
198 log_error("labelsector must be less than %lu",
199 LABEL_SCAN_SECTORS);
200 return EINVALID_CMD_LINE;
201 }
202
203 if (!(cmd->fmt->features & FMT_MDAS) &&
204 (arg_count(cmd, metadatacopies_ARG) ||
205 arg_count(cmd, metadatasize_ARG))) {
206 log_error("Metadata parameters only apply to text format");
207 return EINVALID_CMD_LINE;
208 }
209
210 if (arg_count(cmd, metadatacopies_ARG) &&
211 arg_int_value(cmd, metadatacopies_ARG, -1) > 2) {
212 log_error("Metadatacopies may only be 0, 1 or 2");
213 return EINVALID_CMD_LINE;
214 }
215
216 return process_each_vg(cmd, argc, argv, LCK_VG_WRITE, 0, NULL,
217 &vgconvert_single);
218}
