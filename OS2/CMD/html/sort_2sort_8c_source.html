46/*
47NLS is int 21.65 -> get tables... country=-1 is current... The data:
4802 - upcase table: 128 bytes of "what is that 0x80..0xff char in uppercase?"
4903 - downcase table: 256 bytes of "what is that 0x00..0xff char in lowercase?"
50 (only DOS 6.2+ with country.sys loaded, probably only codepage 866)
5104 - filename uppercase table
5205 - filename terminator table
5306 - collating: 256 "values used to sort char 0x00..0xff" (I think you can
54 use those byte values to "XLAT" (ASM) / look up (C) a number for each
55 char, and then use that number for sorting comparisons)
5607 - DBCS is somewhat different, see RBIL (double byte chars, DOS 4.0+)
57
58The returned buffer contents are NOT the tables but a byte followed by
59a far POINTER to the actual tables!
60All tables are prefixed by a word indicating the table size (128, 256...)!
61Minimum DOS version for NLS stuff is 3.3 ... For table request, give BX
62and DX as codepage / country, or -1 to use current. You must provide a
63buffer at ES:DI and tell its length in CX. On return, CY / buffer / CX set.
64Only if NLSFUNC loaded, other codepages / countries than current are avail.
65
66From DOS 4.0 on, you also have:
67Upcase char: 21.6520.dl -> dl
68Upcase array: 21.6521.ds:dx.cx (string at ds:dx, length cx)
69Upcase string: 21.6522.ds:dx (string terminated by 0 byte)
70*/

96int
97cmpr(void *a, void *b)
98{
99 unsigned char *A, *B, *C;
100
101 A = *(unsigned char **) a;
102 B = *(unsigned char **) b;
103
104 if (sortcol > 0) { /* "sort from column... " */
105 if (strlen(A) > sortcol)
106 A += sortcol;
107 else
108 A = "";
109 if (strlen(B) > sortcol)
110 B += sortcol;
111 else
112 B = "";
113 }
114
115 if (rev) { /* reverse sort: swap strings */
116 /* (or swap sign of result, of course) */
117 C = A;
118 A = B;
119 B = C;
120 }
121
122#ifdef __MSDOS__
123 if (nls) {
124 while (collate[A[0]] == collate[B[0]]) {
125 /* we use collate in the while as well: two different *
126 * bytes may have the same collate position... */
127 if (A[0] == '\0') return 0; /* both at end */
128 A++;
129 B++;
130 }
131 if (collate[A[0]] < collate[B[0]]) return -1;
132 return 1;
133 } else {
134#endif
135 return strcmp(A, B);
136#ifdef __MSDOS__
137 }
138#endif
139}

163int main(int argc, char **argv)
164{
165 char filename[MAXPATH];
166 char temp[MAXLEN + 1];
167 char *list[MAXRECORDS];
168 char *cp; /* option character pointer */
169 int nr;
170 int i;
171 /* FILE *fi; */ /* file descriptor */
172 int fi; /* file HANDLE (fopen/... are big) */
173 nl_catd cat; /* handle for localized messages (catalog) */
174
175#ifdef __MSDOS__
176 /* MAKE SURE THAT YOU USE BYTE ALIGNMENT HERE! */
177 struct NLSBUF {
178 char id;
179 unsigned char far * content;
180 } collbuf;
181 union REGS dosr;
182 struct SREGS doss;
183#endif
184
185 cat = catopen ("sort", 0);
186
187 sortcol = 0;
188 strcpy(filename, "");
189 rev = 0;
190 nls = 0;
191 while (--argc) {
192 if (*(cp = *++argv) == '/') {
193 switch (cp[1]) {
194 case 'R':
195 case 'r':
196 rev = 1;
197 break;
198#ifdef __MSDOS__
199 case 'N':
200 case 'n':
201 if ( ((_osmajor >= 3) && (_osminor >= 3)) ||
202 (_osmajor > 3) ) {
203 dosr.x.ax = 0x6506; /* get collate table */
204 dosr.x.bx = 0xffff; /* default codepage */
205 dosr.x.dx = 0xffff; /* default country */
206 dosr.x.cx = 5; /* buffer size */
207 doss.es = FP_SEG(&collbuf);
208 dosr.x.di = FP_OFF(&collbuf);
209 intdosx(&dosr,&dosr,&doss);
210 if ((dosr.x.flags & 1) == 1) {
211 WriteString(catgets(cat, 2, 1,
212/* catgets ... */ "Error reading NLS collate table\r\n"),
213 StdERR);
214 nls = 0;
215 } else {
216 /* ... CX is returned as table length ... */
217 collate = collbuf.content; /* table pointer */
218 collate++; /* skip leading word, which is */
219 collate++; /* not part of the table */
220 nls = 1;
221 }
222 } else {
223 WriteString(catgets(cat, 2, 2,
224/* catgets ... */ "Only DOS 3.3 or newer supports NLS!\r\n"),
225 StdERR);
226 }
227 break;
228#endif
229 case '?':
230 case 'h':
231 case 'H':
232 help = 1;
233 break;
234 case '+':
235 sortcol = atoi(cp + 1);
236 if (sortcol)
237 sortcol--;
238 break;
239 default:
240 err++;
241 }
242 } else { /* must be a file name */
243 strcpy(filename, *argv);
244 }
245 }
246 if (err || help) {
247 usage(cat);
248 catclose(cat);
249 exit(1);
250 }
251 fi = StdIN; /* just in case */
252 if (strlen(filename)) {
253 if ((fi = open(filename, O_RDONLY)) == -1) {
254 /* was: ... fopen(...,"r") ... == NULL ... */
255 WriteString(catgets(cat, 2, 3, "SORT: Can't open "), StdERR);
256 WriteString(filename,StdERR);
257 WriteString(catgets(cat, 2, 4, " for read\r\n"), StdERR);
258 /* avoided 1.5k-2k overhead of *printf()... */
259 catclose(cat);
260 exit(2);
261 }
262 }
263
264 (void)get_line(fi, NULL, 0); /* initialize buffers */
265
266 for (nr = 0; nr < MAXRECORDS; nr++) {
267 if (!get_line(fi, temp, MAXLEN))
268 /* was: fgets(temp, MAXLEN, fi) == NULL */
269 break;
270 list[nr] = (char *) malloc(strlen(temp) + 1);
271 /* malloc might have big overhead, but we cannot avoid it */
272 if (list[nr] == NULL) {
273 WriteString(catgets(cat, 2, 5, "SORT: Insufficient memory\r\n"),
274 StdERR);
275 catclose(cat);
276 exit(3);
277 }
278 strcpy(list[nr], temp);
279 }
280 if (nr == MAXRECORDS) {
281 WriteString(catgets(cat, 2, 6,
282 "SORT: number of records exceeds maximum\r\n"), /* catgets ... */
283 StdERR);
284 catclose(cat);
285 exit(4);
286 }
287 qsort((void *) list, nr, sizeof(char *), (cmpr_t)cmpr);
288 for (i = 0; i < nr; i++) {
289 WriteString(list[i], StdOUT);
290 WriteString("\r\n",StdOUT);
291 }
292 catclose(cat);
293 return 0;
294}
