14#include <os2.h>
15#include <acpi.h>
16#include <accommon.h>
17#include <acpiapi.h>
18#include <stdio.h>
19#include <stdlib.h>
20#include <string.h>

22void main(
23 void)
24{
25 ACPI_STATUS Status;
26 UINT32 StartTicks, EndTicks, Resolution, TimeElapsed, DeltaTicks, TimeElapsed2;
27 ACPI_TK_VERSION ApiBuffer;
28
29 if (AcpiTkValidateVersion(ACPI_TK_VERSION_MAJOR, ACPI_TK_VERSION_MINOR)) {
30 printf("This program is not compatible with the version of ACPI that is installed.\n");
31
32 /* display the version information */
33 ApiBuffer.Size = sizeof(ACPI_TK_VERSION);
34 if (AcpiTkGetVersion(&ApiBuffer)) exit(1);
35 if (ApiBuffer.Size != sizeof(ACPI_TK_VERSION)) exit(1);
36
37 printf("The version of ACPI installed is %u.%u\n", ApiBuffer.PSD.Major, ApiBuffer.PSD.Minor);
38 printf("The API version is %u.%u, but version %u.%u is needed.\n",
39 ApiBuffer.Api.Major, ApiBuffer.Api.Minor, ACPI_TK_VERSION_MAJOR, ACPI_TK_VERSION_MINOR);
40 }
41
42 Status = AcpiTkGetTimerResolution(&Resolution);
43 printf("Status 0x%x %d - %s\n",Status,Status,AcpiTkStatusToStr(Status));
44 printf("Timer\'s resolution %d bit\n",Resolution);
45
46 Status = AcpiTkGetTimer(&StartTicks);
47 printf("Status 0x%x %d - %s\n",Status,Status,AcpiTkStatusToStr(Status));
48 printf("Start ticks - %d\n Call DosSleep(1000)...\n",StartTicks);
49 DosSleep(1000);
50
51 Status = AcpiTkGetTimer(&EndTicks);
52 printf("Status 0x%x %d - %s\n",Status,Status,AcpiTkStatusToStr(Status));
53 printf("End ticks - %d\n",EndTicks);
54 /*
55 * Compute Tick Delta:
56 * Handle (max one) timer rollovers on 24-bit versus 32-bit timers.
57 */
58 if (StartTicks < EndTicks)
59 {
60 DeltaTicks = EndTicks - StartTicks;
61 }
62 else
63 if (StartTicks > EndTicks)
64 {
65 if (Resolution == 24)
66 { /* 24-bit Timer */
67 DeltaTicks = (((0x00FFFFFF - StartTicks) + EndTicks) & 0x00FFFFFF);
68 }
69 else
70 { /* 32-bit Timer */
71 DeltaTicks = (0xFFFFFFFF - StartTicks) + EndTicks;
72 }
73 }
74 else /* StartTicks == EndTicks */
75 {
76 TimeElapsed = 0;
77 TimeElapsed2 = 0;
78 }
79 /*
80 * Compute Duration (Requires a 64-bit multiply and divide):
81 *
82 * TimeElapsed = (DeltaTicks * 1000000) / PM_TIMER_FREQUENCY_SEC;
83 */
84 TimeElapsed = ( ((UINT64) DeltaTicks) * 1000000LL) / PM_TIMER_FREQUENCY_SEC;
85 printf("Time elapsed - %d mks\n",TimeElapsed);
86
87}
