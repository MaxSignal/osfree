16#include "tools.h"
17
18#define DEFAULT_EXTENT 4096 /* In KB */
19
20int vgcreate(struct cmd_context *cmd, int argc, char **argv)
21{
22 size_t max_lv, max_pv;
23 uint32_t extent_size;
24 char *vg_name;
25 struct volume_group *vg;
26 const char *tag;
27 alloc_policy_t alloc;
28 int clustered;
29
30 if (!argc) {
31 log_error("Please provide volume group name and "
32 "physical volumes");
33 return EINVALID_CMD_LINE;
34 }
35
36 if (argc == 1) {
37 log_error("Please enter physical volume name(s)");
38 return EINVALID_CMD_LINE;
39 }
40
41 vg_name = skip_dev_dir(cmd, argv[0], NULL);
42 max_lv = arg_uint_value(cmd, maxlogicalvolumes_ARG, 0);
43 max_pv = arg_uint_value(cmd, maxphysicalvolumes_ARG, 0);
44 alloc = arg_uint_value(cmd, alloc_ARG, ALLOC_NORMAL);
45
46 if (alloc == ALLOC_INHERIT) {
47 log_error("Volume Group allocation policy cannot inherit "
48 "from anything");
49 return EINVALID_CMD_LINE;
50 }
51
52 if (!(cmd->fmt->features & FMT_UNLIMITED_VOLS)) {
53 if (!max_lv)
54 max_lv = 255;
55 if (!max_pv)
56 max_pv = 255;
57 if (max_lv > 255 || max_pv > 255) {
58 log_error("Number of volumes may not exceed 255");
59 return EINVALID_CMD_LINE;
60 }
61 }
62
63 if (arg_sign_value(cmd, physicalextentsize_ARG, 0) == SIGN_MINUS) {
64 log_error("Physical extent size may not be negative");
65 return EINVALID_CMD_LINE;
66 }
67
68 if (arg_sign_value(cmd, maxlogicalvolumes_ARG, 0) == SIGN_MINUS) {
69 log_error("Max Logical Volumes may not be negative");
70 return EINVALID_CMD_LINE;
71 }
72
73 if (arg_sign_value(cmd, maxphysicalvolumes_ARG, 0) == SIGN_MINUS) {
74 log_error("Max Physical Volumes may not be negative");
75 return EINVALID_CMD_LINE;
76 }
77
78 /* Units of 512-byte sectors */
79 extent_size =
80 arg_uint_value(cmd, physicalextentsize_ARG, DEFAULT_EXTENT) * 2;
81
82 if (!extent_size) {
83 log_error("Physical extent size may not be zero");
84 return EINVALID_CMD_LINE;
85 }
86
87 if (!validate_vg_name(cmd, vg_name)) {
88 log_error("New volume group name \"%s\" is invalid", vg_name);
89 return ECMD_FAILED;
90 }
91
92 /* Create the new VG */
93 if (!(vg = vg_create(cmd, vg_name, extent_size, max_pv, max_lv, alloc,
94 argc - 1, argv + 1)))
95 return ECMD_FAILED;
96
97 if (max_lv != vg->max_lv)
98 log_warn("WARNING: Setting maxlogicalvolumes to %d "
99 "(0 means unlimited)", vg->max_lv);
100
101 if (max_pv != vg->max_pv)
102 log_warn("WARNING: Setting maxphysicalvolumes to %d "
103 "(0 means unlimited)", vg->max_pv);
104
105 if (arg_count(cmd, addtag_ARG)) {
106 if (!(tag = arg_str_value(cmd, addtag_ARG, NULL))) {
107 log_error("Failed to get tag");
108 return ECMD_FAILED;
109 }
110
111 if (!(vg->fid->fmt->features & FMT_TAGS)) {
112 log_error("Volume group format does not support tags");
113 return ECMD_FAILED;
114 }
115
116 if (!str_list_add(cmd->mem, &vg->tags, tag)) {
117 log_error("Failed to add tag %s to volume group %s",
118 tag, vg_name);
119 return ECMD_FAILED;
120 }
121 }
122
123 if (arg_count(cmd, clustered_ARG))
124 clustered = !strcmp(arg_str_value(cmd, clustered_ARG, "n"), "y");
125 else
126 /* Default depends on current locking type */
127 clustered = locking_is_clustered();
128
129 if (clustered)
130 vg->status |= CLUSTERED;
131 else
132 vg->status &= ~CLUSTERED;
133
134 if (!lock_vol(cmd, ORPHAN, LCK_VG_WRITE)) {
135 log_error("Can't get lock for orphan PVs");
136 return ECMD_FAILED;
137 }
138
139 if (!lock_vol(cmd, vg_name, LCK_VG_WRITE | LCK_NONBLOCK)) {
140 log_error("Can't get lock for %s", vg_name);
141 unlock_vg(cmd, ORPHAN);
142 return ECMD_FAILED;
143 }
144
145 if (!archive(vg)) {
146 unlock_vg(cmd, vg_name);
147 unlock_vg(cmd, ORPHAN);
148 return ECMD_FAILED;
149 }
150
151 /* Store VG on disk(s) */
152 if (!vg_write(vg) || !vg_commit(vg)) {
153 unlock_vg(cmd, vg_name);
154 unlock_vg(cmd, ORPHAN);
155 return ECMD_FAILED;
156 }
157
158 unlock_vg(cmd, vg_name);
159 unlock_vg(cmd, ORPHAN);
160
161 backup(vg);
162
163 log_print("Volume group \"%s\" successfully created", vg->name);
164
165 return ECMD_PROCESSED;
166}
