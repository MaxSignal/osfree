27// Top-level notebook dialog procedure -- for WM_INITDLG, expects mp2 to be a
28// pointer to a notebook header structure which defines the notebook
29MRESULT EXPENTRY NotebookDlgProc(HWND hDlg, ULONG msg, MPARAM mp1, MPARAM mp2)
30{
31 int nPage;
32 PNBHDR pNotebookHdr;
33
34 if (msg != WM_INITDLG)
35 pNotebookHdr = (PNBHDR)WinQueryWindowULong(hDlg, QWL_USER);
36
37 switch(msg) {
38 case WM_INITDLG:
39
40 pNotebookHdr = (PNBHDR)(PVOIDFROMMP(mp2));
41 pNotebookHdr->hwndNB = WinWindowFromID(hDlg, pNotebookHdr->ulNotebookID);
42
43 // Save upper-level dialog handle for use by page dialog code
44 pNotebookHdr->hwndNBDlg = hDlg;
45
46 // Save header pointer in window data so WM_SIZE code can use it
47 WinSetWindowULong(hDlg, QWL_USER, (ULONG)pNotebookHdr);
48
49 // Create the notebook
50 if (!CreateNotebook(pNotebookHdr)) {
51 WinDismissDlg(hDlg, 0);
52 return 0;
53 }
54
55 // Post a message to open the notebook after initialization is complete
56 WinPostMsg(hDlg, WM_OPEN_NOTEBOOK, MP0, MP0);
57 break;
58
59 case WM_OPEN_NOTEBOOK:
60
61 // Select the first page, and set the notebook size
62 TurnToFirstPage(pNotebookHdr->hwndNB);
63 SetNotebookSize(pNotebookHdr);
64 break;
65
66 case WM_CONTROL:
67 // If user selected a new page, switch to it
68 if (((ULONG)pNotebookHdr != 0L) && (SHORT1FROMMP(mp1) == pNotebookHdr->ulNotebookID) && (SHORT2FROMMP(mp1) == BKN_PAGESELECTED))
69 SetNBPage(hDlg, (PPAGESELECTNOTIFY)mp2);
70 break;
71
72 case WM_CLOSE_NOTEBOOK:
73 // clear the notebook page array dialog handles to show the pages are no longer in use
74 for (nPage = 0; (nPage < pNotebookHdr->nPages); nPage++)
75 pNotebookHdr->PageArray[nPage].hwndPageDlg = (HWND)0L;
76 WinDismissDlg(hDlg, 0);
77 break;
78
79 default:
80 return WinDefDlgProc(hDlg, msg, mp1, mp2);
81 }
82
83 return 0;
84
85}

92

99// Set the notebook and top-level dialog sizes based on the page sizes
100void SetNotebookSize(PNBHDR pNotebookHdr)
101{
102 int i;
103 RECTL rcl;
104 SWP swp;
105 HWND hwndNB = pNotebookHdr->hwndNB;
106 LONG cx, cy, px, py;
107
108 (void) memset(&rcl, 0, sizeof(RECTL));
109
110 // Get the size of a notebook page by finding the dimensions of the largest
111 // dialog.
112
113 for (i = 0; i < pNotebookHdr->nPages; i++) {
114
115 // Get the height and width of the dialog. Also fill in the id of the
116 // item that is to get focus when the dialog is brought up.
117 if (pNotebookHdr->PageArray[i].idDlg) {
118 GetDialogDimensions(pNotebookHdr->PageArray[i].idDlg, &cx, &cy, &(pNotebookHdr->PageArray[i].idFocus));
119 if (cx > rcl.xRight)
120 rcl.xRight = cx;
121 if (cy > rcl.yTop)
122 rcl.yTop = cy;
123 }
124 }
125
126 // Adjust the notebook so it is big enough for the pages
127 WinMapDlgPoints(HWND_DESKTOP, (PPOINTL) &rcl, 2, TRUE);
128 WinSendMsg(hwndNB, BKM_CALCPAGERECT, MPFROMP(&rcl), MPFROMLONG(FALSE));
129 WinSetWindowPos(hwndNB, NULLHANDLE, 0, 0, rcl.xRight-rcl.xLeft, rcl.yTop-rcl.yBottom, SWP_SIZE);
130
131 // Calculate the size required for the dialog window
132 WinQueryWindowPos(hwndNB, &swp);
133 rcl.xLeft = 0;
134 rcl.yBottom = 0;
135 rcl.xRight = swp.x + swp.cx;
136 rcl.yTop = swp.y + swp.cy;
137 WinCalcFrameRect(pNotebookHdr->hwndNBDlg, &rcl, FALSE );
138
139 // Find the centered position and size, and adjust the dialog window so
140 // it is centered, and big enough for the notebook
141 cx = rcl.xRight - rcl.xLeft;
142 cy = rcl.yTop - rcl.yBottom;
143 px = (WinQuerySysValue(HWND_DESKTOP, SV_CXSCREEN) - cx) / 2;
144 py = (WinQuerySysValue(HWND_DESKTOP, SV_CYSCREEN) - cy) / 2;
145 WinSetWindowPos(pNotebookHdr->hwndNBDlg, NULLHANDLE, px, py, cx, cy, SWP_SIZE | SWP_MOVE | SWP_SHOW | SWP_ACTIVATE);
146}

149// RETURN THE WIDTH AND HEIGHT OF A DIALOG BOX.
150BOOL GetDialogDimensions(ULONG idDlg, PLONG plCx, PLONG plCy, PULONG pidFocus)
151{
152 APIRET rc;
153 PDLGTEMPLATE pDlgTemplate = NULL;
154
155 rc = DosGetResource(0, RT_DIALOG, idDlg, (PPVOID) &pDlgTemplate);
156
157 if (!rc) {
158 PDLGTITEM pDlgItem;
159
160 // Get offset to the item table
161 pDlgItem = (PDLGTITEM)((PBYTE) pDlgTemplate + pDlgTemplate->offadlgti);
162
163 *plCx = (LONG)pDlgItem->cx;
164 *plCy = (LONG)pDlgItem->cy;
165
166 // Currently the focus item doesn't seem to be set properly so it will
167 // be hard-coded in the constant array.
168 //*pidFocus = (ULONG) (pDlgItem + pDlgTemplate->iItemFocus)->id;
169 *pidFocus = *pidFocus;
170 return TRUE;
171 }
172
173 return FALSE;
174}

177// CREATE THE NOTEBOOK WINDOW
178BOOL CreateNotebook(PNBHDR pNotebookHdr)
179{
180 int i;
181
182 // Set the page background and tab background colors to grey so they are the same as a dlg box.
183 WinSendMsg(pNotebookHdr->hwndNB, BKM_SETNOTEBOOKCOLORS, MPFROMLONG(SYSCLR_FIELDBACKGROUND), MPFROMSHORT(BKA_BACKGROUNDPAGECOLORINDEX));
184 WinSendMsg(pNotebookHdr->hwndNB, BKM_SETNOTEBOOKCOLORS, MPFROMLONG(SYSCLR_FIELDBACKGROUND), MPFROMSHORT(BKA_BACKGROUNDMAJORCOLORINDEX));
185 WinSendMsg(pNotebookHdr->hwndNB, BKM_SETNOTEBOOKCOLORS, MPFROMLONG(SYSCLR_FIELDBACKGROUND), MPFROMSHORT(BKA_BACKGROUNDMINORCOLORINDEX));
186
187 if (!SetTabDimensions(pNotebookHdr))
188 return FALSE;
189
190 // Insert all the pages into the notebook and configure them. The dialog
191 // boxes are not going to be loaded and associated with those pages yet.
192 for ( i = 0; ( i < pNotebookHdr->nPages ); i++ ) {
193 if ( !SetUpPage( pNotebookHdr, i ))
194 return FALSE;
195 }
196
197 return TRUE;
198}

201// SET UP A NOTEBOOK PAGE.
202BOOL SetUpPage(PNBHDR pNotebookHdr, INT iPage)
203{
204 ULONG ulPageId;
205 HWND hwndNB = pNotebookHdr->hwndNB;
206
207 // Insert a page into the notebook. Specify that it is to have status text
208 // and the window associated with each page will be automatically sized by
209 // the notebook according to the size of the page.
210
211 if (!(ulPageId = (ULONG) WinSendMsg(hwndNB, BKM_INSERTPAGE, NULL, MPFROM2SHORT(pNotebookHdr->PageArray[iPage].usTabType | BKA_STATUSTEXTON | BKA_AUTOPAGESIZE, BKA_LAST))))
212 return FALSE;
213
214 // Insert a pointer to this page's info into the space available
215 // in each page (its PAGE DATA that is available to the application).
216 if (!WinSendMsg(hwndNB, BKM_SETPAGEDATA, MPFROMLONG(ulPageId), MPFROMP(&(pNotebookHdr->PageArray[iPage]))))
217 return FALSE;
218
219 // Set the text into the status line.
220 if (!WinSendMsg(hwndNB, BKM_SETSTATUSLINETEXT, MPFROMP(ulPageId), MPFROMP(pNotebookHdr->PageArray[iPage].pszStatusLineText)))
221 return FALSE;
222
223 // Set the text into the tab for this page.
224 if (!WinSendMsg(hwndNB, BKM_SETTABTEXT, MPFROMLONG(ulPageId), MPFROMP(pNotebookHdr->PageArray[iPage].pszTabText)))
225 return FALSE;
226
227 return TRUE;
228}

231// SET THE DIMENSIONS OF THE NOTEBOOK TABS.
232BOOL SetTabDimensions(PNBHDR pNotebookHdr)
233{
234 HWND hwndNB = pNotebookHdr->hwndNB;
235 HPS hps = WinGetPS(hwndNB);
236 FONTMETRICS fm;
237 POINTL aptl[TXTBOX_COUNT];
238 INT i, iSize, iLongestMajText = 0, iLongestMinText = 0;
239
240 if (!hps)
241 return FALSE;
242
243 memset(&fm, 0, sizeof(FONTMETRICS));
244
245 // Calculate the height of a tab as the height of an average font character
246 // plus a margin value.
247 if (GpiQueryFontMetrics(hps, sizeof(FONTMETRICS), &fm))
248 fm.lMaxBaselineExt += (TAB_HEIGHT_MARGIN * 2);
249 else
250 fm.lMaxBaselineExt = DEFAULT_NB_TAB_HEIGHT + (TAB_HEIGHT_MARGIN * 2);
251
252 // Calculate the longest tab text for both the MAJOR and MINOR pages
253 for (i = 0; i < pNotebookHdr->nPages; i++) {
254
255 if (!GpiQueryTextBox(hps, strlen(pNotebookHdr->PageArray[i].pszTabText), pNotebookHdr->PageArray[i].pszTabText, TXTBOX_COUNT, aptl))
256 iSize = 0;
257 else
258 iSize = aptl[TXTBOX_CONCAT].x;
259
260 if (pNotebookHdr->PageArray[i].usTabType == BKA_MAJOR) {
261 if (iSize > iLongestMajText)
262 iLongestMajText = iSize;
263 } else {
264 if (iSize > iLongestMinText)
265 iLongestMinText = iSize;
266 }
267 }
268
269 WinReleasePS(hps);
270
271 // Add a margin amount to the longest tab text
272
273 if (iLongestMajText)
274 iLongestMajText += TAB_WIDTH_MARGIN;
275
276 if (iLongestMinText)
277 iLongestMinText += TAB_WIDTH_MARGIN;
278
279 // Set the tab dimensions for the MAJOR and MINOR pages. Note that the
280 // docs as of this writing say to use BKA_MAJOR and BKA_MINOR in mp2 but
281 // you really need BKA_MAJORTAB and BKA_MINORTAB.
282
283 if (iLongestMajText) {
284 if (!WinSendMsg(hwndNB, BKM_SETDIMENSIONS, MPFROM2SHORT(iLongestMajText, (SHORT)fm.lMaxBaselineExt), MPFROMSHORT(BKA_MAJORTAB)))
285 return FALSE;
286 }
287
288 // If no minor tab text, set dimensions to 0
289 if (!WinSendMsg(hwndNB, BKM_SETDIMENSIONS, (iLongestMinText ? MPFROM2SHORT(iLongestMinText, (SHORT)fm.lMaxBaselineExt) : MP0), MPFROMSHORT(BKA_MINORTAB)))
290 return FALSE;
291
292 return TRUE;
293}

296// SET THE PAGE IN THE NOTEBOOK CONTROL.
297VOID SetNBPage(HWND hDlg, PPAGESELECTNOTIFY ppsn)
298{
299 PNBPAGE pnbp;
300 HWND hPageDlg;
301
302 // Get a pointer to the page information that is associated with this page.
303 // It was stored in the page's PAGE DATA in the SetUpPage function.
304 pnbp = (PNBPAGE) WinSendMsg(ppsn->hwndBook, BKM_QUERYPAGEDATA, MPFROMLONG(ppsn->ulPageIdNew), NULL);
305
306 if (!pnbp)
307 return;
308
309 // If this is a BKA_MAJOR page and it is what this app terms a 'parent'
310 // page, that means when the user selects this page we actually want to go
311 // to its first MINOR page. So in effect the MAJOR page is just a dummy page
312 // that has a tab that acts as a placeholder for its MINOR pages. If the
313 // user is using the left arrow to scroll thru the pages and they hit this
314 // dummy MAJOR page, that means they have already been to its MINOR pages in
315 // reverse order. They would now expect to see the page before the dummy
316 // MAJOR page, so we skip the dummy page. Otherwise the user is going the
317 // other way and wants to see the first MINOR page associated with this
318 // 'parent' page so we skip the dummy page and show its first MINOR page.
319
320 if (pnbp->fParent) {
321
322 ULONG ulPageFwd, ulPageNew;
323
324 ulPageFwd = (ULONG) WinSendMsg(ppsn->hwndBook, BKM_QUERYPAGEID, MPFROMLONG(ppsn->ulPageIdNew), MPFROM2SHORT(BKA_NEXT, BKA_MINOR));
325
326 // If this is true, the user is going in reverse order
327 if (ulPageFwd == ppsn->ulPageIdCur)
328 ulPageNew = (ULONG) WinSendMsg(ppsn->hwndBook, BKM_QUERYPAGEID, MPFROMLONG(ppsn->ulPageIdNew), MPFROM2SHORT(BKA_PREV, BKA_MAJOR));
329 else
330 ulPageNew = ulPageFwd;
331
332 if (ulPageNew)
333 WinSendMsg(ppsn->hwndBook, BKM_TURNTOPAGE, MPFROMLONG(ulPageNew), NULL);
334
335 } else {
336
337 hPageDlg = (HWND) WinSendMsg(ppsn->hwndBook, BKM_QUERYPAGEWINDOWHWND, MPFROMLONG(ppsn->ulPageIdNew), NULL);
338
339 if (!hPageDlg) {
340
341 // It is time to load this dialog because the user has flipped pages
342 // to a page that hasn't yet had the dialog associated with it. We pass
343 // a pointer to the notebook handle so the dialog procedure can use it
344 // if necessary.
345 hPageDlg = WinLoadDlg(hDlg, hDlg, pnbp->pfnwpDlg, 0, pnbp->idDlg, &(ppsn->hwndBook));
346
347 if (hPageDlg) {
348
349 // Associate the dialog with the page.
350 if (!WinSendMsg(ppsn->hwndBook, BKM_SETPAGEWINDOWHWND, MPFROMP(ppsn->ulPageIdNew), MPFROMLONG(hPageDlg))) {
351 WinDestroyWindow(hPageDlg);
352 hPageDlg = NULLHANDLE;
353 }
354
355 // Save the dialog window handle in case the dialog procedure needs it
356 pnbp->hwndPageDlg = hPageDlg;
357
358 }
359 }
360 }
361
362 // Set focus to the first control in the dialog. This is not automatically
363 // done by the notebook.
364
365 if (hPageDlg && !pnbp->fParent)
366 WinSetFocus(HWND_DESKTOP, WinWindowFromID(hPageDlg, pnbp->idFocus));
367}
