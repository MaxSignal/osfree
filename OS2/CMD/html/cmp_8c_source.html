56int main(int argc, char **argv)
57{
58 char * psz;
59 int argi;
60 int argi1st = -1;
61 int i;
62 APIRET rc;
63
64
65 /*
66 * Parse commandline
67 */
68 if (argc == 1)
69 {
70 syntax();
71 return -1;
72 }
73
74 for (argi = 1; argi < argc; argi++)
75 {
76 if ((argv[argi][0] == '-')||(argv[argi][0] == '/'))
77 {
78 switch (argv[argi][1])
79 {
80 case 'h':
81 case 'H':
82 case '?':
83 case '-':
84 syntax();
85 return 0;
86
87
88 case 's':
89 case 'S':
90 case 'q':
91 case 'Q':
92 options.fQuiet = argv[argi][2] != '-';
93 break;
94
95 default:
96 // Invalid parameter
97 cmd_ShowSystemMessage(MSG_BAD_PARM1, 0);
98 return -1;
99 }
100 }
101 else if (argi1st == -1)
102 {
103 argi1st = argi;
104 }
105 else
106 {
107 /*
108 * Bind files.
109 */
110 rc = ProcessFiles(argv[argi1st], argv[argi]);
111 if (rc)
112 {
113 fprintf(stderr, "Failed to process files '%s' / '%s'. SYS%04d\n",
114 argv[argi1st], argv[argi], rc);
115 return -4;
116 }
117 argi1st = -1;
118 }
119 }
120
121
122 /*
123 * Check if no second filepattern.
124 */
125 if (argi1st != -1)
126 {
127 fprintf(stderr, "Must specify an even number of filepatterns.");
128 return -4;
129 }
130
131 return cErrors;
132}

149int ProcessFiles(const char *pszFilePattern1, const char *pszFilePattern2)
150{
151 int cPrevTotalFiles = cTotalFiles;
152 char szFilename[CCHMAXPATH];
153 char * pszFilename;
154 const char * pszPath;
155 const char * pszPath1;
156 const char * pszPath2;
157 APIRET rc;
158 ULONG cFiles = 500;
159 HDIR hDir = HDIR_CREATE;
160 char achfindbuf[1024*16];
161 PFILEFINDBUF3 pfindbuf = (PFILEFINDBUF3)&achfindbuf[0];
162
163
164 /*
165 * Extract path from pattern.
166 */
167 pszPath1 = strrchr(pszFilePattern1, '\\');
168 pszPath2 = strrchr(pszFilePattern1, '/');
169 if (pszPath1 != NULL && pszPath1 > pszPath2)
170 pszPath = pszPath1+1;
171 else if (pszPath2 != NULL && pszPath2 > pszPath1)
172 pszPath = pszPath2+1;
173 else if (pszFilePattern1[1] == ':')
174 pszPath = &pszFilePattern1[2];
175 else
176 pszPath = &pszFilePattern1[0];
177 memcpy(szFilename, pszFilePattern1, pszPath - pszFilePattern1);
178 pszFilename = &szFilename[pszPath - pszFilePattern1];
179 *pszFilename = '\0';
180
181 /*
182 * First search.
183 */
184 rc = DosFindFirst((PSZ)pszFilePattern1,
185 &hDir,
186 0,
187 pfindbuf,
188 sizeof(achfindbuf),
189 &cFiles,
190 FIL_STANDARD);
191 while (rc == NO_ERROR && cFiles > 0)
192 {
193 PFILEFINDBUF3 pCur = pfindbuf;
194
195 /*
196 * Loop thru the result.
197 */
198 while (pCur && cFiles--)
199 {
200 /*
201 * Make filename.
202 * Process the file.
203 * Maintain statistics.
204 */
205 strcpy(pszFilename, pCur->achName);
206 rc = ProcessFile(szFilename, pCur->cbFile, pszFilePattern2);
207 if (rc == 0)
208 cSuccess++;
209 else
210 cErrors++;
211 cTotalFiles++;
212
213 /*
214 * Next file.
215 */
216 pCur = pCur->oNextEntryOffset == 0
217 ? NULL
218 : (PFILEFINDBUF3)((char*)pCur + pCur->oNextEntryOffset);
219 }
220
221 /*
222 * Next chunk.
223 */
224 cFiles = 500;
225 rc = DosFindNext(hDir, pfindbuf, sizeof(achfindbuf), &cFiles);
226 }
227
228 /*
229 * Terminate find.
230 */
231 DosFindClose(hDir);
232
233 /*
234 * Warning about no files and return.
235 */
236 if (cTotalFiles == cPrevTotalFiles)
237 fprintf(stderr, "No files found matching the pattern: %s\n", pszFilePattern1);
238 return rc == ERROR_NO_MORE_FILES ? NO_ERROR : rc;
239}

251int ProcessFile(const char *pszFile1, unsigned long cbFile1, const char *pszFilePattern)
252{
253 int rc = 0;
254 FILESTATUS3 fst3;
255 int cchPattern = strlen(pszFilePattern);
256 char szFile2[CCHMAXPATH];
257 char szPattern[CCHMAXPATH];
258 const char * pszFileName;
259 const char * pszPatternName;
260
261
262 /*
263 * Correct the pattern.
264 */
265 memcpy(szPattern, pszFilePattern, cchPattern+1);
266 if (szPattern[cchPattern - 1] == '\\' || szPattern[cchPattern - 1] == '/')
267 szPattern[cchPattern++] = '*';
268 else
269 {
270 if (!DosQueryPathInfo((PSZ)pszFilePattern, FIL_STANDARD, &fst3, sizeof(fst3)))
271 {
272 if (fst3.attrFile & FILE_DIRECTORY)
273 {
274 szPattern[cchPattern++] = '\\';
275 szPattern[cchPattern++] = '*';
276 }
277 }
278 }
279 szPattern[cchPattern] = '\0';
280
281
282 /*
283 * Get the filename part of pszFile1.
284 */
285 pszFileName = strrchr(pszFile1, '\\');
286 if (!pszFileName)
287 pszFileName = strrchr(pszFile1, '/');
288 if (!pszFileName)
289 pszFileName = strchr(pszFile1, ':');
290 if (!pszFileName)
291 pszFileName = pszFile1 - 1;
292 pszFileName++;
293
294
295 /*
296 * Get the filename portion of szPattern.
297 */
298 pszPatternName = strrchr(szPattern, '\\');
299 if (!pszPatternName)
300 pszPatternName = strrchr(szPattern, '/');
301 if (!pszPatternName)
302 pszPatternName = strchr(szPattern, ':');
303 if (!pszPatternName)
304 pszPatternName = szPattern - 1;
305 pszPatternName++;
306
307
308 /*
309 * Edit the input name according to the pattern.
310 */
311 rc = DosEditName(1, (PSZ)pszFileName, (PSZ)pszPatternName, szFile2, sizeof(szFile2));
312 if (!rc)
313 {
314 /*
315 * Make fully qualified name.
316 */
317 if (pszPatternName == szPattern)
318 {
319 memmove(szFile2 + (int)pszFileName - (int)pszFile1, szFile2, strlen(szFile2) + 1);
320 memcpy(szFile2, pszFile1, (int)pszFileName - (int)pszFile1);
321 }
322 else
323 {
324 memmove(szFile2 + (int)pszPatternName - (int)szPattern, szFile2, strlen(szFile2) + 1);
325 memcpy(szFile2, szPattern, (int)pszPatternName - (int)szPattern);
326 }
327
328 /*
329 * Check for existance of target.
330 */
331 rc = DosQueryPathInfo(szFile2, FIL_STANDARD, &fst3, sizeof(fst3));
332 if (rc == NO_ERROR && !(fst3.attrFile & FILE_DIRECTORY))
333 {
334 ULONG ulAction = 0;
335 ULONG cbFile2 = fst3.cbFile;
336 HFILE hFile1 = NULLHANDLE;
337 HFILE hFile2 = NULLHANDLE;
338
339 rc = DosOpen(szFile2, &hFile2, &ulAction, 0, FILE_NORMAL,
340 OPEN_ACTION_FAIL_IF_NEW | OPEN_ACTION_OPEN_IF_EXISTS,
341 OPEN_FLAGS_SEQUENTIAL | OPEN_ACCESS_READONLY | OPEN_SHARE_DENYNONE,
342 NULL);
343 if (!rc)
344 {
345 ulAction = 0;
346 rc = DosOpen((PSZ)pszFile1, &hFile1, &ulAction, 0, FILE_NORMAL,
347 OPEN_ACTION_FAIL_IF_NEW | OPEN_ACTION_OPEN_IF_EXISTS,
348 OPEN_FLAGS_SEQUENTIAL | OPEN_ACCESS_READONLY | OPEN_SHARE_DENYNONE,
349 NULL);
350 if (!rc)
351 {
352 unsigned long cMismatches = 0;
353 unsigned long offRead = 0;
354
355 while (cbFile1 > 0 && cbFile2 > 0)
356 {
357 static char szBuffer1[0x10000];
358 static char szBuffer2[0x10000];
359 ULONG cbRead1;
360 ULONG cbRead2;
361 APIRET rc1;
362 APIRET rc2;
363
364 rc1 = DosRead(hFile1, &szBuffer1[0], min(sizeof(szBuffer1), cbFile1), &cbRead1);
365 cbFile1 -= cbRead1;
366
367 rc2 = DosRead(hFile2, &szBuffer2[0], min(sizeof(szBuffer2), cbFile2), &cbRead2);
368 cbFile2 -= cbRead2;
369
370 if (!rc1 && !rc2)
371 {
372 int cbCmp = min(cbRead1, cbRead2);
373
374 /*
375 * Check if equal using fast compare.
376 */
377 if (memcmp(&szBuffer1[0], &szBuffer2[0], cbCmp))
378 {
379 int i;
380 /*
381 * Wasn't equal. Display mismatches using slow byte by byte compare.
382 */
383 for (i = 0; i < cbCmp; i++)
384 {
385 if (szBuffer1[i] != szBuffer2[i])
386 {
387 if (!options.fQuiet)
388 {
389 if (cMismatches++ == 0)
390 printf("Mismatch comparing '%s' and '%s'.\n", pszFile1, szFile2);
391 printf("at 0x%08x 0x%02x (%c) != 0x%02 (%c)\n",
392 offRead + i,
393 szBuffer1[i], isprint(szBuffer1[i]) ? szBuffer1[i] : ' ',
394 szBuffer2[i], isprint(szBuffer2[i]) ? szBuffer2[i] : ' ');
395 }
396 else
397 cMismatches++;
398 }
399 }
400 }
401
402 /*
403 * Check if last read and size is different.
404 */
405 if (cbRead1 != cbRead2)
406 {
407 if (!options.fQuiet)
408 {
409 if (cMismatches == 0)
410 printf("Mismatch comparing '%s' and '%s'.\n", pszFile1, szFile2);
411 printf("Length differs. %d != %d\n",
412 offRead + cbRead1 + cbFile1,
413 offRead + cbRead2 + cbFile2);
414 printf("%d differences so far\n", cMismatches);
415 }
416 rc = 1;
417 break;
418 }
419
420 /*
421 * Last read?
422 */
423 if (!cbFile1 || !cbFile2)
424 {
425 if (!options.fQuiet)
426 {
427 if (cMismatches > 0)
428 {
429 if (cMismatches == 0)
430 printf("Mismatch comparing '%s' and '%s'.\n", pszFile1, szFile2);
431 printf("%d differences\n", cMismatches);
432 }
433 else
434 printf("Successfull. '%s' matches '%s'.\n", pszFile1, szFile2);
435 }
436 }
437 }
438 else
439 {
440 /*
441 * Read Error.
442 */
443 if (rc1)
444 printf("Error: failed to read from '%s'. rc=%d\n", pszFile1, rc1);
445 if (rc2)
446 printf("Error: failed to read from '%s'. rc=%d\n", szFile2, rc2);
447 rc = rc1 ? rc1 : rc2;
448 break;
449 }
450 }
451
452 DosClose(hFile1);
453
454 /*
455 * Return number of errors.
456 */
457 if (cMismatches && !rc)
458 rc = cMismatches;
459 }
460 else
461 fprintf(stderr, "(SYS%04d): Could not open file '%s'.\n", rc, pszFile1);
462
463 DosClose(hFile2);
464 }
465 else
466 fprintf(stderr, "(SYS%04d): Could not open file '%s'.\n", rc, szFile2);
467 }
468 else
469 {
470 if (rc)
471 fprintf(stderr, "(SYS%04d): File '%s' was not found.\n", rc, szFile2);
472 else
473 fprintf(stderr, "Error: Can't compare a file with a directory (%s).\n", szFile2);
474 }
475 }
476 else
477 fprintf(stderr, "Internal error DosEditName(1, '%s','%s',..) -> rc=%d\n", pszFileName, pszPatternName, rc);
478
479 return rc;
480}
