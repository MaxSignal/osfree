13// store a string; remove any previous string for the same item
14int ini_string(INIFILE *InitData, int *dataptr, char *string, int slen)
15{
16 unsigned int i;
17 int old_len, ptype, move_cnt;
18 unsigned int offset;
19 unsigned int *fixptr;
20 char *old_string;
21
22 // calculate length of previous string, change in string space
23 old_len = ((offset = (unsigned int)*dataptr) == INI_EMPTYSTR) ? 0 : (strlen((old_string = InitData->StrData + offset)) + 1);
24
25 // holler if no room
26 if ((InitData->StrUsed + slen + 1 - old_len) > InitData->StrMax)
27 return 1;
28
29 // if there is an old string in the middle of the string space, collapse
30 // it out of the way and adjust all pointers
31 if (offset != INI_EMPTYSTR) {
32
33 if ((move_cnt = InitData->StrUsed - (offset + old_len)) > 0) {
34
35 memmove(old_string, old_string + old_len, move_cnt);
36 for (i = 0; (i < guINIItemCount); i++) {
37
38 fixptr = (unsigned int *)((char *)InitData + ((char *)gaINIItemList[i].pItemData - (char *)&gaInifile));
39 ptype = (int)(INI_PTMASK & gaINIItemList[i].cParseType);
40
41 if (((ptype == INI_STR) || (ptype == INI_PATH)) && (*fixptr != INI_EMPTYSTR) && (*fixptr > offset))
42 *fixptr -= old_len;
43 }
44 }
45
46 InitData->StrUsed -= old_len;
47 }
48
49 // put new string in place and adjust space in use
50 if (slen > 0) {
51 memmove(InitData->StrData + InitData->StrUsed, string, slen + 1);
52 *dataptr = InitData->StrUsed;
53 InitData->StrUsed += (slen + 1);
54 } else
55 *dataptr = INI_EMPTYSTR;
56
57 return 0;
58}
