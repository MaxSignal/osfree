41static int _lvconvert_name_params(struct lvconvert_params *lp,
42 struct cmd_context *cmd,
43 int *pargc, char ***pargv)
44{
45 char *ptr;
46 const char *vg_name = NULL;
47
48 if (lp->snapshot) {
49 if (!*pargc) {
50 log_error("Please specify a logical volume to act as "
51 "the snapshot origin.");
52 return 0;
53 }
54
55 lp->origin = *pargv[0];
56 (*pargv)++, (*pargc)--;
57 if (!(lp->vg_name = extract_vgname(cmd, lp->origin))) {
58 log_error("The origin name should include the "
59 "volume group.");
60 return 0;
61 }
62
63 /* Strip the volume group from the origin */
64 if ((ptr = strrchr(lp->origin, (int) '/')))
65 lp->origin = ptr + 1;
66 }
67
68 if (!*pargc) {
69 log_error("Please provide logical volume path");
70 return 0;
71 }
72
73 lp->lv_name = (*pargv)[0];
74 (*pargv)++, (*pargc)--;
75
76 if (strchr(lp->lv_name, '/') &&
77 (vg_name = extract_vgname(cmd, lp->lv_name)) &&
78 lp->vg_name && strcmp(vg_name, lp->vg_name)) {
79 log_error("Please use a single volume group name "
80 "(\"%s\" or \"%s\")", vg_name, lp->vg_name);
81 return 0;
82 }
83
84 if (!lp->vg_name)
85 lp->vg_name = vg_name;
86
87 if (!validate_name(lp->vg_name)) {
88 log_error("Please provide a valid volume group name");
89 return 0;
90 }
91
92 if ((ptr = strrchr(lp->lv_name, '/')))
93 lp->lv_name = ptr + 1;
94
95 if (!apply_lvname_restrictions(lp->lv_name))
96 return_0;
97
98 return 1;
99}

101static int _read_params(struct lvconvert_params *lp, struct cmd_context *cmd,
102 int argc, char **argv)
103{
104 int region_size;
105 int pagesize = lvm_getpagesize();
106
107 memset(lp, 0, sizeof(*lp));
108
109 if (arg_count(cmd, mirrors_ARG) + arg_count(cmd, snapshot_ARG) != 1) {
110 log_error("Exactly one of --mirrors or --snapshot arguments "
111 "required.");
112 return 0;
113 }
114
115 if (arg_count(cmd, snapshot_ARG))
116 lp->snapshot = 1;
117
118 if (arg_count(cmd, mirrors_ARG)) {
119 lp->mirrors = arg_uint_value(cmd, mirrors_ARG, 0);
120 lp->mirrors_sign = arg_sign_value(cmd, mirrors_ARG, 0);
121 }
122
123 lp->alloc = ALLOC_INHERIT;
124 if (arg_count(cmd, alloc_ARG))
125 lp->alloc = arg_uint_value(cmd, alloc_ARG, lp->alloc);
126
127 if (lp->snapshot) {
128 if (arg_count(cmd, regionsize_ARG)) {
129 log_error("--regionsize is only available with mirrors");
130 return 0;
131 }
132
133 if (arg_sign_value(cmd, chunksize_ARG, 0) == SIGN_MINUS) {
134 log_error("Negative chunk size is invalid");
135 return 0;
136 }
137 lp->chunk_size = 2 * arg_uint_value(cmd, chunksize_ARG, 8);
138 if (lp->chunk_size < 8 || lp->chunk_size > 1024 ||
139 (lp->chunk_size & (lp->chunk_size - 1))) {
140 log_error("Chunk size must be a power of 2 in the "
141 "range 4K to 512K");
142 return 0;
143 }
144 log_verbose("Setting chunksize to %d sectors.", lp->chunk_size);
145
146 if (!(lp->segtype = get_segtype_from_string(cmd, "snapshot")))
147 return_0;
148
149 lp->zero = strcmp(arg_str_value(cmd, zero_ARG,
150 (lp->segtype->flags &
151 SEG_CANNOT_BE_ZEROED) ?
152 "n" : "y"), "n");
153
154 } else { /* Mirrors */
155 if (arg_count(cmd, chunksize_ARG)) {
156 log_error("--chunksize is only available with "
157 "snapshots");
158 return 0;
159 }
160
161 if (arg_count(cmd, zero_ARG)) {
162 log_error("--zero is only available with snapshots");
163 return 0;
164 }
165
166 /*
167 * --regionsize is only valid if converting an LV into a mirror.
168 * Checked when we know the state of the LV being converted.
169 */
170
171 if (arg_count(cmd, regionsize_ARG)) {
172 if (arg_sign_value(cmd, regionsize_ARG, 0) ==
173 SIGN_MINUS) {
174 log_error("Negative regionsize is invalid");
175 return 0;
176 }
177 lp->region_size = 2 * arg_uint_value(cmd,
178 regionsize_ARG, 0);
179 } else {
180 region_size = 2 * find_config_tree_int(cmd,
181 "activation/mirror_region_size",
182 DEFAULT_MIRROR_REGION_SIZE);
183 if (region_size < 0) {
184 log_error("Negative regionsize in "
185 "configuration file is invalid");
186 return 0;
187 }
188 lp->region_size = region_size;
189 }
190
191 if (lp->region_size % (pagesize >> SECTOR_SHIFT)) {
192 log_error("Region size (%" PRIu32 ") must be "
193 "a multiple of machine memory "
194 "page size (%d)",
195 lp->region_size, pagesize >> SECTOR_SHIFT);
196 return 0;
197 }
198
199 if (lp->region_size & (lp->region_size - 1)) {
200 log_error("Region size (%" PRIu32
201 ") must be a power of 2", lp->region_size);
202 return 0;
203 }
204
205 if (!lp->region_size) {
206 log_error("Non-zero region size must be supplied.");
207 return 0;
208 }
209
210 if (!(lp->segtype = get_segtype_from_string(cmd, "striped")))
211 return_0;
212 }
213
214 if (activation() && lp->segtype->ops->target_present &&
215 !lp->segtype->ops->target_present(NULL)) {
216 log_error("%s: Required device-mapper target(s) not "
217 "detected in your kernel", lp->segtype->name);
218 return 0;
219 }
220
221 if (!_lvconvert_name_params(lp, cmd, &argc, &argv))
222 return_0;
223
224 lp->pv_count = argc;
225 lp->pvs = argv;
226
227 return 1;
228}

230static int lvconvert_mirrors(struct cmd_context * cmd, struct logical_volume * lv,
231 struct lvconvert_params *lp)
232{
233 struct lv_segment *seg;
234 uint32_t existing_mirrors;
235 struct alloc_handle *ah = NULL;
236 struct logical_volume *log_lv;
237 struct list *parallel_areas;
238 struct segment_type *segtype; /* FIXME: could I just use lp->segtype */
239 float sync_percent;
240
241 seg = first_seg(lv);
242 existing_mirrors = seg->area_count;
243
244 /* Adjust required number of mirrors */
245 if (lp->mirrors_sign == SIGN_PLUS)
246 lp->mirrors = existing_mirrors + lp->mirrors;
247 else if (lp->mirrors_sign == SIGN_MINUS) {
248 if (lp->mirrors >= existing_mirrors) {
249 log_error("Logical volume %s only has %" PRIu32 " mirrors.",
250 lv->name, existing_mirrors);
251 return 0;
252 }
253 lp->mirrors = existing_mirrors - lp->mirrors;
254 } else
255 lp->mirrors += 1;
256
257 if (arg_count(cmd, regionsize_ARG) && (lv->status & MIRRORED) &&
258 (lp->region_size != seg->region_size)) {
259 log_error("Mirror log region size cannot be changed on "
260 "an existing mirror.");
261 return 0;
262 }
263
264 if ((lp->mirrors == 1)) {
265 if (!(lv->status & MIRRORED)) {
266 log_error("Logical volume %s is already not mirrored.",
267 lv->name);
268 return 1;
269 }
270
271 if (!remove_mirror_images(seg, 1,
272 lp->pv_count ? lp->pvh : NULL, 1))
273 return_0;
274 } else { /* mirrors > 1 */
275 if ((lv->status & MIRRORED)) {
276 if (list_size(&lv->segments) != 1) {
277 log_error("Logical volume %s has multiple "
278 "mirror segments.", lv->name);
279 return 0;
280 }
281 if (lp->mirrors == existing_mirrors) {
282 if (!seg->log_lv && !arg_count(cmd, corelog_ARG)) {
283 /* No disk log present, add one. */
284 if (!(parallel_areas = build_parallel_areas_from_lv(cmd, lv)))
285 return_0;
286 if (!lv_mirror_percent(cmd, lv, 0, &sync_percent, NULL)) {
287 log_error("Unable to determine mirror sync status.");
288 return 0;
289 }
290
291 segtype = get_segtype_from_string(cmd, "striped");
292
293 if (!(ah = allocate_extents(lv->vg, NULL, segtype, 0,
294 0, 1, 0,
295 NULL, 0, 0, lp->pvh,
296 lp->alloc,
297 parallel_areas))) {
298 stack;
299 return 0;
300 }
301
302 if (sync_percent >= 100.0)
303 init_mirror_in_sync(1);
304 else
305 init_mirror_in_sync(0);
306
307 if (!(log_lv = create_mirror_log(cmd, lv->vg, ah,
308 lp->alloc, lv->name,
309 (sync_percent >= 100.0) ?
310 1 : 0, &lv->tags))) {
311 log_error("Failed to create mirror log.");
312 return 0;
313 }
314 seg->log_lv = log_lv;
315 log_lv->status |= MIRROR_LOG;
316 first_seg(log_lv)->mirror_seg = seg;
317 } else if (seg->log_lv && arg_count(cmd, corelog_ARG)) {
318 /* Had disk log, switch to core. */
319 if (!lv_mirror_percent(cmd, lv, 0, &sync_percent, NULL)) {
320 log_error("Unable to determine mirror sync status.");
321 return 0;
322 }
323
324 if (sync_percent >= 100.0)
325 init_mirror_in_sync(1);
326 else
327 init_mirror_in_sync(0);
328
329 if (!remove_mirror_images(seg, lp->mirrors,
330 lp->pv_count ?
331 lp->pvh : NULL, 1))
332 return_0;
333 } else {
334 /* No change */
335 log_error("Logical volume %s already has %"
336 PRIu32 " mirror(s).", lv->name,
337 lp->mirrors - 1);
338 return 1;
339 }
340 } else if (lp->mirrors > existing_mirrors) {
341 /* FIXME Unless anywhere, remove PV of log_lv
342 * from allocatable_pvs & allocate
343 * (mirrors - existing_mirrors) new areas
344 */
345 /* FIXME Create mirror hierarchy to sync */
346 log_error("Adding mirror images is not "
347 "supported yet.");
348 return 0;
349 } else {
350 /* Reduce number of mirrors */
351 if (!remove_mirror_images(seg, lp->mirrors,
352 lp->pv_count ?
353 lp->pvh : NULL, 0))
354 return_0;
355 }
356 } else {
357 /* Make existing LV into mirror set */
358 /* FIXME Share code with lvcreate */
359
360 /* FIXME Why is this restriction here? Fix it! */
361 list_iterate_items(seg, struct lv_segment, &lv->segments) {
362 if (seg_is_striped(seg) && seg->area_count > 1) {
363 log_error("Mirrors of striped volumes are not yet supported.");
364 return 0;
365 }
366 }
367
368 if (!(parallel_areas = build_parallel_areas_from_lv(cmd, lv)))
369 return_0;
370
371 if (!(ah = allocate_extents(lv->vg, NULL, lp->segtype,
372 1, lp->mirrors - 1,
373 arg_count(cmd, corelog_ARG) ? 0 : 1,
374 lv->le_count * (lp->mirrors - 1),
375 NULL, 0, 0, lp->pvh,
376 lp->alloc,
377 parallel_areas)))
378 return_0;
379
380 lp->region_size = adjusted_mirror_region_size(lv->vg->extent_size,
381 lv->le_count,
382 lp->region_size);
383
384 log_lv = NULL;
385 if (!arg_count(cmd, corelog_ARG) &&
386 !(log_lv = create_mirror_log(cmd, lv->vg, ah,
387 lp->alloc,
388 lv->name, 0, &lv->tags))) {
389 log_error("Failed to create mirror log.");
390 return 0;
391 }
392
393 if (!create_mirror_layers(ah, 1, lp->mirrors, lv,
394 lp->segtype, 0,
395 lp->region_size,
396 log_lv))
397 return_0;
398 }
399 }
400
401 log_very_verbose("Updating logical volume \"%s\" on disk(s)", lv->name);
402
403 if (!vg_write(lv->vg))
404 return_0;
405
406 backup(lv->vg);
407
408 if (!suspend_lv(cmd, lv)) {
409 log_error("Failed to lock %s", lv->name);
410 vg_revert(lv->vg);
411 return 0;
412 }
413
414 if (!vg_commit(lv->vg)) {
415 resume_lv(cmd, lv);
416 return 0;
417 }
418
419 log_very_verbose("Updating \"%s\" in kernel", lv->name);
420
421 if (!resume_lv(cmd, lv)) {
422 log_error("Problem reactivating %s", lv->name);
423 return 0;
424 }
425
426 log_print("Logical volume %s converted.", lv->name);
427
428 return 1;
429}

431static int lvconvert_snapshot(struct cmd_context *cmd,
432 struct logical_volume *lv,
433 struct lvconvert_params *lp)
434{
435 struct logical_volume *org;
436
437 if (!(org = find_lv(lv->vg, lp->origin))) {
438 log_error("Couldn't find origin volume '%s'.", lp->origin);
439 return 0;
440 }
441
442 if (org->status & (LOCKED|PVMOVE) || lv_is_cow(org)) {
443 log_error("Unable to create a snapshot of a %s LV.",
444 org->status & LOCKED ? "locked" :
445 org->status & PVMOVE ? "pvmove" : "snapshot");
446 return 0;
447 }
448
449 if (!lp->zero || !(lv->status & LVM_WRITE))
450 log_warn("WARNING: \"%s\" not zeroed", lv->name);
451 else if (!set_lv(cmd, lv, 0, 0)) {
452 log_error("Aborting. Failed to wipe snapshot "
453 "exception store.");
454 return 0;
455 }
456
457 if (!deactivate_lv(cmd, lv)) {
458 log_error("Couldn't deactivate LV %s.", lv->name);
459 return 0;
460 }
461
462 if (!vg_add_snapshot(lv->vg->fid, NULL, org, lv, NULL, org->le_count,
463 lp->chunk_size)) {
464 log_error("Couldn't create snapshot.");
465 return 0;
466 }
467
468 /* store vg on disk(s) */
469 if (!vg_write(lv->vg))
470 return_0;
471
472 backup(lv->vg);
473
474 if (!suspend_lv(cmd, org)) {
475 log_error("Failed to suspend origin %s", org->name);
476 vg_revert(lv->vg);
477 return 0;
478 }
479
480 if (!vg_commit(lv->vg))
481 return_0;
482
483 if (!resume_lv(cmd, org)) {
484 log_error("Problem reactivating origin %s", org->name);
485 return 0;
486 }
487
488 log_print("Logical volume %s converted to snapshot.", lv->name);
489
490 return 1;
491}

493static int lvconvert_single(struct cmd_context *cmd, struct logical_volume *lv,
494 void *handle)
495{
496 struct lvconvert_params *lp = handle;
497
498 if (lv->status & LOCKED) {
499 log_error("Cannot convert locked LV %s", lv->name);
500 return ECMD_FAILED;
501 }
502
503 if (lv_is_origin(lv)) {
504 log_error("Can't convert logical volume \"%s\" under snapshot",
505 lv->name);
506 return ECMD_FAILED;
507 }
508
509 if (lv_is_cow(lv)) {
510 log_error("Can't convert snapshot logical volume \"%s\"",
511 lv->name);
512 return ECMD_FAILED;
513 }
514
515 if (lv->status & PVMOVE) {
516 log_error("Unable to convert pvmove LV %s", lv->name);
517 return ECMD_FAILED;
518 }
519
520 if (arg_count(cmd, mirrors_ARG)) {
521 if (!archive(lv->vg))
522 return ECMD_FAILED;
523 if (!lvconvert_mirrors(cmd, lv, lp))
524 return ECMD_FAILED;
525 } else if (lp->snapshot) {
526 if (!archive(lv->vg))
527 return ECMD_FAILED;
528 if (!lvconvert_snapshot(cmd, lv, lp))
529 return ECMD_FAILED;
530 }
531
532 return ECMD_PROCESSED;
533}

535int lvconvert(struct cmd_context * cmd, int argc, char **argv)
536{
537 int consistent = 1;
538 struct volume_group *vg;
539 struct lv_list *lvl;
540 struct lvconvert_params lp;
541 int ret = ECMD_FAILED;
542
543 if (!_read_params(&lp, cmd, argc, argv)) {
544 stack;
545 return EINVALID_CMD_LINE;
546 }
547
548 log_verbose("Checking for existing volume group \"%s\"", lp.vg_name);
549
550 if (!lock_vol(cmd, lp.vg_name, LCK_VG_WRITE)) {
551 log_error("Can't get lock for %s", lp.vg_name);
552 return ECMD_FAILED;
553 }
554
555 if (!(vg = vg_read(cmd, lp.vg_name, NULL, &consistent))) {
556 log_error("Volume group \"%s\" doesn't exist", lp.vg_name);
557 goto error;
558 }
559
560 if (!vg_check_status(vg, CLUSTERED | EXPORTED_VG | LVM_WRITE))
561 goto error;
562
563 if (!(lvl = find_lv_in_vg(vg, lp.lv_name))) {
564 log_error("Logical volume \"%s\" not found in "
565 "volume group \"%s\"", lp.lv_name, lp.vg_name);
566 goto error;
567 }
568
569 if (lp.pv_count) {
570 if (!(lp.pvh = create_pv_list(cmd->mem, vg, lp.pv_count,
571 lp.pvs, 1))) {
572 stack;
573 goto error;
574 }
575 } else
576 lp.pvh = &vg->pvs;
577
578 ret = lvconvert_single(cmd, lvl->lv, &lp);
579
580error:
581 unlock_vg(cmd, lp.vg_name);
582 return ret;
583}
