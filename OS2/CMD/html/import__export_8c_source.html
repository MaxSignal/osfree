32int import_pool_vg(struct volume_group *vg, struct dm_pool *mem, struct list *pls)
33{
34 struct pool_list *pl;
35
36 list_iterate_items(pl, struct pool_list, pls) {
37 vg->extent_count +=
38 ((pl->pd.pl_blocks) / POOL_PE_SIZE);
39
40 vg->pv_count++;
41
42 if (vg->name)
43 continue;
44
45 vg->name = dm_pool_strdup(mem, pl->pd.pl_pool_name);
46 get_pool_vg_uuid(&vg->id, &pl->pd);
47 vg->extent_size = POOL_PE_SIZE;
48 vg->status |= LVM_READ | LVM_WRITE | CLUSTERED | SHARED;
49 vg->free_count = vg->extent_count;
50 vg->max_lv = 1;
51 vg->max_pv = POOL_MAX_DEVICES;
52 vg->alloc = ALLOC_NORMAL;
53 vg->lv_count = 0;
54 }
55
56 return 1;
57}

59int import_pool_lvs(struct volume_group *vg, struct dm_pool *mem, struct list *pls)
60{
61 struct pool_list *pl;
62 struct lv_list *lvl = dm_pool_zalloc(mem, sizeof(*lvl));
63 struct logical_volume *lv;
64
65 if (!lvl) {
66 log_error("Unable to allocate lv list structure");
67 return 0;
68 }
69
70 if (!(lvl->lv = dm_pool_zalloc(mem, sizeof(*lvl->lv)))) {
71 log_error("Unable to allocate logical volume structure");
72 return 0;
73 }
74
75 lv = lvl->lv;
76 lv->status = 0;
77 lv->vg = vg;
78 lv->alloc = ALLOC_NORMAL;
79 lv->size = 0;
80 lv->name = NULL;
81 lv->le_count = 0;
82 lv->read_ahead = 0;
83 lv->snapshot = NULL;
84 list_init(&lv->snapshot_segs);
85 list_init(&lv->segments);
86 list_init(&lv->tags);
87
88 list_iterate_items(pl, struct pool_list, pls) {
89 lv->size += pl->pd.pl_blocks;
90
91 if (lv->name)
92 continue;
93
94 if (!(lv->name = dm_pool_strdup(mem, pl->pd.pl_pool_name))) {
95 stack;
96 return 0;
97 }
98
99 get_pool_lv_uuid(lv->lvid.id, &pl->pd);
100 log_debug("Calculated lv uuid for lv %s: %s", lv->name,
101 lv->lvid.s);
102
103 lv->status |= VISIBLE_LV | LVM_READ | LVM_WRITE;
104 lv->major = POOL_MAJOR;
105
106 /* for pool a minor of 0 is dynamic */
107 if (pl->pd.pl_minor) {
108 lv->status |= FIXED_MINOR;
109 lv->minor = pl->pd.pl_minor + MINOR_OFFSET;
110 } else {
111 lv->minor = -1;
112 }
113 lv->snapshot = NULL;
114 list_init(&lv->snapshot_segs);
115 list_init(&lv->segments);
116 list_init(&lv->tags);
117 }
118
119 lv->le_count = lv->size / POOL_PE_SIZE;
120 lvl->lv = lv;
121 list_add(&vg->lvs, &lvl->list);
122 vg->lv_count++;
123
124 return 1;
125}

128 struct list *pvs, struct dm_pool *mem, struct list *pls)
129{
130 struct pv_list *pvl;

148 list_add(pvs, &pvl->list);
149 }
150
151 return 1;

154int import_pool_pv(const struct format_type *fmt, struct dm_pool *mem,
155 struct volume_group *vg, struct physical_volume *pv,
156 struct pool_list *pl)
157{
158 struct pool_disk *pd = &pl->pd;
159
160 memset(pv, 0, sizeof(*pv));
161
162 get_pool_pv_uuid(&pv->id, pd);
163 pv->fmt = fmt;
164
165 pv->dev = pl->dev;
166 if (!(pv->vg_name = dm_pool_strdup(mem, pd->pl_pool_name))) {
167 log_error("Unable to duplicate vg_name string");
168 return 0;
169 }
170 memcpy(&pv->vgid, &vg->id, sizeof(vg->id));
171 pv->status = 0;
172 pv->size = pd->pl_blocks;
173 pv->pe_size = POOL_PE_SIZE;
174 pv->pe_start = POOL_PE_START;
175 pv->pe_count = pv->size / POOL_PE_SIZE;
176 pv->pe_alloc_count = 0;
177
178 list_init(&pv->tags);
179 list_init(&pv->segments);
180
181 if (!alloc_pv_segment_whole_pv(mem, pv)) {
182 stack;
183 return 0;
184 }
185
186 return 1;
187}

189static const char *_cvt_sptype(uint32_t sptype)
190{
191 int i;
192 for (i = 0; sptype_names[i].name[0]; i++) {
193 if (sptype == sptype_names[i].label) {
194 break;
195 }

197 log_debug("Found sptype %X and converted it to %s",

202static int _add_stripe_seg(struct dm_pool *mem,
203 struct user_subpool *usp, struct logical_volume *lv,
204 uint32_t *le_cur)
205{
206 struct lv_segment *seg;
207 struct segment_type *segtype;
208 unsigned j;
209 uint32_t area_len;
210
211 if (usp->striping & (usp->striping - 1)) {
212 log_error("Stripe size must be a power of 2");
213 return 0;
214 }
215
216 area_len = (usp->devs[0].blocks) / POOL_PE_SIZE;
217
218 if (!(segtype = get_segtype_from_string(lv->vg->cmd,
219 "striped"))) {
220 stack;
221 return 0;
222 }
223
224 if (!(seg = alloc_lv_segment(mem, segtype, lv, *le_cur,
225 area_len * usp->num_devs, 0,
226 usp->striping, NULL, usp->num_devs,
227 area_len, 0, 0, 0))) {
228 log_error("Unable to allocate striped lv_segment structure");
229 return 0;
230 }
231
232 for (j = 0; j < usp->num_devs; j++)
233 if (!set_lv_segment_area_pv(seg, j, usp->devs[j].pv, 0)) {
234 stack;
235 return 0;
236 }
237
238 /* add the subpool type to the segment tag list */
239 str_list_add(mem, &seg->tags, _cvt_sptype(usp->type));
240
241 list_add(&lv->segments, &seg->list);
242
243 *le_cur += seg->len;
244
245 return 1;
246}

248static int _add_linear_seg(struct dm_pool *mem,
249 struct user_subpool *usp, struct logical_volume *lv,
250 uint32_t *le_cur)
251{
252 struct lv_segment *seg;
253 struct segment_type *segtype;
254 unsigned j;
255 uint32_t area_len;
256
257 if (!(segtype = get_segtype_from_string(lv->vg->cmd, "striped"))) {
258 stack;
259 return 0;
260 }
261
262 for (j = 0; j < usp->num_devs; j++) {
263 area_len = (usp->devs[j].blocks) / POOL_PE_SIZE;
264
265 if (!(seg = alloc_lv_segment(mem, segtype, lv, *le_cur,
266 area_len, 0, usp->striping,
267 NULL, 1, area_len,
268 POOL_PE_SIZE, 0, 0))) {
269 log_error("Unable to allocate linear lv_segment "
270 "structure");
271 return 0;
272 }
273
274 /* add the subpool type to the segment tag list */
275 str_list_add(mem, &seg->tags, _cvt_sptype(usp->type));
276
277 if (!set_lv_segment_area_pv(seg, 0, usp->devs[j].pv, 0)) {
278 stack;
279 return 0;
280 }
281 list_add(&lv->segments, &seg->list);
282
283 *le_cur += seg->len;
284 }
285
286 return 1;
287}

289int import_pool_segments(struct list *lvs, struct dm_pool *mem,
290 struct user_subpool *usp, int subpools)
291{
292 struct lv_list *lvl;
293 struct logical_volume *lv;
294 uint32_t le_cur = 0;
295 int i;
296
297 list_iterate_items(lvl, struct lv_list, lvs) {
298 lv = lvl->lv;
299
300 if (lv->status & SNAPSHOT)
301 continue;
302
303 for (i = 0; i < subpools; i++) {
304 if (usp[i].striping) {
305 if (!_add_stripe_seg(mem, &usp[i], lv, &le_cur)) {
306 stack;
307 return 0;
308 }
309 } else {
310 if (!_add_linear_seg(mem, &usp[i], lv, &le_cur)) {
311 stack;
312 return 0;
313 }
314 }
315 }
316 }
317
318 return 1;
319}
