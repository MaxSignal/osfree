34 *
35 * Description: Since the LVM Engine Interface is very

38 * implementation of the LVM Engine Interface was divided
39 * among several C files which would have access to a
40 * core set of data. Engine.H and Engine.C define,

43 * functions described in the LVM Engine Interface.

52 * / |-------------------------------------| \
53 * / | \
54 * BootManager.C Logging.H Handle_Manager.C

56 * Notes: This module is used to maintain a copy of the original partitioning
57 * information for inclusion in a log file if logging is active. It

113/*********************************************************************/
114/* */
115/* Function Name: Log_Current_Configuration */
116/* Descriptive Name: */
117/* Input: */
118/* Output: */
119/* Error Handling: */
120/* Side Effects: */
121/* Notes: None. */
122/* */
123/*********************************************************************/
124void _System Log_Current_Configuration( void )
125{
126
127 CARDINAL32 Index; /* Used to walk the DriveArray. */
128 CARDINAL32 Error_Code; /* Used when calling the ForEachItem function. */
129
130 /* Is logging enabled? */
131 if ( Logging_Enabled )
132 {
133
134 /* Logging is enabled. Log the current configuration. */
135
136 /* Is the LVM Engine open? */
137 if ( DriveArray != NULL )
138 {
139
140 fprintf(Log_File,"\n\n*****************************************************************************\n");
141 fprintf(Log_File,"*****************************************************************************\n\n");
142
143 fprintf(Log_File," LVM Engine Status Report\n\n");
144
145 fprintf(Log_File,"\n\n*****************************************************************************\n");
146 fprintf(Log_File,"*****************************************************************************\n\n");
147
148 /* Now output the volume information for the system. */
149 fprintf(Log_File,"\n\n============================ Volume Status Report ============================\n\n");
150
151 /* Traverse the list of volumes and log each volume's data. */
152 ForEachItem(Volumes,&Log_Volumes_And_Partitions,NULL,TRUE,&Error_Code);
153
154 /* Now we will output the data on the physical drives in the system. */
155 fprintf(Log_File,"\n\n============================ Drive Status Report =============================\n\n");
156
157 /* For each drive in the drive array, output all the information that has been gathered so far. */
158 for (Index = 0; Index < DriveCount; Index++ )
159 {
160
161 fprintf(Log_File,"\n\n+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++\n\n");
162
163 fprintf(Log_File," Drive Information for drive %d.\n\n",Index + 1);
164
165 fprintf(Log_File,"Drive Array Index: %lu\n",DriveArray[Index].DriveArrayIndex);
166 fprintf(Log_File,"External Handle: %lx\n",(CARDINAL32) DriveArray[Index].External_Handle);
167 fprintf(Log_File,"Drive Size ( in sectors ): %lu\n",DriveArray[Index].Drive_Size);
168 fprintf(Log_File,"Sectors per Cylinder: %lu\n",DriveArray[Index].Sectors_Per_Cylinder);
169 fprintf(Log_File,"1024 Cylinder Limit LBA: %lu\n",DriveArray[Index].Cylinder_Limit);
170 fprintf(Log_File,"Drive Serial Number assigned by LVM: %lu\n",DriveArray[Index].Drive_Serial_Number);
171 fprintf(Log_File,"Boot Drive Serial Number: %lu\n",DriveArray[Index].Boot_Drive_Serial_Number);
172 fprintf(Log_File,"Installation Flags ( only valid for drive 1 ): %lx\n",DriveArray[Index].Install_Flags);
173 fprintf(Log_File,"Number of primary partitions on this drive: %lu\n",DriveArray[Index].Primary_Partition_Count);
174 fprintf(Log_File,"Number of logical partitions on this drive: %lu\n",DriveArray[Index].Logical_Partition_Count);
175 fprintf(Log_File,"Drive Geometry: %u Cylinders, %u Heads, %u Sectors per track.\n", DriveArray[Index].Geometry.Cylinders, DriveArray[Index].Geometry.Heads, DriveArray[Index].Geometry.Sectors);
176 fprintf(Log_File,"Drive Name: %#20.20s\n",DriveArray[Index].Drive_Name);
177
178 fprintf(Log_File,"\nDrive Attributes:\n");
179
180 if ( DriveArray[Index].Cylinder_Limit_Applies )
181 fprintf(Log_File," The 1024 Cylinder Limit applies.\n");
182 else
183 fprintf(Log_File," The 1024 Cylinder Limit does NOT apply.\n");
184
185 if ( DriveArray[Index].ChangesMade )
186 fprintf(Log_File," Changes have been made to this drive.\n");
187 else
188 fprintf(Log_File," This drive has not been altered yet.\n");
189
190 if ( DriveArray[Index].IO_Error )
191 fprintf(Log_File," There have been I/O errors on this drive!\n");
192 else
193 fprintf(Log_File," This drive has had NO I/O errors.\n");
194
195 if ( DriveArray[Index].Corrupt )
196 {
197 if( DriveArray[Index].NonFatalCorrupt)
198 { fprintf(Log_File," The partition tables on this drive seems nonfatal corrupt!\n");
199 } else {
200 fprintf(Log_File," The partition tables on this drive are corrupt!\n");
201 }
202 }
203 else
204 fprintf(Log_File," The partition tables on this drive are valid.\n");
205
206 if ( DriveArray[Index].Record_Initialized )
207 fprintf(Log_File," This drive array entry has been fully initialized.\n");
208 else
209 fprintf(Log_File," This drive array entry has NOT been fully initialized.\n");
210
211 if ( DriveArray[Index].Is_PRM )
212 fprintf(Log_File," This drive is a PRM.\n");
213 else
214 fprintf(Log_File," This drive is NOT a PRM.\n");
215
216 if ( DriveArray[Index].Is_Big_Floppy )
217 fprintf(Log_File," This drive is formatted as a big floppy.\n");
218 else
219 fprintf(Log_File," This drive is NOT formatted as a big floppy.\n");
220
221 if ( DriveArray[Index].Reboot_Flag )
222 fprintf(Log_File," The Reboot_Flag used by Install is set.\n");
223 else
224 fprintf(Log_File," The Reboot_Flag used by Install is clear.\n");
225
226 if ( DriveArray[Index].Unusable )
227 fprintf(Log_File," This drive is UNUSABLE due to I/O errors on the MBR!\n");
228 else
229 fprintf(Log_File," This drive is usable.\n");
230
231 fprintf(Log_File,"\nThe partitions on this drive are as follows: \n");
232
233 /* Now output the partition information for this drive. */
234 if ( DriveArray[Index].Partitions != NULL)
235 ForEachItem(DriveArray[Index].Partitions, &Log_Volumes_And_Partitions, NULL, TRUE, &Error_Code);
236
237 }
238
239 /* Now indicate that our LVM Engine Status Report is complete. */
240 fprintf(Log_File,"\n\n***********************************************************\n");
241 fprintf(Log_File,"###############################################################\n\n");
242
243 fprintf(Log_File," End of LVM Engine Status Report\n\n");
244
245 fprintf(Log_File,"\n\n###########################################################\n");
246 fprintf(Log_File,"***************************************************************\n\n");
247
248 fflush(Log_File);
249
250 }
251 else
252 fprintf(Log_File,"\n\nThe LVM Engine is currently CLOSED.\n\n");
253
254 }
255
256 return;
257
258}

300/*********************************************************************/
301/* */
302/* Function Name: Start_Logging */
303/* */
304/* Descriptive Name: Enables the LVM Engine logging. Once enabled,*/
305/* the LVM Engine logging function will log all */
306/* LVM Engine activity to the specified log file.*/
307/* The data is logged in a binary format for */
308/* compactness and speed. */
309/* */
310/* Input: char * Filename - The filename of the file to use as the */
311/* log file. */
312/* CARDINAL32 * Error_Code - The address of a CARDINAL32 in */
313/* in which to store an error code*/
314/* should an error occur. */
315/* int level - debug level: 0-full, 1- partial */
316/* */
317/* Output: If the logging file was successfully created, then */
318/* *Error_Code will be 0. If the log file could not be */
319/* created, then *Error_Code will be > 0. */
320/* */
321/* Error Handling: If the log file can not be created, then */
322/* *Error_Code will be > 0. */
323/* */
324/* Side Effects: A file may be created/opened for logging of */
325/* LVM Engine actions. */
326/* */
327/* Notes: None. */
328/* */
329/*********************************************************************/
330void Start_Logging( char * Filename, CARDINAL32 * Error_Code, int level )
331{
332
333 time_t Current_Time; /* Used to get the current time. */
334 int IO_Result; /* Used to hold the return value from fprintf. */
335
336 /* Try to open the log file. */
337 Log_File = fopen(Filename, "a");
338
339 /* Did we succeed? */
340 if ( Log_File == (FILE *) NULL )
341 {
342
343 /* We can not open the log file! Indicate the error and abort. */
344 *Error_Code = LVM_ENGINE_CAN_NOT_OPEN_LOG_FILE;
345
346 return;
347
348 }
349
350 /* Disable buffering of the log file. */
351 setbuf(Log_File,NULL);
352
353 /* Since the log file was created successfully, indicate that logging is enabled and return to caller. */
354 Logging_Enabled = level+1;
355
356
357 /* Now write out a message to mark the beginning of a new section in the log file. This is to cover the
358 case where we are appending to an existing log file. This allows us to distinguish where one log
359 ends and another begins. */
360
361
362 if ( ( fprintf(Log_File, "\n\n****************************************************************\n") < 0 ) ||
363 ( fprintf(Log_File, "* ALVM vers %s Log File *\n", ALVM_VERSION) < 0 ) ||
364 ( fprintf(Log_File, "*****************************************************************\n\n") < 0 ) ||
365 ( fprintf(Log_File, "This log file was created on ") < 0 )
366 )
367 {
368
369 /* We had an error writing to the output file! Close the output file and abort! */
370 fclose(Log_File);
371
372 /* Disable logging. */
373 Logging_Enabled = 0;
374
375 /* Set an error code. */
376 *Error_Code = LVM_ENGINE_CAN_NOT_WRITE_TO_LOG_FILE;
377
378 return;
379
380 }
381
382 /* Now output the date and time at which the log file was created. */
383
384 if ( time( &Current_Time ) != (time_t) -1 )
385 {
386
387 IO_Result = fprintf(Log_File,"%s\n\n", ctime(&Current_Time) );
388
389 }
390 else
391 {
392
393 IO_Result = fprintf(Log_File,"< Date and Time not available! >\n\n");
394
395 }
396
397 /* Did we succeed in writing the date and time to the log file? */
398 if ( IO_Result < 0 )
399 {
400
401 /* We had an error writing to the output file! Close the output file and abort! */
402 fclose(Log_File);
403
404 /* Disable logging. */
405 Logging_Enabled = FALSE;
406
407 /* Set an error code. */
408 *Error_Code = LVM_ENGINE_CAN_NOT_WRITE_TO_LOG_FILE;
409
410 return;
411
412 }
413
414 /* If the LVM Engine is already open, then log its current state. */
415 if ( DriveArray != NULL )
416 {
417
418 /* Indicate in the log file that what we are going to be putting in the log is not the initial LVM Engine state but
419 the state of the LVM Engine as we found it since it was already open when logging started. */
420 if ( fprintf(Log_File,"The LVM Engine was opened prior to the start of logging.\n\nThe current state of the LVM Engine follows: \n\n") >= 0 )
421 Log_Current_Configuration();
422 else
423 {
424
425 /* We had an error writing to the output file! Close the output file and abort! */
426 fclose(Log_File);
427
428 /* Disable logging. */
429 Logging_Enabled = 0;
430
431 /* Set an error code. */
432 *Error_Code = LVM_ENGINE_CAN_NOT_WRITE_TO_LOG_FILE;
433
434 return;
435
436 }
437
438 }
439
440 /* All done. Indicate success and return to caller. */
441 *Error_Code = LVM_ENGINE_NO_ERROR;
442
443 return;
444
445}

448/*********************************************************************/
449/* */
450/* Function Name: Stop_Logging */
451/* */
452/* Descriptive Name: This function ends LVM Engine logging and */
453/* closes the log file. */
454/* */
455/* Input: CARDINAL32 * Error_Code - The address of a CARDINAL32 in */
456/* in which to store an error code*/
457/* should an error occur. */
458/* */
459/* Output: *Error_Code will be 0 if this function completes */
460/* successfully; otherwise it will be > 0. */
461/* */
462/* Error Handling: If the log file is not currently opened, or if */
463/* the close operation fails on the log file, then */
464/* *Error_Code will be > 0. */
465/* */
466/* Side Effects: The log file may be closed. */
467/* */
468/* Notes: None. */
469/* */
470/*********************************************************************/
471void Stop_Logging ( CARDINAL32 * Error_Code )
472{
473
474 /* Is logging enabled? */
475 if ( Logging_Enabled )
476 {
477
478 /* Indicate that logging has been disabled. */
479 Logging_Enabled = 0;
480
481 /* Close the log file. */
482 *Error_Code = fclose(Log_File);
483
484 /* We don't care about whether or not the close was successful, so indicate success and exit. */
485 *Error_Code = LVM_ENGINE_NO_ERROR;
486
487 }
488
489 return;
490
491}

501/* */
502/* Function Name: Log_Volumes_And_Partitions */

504/* Input: */

510/*********************************************************************/
511void _System Log_Volumes_And_Partitions(ADDRESS Object, TAG ObjectTag, CARDINAL32 ObjectSize, ADDRESS ObjectHandle, ADDRESS Parameters, CARDINAL32 * Error_Code)
512{
513
514 /* Declare a local variable so that we can access the Partition_Data without having to typecast each time. */
515 Partition_Data * PartitionRecord = (Partition_Data *) Object;
516
517 /* Declare a local variable so that we can access the Volume_Data without having to typecast each time. */
518 Volume_Data * VolumeRecord = ( Volume_Data *) Object;
519
520 /* Assume success. */
521 *Error_Code = DLIST_SUCCESS;
522
523 /* Lets see what we have to work with! */
524 switch ( ObjectTag )
525 {
526 case PARTITION_DATA_TAG : /* We must log the contents of this partition record. */
527 fprintf(Log_File,"\n\n----------------------------------PARTITION----------------------------------\n\n");
528
529 fprintf(Log_File,"This Partition's External Handle: %lx (HEX)\n",PartitionRecord->External_Handle);
530 fprintf(Log_File,"This Partition's Drive Index: %lu (DEC)\n",PartitionRecord->Drive_Index);
531 fprintf(Log_File,"This Partition's Internal Handle: %lx (HEX)\n",PartitionRecord->Drive_Partition_Handle);
532 fprintf(Log_File,"The External Handle of the Volume this partition belongs to: %lx (HEX)\n",PartitionRecord->External_Volume_Handle);
533 fprintf(Log_File,"The Internal Handle of the Volume this partition belongs to: %lx (HEX)\n",PartitionRecord->Volume_Handle);
534 fprintf(Log_File,"The LBA of the first sector of this partition: %lx (HEX)\n",PartitionRecord->Starting_Sector);
535 fprintf(Log_File,"The size of this partition, in sectors: %lu (DEC)\n",PartitionRecord->Partition_Size);
536 fprintf(Log_File,"The usable size of this partition, in sectors: %lu (DEC)\n",PartitionRecord->Usable_Size);
537 fprintf(Log_File,"The name of this partition is: %#20.20s\n",PartitionRecord->Partition_Name);
538 fprintf(Log_File,"The name of the filesystem used on this partition is: %#20.20s\n",PartitionRecord->File_System_Name);
539
540 fprintf(Log_File,"\nThe partition table entry for this partition contains the following:\n");
541 fprintf(Log_File," The Boot Indicator is: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Boot_Indicator);
542 fprintf(Log_File," The Starting Head field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Starting_Head);
543 fprintf(Log_File," The Starting Sector field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Starting_Sector);
544 fprintf(Log_File," The Starting Cylinder field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Starting_Cylinder);
545 fprintf(Log_File," The Format Indicator is: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Format_Indicator);
546 fprintf(Log_File," The Ending Head field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Ending_Head);
547 fprintf(Log_File," The Ending Sector field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Ending_Sector);
548 fprintf(Log_File," The Ending Cylinder field contains: %lx (HEX)\n", (CARDINAL32) PartitionRecord->Partition_Table_Entry.Ending_Cylinder);
549 fprintf(Log_File," The Sector Offset is: %lu (DEC)\n", PartitionRecord->Partition_Table_Entry.Sector_Offset);
550 fprintf(Log_File," The Sector Count is: %lu (DEC)\n", PartitionRecord->Partition_Table_Entry.Sector_Count);
551
552 fprintf(Log_File,"\nThe DLA Table Entry for this partition contains the following:\n");
553 fprintf(Log_File," Volume Serial Number = %lu (DEC)\n",PartitionRecord->DLA_Table_Entry.Volume_Serial_Number);
554 fprintf(Log_File," Partition Serial Number = %lu (DEC)\n",PartitionRecord->DLA_Table_Entry.Partition_Serial_Number);
555 fprintf(Log_File," Partition Size = %lu (DEC)\n",PartitionRecord->DLA_Table_Entry.Partition_Size);
556 fprintf(Log_File," The LBA of the first sector of the partition is %lu (DEC)\n",PartitionRecord->DLA_Table_Entry.Partition_Start);
557 fprintf(Log_File," The drive letter assigned to this partition: %c\n",PartitionRecord->DLA_Table_Entry.Drive_Letter);
558 fprintf(Log_File," Volume Name = %#20.20s\n",PartitionRecord->DLA_Table_Entry.Volume_Name);
559 fprintf(Log_File," Partition Name = %#20.20s\n",PartitionRecord->DLA_Table_Entry.Partition_Name);
560
561 if ( PartitionRecord->DLA_Table_Entry.On_Boot_Manager_Menu )
562 fprintf(Log_File," On Boot Manager Menu is set to TRUE.\n");
563 else
564 fprintf(Log_File," On Boot Manager Menu is set to FALSE.\n");
565
566 if ( PartitionRecord->DLA_Table_Entry.Installable )
567 fprintf(Log_File," Installable is set to TRUE.\n");
568 else
569 fprintf(Log_File," Installable set to FALSE.\n");
570
571
572 fprintf(Log_File,"\nThis partition has the following attributes:\n");
573
574 if ( PartitionRecord->Partition_Type == FreeSpace )
575 fprintf(Log_File," This partition record represents free space on the drive.\n");
576 else
577 if ( PartitionRecord->Partition_Type == Partition )
578 fprintf(Log_File," This partition record represents a partition on the drive.\n");
579 else
580 if ( PartitionRecord->Partition_Type == MBR_EBR )
581 fprintf(Log_File," This partition record represents an MBR or EBR.\n");
582 else
583 fprintf(Log_File," THE PARTITION_TYPE FIELD FOR THIS PARTITION RECORD IS INVALID!\n");
584
585 if ( PartitionRecord->New_Partition )
586 fprintf(Log_File," This partition record represents a new partition.\n");
587 else
588 fprintf(Log_File," This partition record represents a pre-existing partition.\n");
589
590 if ( PartitionRecord->Primary_Partition )
591 fprintf(Log_File," This partition record represents a primary partition.\n");
592 else
593 fprintf(Log_File," This partition record represents a logical drive.\n");
594
595 if ( PartitionRecord->Spanned_Volume)
596 fprintf(Log_File," This partition is part of a spanned volume.\n");
597 else
598 fprintf(Log_File," This partition is NOT part of a spanned volume.\n");
599
600 if ( PartitionRecord->Migration_Needed )
601 fprintf(Log_File," This partition was placed on the Boot Manager menu by FDISK and needed to be migrated.\n");
602 else
603 fprintf(Log_File," This partition does NOT need to be migrated by LVM.\n");
604
605 /* Is this partition record represent a real partition, or is it a composite? */
606 if ( ( PartitionRecord->Feature_Data != NULL ) &&
607 ( PartitionRecord->Drive_Index == (CARDINAL32) -1L )
608 )
609 {
610
611 fprintf(Log_File,"\n\nThis partition record is a composite. It represents the following partitions:\n\n");
612
613 ForEachItem(PartitionRecord->Feature_Data->Partitions,&Log_Volumes_And_Partitions,NULL,TRUE,Error_Code);
614
615 }
616
617 break; /* Keep the compiler happy. */
618
619 case VOLUME_DATA_TAG : /* We must log the contents of this volume record. */
620 fprintf(Log_File,"\n\n-------------------------------------VOLUME----------------------------------\n\n");
621
622 fprintf(Log_File,"The Serial Number of this Volume is: %lu (DEC)\n",VolumeRecord->Volume_Serial_Number);
623 fprintf(Log_File,"The external handle of this Volume is: %lx (HEX)\n",VolumeRecord->External_Handle);
624 fprintf(Log_File,"The internal handle of this volume is: %lx (HEX)\n",VolumeRecord->Volume_Handle);
625 fprintf(Log_File,"The size of this volume (in sectors) is: %lu (DEC)\n",VolumeRecord->Volume_Size);
626 fprintf(Log_File,"This volume is comprised of %lu (DEC) partitions.\n",VolumeRecord->Partition_Count);
627 fprintf(Log_File,"The Drive Letter Conflict Code for this volume is: %lu (DEC)\n",VolumeRecord->Drive_Letter_Conflict);
628 fprintf(Log_File,"The name of this volume is: %.20s\n",VolumeRecord->Volume_Name);
629 fprintf(Log_File,"The filesystem used on this volume is: %.20s\n",VolumeRecord->File_System_Name);
630 fprintf(Log_File,"The drive letter preference for this volume is: %c\n",VolumeRecord->Drive_Letter_Preference);
631 fprintf(Log_File,"The current drive letter for this volume is: %c\n",VolumeRecord->Current_Drive_Letter);
632 fprintf(Log_File,"The initial drive letter for this volume is: %c\n",VolumeRecord->Initial_Drive_Letter);
633 fprintf(Log_File,"The device type for this volume is %u (DEC)\n", VolumeRecord->Device_Type);
634 fprintf(Log_File,"\nThis Volume has the following attributes:\n");
635
636 if ( VolumeRecord->New_Volume)
637 fprintf(Log_File," This is a newly created volume.\n");
638 else
639 fprintf(Log_File," This is a prior existing volume.\n");
640
641 if ( VolumeRecord->Can_Be_Altered )
642 fprintf(Log_File," This volume can be altered.\n");
643 else
644 fprintf(Log_File," This volume can NOT be altered.\n");
645
646 if ( VolumeRecord->Compatibility_Volume )
647 fprintf(Log_File," This is a Compatibility Volume.\n");
648 else
649 fprintf(Log_File," This is an LVM Volume.\n");
650
651 if (VolumeRecord->Foreign_Volume)
652 fprintf(Log_File," This is a foreign volume.\n");
653 else
654 fprintf(Log_File," This is NOT a foreign volume.\n");
