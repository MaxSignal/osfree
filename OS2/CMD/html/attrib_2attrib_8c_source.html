248{

543int ModifyFileAttribs(char *file,ATTRIB *flags)
544{
545 FILESTATUS3 fileStatus;
546 APIRET rc;
547
548 rc=DosQueryPathInfo(file,FIL_STANDARD,&fileStatus,sizeof(FILESTATUS3));
549
550 if (rc) return rc;
551
552 if ((flags->bReadOnly==ATR_ACTION_IGN) &&
553 (flags->bHidden==ATR_ACTION_IGN) &&
554 (flags->bSystem==ATR_ACTION_IGN) &&
555 (flags->bArchived==ATR_ACTION_IGN))
556 {
557 if ((fileStatus.attrFile&FILE_READONLY)==FILE_READONLY) printf("R"); else printf("_");
558 if ((fileStatus.attrFile&FILE_HIDDEN)==FILE_HIDDEN) printf("H"); else printf("_");
559 if ((fileStatus.attrFile&FILE_SYSTEM)==FILE_SYSTEM) printf("S"); else printf("_");
560 if ((fileStatus.attrFile&FILE_ARCHIVED)==FILE_ARCHIVED) printf("A"); else printf("_");
561 if ((fileStatus.attrFile&FILE_DIRECTORY)==FILE_DIRECTORY) printf("D"); else printf("_");
562 printf(" %s\n", file);
563 } else {
564
565
566 /* modify it's attributes according to flags */
567
568 /* Hidden */
569 MODIFYATTRS(flags->bHidden,FILE_HIDDEN);
570
571 /* Archived */
572 MODIFYATTRS(flags->bArchived,FILE_ARCHIVED);
573
574 /* System */
575 MODIFYATTRS(flags->bSystem,FILE_SYSTEM);
576
577 /* Readonly */
578 MODIFYATTRS(flags->bReadOnly,FILE_READONLY);
579
580 rc=DosSetPathInfo(file,FIL_STANDARD,&fileStatus,sizeof(FILESTATUS3),
581 DSPI_WRTTHRU);
582 }
