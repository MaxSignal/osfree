135/*
136 * build the help text_line array from the help message file entry.
137 * caller frees memory.
138 * displays a default help panel if error.
139 *
140 */
141
142PRIVATE
143void *
144get_help_message ( uint message_number )
145{
146 register
147 char *line;
148 char *memory,
149 *text_line;
150 uint rc,
151 index,
152 length;
153 bool no_help = TRUE;
154
155 memory = malloc ( MAX_HELP_MESSAGE_SIZE );
156
157 if ( memory ) {
158 rc = DosGetMessage ( NULL, 0, memory, MAX_HELP_MESSAGE_SIZE - 1,
159 message_number, HELP_MESSAGE_FILE, &length );
160
161 if ( rc == 0 ) {
162 no_help = FALSE;
163 memory [ length ] = '\0';
164 text_line = memory;
165 index = 0;
166 for ( line = memory; *line; ++line ) {
167 if ( *line == '\r' ) {
168 *line = ' ';
169 } else if ( *line == '\n' ) {
170 *line = '\0';
171 if ( index < MAX_HELP_TEXT_LINES ) {
172 Help_panel_text [ index ] = text_line;
173 ++index;
174 text_line = line + 1;
175 }
176 }
177 }
178 Help_panel_text [ index ] = NULL;
179 }
180 }
181
182 if ( no_help ) { /* no help available */
183 Help_panel_text [ 0 ] = " ";
184 Help_panel_text [ 1 ] = No_help_available;
185 Help_panel_text [ 2 ] = " ";
186 Help_panel_text [ 3 ] = 0;
187 }
188
189 return memory;
190}

193/*
194 * DoHelpPanel shows the panel and acts on input events
195 *
196 */
197
198PUBLIC
199void _System
200DoHelpPanel ( register
201 panel_t *panel )
202{
203 uint message_number,
204 save_control;
205 void *memory;
206
207 message_number = panel->help.default_message;
208
209 if ( panel->help.array ) {
210 if ( panel->choice < panel->help.array_size ) {
211 message_number = panel->help.array [ panel->choice ];
212 }
213 }
214
215 memory = get_help_message ( message_number );
216
217 Help_panel.control |= NOT_SIZED;
218 Help_panel.box.row_length = 0;
219 SetPanelSize ( &Help_panel );
220 Help_panel.box.column = SCREEN_WIDTH - min( Help_panel.box.row_length, SCREEN_WIDTH );
221
222 save_control = Message_bar_panel.control;
223 Message_bar_panel.control |= SAVE_RESTORE;
224
225 MESSAGE_BAR_PANEL ( Help_message_line );
226 ScrollingPanel ( &Help_panel );
227
228 ErasePanel ( &Help_panel );
229 ErasePanel ( &Message_bar_panel );
230 Message_bar_panel.control = save_control;
231
232 if ( memory ) {
233 free ( memory );
234 }
235
236}
